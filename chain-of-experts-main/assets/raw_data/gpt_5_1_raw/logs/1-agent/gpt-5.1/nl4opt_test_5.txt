import pulp

# Define the problem
prob = pulp.LpProblem("Flooring_Production_Optimization", pulp.LpMaximize)

# Decision variables: square feet of hardwood (H) and vinyl (V)
H = pulp.LpVariable('Hardwood', lowBound=0)
V = pulp.LpVariable('Vinyl', lowBound=0)

# Objective function: maximize profit
prob += 2.5 * H + 3 * V, "Total_Profit"

# Constraints
prob += H >= 20000, "Min_Hardwood_Demand"
prob += V >= 10000, "Min_Vinyl_Demand"
prob += H + V >= 60000, "Min_Total_Shipment"
prob += H <= 50000, "Max_Hardwood_Production"
prob += V <= 30000, "Max_Vinyl_Production"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Get results
H_opt = pulp.value(H)
V_opt = pulp.value(V)
profit_opt = pulp.value(prob.objective)

# Print detailed solution (optional)
print(f"Optimal Hardwood Production (sq ft): {H_opt}")
print(f"Optimal Vinyl Production (sq ft): {V_opt}")
print(f"Maximum Weekly Profit: {profit_opt}")

# Required final objective value print
print(f"FINAL_OBJ: {profit_opt}")
