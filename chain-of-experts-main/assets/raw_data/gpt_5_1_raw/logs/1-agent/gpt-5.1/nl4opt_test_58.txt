import pulp

# Define the problem
prob = pulp.LpProblem("Minimize_Medication_Time", pulp.LpMinimize)

# Decision variables: number of calcium pills (c) and vitamin D pills (v)
c = pulp.LpVariable('calcium_pills', lowBound=0, cat='Integer')
v = pulp.LpVariable('vitaminD_pills', lowBound=0, cat='Integer')

# Objective: Minimize total time until all pills are effective
# Total time = 5*c + 6*v
prob += 5 * c + 6 * v, "Total_Effective_Time"

# Constraints
# At least 130 total pills
prob += c + v >= 130, "Total_Pills_Constraint"

# At least 40 vitamin D pills
prob += v >= 40, "VitaminD_Min_Constraint"

# Calcium pills must be more than vitamin D pills
prob += c >= v + 1, "Calcium_More_Than_VitaminD"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=0))

# Extract solution
calcium_pills = int(pulp.value(c))
vitaminD_pills = int(pulp.value(v))
final_obj = pulp.value(prob.objective)

# Print solution details (optional for user insight)
print(f"Calcium pills: {calcium_pills}")
print(f"Vitamin D pills: {vitaminD_pills}")
print(f"Total time (minutes): {final_obj}")

# Required final objective print
print(f"FINAL_OBJ: {final_obj}")
