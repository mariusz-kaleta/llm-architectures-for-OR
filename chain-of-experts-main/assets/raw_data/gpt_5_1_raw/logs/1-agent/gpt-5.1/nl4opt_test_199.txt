import pulp

# Define the problem
prob = pulp.LpProblem("Maximize_Audience", pulp.LpMaximize)

# Decision variables: number of commercials on each platform
x_pi = pulp.LpVariable('x_pi', lowBound=0, cat='Integer')
x_beta = pulp.LpVariable('x_beta', lowBound=0, upBound=8, cat='Integer')
x_gamma = pulp.LpVariable('x_gamma', lowBound=0, cat='Integer')

# Objective: maximize audience
prob += 2000 * x_pi + 5000 * x_beta + 9000 * x_gamma

# Constraints

# Budget constraint
prob += 1200 * x_pi + 2000 * x_beta + 4000 * x_gamma <= 20000

# Total commercials
t = pulp.LpVariable('t', lowBound=0, cat='Continuous')
prob += t == x_pi + x_beta + x_gamma

# At most a third on Gamma Live: x_gamma <= (1/3) * t  ->  3*x_gamma <= t
prob += 3 * x_gamma <= t

# At least 20% on Pi TV: x_pi >= 0.2 * t  ->  5*x_pi >= t
prob += 5 * x_pi >= t

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=0))

# Extract the solution
x_pi_val = pulp.value(x_pi)
x_beta_val = pulp.value(x_beta)
x_gamma_val = pulp.value(x_gamma)
obj_val = pulp.value(prob.objective)

# Print solution details (can be commented out if only final objective is needed)
print(f"Pi TV commercials: {x_pi_val}")
print(f"Beta Video commercials: {x_beta_val}")
print(f"Gamma Live commercials: {x_gamma_val}")
print(f"Maximum audience: {obj_val}")

# Final required output
print(f"FINAL_OBJ: {obj_val}")
