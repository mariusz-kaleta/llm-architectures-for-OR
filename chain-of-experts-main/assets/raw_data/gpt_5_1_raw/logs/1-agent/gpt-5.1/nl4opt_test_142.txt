import pulp

# Initialize the problem
prob = pulp.LpProblem("Art_Store_Production", pulp.LpMaximize)

# Decision variables: number of large (L) and small (S) art pieces
L = pulp.LpVariable('Large', lowBound=5, cat='Continuous')
S = pulp.LpVariable('Small', lowBound=5, cat='Continuous')

# Objective function: Maximize profit
prob += 30 * L + 15 * S

# Constraints
# Paint: 4L + 2S <= 100
prob += 4 * L + 2 * S <= 100, "Paint_Constraint"

# Glitter: 3L + 1S <= 50
prob += 3 * L + 1 * S <= 50, "Glitter_Constraint"

# Glue: 5L + 2S <= 70
prob += 5 * L + 2 * S <= 70, "Glue_Constraint"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=0))

# Get results
L_opt = pulp.value(L)
S_opt = pulp.value(S)
final_obj = pulp.value(prob.objective)

# Print decision variables (optional intermediate output)
print(f"Optimal number of large art pieces: {L_opt}")
print(f"Optimal number of small art pieces: {S_opt}")

# Final required output
print(f"FINAL_OBJ: {final_obj}")
