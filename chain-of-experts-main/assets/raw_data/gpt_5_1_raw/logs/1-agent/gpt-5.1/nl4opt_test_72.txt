import pulp

# Initialize the problem: maximize total slime
prob = pulp.LpProblem("Slime_Production", pulp.LpMaximize)

# Decision variables: number of times each table is set up (can be fractional)
x1 = pulp.LpVariable('Table1', lowBound=0)
x2 = pulp.LpVariable('Table2', lowBound=0)

# Objective: maximize total slime produced
# Table 1: 4 units slime per use, Table 2: 5 units slime per use
prob += 4 * x1 + 5 * x2, "Total_Slime"

# Constraints:

# Powder constraint: 3 units for table 1, 8 units for table 2, at most 100 units
prob += 3 * x1 + 8 * x2 <= 100, "Powder"

# Glue constraint: 5 units for table 1, 6 units for table 2, at most 90 units
prob += 5 * x1 + 6 * x2 <= 90, "Glue"

# Mess constraint: 2 units for table 1, 4 units for table 2, at most 30 units
prob += 2 * x1 + 4 * x2 <= 30, "Mess"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Retrieve values
table1_val = pulp.value(x1)
table2_val = pulp.value(x2)
final_obj = pulp.value(prob.objective)

# Print solution details (optional for inspection)
print(f"Table 1 setups: {table1_val}")
print(f"Table 2 setups: {table2_val}")
print(f"Max slime produced: {final_obj}")

# Required final objective print
print(f"FINAL_OBJ: {final_obj}")
