import pulp

# Define the problem
prob = pulp.LpProblem("Water_Salesman_Problem", pulp.LpMaximize)

# Decision variables
s = pulp.LpVariable('small_kegs', lowBound=0, upBound=30, cat='Integer')
l = pulp.LpVariable('large_kegs', lowBound=0, upBound=10, cat='Integer')

# Objective function: maximize total liters transported
prob += 40 * s + 100 * l

# Constraints
prob += s + l <= 25               # At most 25 kegs total
prob += l >= 5                    # At least 5 large kegs
prob += s >= 2 * l                # At least twice as many small as large

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract results
small_kegs_opt = int(pulp.value(s))
large_kegs_opt = int(pulp.value(l))
total_water = 40 * small_kegs_opt + 100 * large_kegs_opt

# Print solution details (optional)
print(f"Optimal number of small kegs: {small_kegs_opt}")
print(f"Optimal number of large kegs: {large_kegs_opt}")
print(f"Total water transported (liters): {total_water}")

# Final objective value print as required
print(f"FINAL_OBJ: {total_water}")
