import pulp

# Create the LP problem: minimize number of vehicles
prob = pulp.LpProblem("Airport_Vehicle_Optimization", pulp.LpMinimize)

# Decision variables: number of 4-wheelers (x) and 3-wheelers (y)
x = pulp.LpVariable('x', lowBound=0, cat='Integer')  # 4-wheelers
y = pulp.LpVariable('y', lowBound=0, cat='Integer')  # 3-wheelers

# Objective: minimize total number of vehicles
prob += x + y, "Total_Vehicles"

# Constraints:
# Luggage capacity: 60x + 40y >= 1000
prob += 60 * x + 40 * y >= 1000, "Luggage_Requirement"

# Pollution limit: 30x + 15y <= 430
prob += 30 * x + 15 * y <= 430, "Pollution_Limit"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract the results
x_opt = pulp.value(x)
y_opt = pulp.value(y)
obj_value = pulp.value(prob.objective)

# Print solution details
print(f"Optimal number of 4-wheelers (x): {x_opt}")
print(f"Optimal number of 3-wheelers (y): {y_opt}")
print(f"Minimum total number of vehicles: {obj_value}")

print(f"FINAL_OBJ: {obj_value}")
