import pulp

# Create the LP problem: maximize profit
prob = pulp.LpProblem("Bakery_Profit_Maximization", pulp.LpMaximize)

# Decision variables: batches of bagels (x) and croissants (y)
x = pulp.LpVariable('bagels', lowBound=0, cat='Continuous')
y = pulp.LpVariable('croissants', lowBound=0, cat='Continuous')

# Parameters
oven_hours = 70
chef_hours = 32

# Objective function: Maximize profit = 20x + 40y
prob += 20 * x + 40 * y, "Total_Profit"

# Constraints
# Oven time: 2x + 1y <= 70
prob += 2 * x + 1 * y <= oven_hours, "Oven_Time"

# Pastry chef time: 0.25x + 2y <= 32
prob += 0.25 * x + 2 * y <= chef_hours, "Chef_Time"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Get the final objective value
final_obj = pulp.value(prob.objective)

# print(f"Maximum Profit: {final_obj}")

print(f"FINAL_OBJ: {final_obj}")
