import pulp

# Create the problem
prob = pulp.LpProblem("Mango_Guava_Profit_Max", pulp.LpMaximize)

# Decision variables: number of mangos (m) and guavas (g)
m = pulp.LpVariable('m', lowBound=0)
g = pulp.LpVariable('g', lowBound=0)

# Objective function: maximize profit = 3*m + 4*g
prob += 3 * m + 4 * g, "Total_Profit"

# Constraints
# Budget constraint: 5*m + 3*g <= 20000
prob += 5 * m + 3 * g <= 20000, "Budget"

# Mango sales bounds: 100 <= m <= 150
prob += m >= 100, "Mango_Min"
prob += m <= 150, "Mango_Max"

# Guava constraint: g <= (1/3)*m
prob += g <= (1.0 / 3.0) * m, "Guava_Constraint"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=0))

# Extract results
m_val = pulp.value(m)
g_val = pulp.value(g)
profit_val = pulp.value(prob.objective)

# Print detailed solution
print(f"Optimal number of mangos to sell: {m_val}")
print(f"Optimal number of guavas to sell: {g_val}")
print(f"Maximum profit: {profit_val}")

# Required final objective print
print(f"FINAL_OBJ: {profit_val}")
