import pulp

# Create the problem
prob = pulp.LpProblem("Patient_Transport", pulp.LpMinimize)

# Decision variables
H = pulp.LpVariable('Helicopter_Trips', lowBound=0, cat='Integer')
B = pulp.LpVariable('Bus_Trips', lowBound=0, upBound=10, cat='Integer')

# Objective: Minimize total time
prob += 1 * H + 3 * B

# Constraints
# At least 120 patients
prob += 5 * H + 8 * B >= 120

# At least 30% of trips by helicopter
prob += H >= 0.3 * (H + B)

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

H_opt = pulp.value(H)
B_opt = pulp.value(B)
obj_val = pulp.value(prob.objective)

print(f"Optimal Helicopter Trips: {H_opt}")
print(f"Optimal Bus Trips: {B_opt}")
print(f"Total Time: {obj_val}")
print(f"FINAL_OBJ: {obj_val}")
