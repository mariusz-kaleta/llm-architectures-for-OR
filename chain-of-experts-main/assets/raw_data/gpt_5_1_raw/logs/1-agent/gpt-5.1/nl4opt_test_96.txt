import pulp

# Define the problem
prob = pulp.LpProblem("Minimize_Cholesterol", pulp.LpMinimize)

# Decision variables: number of burgers (b) and pizza slices (p)
b = pulp.LpVariable('burgers', lowBound=0, cat='Continuous')
p = pulp.LpVariable('pizza_slices', lowBound=0, cat='Continuous')

# Objective function: Minimize cholesterol
# Each burger: 12 units cholesterol, each pizza slice: 10 units cholesterol
prob += 12 * b + 10 * p

# Constraints:
# Fat: Each burger 10 units, each slice 8 units, need at least 130 units
prob += 10 * b + 8 * p >= 130

# Calories: Each burger 300, each slice 250, need at least 3000 calories
prob += 300 * b + 250 * p >= 3000

# Pizza at least twice burgers: p >= 2b
prob += p >= 2 * b

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Get solution values
b_value = pulp.value(b)
p_value = pulp.value(p)
final_obj = pulp.value(prob.objective)

# Print results
print(f"Optimal number of burgers: {b_value}")
print(f"Optimal number of pizza slices: {p_value}")
print(f"Minimum cholesterol intake: {final_obj}")
print(f"FINAL_OBJ: {final_obj}")
