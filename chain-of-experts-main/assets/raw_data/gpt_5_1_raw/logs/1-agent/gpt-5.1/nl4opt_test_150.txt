import pulp

# Define the problem
prob = pulp.LpProblem("Pharma_Production_Time_Minimization", pulp.LpMinimize)

# Decision variables: hours run at lab 1 and lab 2
x1 = pulp.LpVariable('Hours_Lab1', lowBound=0)
x2 = pulp.LpVariable('Hours_Lab2', lowBound=0)

# Objective: minimize total time (hours at lab 1 + hours at lab 2)
prob += x1 + x2, "Total_Time"

# Constraints

# Heart medication pills: at least 20000
# Lab 1: 20 heart pills/hour, Lab 2: 30 heart pills/hour
prob += 20 * x1 + 30 * x2 >= 20000, "Heart_Demand"

# Lung medication pills: at least 30000
# Lab 1: 30 lung pills/hour, Lab 2: 40 lung pills/hour
prob += 30 * x1 + 40 * x2 >= 30000, "Lung_Demand"

# Worker hours: each session hour at lab 1 requires 3 worker hours, at lab 2 requires 5 worker hours
prob += 3 * x1 + 5 * x2 <= 1500, "Worker_Hours"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Retrieve results
hours_lab1 = pulp.value(x1)
hours_lab2 = pulp.value(x2)
objective_value = pulp.value(prob.objective)

# Print results
print(f"Hours at Lab 1: {hours_lab1}")
print(f"Hours at Lab 2: {hours_lab2}")
print(f"Minimum total time (hours): {objective_value}")

# Required final objective print
print(f"FINAL_OBJ: {objective_value}")
