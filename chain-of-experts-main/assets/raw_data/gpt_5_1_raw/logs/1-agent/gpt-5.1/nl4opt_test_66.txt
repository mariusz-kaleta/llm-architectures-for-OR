import pulp

# Define the problem
prob = pulp.LpProblem("Corn_Transport_Minimization", pulp.LpMinimize)

# Decision variables: number of tractors (T) and cars (C)
T = pulp.LpVariable('T', lowBound=0, cat='Integer')
C = pulp.LpVariable('C', lowBound=0, cat='Integer')

# Objective: minimize total number of vehicles
prob += T + C, "Total_Vehicles"

# Constraints:
# Capacity: 40 kg per tractor, 20 kg per car, at least 500 kg
prob += 40 * T + 20 * C >= 500, "Capacity_Constraint"

# Cars at least twice the tractors
prob += C >= 2 * T, "Car_Tractor_Ratio"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract the objective value
final_obj_value = pulp.value(prob.objective)

# Print solution details (optional)
print("Status:", pulp.LpStatus[prob.status])
print("Number of tractors (T):", int(pulp.value(T)))
print("Number of cars (C):", int(pulp.value(C)))
print("Minimum total vehicles:", int(final_obj_value))

# Final required print
print(f"FINAL_OBJ: {final_obj_value}")
