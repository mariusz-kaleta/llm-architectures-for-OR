import pulp

# Define the problem: minimize total number of vehicles
problem = pulp.LpProblem("TourismFleetPlanning", pulp.LpMinimize)

# Decision variables: number of sedans (s) and buses (b)
s = pulp.LpVariable('sedans', lowBound=0, cat='Integer')
b = pulp.LpVariable('buses', lowBound=0, cat='Integer')

# Parameters from the problem
seats_sedan = 50
pollution_sedan = 10

seats_bus = 250
pollution_bus = 40

max_pollution = 800
min_customers = 4600

# Objective: minimize total number of vehicles
problem += s + b, "MinimizeNumberOfVehicles"

# Constraints:
# 1. Customer capacity constraint
problem += seats_sedan * s + seats_bus * b >= min_customers, "CustomerCapacity"

# 2. Pollution constraint
problem += pollution_sedan * s + pollution_bus * b <= max_pollution, "PollutionLimit"

# Solve the problem
problem.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract results
sedans_opt = int(pulp.value(s))
buses_opt = int(pulp.value(b))
final_obj = pulp.value(problem.objective)

# Print results
print(f"Optimal number of sedans: {sedans_opt}")
print(f"Optimal number of buses: {buses_opt}")
print(f"FINAL_OBJ: {final_obj}")
