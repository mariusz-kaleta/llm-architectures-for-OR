import pulp

# Initialize the problem
prob = pulp.LpProblem("Electricity_Production", pulp.LpMaximize)

# Decision variables: number of times to conduct each experiment
alpha = pulp.LpVariable('alpha', lowBound=0, cat='Continuous')
beta = pulp.LpVariable('beta', lowBound=0, cat='Continuous')

# Objective function: maximize total electricity produced
prob += 8 * alpha + 10 * beta

# Constraints
prob += 3 * alpha + 5 * beta <= 800   # Metal constraint
prob += 5 * alpha + 4 * beta <= 750   # Acid constraint

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=0))

# Extract results
alpha_val = pulp.value(alpha)
beta_val = pulp.value(beta)
objective_val = pulp.value(prob.objective)

# Print solution details
print(f"Optimal number of alpha experiments: {alpha_val}")
print(f"Optimal number of beta experiments: {beta_val}")
print(f"Maximum total electricity produced: {objective_val}")

# Final required print
print(f"FINAL_OBJ: {objective_val}")
