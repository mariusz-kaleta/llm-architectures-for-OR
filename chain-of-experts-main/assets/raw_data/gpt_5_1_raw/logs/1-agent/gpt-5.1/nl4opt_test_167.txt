import pulp

# Define the problem
prob = pulp.LpProblem("Minimum_Cost_Diet", pulp.LpMinimize)

# Decision variables: number of hamburgers (h) and chicken wraps (c)
h = pulp.LpVariable('hamburgers', lowBound=0, cat='Continuous')
c = pulp.LpVariable('chicken_wraps', lowBound=0, cat='Continuous')

# Objective function: Minimize total cost
prob += 6.5 * h + 4.0 * c

# Constraints
# Calories: at least 2200
prob += 800 * h + 450 * c >= 2200, "CalorieRequirement"

# Protein: at least 50 grams
prob += 19 * h + 12 * c >= 50, "ProteinRequirement"

# Carbs: at least 70 grams
prob += 20 * h + 10 * c >= 70, "CarbRequirement"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract results
h_opt = pulp.value(h)
c_opt = pulp.value(c)
obj_value = pulp.value(prob.objective)

# Print solution details (optional)
print(f"Optimal number of hamburgers: {h_opt}")
print(f"Optimal number of chicken wraps: {c_opt}")
print(f"Minimum total cost: {obj_value}")

# Final required print
print(f"FINAL_OBJ: {obj_value}")
