import pulp as pl

# Initialize the problem
prob = pl.LpProblem("Pipe_Optimization", pl.LpMinimize)

# Decision variables: number of wide and narrow pipes
W = pl.LpVariable('Wide_Pipes', lowBound=0, cat='Integer')
N = pl.LpVariable('Narrow_Pipes', lowBound=0, cat='Integer')

# Objective: minimize total number of pipes
prob += W + N, "Total_Pipes"

# Constraints
# Water transport requirement: 25W + 15N >= 900
prob += 25 * W + 15 * N >= 900, "Water_Requirement"

# Wide pipes at most a third of narrow pipes: W <= N/3
prob += W <= N / 3, "Wide_to_Narrow_Ratio"

# At least 5 wide pipes
prob += W >= 5, "Min_Wide_Pipes"

# Solve the problem
prob.solve(pl.PULP_CBC_CMD(msg=0))

# Extract the objective value
final_obj = pl.value(prob.objective)

# Print the final objective value in required format
print(f"FINAL_OBJ: {final_obj}")
