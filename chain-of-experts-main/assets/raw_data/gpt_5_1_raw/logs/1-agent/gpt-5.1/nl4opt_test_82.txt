import pulp

# Create the problem
prob = pulp.LpProblem("Delivery_Shift_Optimization", pulp.LpMaximize)

# Decision variables
x_bike = pulp.LpVariable('x_bike', lowBound=0, cat='Integer')
x_scooter = pulp.LpVariable('x_scooter', lowBound=5, cat='Integer')  # at least 5 scooter shifts

# Objective: Maximize tips
prob += 50 * x_bike + 43 * x_scooter

# Constraints
# Total shifts limit
prob += x_bike + x_scooter <= 40

# Energy limit
prob += 5 * x_bike + 6 * x_scooter <= 230

# Orders requirement
prob += 10 * x_bike + 7 * x_scooter >= 320

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Get results
x_bike_val = pulp.value(x_bike)
x_scooter_val = pulp.value(x_scooter)
final_obj = pulp.value(prob.objective)

# Print solution details (optional for understanding)
print(f"Bike shifts: {x_bike_val}")
print(f"Scooter shifts: {x_scooter_val}")
print(f"Maximum tips: {final_obj}")

# Final required output
print(f"FINAL_OBJ: {final_obj}")
