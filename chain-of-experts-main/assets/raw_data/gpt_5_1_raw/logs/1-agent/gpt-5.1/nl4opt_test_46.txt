import pulp

# Create the problem
prob = pulp.LpProblem("TeaEstateOptimization", pulp.LpMaximize)

# Decision variables
x_traditional = pulp.LpVariable('Acres_Traditional', lowBound=0)
x_modern = pulp.LpVariable('Acres_Modern', lowBound=0)

# Parameters
total_acres = 500
max_fuel = 9000
max_waste = 6000

# Objective: maximize tea leaves picked
# Traditional: 30 kg/acre, Modern: 40 kg/acre
prob += 30 * x_traditional + 40 * x_modern, "Total_Tea_Leaves"

# Constraints

# Land constraint
prob += x_traditional + x_modern <= total_acres, "Land_Constraint"

# Fuel constraint
# Traditional: 20 liters/acre, Modern: 15 liters/acre
prob += 20 * x_traditional + 15 * x_modern <= max_fuel, "Fuel_Constraint"

# Waste constraint
# Traditional: 10 kg/acre, Modern: 15 kg/acre
prob += 10 * x_traditional + 15 * x_modern <= max_waste, "Waste_Constraint"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Retrieve results
acres_traditional = pulp.value(x_traditional)
acres_modern = pulp.value(x_modern)
max_tea_leaves = pulp.value(prob.objective)

# Print detailed results
print(f"Acres using traditional machine: {acres_traditional}")
print(f"Acres using modern machine: {acres_modern}")
print(f"Maximum tea leaves picked (kg): {max_tea_leaves}")

# Final objective value print as required
print(f"FINAL_OBJ: {max_tea_leaves}")
