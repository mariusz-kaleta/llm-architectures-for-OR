import pulp

# Create the problem
prob = pulp.LpProblem("Hand_Sanitizer_Production", pulp.LpMaximize)

# Decision variables: number of liquid (L) and foam (F) hand sanitizers
L = pulp.LpVariable('L', lowBound=0, cat='Integer')
F = pulp.LpVariable('F', lowBound=0, cat='Integer')

# Objective: maximize number of hands cleaned
# Each liquid cleans 30 hands, each foam cleans 20 hands
prob += 30 * L + 20 * F

# Constraints
# Water: 40L + 60F <= 2000
prob += 40 * L + 60 * F <= 2000

# Alcohol: 50L + 40F <= 2100
prob += 50 * L + 40 * F <= 2100

# Number of foam sanitizers must exceed number of liquid sanitizers: F >= L + 1
prob += F >= L + 1

# At most 30 liquid sanitizers: L <= 30
prob += L <= 30

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Get results
L_opt = pulp.value(L)
F_opt = pulp.value(F)
objective_value = pulp.value(prob.objective)

# Print results
print(f"Optimal number of liquid sanitizers (L): {L_opt}")
print(f"Optimal number of foam sanitizers (F): {F_opt}")
print(f"Maximum number of hands that can be cleaned: {objective_value}")
print(f"FINAL_OBJ: {objective_value}")
