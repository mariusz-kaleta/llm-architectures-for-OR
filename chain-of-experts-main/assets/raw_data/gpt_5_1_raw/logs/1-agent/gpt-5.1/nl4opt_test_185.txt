import pulp

# Define the problem
prob = pulp.LpProblem("FastFoodProductionTimeMinimization", pulp.LpMinimize)

# Decision variables: number of wraps (w) and platters (p)
w = pulp.LpVariable('wraps', lowBound=0, cat='Continuous')
p = pulp.LpVariable('platters', lowBound=0, cat='Continuous')

# Objective function: minimize total production time
# Each wrap takes 10 minutes, each platter takes 8 minutes
prob += 10 * w + 8 * p, "Total_Production_Time"

# Constraints:
# Meat: 5 units per wrap, 7 units per platter, at least 3000 units used
prob += 5 * w + 7 * p >= 3000, "Meat_Constraint"

# Rice: 3 units per wrap, 5 units per platter, at least 2500 units used
prob += 3 * w + 5 * p >= 2500, "Rice_Constraint"

# Wraps at least 3 times as many as platters: w >= 3p
prob += w >= 3 * p, "Wraps_Platter_Ratio"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=0))

# Get the results
wraps_opt = pulp.value(w)
platters_opt = pulp.value(p)
final_obj = pulp.value(prob.objective)

# Print the solution
print(f"Optimal number of wraps: {wraps_opt}")
print(f"Optimal number of platters: {platters_opt}")
print(f"Minimum total production time (minutes): {final_obj}")

# Required final objective print
print(f"FINAL_OBJ: {final_obj}")
