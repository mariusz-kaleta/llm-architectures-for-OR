import pulp

# Create the LP problem: minimize cost
prob = pulp.LpProblem("Medicine_Requirement_Problem", pulp.LpMinimize)

# Decision variables: number of Zodiac (x) and Sunny (y) pills
x = pulp.LpVariable('Zodiac_pills', lowBound=0, cat='Continuous')
y = pulp.LpVariable('Sunny_pills', lowBound=0, cat='Continuous')

# Objective function: minimize total cost
prob += 1 * x + 3 * y, "Total_Cost"

# Constraints:
# Z1 requirement: 1.3x + 1.2y >= 5
prob += 1.3 * x + 1.2 * y >= 5, "Z1_requirement"

# D3 requirement: 1.5x + 5y >= 10
prob += 1.5 * x + 5 * y >= 10, "D3_requirement"

# Solve the LP
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Get the objective value
final_obj_value = pulp.value(prob.objective)

# Print the final objective value in the required format
print(f"FINAL_OBJ: {final_obj_value}")
