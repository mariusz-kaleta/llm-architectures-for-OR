import pulp

# Initialize the problem
prob = pulp.LpProblem("Supplement_Optimization", pulp.LpMinimize)

# Decision variables: number of pills of A and B (non-negative continuous; can be changed to Integer if needed)
A = pulp.LpVariable('A', lowBound=0, cat='Continuous')
B = pulp.LpVariable('B', lowBound=0, cat='Continuous')

# Objective function: Minimize cost = 2A + 3B
prob += 2 * A + 3 * B, "Total_Cost"

# Constraints:
# Iron: 5A + 4B >= 40
prob += 5 * A + 4 * B >= 40, "Iron_Requirement"

# Calcium: 10A + 15B >= 50
prob += 10 * A + 15 * B >= 50, "Calcium_Requirement"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract results
A_val = pulp.value(A)
B_val = pulp.value(B)
obj_val = pulp.value(prob.objective)

# Print solution details (optional)
print(f"Optimal number of pills of supplement A: {A_val}")
print(f"Optimal number of pills of supplement B: {B_val}")
print(f"Minimum total cost: {obj_val}")

# Final required output
print(f"FINAL_OBJ: {obj_val}")
