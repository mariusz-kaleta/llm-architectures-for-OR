import pulp

# Define the problem
prob = pulp.LpProblem("Sailor_Diet_Problem", pulp.LpMinimize)

# Decision variables: number of crab cakes (x) and lobster rolls (y)
x = pulp.LpVariable('CrabCakes', lowBound=0)
y = pulp.LpVariable('LobsterRolls', lowBound=0)

# Objective: Minimize total unsaturated fat: 4x + 6y
prob += 4 * x + 6 * y

# Constraints:
# Vitamin A: 5x + 8y >= 80
prob += 5 * x + 8 * y >= 80, "Vitamin_A"

# Vitamin C: 7x + 4y >= 100
prob += 7 * x + 4 * y >= 100, "Vitamin_C"

# Lobster rolls at most 40% of meals: y <= 0.4(x + y) -> 0.6y <= 0.4x -> 3y <= 2x
prob += 3 * y <= 2 * x, "Lobster_Ratio"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Get the results
x_val = pulp.value(x)
y_val = pulp.value(y)
obj_val = pulp.value(prob.objective)

# Print the solution
print(f"Optimal number of crab cakes: {x_val}")
print(f"Optimal number of lobster rolls: {y_val}")
print(f"Minimum total unsaturated fat: {obj_val}")

print(f"FINAL_OBJ: {obj_val}")
