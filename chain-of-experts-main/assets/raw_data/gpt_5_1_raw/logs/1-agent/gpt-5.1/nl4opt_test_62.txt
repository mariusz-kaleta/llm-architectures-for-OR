import pulp

# Define the problem
prob = pulp.LpProblem("Handbag_Profit_Maximization", pulp.LpMaximize)

# Decision variables: number of regular (R) and premium (P) handbags
R = pulp.LpVariable('Regular', lowBound=0, cat='Integer')
P = pulp.LpVariable('Premium', lowBound=0, cat='Integer')

# Parameters
profit_R = 30
profit_P = 180
cost_R = 200
cost_P = 447
budget = 250000
max_total = 475

# Objective function: maximize profit
prob += profit_R * R + profit_P * P

# Constraints
prob += cost_R * R + cost_P * P <= budget           # Budget constraint
prob += R + P <= max_total                           # Sales limit constraint

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract results
R_opt = int(pulp.value(R))
P_opt = int(pulp.value(P))
objective_value = pulp.value(prob.objective)

# Print solution details (optional)
print(f"Optimal number of regular handbags: {R_opt}")
print(f"Optimal number of premium handbags: {P_opt}")
print(f"Maximum monthly profit: {objective_value}")

# Final required print
print(f"FINAL_OBJ: {objective_value}")
