import pulp

# Define the problem
prob = pulp.LpProblem("Minimize_Animals", pulp.LpMinimize)

# Decision variables
cows = pulp.LpVariable('cows', lowBound=0, cat='Integer')
elephants = pulp.LpVariable('elephants', lowBound=0, cat='Integer')

# Objective: Minimize total number of animals
prob += cows + elephants

# Constraints
# Capacity constraint: at least 1000 bricks
prob += 20 * cows + 50 * elephants >= 1000

# Traffic constraint: elephants cannot exceed cows
prob += elephants <= cows

# Balance constraint: at most twice the number of cows as elephants
prob += cows <= 2 * elephants

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=0))

# Extract results
cows_val = int(pulp.value(cows))
elephants_val = int(pulp.value(elephants))
obj_val = int(pulp.value(prob.objective))

# Print solution details (optional)
print(f"Cows: {cows_val}")
print(f"Elephants: {elephants_val}")
print(f"Total animals: {obj_val}")
print(f"Total bricks carried: {20 * cows_val + 50 * elephants_val}")

# Final objective value
print(f"FINAL_OBJ: {obj_val}")
