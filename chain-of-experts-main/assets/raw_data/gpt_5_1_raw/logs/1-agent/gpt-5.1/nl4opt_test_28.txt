import pulp

# Data initialization
DogCapability = 100
TruckCapability = 300
DogCost = 50
TruckCost = 100
MaxBudget = 1000

# Define the problem: maximize total fish transported
prob = pulp.LpProblem("Fish_Transport_Maximization", pulp.LpMaximize)

# Decision variables: number of sled dog trips and truck trips (continuous, non-negative)
dog_trips = pulp.LpVariable('dog_trips', lowBound=0, cat='Continuous')
truck_trips = pulp.LpVariable('truck_trips', lowBound=0, cat='Continuous')

# Objective function: maximize total fish transported
prob += DogCapability * dog_trips + TruckCapability * truck_trips

# Budget constraint
prob += DogCost * dog_trips + TruckCost * truck_trips <= MaxBudget, "Budget_Constraint"

# Constraint: number of sled dog trips must be less than number of truck trips
prob += dog_trips <= truck_trips, "Dog_Less_Than_Truck"

# Solve the LP
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract final objective value
final_obj_value = pulp.value(prob.objective)

# Print the final objective value in required format
print(f"FINAL_OBJ: {final_obj_value}")
