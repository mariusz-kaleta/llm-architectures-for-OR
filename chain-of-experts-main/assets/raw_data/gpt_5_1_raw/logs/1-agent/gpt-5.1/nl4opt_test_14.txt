import pulp

# Define the problem
prob = pulp.LpProblem("ThemeParkTransport", pulp.LpMinimize)

# Decision variables
scooters = pulp.LpVariable('scooters', lowBound=0, cat='Integer')
rickshaws = pulp.LpVariable('rickshaws', lowBound=0, cat='Integer')

# Objective: Minimize the number of scooters
prob += scooters, "Minimize_Number_of_Scooters"

# Constraints

# Capacity constraint: 2 per scooter, 3 per rickshaw, at least 300 visitors
prob += 2 * scooters + 3 * rickshaws >= 300, "CapacityConstraint"

# Pollution constraint: at most 40% of vehicles can be rickshaws
# rickshaws <= 0.4 * (scooters + rickshaws)
# Equivalent linear form: rickshaws <= (2/3) * scooters
prob += rickshaws <= (2.0/3.0) * scooters, "PollutionConstraint"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract the solution
opt_scooters = pulp.value(scooters)
opt_rickshaws = pulp.value(rickshaws)

# Print results
print(f"Optimal number of scooters: {opt_scooters}")
print(f"Optimal number of rickshaws: {opt_rickshaws}")

# Final objective value (minimum number of scooters)
FINAL_OBJ = opt_scooters
print(f"FINAL_OBJ: {FINAL_OBJ}")
