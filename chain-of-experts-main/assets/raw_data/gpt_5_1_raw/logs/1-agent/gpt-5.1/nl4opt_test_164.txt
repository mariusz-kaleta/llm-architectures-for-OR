import pulp

# Define the problem
prob = pulp.LpProblem("Appliance_Profit_Maximization", pulp.LpMaximize)

# Decision variables
refrigerators = pulp.LpVariable('Refrigerators', lowBound=0, cat='Integer')
stoves = pulp.LpVariable('Stoves', lowBound=0, cat='Integer')

# Parameters
mover_time_refrigerator = 60
setup_time_refrigerator = 20
mover_time_stove = 45
setup_time_stove = 25
total_mover_time = 20000
total_setup_time = 13000
profit_refrigerator = 400
profit_stove = 260

# Objective function
prob += profit_refrigerator * refrigerators + profit_stove * stoves, "Total_Profit"

# Constraints
prob += mover_time_refrigerator * refrigerators + mover_time_stove * stoves <= total_mover_time, "Mover_Time_Constraint"
prob += setup_time_refrigerator * refrigerators + setup_time_stove * stoves <= total_setup_time, "Setup_Time_Constraint"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=0))

# Retrieve results
opt_refrigerators = pulp.value(refrigerators)
opt_stoves = pulp.value(stoves)
final_profit = pulp.value(prob.objective)

# Print results
print(f"Optimal number of refrigerators: {opt_refrigerators}")
print(f"Optimal number of stoves: {opt_stoves}")
print(f"FINAL_OBJ: {final_profit}")
