import pulp

# Create the problem
prob = pulp.LpProblem("Headphone_Production", pulp.LpMaximize)

# Decision variables
wired = pulp.LpVariable('wired', lowBound=0, cat='Continuous')
wireless = pulp.LpVariable('wireless', lowBound=0, cat='Continuous')

# Objective function: Maximize profit
prob += 50 * wired + 20 * wireless

# Constraints
prob += wired <= 100          # Wired production limit
prob += wireless <= 170       # Wireless production limit
prob += wired + wireless <= 150  # Shared testing machine capacity

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract results
wired_opt = pulp.value(wired)
wireless_opt = pulp.value(wireless)
final_obj = pulp.value(prob.objective)

# Print solution details
print(f"Optimal wired headphones: {wired_opt}")
print(f"Optimal wireless headphones: {wireless_opt}")
print(f"Maximum profit: {final_obj}")

# Required final objective print
print(f"FINAL_OBJ: {final_obj}")
