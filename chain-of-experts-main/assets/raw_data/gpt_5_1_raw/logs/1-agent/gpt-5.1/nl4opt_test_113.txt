import pulp

# Initialize the problem
prob = pulp.LpProblem("Ben_Orchard_Optimization", pulp.LpMaximize)

# Decision variables: acres of apples (A) and pears (P)
A = pulp.LpVariable('Apples', lowBound=0, cat='Continuous')
P = pulp.LpVariable('Pears', lowBound=0, cat='Continuous')

# Objective function: Maximize profit = 2A + 4P
prob += 2 * A + 4 * P

# Constraints
prob += A + P <= 50          # Total land available
prob += A >= 5               # Minimum apples
prob += P >= 10              # Minimum pears
prob += P >= A               # More pears than apples
prob += P <= 2 * A           # At most twice as many pears as apples

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract results
opt_A = pulp.value(A)
opt_P = pulp.value(P)
opt_profit = pulp.value(prob.objective)

# Print solution details
print(f"Optimal acres of apples: {opt_A}")
print(f"Optimal acres of pears: {opt_P}")
print(f"Maximum profit: {opt_profit}")

# Required final objective print
print(f"FINAL_OBJ: {opt_profit}")
