import pulp

# Define the problem
prob = pulp.LpProblem("Appliance_Repair_Optimization", pulp.LpMaximize)

# Decision variables: number of washing machines (x) and freezers (y)
x = pulp.LpVariable('WashingMachines', lowBound=0, cat='Continuous')
y = pulp.LpVariable('Freezers', lowBound=0, cat='Continuous')

# Objective function: maximize earnings
prob += 250 * x + 375 * y, "Total_Earnings"

# Constraints
# Inspection time: 30x + 20y ≤ 5000
prob += 30 * x + 20 * y <= 5000, "Inspection_Time"

# Total schedule time: inspection + fixing
# Washing machine total: 30 + 90 = 120 minutes
# Freezer total: 20 + 125 = 145 minutes
prob += 120 * x + 145 * y <= 20000, "Total_Time"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=0))

# Retrieve the results
x_val = pulp.value(x)
y_val = pulp.value(y)
obj_val = pulp.value(prob.objective)

# Print solution details (optional, can be commented out if not needed)
print(f"Optimal number of washing machines to fix: {x_val}")
print(f"Optimal number of freezers to fix: {y_val}")
print(f"Maximum earnings: {obj_val}")

# Required final objective value print
print(f"FINAL_OBJ: {obj_val}")
