import pulp as pl

# Define the problem
prob = pl.LpProblem("Cough_Syrup_Bottles", pl.LpMaximize)

# Decision variables: number of kids and adult bottles
K = pl.LpVariable('Kids_Bottles', lowBound=50, cat='Integer')   # At least 50 kids bottles
A = pl.LpVariable('Adult_Bottles', lowBound=0, cat='Integer')

# Objective: maximize total number of bottles
prob += K + A, "Total_Bottles"

# Constraints:

# Syrup availability: 100K + 300A ≤ 25000
prob += 100 * K + 300 * A <= 25000, "Syrup_Constraint"

# At least 3 times as many adult size bottles as kids size bottles: A ≥ 3K
prob += A >= 3 * K, "Adult_to_Kids_Constraint"

# Solve the problem
prob.solve(pl.PULP_CBC_CMD(msg=False))

# Retrieve results
K_opt = pl.value(K)
A_opt = pl.value(A)
obj_value = pl.value(prob.objective)

# Print the solution
print(f"Kids_Bottles: {int(K_opt)}")
print(f"Adult_Bottles: {int(A_opt)}")
print(f"Total_Bottles: {int(obj_value)}")
print(f"FINAL_OBJ: {obj_value}")
