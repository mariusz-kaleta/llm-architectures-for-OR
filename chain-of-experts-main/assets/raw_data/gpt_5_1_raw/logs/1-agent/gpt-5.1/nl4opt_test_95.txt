import pulp as pl

# Define the problem
prob = pl.LpProblem("Dog_Bones_Optimization", pl.LpMinimize)

# Decision variables
S = pl.LpVariable('Small_Bones', lowBound=0, cat='Continuous')
L = pl.LpVariable('Large_Bones', lowBound=30, cat='Continuous')  # At least 30 large bones

# Objective: Minimize meat usage
# Small bone: 12 units meat, Large bone: 15 units meat
prob += 12 * S + 15 * L, "Total_Meat_Used"

# Constraints

# Tooth medication limit: 10S + 15L <= 2000
prob += 10 * S + 15 * L <= 2000, "Tooth_Medication_Limit"

# At least 50% of bones are small: S >= L
prob += S >= L, "Small_Bones_At_Least_Half"

# Solve the problem
prob.solve(pl.PULP_CBC_CMD(msg=False))

# Get results
S_opt = pl.value(S)
L_opt = pl.value(L)
obj_value = pl.value(prob.objective)

# Print detailed solution (optional)
print(f"Optimal number of small bones: {S_opt}")
print(f"Optimal number of large bones: {L_opt}")
print(f"Minimum meat required: {obj_value}")

# Required final objective value print
print(f"FINAL_OBJ: {obj_value}")
