import pulp

# Initialize the problem: minimize pollution
problem = pulp.LpProblem("School_Transport_Pollution_Minimization", pulp.LpMinimize)

# Decision variables
vans = pulp.LpVariable('vans', lowBound=0, cat='Integer')
minibuses = pulp.LpVariable('minibuses', lowBound=0, cat='Integer')

# Objective function: minimize total pollution
# Van: 7 units, Minibus: 10 units
problem += 7 * vans + 10 * minibuses, "Total_Pollution"

# Constraints

# Capacity: at least 150 kids
# Van capacity: 6, Minibus capacity: 10
problem += 6 * vans + 10 * minibuses >= 150, "Capacity_Constraint"

# At most 10 minibuses
problem += minibuses <= 10, "Minibus_Limit"

# Number of vans must exceed number of minibuses
problem += vans >= minibuses + 1, "Vans_Exceed_Minibuses"

# Solve the problem
problem.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract solution
opt_vans = int(pulp.value(vans))
opt_minibuses = int(pulp.value(minibuses))
final_obj = pulp.value(problem.objective)

# Print solution details (optional for understanding; not required by instructions)
print(f"Optimal number of vans: {opt_vans}")
print(f"Optimal number of minibuses: {opt_minibuses}")
print(f"Minimum total pollution: {final_obj}")

# Required final objective print
print(f"FINAL_OBJ: {final_obj}")
