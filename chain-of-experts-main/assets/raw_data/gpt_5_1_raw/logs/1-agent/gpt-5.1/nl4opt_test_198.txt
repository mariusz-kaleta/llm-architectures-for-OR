import pulp

# Initialize the problem
problem = pulp.LpProblem("Railway_Planning", pulp.LpMinimize)

# Decision variables
trains = pulp.LpVariable('trains', lowBound=0, cat='Integer')
trams = pulp.LpVariable('trams', lowBound=0, cat='Integer')

# Objective: minimize total number of transportation units
problem += trains + trams, "Total_Units"

# Constraints
# Capacity constraint: at least 600 people per hour
problem += 120 * trains + 30 * trams >= 600, "Capacity_Constraint"

# Trams at least twice the number of trains
problem += trams >= 2 * trains, "Tram_Train_Ratio"

# Solve the problem
problem.solve(pulp.PULP_CBC_CMD(msg=False))

# Get the objective value
final_obj = pulp.value(problem.objective)

# Print the results (optional detailed output)
print(f"Number of trains: {int(pulp.value(trains))}")
print(f"Number of trams: {int(pulp.value(trams))}")
print(f"Total units: {int(final_obj)}")

# Required final objective print
print(f"FINAL_OBJ: {final_obj}")
