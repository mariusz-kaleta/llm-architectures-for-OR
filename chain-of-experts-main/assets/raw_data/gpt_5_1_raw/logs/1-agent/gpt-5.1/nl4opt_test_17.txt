import pulp

# Define the problem
prob = pulp.LpProblem("Cleaning_Time_Minimization", pulp.LpMinimize)

# Decision variables
C = pulp.LpVariable('Cleansing', lowBound=0)       # units of cleansing chemical
O = pulp.LpVariable('OdorRemoving', lowBound=0)    # units of odor-removing chemical

# Objective function: minimize total time
prob += 4 * C + 6 * O

# Constraints
prob += C >= 100                     # at least 100 units of cleansing chemical
prob += C + O >= 300                 # at least 300 total units of chemicals
prob += C <= 2 * O                   # at most twice as much cleansing as odor-removing

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract results
C_opt = pulp.value(C)
O_opt = pulp.value(O)
obj_val = pulp.value(prob.objective)

print(f"Optimal Cleansing Units: {C_opt}")
print(f"Optimal Odor-Removing Units: {O_opt}")
print(f"Minimum Total Time: {obj_val}")
print(f"FINAL_OBJ: {obj_val}")
