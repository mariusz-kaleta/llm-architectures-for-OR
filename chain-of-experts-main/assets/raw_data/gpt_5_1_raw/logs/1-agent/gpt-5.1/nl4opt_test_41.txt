import pulp

# Initialize the problem
prob = pulp.LpProblem("Bakery_Production", pulp.LpMaximize)

# Decision variables: number of each type of cake
crepe = pulp.LpVariable('crepe', lowBound=0, cat='Continuous')
sponge = pulp.LpVariable('sponge', lowBound=0, cat='Continuous')
birthday = pulp.LpVariable('birthday', lowBound=0, cat='Continuous')

# Parameters
batter_total = 20000
milk_total = 14000

batter_crepe = 400
milk_crepe = 200

batter_sponge = 500
milk_sponge = 300

batter_birthday = 450
milk_birthday = 350

profit_crepe = 12
profit_sponge = 10
profit_birthday = 15

# Objective function: Maximize profit
prob += (profit_crepe * crepe +
         profit_sponge * sponge +
         profit_birthday * birthday), "Total_Profit"

# Constraints
prob += (batter_crepe * crepe +
         batter_sponge * sponge +
         batter_birthday * birthday <= batter_total), "Batter_Constraint"

prob += (milk_crepe * crepe +
         milk_sponge * sponge +
         milk_birthday * birthday <= milk_total), "Milk_Constraint"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Get the results
crepe_val = pulp.value(crepe)
sponge_val = pulp.value(sponge)
birthday_val = pulp.value(birthday)
final_obj = pulp.value(prob.objective)

# Print solution details (optional for user visibility)
print(f"Crepe cakes: {crepe_val}")
print(f"Sponge cakes: {sponge_val}")
print(f"Birthday cakes: {birthday_val}")
print(f"Maximum Profit: {final_obj}")

# Required final objective output
print(f"FINAL_OBJ: {final_obj}")
