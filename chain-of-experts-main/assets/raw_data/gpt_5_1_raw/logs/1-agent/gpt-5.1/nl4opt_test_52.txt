import pulp

# Define the problem
prob = pulp.LpProblem("Snack_Export_Maximization", pulp.LpMaximize)

# Decision variables
S = pulp.LpVariable('Small_Suitcases', lowBound=0, cat='Integer')
L = pulp.LpVariable('Large_Suitcases', lowBound=0, cat='Integer')

# Objective function: Maximize total snacks = 50*S + 80*L
prob += 50 * S + 80 * L, "Total_Snacks"

# Constraints

# At least twice as many small as large suitcases: S >= 2L
prob += S >= 2 * L, "Preference_Constraint"

# Availability of suitcases
prob += S <= 70, "Small_Availability"
prob += L <= 50, "Large_Availability"

# At least 15 large suitcases
prob += L >= 15, "Min_Large"

# At most 70 suitcases in total
prob += S + L <= 70, "Total_Suitcases"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=0))

# Retrieve results
optimal_S = int(pulp.value(S))
optimal_L = int(pulp.value(L))
optimal_obj = pulp.value(prob.objective)

# Print solution details (optional)
print(f"Optimal number of small suitcases: {optimal_S}")
print(f"Optimal number of large suitcases: {optimal_L}")
print(f"Maximum number of snacks delivered: {int(optimal_obj)}")

# Final required output
print(f"FINAL_OBJ: {optimal_obj}")
