import pulp

# Create the LP problem (minimization)
model = pulp.LpProblem("Teddy_Bear_Production", pulp.LpMinimize)

# Decision variables: hours to run factory 1 and 2 (continuous, >= 0)
x1 = pulp.LpVariable('Hours_Factory1', lowBound=0, cat='Continuous')
x2 = pulp.LpVariable('Hours_Factory2', lowBound=0, cat='Continuous')

# Objective function: minimize total cost
# Factory 1 cost: $300 per hour
# Factory 2 cost: $600 per hour
model += 300 * x1 + 600 * x2, "Total_Cost"

# Constraints:
# Black bears: Factory 1 produces 5/hour, Factory 2 produces 10/hour
model += 5 * x1 + 10 * x2 >= 20, "Black_Demand"

# White bears: Factory 1 produces 6/hour, Factory 2 produces 10/hour
model += 6 * x1 + 10 * x2 >= 5, "White_Demand"

# Brown bears: Only Factory 1 produces 3/hour
model += 3 * x1 >= 15, "Brown_Demand"

# Solve the model
model.solve(pulp.PULP_CBC_CMD(msg=False))

# Get final objective value
final_obj = pulp.value(model.objective)

# Print the final objective value in required format
print(f"FINAL_OBJ: {final_obj}")
