import pulp as pl

# Define the problem
prob = pl.LpProblem("Magic_School_Letters", pl.LpMaximize)

# Decision variables
p = pl.LpVariable('pigeons', lowBound=20, cat='Integer')  # at least 20 pigeons
o = pl.LpVariable('owls', lowBound=0, cat='Integer')

# Objective: maximize total letters sent
prob += 2 * p + 5 * o

# Constraints

# Treats constraint: 3 treats per pigeon, 5 per owl, max 1000
prob += 3 * p + 5 * o <= 1000

# At most 40% of the birds can be owls: o <= 0.4*(p + o) -> 0.6 o <= 0.4 p
prob += 3 * o <= 2 * p

# Solve
prob.solve(pl.PULP_CBC_CMD(msg=False))

# Get results
p_opt = pl.value(p)
o_opt = pl.value(o)
obj_val = pl.value(prob.objective)

# Print solution
print(f"Optimal number of pigeons: {p_opt}")
print(f"Optimal number of owls: {o_opt}")
print(f"Maximum letters sent: {obj_val}")
print(f"FINAL_OBJ: {obj_val}")
