import pulp

# Initialize the problem: minimize number of workers
prob = pulp.LpProblem("Snow_Removal_Staffing", pulp.LpMinimize)

# Decision variables
x = pulp.LpVariable('seasonal', lowBound=0, cat='Integer')   # number of seasonal workers
y = pulp.LpVariable('permanent', lowBound=0, cat='Integer')  # number of permanent workers

# Parameters
hours_seasonal = 6
hours_permanent = 10
cost_seasonal = 120
cost_permanent = 250
required_hours = 300
budget = 6500

# Objective: minimize total number of workers
prob += x + y, "Total_Workers"

# Constraints
prob += hours_seasonal * x + hours_permanent * y >= required_hours, "Labor_Requirement"
prob += cost_seasonal * x + cost_permanent * y <= budget, "Budget_Constraint"

# Solve
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Retrieve results
seasonal_val = pulp.value(x)
permanent_val = pulp.value(y)
final_obj = pulp.value(prob.objective)

print(f"Seasonal workers: {seasonal_val}")
print(f"Permanent workers: {permanent_val}")
print(f"Total workers (objective): {final_obj}")

print(f"FINAL_OBJ: {final_obj}")
