import pulp

# Define the problem
prob = pulp.LpProblem("Scooters_Bikes_Production", pulp.LpMaximize)

# Decision variables
scooters = pulp.LpVariable('scooters', lowBound=0, cat='Continuous')
bikes = pulp.LpVariable('bikes', lowBound=0, cat='Continuous')

# Objective function: Maximize profit
prob += 200 * scooters + 300 * bikes, "Total_Profit"

# Constraints
# Design time: 2 hours per scooter, 4 hours per bike, max 5000 hours
prob += 2 * scooters + 4 * bikes <= 5000, "Design_Time_Constraint"

# Engineering time: 3 hours per scooter, 5 hours per bike, max 6000 hours
prob += 3 * scooters + 5 * bikes <= 6000, "Engineering_Time_Constraint"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract results
opt_scooters = pulp.value(scooters)
opt_bikes = pulp.value(bikes)
final_obj = pulp.value(prob.objective)

# Print solution details (optional for user insight)
print(f"Optimal number of scooters to produce: {opt_scooters}")
print(f"Optimal number of bikes to produce: {opt_bikes}")
print(f"Maximum Profit: {final_obj}")

# Required final objective print
print(f"FINAL_OBJ: {final_obj}")
