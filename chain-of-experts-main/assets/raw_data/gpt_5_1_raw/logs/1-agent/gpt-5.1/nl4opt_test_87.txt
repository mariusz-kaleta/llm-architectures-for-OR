import pulp

# Define the problem
prob = pulp.LpProblem("Diet_Problem", pulp.LpMinimize)

# Decision variables: number of servings of noodles (x) and protein bars (y)
x = pulp.LpVariable('noodles_servings', lowBound=0, cat='Continuous')
y = pulp.LpVariable('protein_bar_servings', lowBound=0, cat='Continuous')

# Objective function: Minimize total cost
# Noodles: $5 per serving, Protein bars: $2.5 per serving
prob += 5 * x + 2.5 * y, "Total_Cost"

# Constraints
# Calorie constraint: at least 2000 calories
# Noodles: 600 cal/serving, Protein bars: 250 cal/serving
prob += 600 * x + 250 * y >= 2000, "Calorie_Requirement"

# Protein constraint: at least 16 grams of protein
# Noodles: 1.5 g/serving, Protein bars: 5 g/serving
prob += 1.5 * x + 5 * y >= 16, "Protein_Requirement"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Retrieve the optimal objective value
final_obj_value = pulp.value(prob.objective)

# Print the final objective value in the required format
print(f"FINAL_OBJ: {final_obj_value}")
