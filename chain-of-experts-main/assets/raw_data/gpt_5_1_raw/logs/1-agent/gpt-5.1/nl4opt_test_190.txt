import pulp

# Define the problem
prob = pulp.LpProblem("Milk_Tea_Production", pulp.LpMaximize)

# Decision variables: number of bottles of black and matcha milk tea
x_black = pulp.LpVariable('Black_Milk_Tea', lowBound=0, cat='Continuous')
x_matcha = pulp.LpVariable('Matcha_Milk_Tea', lowBound=0, cat='Continuous')

# Parameters
milk_black = 600
honey_black = 10
profit_black = 7.5

milk_matcha = 525
honey_matcha = 5
profit_matcha = 5

milk_available = 30000
honey_available = 500

# Objective function: maximize total profit
prob += profit_black * x_black + profit_matcha * x_matcha

# Constraints
prob += milk_black * x_black + milk_matcha * x_matcha <= milk_available, "Milk_Constraint"
prob += honey_black * x_black + honey_matcha * x_matcha <= honey_available, "Honey_Constraint"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract results
black_opt = pulp.value(x_black)
matcha_opt = pulp.value(x_matcha)
final_obj = pulp.value(prob.objective)

# Print solution details
print(f"Optimal number of black milk tea bottles: {black_opt}")
print(f"Optimal number of matcha milk tea bottles: {matcha_opt}")
print(f"Maximum profit: {final_obj}")

# Final required print
print(f"FINAL_OBJ: {final_obj}")
