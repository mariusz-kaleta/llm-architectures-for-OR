import pulp

# Initialize the problem
problem = pulp.LpProblem("Candy_Mix_Optimization", pulp.LpMaximize)

# Decision variables
R = pulp.LpVariable('Regular_Mix_kg', lowBound=0)
S = pulp.LpVariable('Sour_Surprise_Mix_kg', lowBound=0)

# Objective function: Maximize profit
problem += 3 * R + 5 * S

# Constraints
# Regular candy constraint: 0.8R + 0.1S <= 80
problem += 0.8 * R + 0.1 * S <= 80

# Sour candy constraint: 0.2R + 0.9S <= 60
problem += 0.2 * R + 0.9 * S <= 60

# Solve the problem
problem.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract results
R_opt = pulp.value(R)
S_opt = pulp.value(S)
final_obj = pulp.value(problem.objective)

# Print detailed solution
print(f"Optimal Regular Mix (kg): {R_opt}")
print(f"Optimal Sour Surprise Mix (kg): {S_opt}")
print(f"Maximum Profit: {final_obj}")

# Required final objective print
print(f"FINAL_OBJ: {final_obj}")
