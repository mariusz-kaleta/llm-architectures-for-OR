import pulp

# Initialize the problem
model = pulp.LpProblem("Printer_Production", pulp.LpMaximize)

# Decision variables
color = pulp.LpVariable('color', lowBound=0, upBound=20, cat='Continuous')
bw = pulp.LpVariable('bw', lowBound=0, upBound=30, cat='Continuous')

# Constraints
model += color + bw <= 35  # Paper tray installing machine capacity

# Objective function
model += 200 * color + 70 * bw

# Solve the problem
model.solve(pulp.PULP_CBC_CMD(msg=False))

# Get results
color_opt = pulp.value(color)
bw_opt = pulp.value(bw)
final_obj = pulp.value(model.objective)

# Print results (optional)
print("Optimal production plan:")
print(f"Color printers: {color_opt}")
print(f"Black and white printers: {bw_opt}")
print(f"Maximum profit: {final_obj}")

# Required final objective value print
print(f"FINAL_OBJ: {final_obj}")
