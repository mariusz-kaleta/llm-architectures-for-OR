import itertools

def solve_transport_problem():
    # Parameters
    cows_required = 80
    max_truck_trips = 8
    cows_per_helicopter = 3
    cows_per_truck = 7
    pollution_helicopter = 5
    pollution_truck = 10

    best_solution = None
    best_pollution = float('inf')

    # Brute force over feasible integer numbers of trips
    # Helicopter trips: from 0 up to a safe upper bound
    # Upper bound: in worst case, all cows by helicopter
    max_heli_trips = (cows_required + cows_per_helicopter - 1) // cows_per_helicopter

    for h in range(max_heli_trips + 1):
        for t in range(max_truck_trips + 1):
            transported_cows = cows_per_helicopter * h + cows_per_truck * t
            if transported_cows >= cows_required:
                pollution = pollution_helicopter * h + pollution_truck * t
                if pollution < best_pollution:
                    best_pollution = pollution
                    best_solution = (h, t)

    h_opt, t_opt = best_solution

    # Print solution and final objective
    print(f"Optimal helicopter trips: {h_opt}")
    print(f"Optimal truck trips: {t_opt}")
    print(f"Minimum total pollution: {best_pollution}")
    print(f"FINAL_OBJ: {best_pollution}")

if __name__ == "__main__":
    solve_transport_problem()
