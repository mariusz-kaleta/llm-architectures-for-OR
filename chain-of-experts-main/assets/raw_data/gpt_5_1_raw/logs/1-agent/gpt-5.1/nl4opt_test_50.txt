import pulp

# Define the problem
prob = pulp.LpProblem("Snack_Mix_Optimization", pulp.LpMaximize)

# Decision variables
x1 = pulp.LpVariable('kg_mix1', lowBound=0)
x2 = pulp.LpVariable('kg_mix2', lowBound=0)

# Parameters (from problem text)
cat_paw_available = 20.0
gold_shark_available = 50.0

# Mix composition
# Mix 1: 20% cat, 80% gold
# Mix 2: 35% cat, 65% gold

# Objective: Maximize profit
profit_mix1 = 12.0
profit_mix2 = 15.0
prob += profit_mix1 * x1 + profit_mix2 * x2

# Constraints
# Cat paw constraint
prob += 0.20 * x1 + 0.35 * x2 <= cat_paw_available

# Gold shark constraint
prob += 0.80 * x1 + 0.65 * x2 <= gold_shark_available

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=0))

# Retrieve results
kg_mix1_opt = pulp.value(x1)
kg_mix2_opt = pulp.value(x2)
final_obj = pulp.value(prob.objective)

# Print solution
print(f"Optimal kg of first mix: {kg_mix1_opt}")
print(f"Optimal kg of second mix: {kg_mix2_opt}")
print(f"Maximum profit: {final_obj}")
print(f"FINAL_OBJ: {final_obj}")
