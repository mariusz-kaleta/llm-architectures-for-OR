import pulp as pl

# Create the problem
prob = pl.LpProblem("Desk_Drawer_Production", pl.LpMaximize)

# Decision variables
desks = pl.LpVariable('desks', lowBound=0, cat='Continuous')
drawers = pl.LpVariable('drawers', lowBound=0, cat='Continuous')

# Objective function: Maximize profit
prob += 100 * desks + 90 * drawers

# Constraints
# Assembly time: 40 minutes per desk, 30 minutes per drawer, max 4000
prob += 40 * desks + 30 * drawers <= 4000, "Assembly_Time"

# Sanding time: 20 minutes per desk, 10 minutes per drawer, max 3500
prob += 20 * desks + 10 * drawers <= 3500, "Sanding_Time"

# Solve the problem
prob.solve(pl.PULP_CBC_CMD(msg=False))

# Get results
opt_desks = pl.value(desks)
opt_drawers = pl.value(drawers)
opt_value = pl.value(prob.objective)

# Print results
print(f"Optimal number of desks: {opt_desks}")
print(f"Optimal number of drawers: {opt_drawers}")
print(f"Maximum profit: {opt_value}")

print(f"FINAL_OBJ: {opt_value}")
