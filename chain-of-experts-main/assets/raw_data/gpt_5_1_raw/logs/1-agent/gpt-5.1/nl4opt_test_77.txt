import pulp

# Create the LP problem: minimize total waste
prob = pulp.LpProblem("Medicine_Delivery_Waste_Minimization", pulp.LpMinimize)

# Decision variables: minutes machine 1 (x1) and machine 2 (x2) are used
x1 = pulp.LpVariable('x1', lowBound=0)
x2 = pulp.LpVariable('x2', lowBound=0)

# Objective function: minimize total waste
# Machine 1: 0.3 units waste per minute
# Machine 2: 0.5 units waste per minute
prob += 0.3 * x1 + 0.5 * x2

# Constraints
# Heart: at most 8 units
prob += 0.5 * x1 + 0.3 * x2 <= 8

# Brain: at least 4 units
prob += 0.8 * x1 + 1.0 * x2 >= 4

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Retrieve results
x1_val = pulp.value(x1)
x2_val = pulp.value(x2)
final_obj = pulp.value(prob.objective)

# Print solution details (optional)
print(f"Minutes on Machine 1 (x1): {x1_val}")
print(f"Minutes on Machine 2 (x2): {x2_val}")
print(f"Total waste: {final_obj}")

# Required final objective print
print(f"FINAL_OBJ: {final_obj}")
