import pulp

# Initialize the problem
prob = pulp.LpProblem("Glass_Pane_Production", pulp.LpMaximize)

# Decision variables
regular = pulp.LpVariable('regular', lowBound=0, cat='Continuous')
tempered = pulp.LpVariable('tempered', lowBound=0, cat='Continuous')

# Objective function: maximize profit
prob += 8 * regular + 10 * tempered

# Constraints
# Heating machine time: 3 min per regular, 5 min per tempered, max 300
prob += 3 * regular + 5 * tempered <= 300

# Cooling machine time: 5 min per regular, 8 min per tempered, max 300
prob += 5 * regular + 8 * tempered <= 300

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Get results
optimal_regular = pulp.value(regular)
optimal_tempered = pulp.value(tempered)
max_profit = pulp.value(prob.objective)

# Print results
print(f"Optimal number of regular glass panes: {optimal_regular}")
print(f"Optimal number of tempered glass panes: {optimal_tempered}")
print(f"Maximum profit: {max_profit}")
print(f"FINAL_OBJ: {max_profit}")
