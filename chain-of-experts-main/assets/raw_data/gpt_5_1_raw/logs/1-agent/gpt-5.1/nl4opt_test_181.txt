import pulp

# Initialize the problem
prob = pulp.LpProblem("Medication_Time_Minimization", pulp.LpMinimize)

# Decision variables
x = pulp.LpVariable('Anxiety_Med', lowBound=0, cat='Integer')
y = pulp.LpVariable('AntiDepressant', lowBound=0, cat='Integer')

# Objective function: Minimize total time = 3x + 5y
prob += 3 * x + 5 * y, "Total_Time"

# Constraints
prob += x + y >= 100, "Total_Units_At_Least_100"
prob += x >= 30, "At_Least_30_Anxiety"
prob += x <= 2 * y, "Anxiety_At_Most_Twice_Antidepressant"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=0))

# Extract solution
x_val = pulp.value(x)
y_val = pulp.value(y)
obj_val = pulp.value(prob.objective)

# Print results
print(f"Anxiety medication units (x): {x_val}")
print(f"Anti-depressant units (y): {y_val}")
print(f"Total time: {obj_val} minutes")
print(f"FINAL_OBJ: {obj_val}")
