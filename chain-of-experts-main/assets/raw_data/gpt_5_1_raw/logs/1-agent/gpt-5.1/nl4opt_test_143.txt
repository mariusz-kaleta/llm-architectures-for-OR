import pulp

# Define the problem
prob = pulp.LpProblem("Desert_Transport_Optimization", pulp.LpMinimize)

# Decision variables
# x = number of camel caravans
# y = number of desert trucks
x = pulp.LpVariable('Camels', lowBound=0, cat='Integer')
y = pulp.LpVariable('Trucks', lowBound=0, cat='Integer')

# Parameters
units_camel = 50
time_camel = 12
units_truck = 150
time_truck = 5
demand = 1500

# Objective: Minimize total hours (sum over all trips)
prob += time_camel * x + time_truck * y, "Total_Time"

# Constraints
# Demand constraint: must deliver at least 1500 units
prob += units_camel * x + units_truck * y >= demand, "Demand"

# Preference: more camel caravans than desert trucks
prob += x >= y + 1, "More_Camels_Than_Trucks"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=0))

# Retrieve results
x_val = pulp.value(x)
y_val = pulp.value(y)
obj_val = pulp.value(prob.objective)

# Print solution
print(f"Optimal number of camel caravans: {x_val}")
print(f"Optimal number of desert trucks: {y_val}")
print(f"Minimum total hours: {obj_val}")

print(f"FINAL_OBJ: {obj_val}")
