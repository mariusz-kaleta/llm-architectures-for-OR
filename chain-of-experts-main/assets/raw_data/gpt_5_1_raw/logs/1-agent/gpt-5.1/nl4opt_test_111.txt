import pulp

# Define the problem
prob = pulp.LpProblem("Snow_Removal_Truck_Optimization", pulp.LpMaximize)

# Decision variables
S = pulp.LpVariable('Small_Trucks', lowBound=0, cat='Integer')
L = pulp.LpVariable('Large_Trucks', lowBound=0, cat='Integer')

# Objective function: Maximize total snow transported
prob += 30 * S + 50 * L

# Constraints

# People availability: 2 per small, 4 per large, total 30
prob += 2 * S + 4 * L <= 30

# At least 10 small trucks
prob += S >= 10

# At least 3 large trucks
prob += L >= 3

# Number of small trucks must be twice the number of large trucks
prob += S == 2 * L

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Retrieve results
small_trucks = pulp.value(S)
large_trucks = pulp.value(L)
final_obj = pulp.value(prob.objective)

# Print solution details (optional, but helpful)
print("Small Trucks:", small_trucks)
print("Large Trucks:", large_trucks)
print("Total Snow Transported:", final_obj)

# Final required output
print(f"FINAL_OBJ: {final_obj}")
