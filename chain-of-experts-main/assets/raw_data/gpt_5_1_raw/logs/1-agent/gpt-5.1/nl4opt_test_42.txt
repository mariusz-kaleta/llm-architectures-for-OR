import pulp as pl

# Define the problem
prob = pl.LpProblem("Body_Wash_Optimization", pl.LpMinimize)

# Decision variables
coconut = pl.LpVariable("coconut", lowBound=0)
lavender = pl.LpVariable("lavender", lowBound=0)

# Objective: Minimize total time to be effective
prob += 0.7 * coconut + 0.9 * lavender

# Constraints
prob += coconut >= 300                    # At least 300 units of coconut oil
prob += coconut <= 550                    # At most 550 units of coconut oil
prob += lavender <= 550                   # At most 550 units of lavender
prob += coconut <= 3 * lavender           # At most thrice the amount of coconut oil as lavender

# Solve the problem
prob.solve(pl.PULP_CBC_CMD(msg=False))

# Extract results
coconut_val = pl.value(coconut)
lavender_val = pl.value(lavender)
objective_val = pl.value(prob.objective)

# Print detailed results
print(f"Optimal coconut units: {coconut_val}")
print(f"Optimal lavender units: {lavender_val}")
print(f"Minimum total time: {objective_val}")

# Final required output
print(f"FINAL_OBJ: {objective_val}")
