import pulp as pl

# Initialize the problem
prob = pl.LpProblem("Radiation_Therapy_Optimization", pl.LpMinimize)

# Decision variables: minutes for Beam 1 (x1) and Beam 2 (x2)
x1 = pl.LpVariable('x1', lowBound=0)
x2 = pl.LpVariable('x2', lowBound=0)

# Objective: Minimize total radiation to pancreas
# Pancreas dose per minute: Beam 1 -> 0.3, Beam 2 -> 0.2
prob += 0.3 * x1 + 0.2 * x2, "Pancreas_Dose"

# Constraints:
# Skin dose: 0.2*x1 + 0.1*x2 <= 4
prob += 0.2 * x1 + 0.1 * x2 <= 4, "Skin_Dose"

# Tumor dose: 0.6*x1 + 0.4*x2 >= 3
prob += 0.6 * x1 + 0.4 * x2 >= 3, "Tumor_Dose"

# Solve the problem
prob.solve(pl.PULP_CBC_CMD(msg=False))

# Extract solution
x1_val = pl.value(x1)
x2_val = pl.value(x2)
obj_val = pl.value(prob.objective)

# Print solution details
print(f"Beam 1 minutes (x1): {x1_val}")
print(f"Beam 2 minutes (x2): {x2_val}")
print(f"Minimum pancreas radiation: {obj_val}")

# Required final objective print
print(f"FINAL_OBJ: {obj_val}")
