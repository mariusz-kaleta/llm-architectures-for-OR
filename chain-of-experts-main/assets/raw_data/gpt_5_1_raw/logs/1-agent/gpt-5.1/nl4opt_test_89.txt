import pulp

# Create the problem
prob = pulp.LpProblem("Diet_Sugar_Minimization", pulp.LpMinimize)

# Decision variables: number of packs of blueberries (x) and strawberries (y)
x = pulp.LpVariable('blueberries', lowBound=0, cat='Continuous')
y = pulp.LpVariable('strawberries', lowBound=0, cat='Continuous')

# Objective function: minimize total sugar intake
# 5 grams per pack of blueberries, 7 grams per pack of strawberries
prob += 5 * x + 7 * y

# Constraints:
# Anti-oxidants: 3x + 1y >= 90
prob += 3 * x + 1 * y >= 90

# Minerals: 5x + 7y >= 100
prob += 5 * x + 7 * y >= 100

# Strawberries at least 3 times blueberries: y >= 3x
prob += y >= 3 * x

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Get results
x_val = pulp.value(x)
y_val = pulp.value(y)
obj_val = pulp.value(prob.objective)

# Print solution details (numbers of packs)
print(f"Blueberries (packs): {x_val}")
print(f"Strawberries (packs): {y_val}")

# Print final objective value in required format
print(f"FINAL_OBJ: {obj_val}")
