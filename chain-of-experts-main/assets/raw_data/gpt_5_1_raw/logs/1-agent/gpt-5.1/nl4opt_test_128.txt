import pulp

# Define the problem
prob = pulp.LpProblem("Bottle_Production", pulp.LpMaximize)

# Decision variables: number of glass and plastic bottles
G = pulp.LpVariable('G', lowBound=20, cat='Integer')  # at least 20 glass bottles
P = pulp.LpVariable('P', lowBound=0, cat='Integer')

# Constraints
# Total water used cannot exceed 250000 ml
prob += 500 * G + 750 * P <= 250000

# Number of plastic bottles must be at least 3 times the number of glass bottles
prob += P >= 3 * G

# Objective: maximize total number of bottles
prob += G + P

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=0))

# Extract results
G_opt = pulp.value(G)
P_opt = pulp.value(P)
final_obj = pulp.value(prob.objective)

# Print solution details (optional prints)
print(f"Optimal number of glass bottles (G): {G_opt}")
print(f"Optimal number of plastic bottles (P): {P_opt}")

# Final objective value print as required
print(f"FINAL_OBJ: {final_obj}")
