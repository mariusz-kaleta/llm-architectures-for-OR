import pulp as pl

# Create the linear programming problem
prob = pl.LpProblem("Taco_Stand_Profit_Maximization", pl.LpMaximize)

# Decision variables
x1 = pl.LpVariable('Regular_Tacos', lowBound=0, cat='Continuous')
x2 = pl.LpVariable('Deluxe_Tacos', lowBound=0, cat='Continuous')

# Objective function: Maximize profit
prob += 2.50 * x1 + 3.55 * x2

# Constraints
prob += x1 <= 50          # Demand constraint for regular tacos
prob += x2 <= 40          # Demand constraint for deluxe tacos
prob += x1 + x2 <= 70     # Supply constraint for total tacos

# Solve the problem
prob.solve(pl.PULP_CBC_CMD(msg=False))

# Get results
x1_opt = pl.value(x1)
x2_opt = pl.value(x2)
final_obj = pl.value(prob.objective)

# Print solution details (optional prints; can be commented out)
print(f"Optimal number of regular tacos (x1): {x1_opt}")
print(f"Optimal number of deluxe tacos (x2): {x2_opt}")
print(f"Maximum profit: {final_obj}")

# Required final objective value print
print(f"FINAL_OBJ: {final_obj}")
