import pulp

# Create the LP problem: maximize profit
prob = pulp.LpProblem("RealEstateInvestment", pulp.LpMaximize)

# Decision variables: investment in apartments (A) and townhouses (T)
A = pulp.LpVariable('A', lowBound=0)
T = pulp.LpVariable('T', lowBound=0)

# Objective function: maximize total profit
prob += 0.10 * A + 0.15 * T

# Constraints
prob += A + T == 600000          # Total investment
prob += A <= 200000              # Apartments investment cap
prob += A >= 0.5 * T             # Apartments at least half of townhouses

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract results
A_val = pulp.value(A)
T_val = pulp.value(T)
obj_val = pulp.value(prob.objective)

# Print results
print(f"Investment in apartments: {A_val}")
print(f"Investment in townhouses: {T_val}")
print(f"FINAL_OBJ: {obj_val}")
