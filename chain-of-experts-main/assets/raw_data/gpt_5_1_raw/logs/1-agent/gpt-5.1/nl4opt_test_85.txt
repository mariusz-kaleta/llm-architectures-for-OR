import pulp

# Initialize the problem
prob = pulp.LpProblem("Maple_Oil_Optimization", pulp.LpMaximize)

# Decision variables: number of tanks (can be fractional, non-negative)
light = pulp.LpVariable('light', lowBound=0, cat='Continuous')
non_sticky = pulp.LpVariable('non_sticky', lowBound=0, cat='Continuous')
heavy = pulp.LpVariable('heavy', lowBound=0, cat='Continuous')

# Objective function: maximize net revenue
prob += 550 * light + 750 * non_sticky + 950 * heavy

# Constraints for compound A and B
prob += 3 * light + 6 * non_sticky + 9 * heavy <= 250  # Compound A
prob += 3 * light + 2 * non_sticky + 3 * heavy <= 150  # Compound B

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract results
light_val = pulp.value(light)
non_sticky_val = pulp.value(non_sticky)
heavy_val = pulp.value(heavy)
final_obj = pulp.value(prob.objective)

# Print solution details
print(f"Light oil tanks: {light_val}")
print(f"Non-sticky oil tanks: {non_sticky_val}")
print(f"Heavy oil tanks: {heavy_val}")
print(f"Maximum net revenue: {final_obj}")

# Required final objective print
print(f"FINAL_OBJ: {final_obj}")
