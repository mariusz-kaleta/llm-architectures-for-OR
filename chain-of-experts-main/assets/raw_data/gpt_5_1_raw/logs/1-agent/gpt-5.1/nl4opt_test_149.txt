import pulp

# Initialize the problem
problem = pulp.LpProblem("Office_Desk_Optimization", pulp.LpMaximize)

# Decision variables: number of long desks (L) and short desks (S)
L = pulp.LpVariable('Long_Desks', lowBound=0, cat='Integer')
S = pulp.LpVariable('Short_Desks', lowBound=0, cat='Integer')

# Objective: Maximize seating availability
# Long desk seats 6, Short desk seats 2
problem += 6 * L + 2 * S, "Total_Seating"

# Constraints
# Budget: 300L + 100S <= 2000
problem += 300 * L + 100 * S <= 2000, "Budget_Constraint"

# Space: 10L + 4S <= 200
problem += 10 * L + 4 * S <= 200, "Space_Constraint"

# Solve the problem
problem.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract the optimal values
optimal_L = pulp.value(L)
optimal_S = pulp.value(S)
final_obj = pulp.value(problem.objective)

# Print the solution details (optional, not requested but useful for verification)
print(f"Optimal number of long desks: {optimal_L}")
print(f"Optimal number of short desks: {optimal_S}")
print(f"Maximum seating: {final_obj}")

# Required final objective print
print(f"FINAL_OBJ: {final_obj}")
