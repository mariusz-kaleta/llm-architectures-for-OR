import pulp

# Define the problem
prob = pulp.LpProblem("Science_Show_Optimization", pulp.LpMaximize)

# Decision variables: number of times each demonstration is performed
x1 = pulp.LpVariable('Demo1', lowBound=0, cat='Continuous')
x2 = pulp.LpVariable('Demo2', lowBound=0, cat='Continuous')

# Objective: Maximize total minty foam
# Demo1 produces 25 units, Demo2 produces 18 units
prob += 25 * x1 + 18 * x2

# Constraints:
# Mint: 10 units for Demo1, 12 units for Demo2, total available 120
prob += 10 * x1 + 12 * x2 <= 120

# Active ingredient: 20 units for Demo1, 15 units for Demo2, total available 100
prob += 20 * x1 + 15 * x2 <= 100

# Black tar: 5 units for Demo1, 3 units for Demo2, at most 50
prob += 5 * x1 + 3 * x2 <= 50

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Retrieve results
demo1_val = pulp.value(x1)
demo2_val = pulp.value(x2)
final_obj = pulp.value(prob.objective)

# Print solution
print(f"Number of Demonstration 1: {demo1_val}")
print(f"Number of Demonstration 2: {demo2_val}")
print(f"Maximum Minty Foam Produced: {final_obj}")

print(f"FINAL_OBJ: {final_obj}")
