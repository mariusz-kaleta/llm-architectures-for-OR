import pulp as pl

# Define the problem
prob = pl.LpProblem("Minimize_Fat_Intake", pl.LpMinimize)

# Decision variables: number of fish meals (x) and chicken meals (y)
x = pl.LpVariable('fish_meals', lowBound=0, cat='Continuous')
y = pl.LpVariable('chicken_meals', lowBound=0, cat='Continuous')

# Objective: Minimize total fat intake
# Each fish meal: 7 units of fat
# Each chicken meal: 10 units of fat
prob += 7 * x + 10 * y, "Total_Fat"

# Constraints:
# Protein: 10x + 15y >= 130
prob += 10 * x + 15 * y >= 130, "Protein_Requirement"

# Iron: 12x + 8y >= 120
prob += 12 * x + 8 * y >= 120, "Iron_Requirement"

# At least twice as many chicken meals as fish meals: y >= 2x
prob += y >= 2 * x, "Chicken_to_Fish_Ratio"

# Solve the problem
prob.solve(pl.PULP_CBC_CMD(msg=False))

# Get the results
fish_meals = pl.value(x)
chicken_meals = pl.value(y)
final_obj = pl.value(prob.objective)

# Print the solution details
print(f"Optimal number of fish meals: {fish_meals}")
print(f"Optimal number of chicken meals: {chicken_meals}")
print(f"Minimum total fat intake: {final_obj}")

# Required final objective print
print(f"FINAL_OBJ: {final_obj}")
