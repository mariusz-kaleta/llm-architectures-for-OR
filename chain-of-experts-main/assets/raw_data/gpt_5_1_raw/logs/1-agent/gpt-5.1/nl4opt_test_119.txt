import pulp

# Initialize the problem
prob = pulp.LpProblem("Mail_Delivery_Min_Gas", pulp.LpMinimize)

# Decision variables: number of trips by regular (R) and speed (S) boats
R = pulp.LpVariable('R', lowBound=0, cat='Continuous')
S = pulp.LpVariable('S', lowBound=0, cat='Continuous')

# Objective: Minimize total gas consumption
prob += 10 * R + 20 * S

# Constraints

# Mail capacity: at least 1000 pieces
prob += 20 * R + 30 * S >= 1000, "MailRequirement"

# At most 20 regular boat trips
prob += R <= 20, "RegularTripLimit"

# At least 50% of trips must be speed boats: S >= R
prob += S >= R, "SpeedTripShare"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

R_opt = pulp.value(R)
S_opt = pulp.value(S)
obj_val = pulp.value(prob.objective)

# Print solution details (optional for understanding; not required by prompt)
print(f"Regular boat trips (R): {R_opt}")
print(f"Speed boat trips (S): {S_opt}")
print(f"Total gas used: {obj_val}")

# Final required output
print(f"FINAL_OBJ: {obj_val}")
