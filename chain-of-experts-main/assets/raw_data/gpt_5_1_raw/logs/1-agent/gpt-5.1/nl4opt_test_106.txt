import pulp

# Initialize the problem
prob = pulp.LpProblem("Mark_Farm_Profit_Maximization", pulp.LpMaximize)

# Decision variables: acres of potatoes (P) and cucumbers (C)
P = pulp.LpVariable('Potatoes', lowBound=0, cat='Continuous')
C = pulp.LpVariable('Cucumbers', lowBound=0, cat='Continuous')

# Objective function: maximize profit
profit_per_acre_potatoes = 500
profit_per_acre_cucumbers = 650
prob += profit_per_acre_potatoes * P + profit_per_acre_cucumbers * C

# Constraints

# Total land constraint
prob += P + C <= 50, "Total_Land"

# Contract requirements
prob += P >= 12, "Min_Potatoes"
prob += C >= 15, "Min_Cucumbers"

# Preference: more cucumbers than potatoes
prob += C >= P, "More_Cucumbers_than_Potatoes"

# Resource limit: at most twice the amount of cucumbers as potatoes
prob += C <= 2 * P, "Cucumbers_At_Most_Twice_Potatoes"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=0))

# Extract optimal values
opt_P = pulp.value(P)
opt_C = pulp.value(C)
opt_profit = pulp.value(prob.objective)

# Print solution details (optional, but useful)
print(f"Optimal acres of potatoes: {opt_P}")
print(f"Optimal acres of cucumbers: {opt_C}")
print(f"Maximum profit: {opt_profit}")

# Required final output
print(f"FINAL_OBJ: {opt_profit}")
