import pulp

# Define the problem
prob = pulp.LpProblem("Breakfast_Sandwich_Optimization", pulp.LpMaximize)

# Decision variables
regular = pulp.LpVariable('regular', lowBound=0, cat='Integer')
special = pulp.LpVariable('special', lowBound=0, cat='Integer')

# Objective function: Maximize profit = 3R + 4S
prob += 3 * regular + 4 * special

# Constraints
# Eggs: 2R + 3S <= 40
prob += 2 * regular + 3 * special <= 40

# Bacon: 3R + 5S <= 70
prob += 3 * regular + 5 * special <= 70

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract results
reg_val = pulp.value(regular)
spec_val = pulp.value(special)
final_obj = pulp.value(prob.objective)

# Print solution details (optional for understanding)
print(f"Regular sandwiches: {reg_val}")
print(f"Special sandwiches: {spec_val}")
print(f"Maximum Profit: {final_obj}")

# Required final objective print
print(f"FINAL_OBJ: {final_obj}")
