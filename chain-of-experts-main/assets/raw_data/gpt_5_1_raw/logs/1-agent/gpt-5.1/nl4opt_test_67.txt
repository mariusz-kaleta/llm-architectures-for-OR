import pulp

# Define the problem
prob = pulp.LpProblem("Lawn_Optimization", pulp.LpMinimize)

# Decision variables
fertilizer = pulp.LpVariable('fertilizer', lowBound=50, cat='Continuous')  # at least 50 units
seeds = pulp.LpVariable('seeds', lowBound=0, cat='Continuous')

# Objective function: minimize total time
prob += 0.5 * fertilizer + 1.5 * seeds

# Constraints
prob += fertilizer + seeds <= 300          # at most 300 units combined
prob += fertilizer <= 2 * seeds            # at most twice the amount of fertilizer as seeds

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Retrieve optimal values
opt_fertilizer = pulp.value(fertilizer)
opt_seeds = pulp.value(seeds)
opt_obj = pulp.value(prob.objective)

# Print solution details
print(f"Optimal fertilizer units: {opt_fertilizer}")
print(f"Optimal seed units: {opt_seeds}")
print(f"Minimum total time: {opt_obj}")

# Final required output
print(f"FINAL_OBJ: {opt_obj}")
