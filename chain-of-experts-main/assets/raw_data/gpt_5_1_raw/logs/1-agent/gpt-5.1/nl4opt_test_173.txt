import pulp

# Initialize the problem
prob = pulp.LpProblem("Desktop_Production_Problem", pulp.LpMaximize)

# Decision variables: number of premium (P) and regular (R) desktops
P = pulp.LpVariable('Premium', lowBound=0, cat='Integer')
R = pulp.LpVariable('Regular', lowBound=0, cat='Integer')

# Objective function: maximize profit
# Profit: 500 per premium, 300 per regular
prob += 500 * P + 300 * R, "Total_Profit"

# Constraints
# 1. At most 200 desktops
prob += P + R <= 200, "Total_Desktops_Limit"

# 2. Production cost constraint: at most 300000
# Premium cost: 2000 each, Regular cost: 1000 each
prob += 2000 * P + 1000 * R <= 300000, "Budget_Constraint"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=0))

# Extract results
premium_qty = int(pulp.value(P))
regular_qty = int(pulp.value(R))
final_profit = pulp.value(prob.objective)

# Print detailed solution (optional)
print(f"Premium desktops to produce: {premium_qty}")
print(f"Regular desktops to produce: {regular_qty}")
print(f"Maximum profit: {final_profit}")

# Required final objective print
print(f"FINAL_OBJ: {final_profit}")
