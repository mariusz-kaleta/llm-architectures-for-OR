import pulp

# Define the problem
prob = pulp.LpProblem("Factory_Optimization", pulp.LpMinimize)

# Decision variables: number of medium (M) and small (S) factories
M = pulp.LpVariable('M', lowBound=0, cat='Integer')
S = pulp.LpVariable('S', lowBound=0, cat='Integer')

# Objective: minimize total number of factories
prob += M + S, "Total_Factories"

# Constraints
# Toy production constraint: 50M + 35S >= 250
prob += 50 * M + 35 * S >= 250, "Toy_Production"

# Operator constraint: 3M + 2S <= 16
prob += 3 * M + 2 * S <= 16, "Operator_Limit"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Get results
M_opt = pulp.value(M)
S_opt = pulp.value(S)
obj_val = pulp.value(prob.objective)

# Print solution details (optional)
print(f"Optimal number of medium factories: {M_opt}")
print(f"Optimal number of small factories: {S_opt}")

# Required final objective print
print(f"FINAL_OBJ: {obj_val}")
