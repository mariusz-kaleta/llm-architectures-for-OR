import pulp

# Define the problem
prob = pulp.LpProblem("Pizza_Baking_Time_Minimization", pulp.LpMinimize)

# Decision variables
L = pulp.LpVariable('Large_Pizzas', lowBound=0, cat='Continuous')
M = pulp.LpVariable('Medium_Pizzas', lowBound=0, cat='Continuous')

# Objective: Minimize total baking time
prob += 12 * L + 8 * M

# Constraints
# Dough constraint: at least 10000 units
prob += 12 * L + 8 * M >= 10000

# Toppings constraint: at least 4400 units
prob += 5 * L + 4 * M >= 4400

# At least 200 medium pizzas
prob += M >= 200

# At least two times as many large pizzas as medium pizzas
prob += L >= 2 * M

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Get results
L_opt = pulp.value(L)
M_opt = pulp.value(M)
obj_value = pulp.value(prob.objective)

# Print decision variables and objective
print(f"Optimal number of large pizzas: {L_opt}")
print(f"Optimal number of medium pizzas: {M_opt}")
print(f"Minimum total baking time (minutes): {obj_value}")

print(f"FINAL_OBJ: {obj_value}")
