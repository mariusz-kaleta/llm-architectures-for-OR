import pulp

# Create the problem
prob = pulp.LpProblem("Wine_Bottling_Problem", pulp.LpMaximize)

# Decision variables
v = pulp.LpVariable('Vintage', lowBound=10, cat='Integer')   # at least 10 vintage bottles
r = pulp.LpVariable('Regular', lowBound=0, cat='Integer')

# Objective: maximize total number of bottles
prob += v + r, "Total_Bottles"

# Constraints
# Volume constraint: 500 ml per vintage, 750 ml per regular, total 100000 ml
prob += 500 * v + 750 * r <= 100000, "Volume_Constraint"

# Regular at least 4 times vintage
prob += r >= 4 * v, "Regular_At_Least_4_Times_Vintage"

# Solve
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Retrieve results
v_opt = pulp.value(v)
r_opt = pulp.value(r)
obj_val = pulp.value(prob.objective)

# Print detailed solution
print(f"Optimal number of vintage bottles: {v_opt}")
print(f"Optimal number of regular bottles: {r_opt}")
print(f"Maximum total number of bottles: {obj_val}")

# Required final objective print
print(f"FINAL_OBJ: {obj_val}")
