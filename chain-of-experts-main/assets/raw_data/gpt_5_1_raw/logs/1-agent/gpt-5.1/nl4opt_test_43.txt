import itertools

def solve():
    # Decision variables:
    # x = number of dual model machines (integer, >= 0)
    # y = number of single model machines (integer, >= 0, and y > x)

    best_solution = None
    best_obj = None

    # We know we must stamp at least 300 letters per minute:
    # 50x + 30y >= 300
    # And glue constraint: 20x + 15y <= 135
    #
    # Rough bounds: 
    # Max machines if all singles under glue: 15*y <= 135 -> y <= 9
    # Max machines if all duals under glue: 20*x <= 135 -> x <= 6
    # So x in [0,6], y in [0,9] is enough to enumerate.

    for x in range(0, 7):
        for y in range(0, 10):
            # Constraint: number of single model machines must be more than dual
            if y <= x:
                continue
            # Letters per minute constraint
            if 50 * x + 30 * y < 300:
                continue
            # Glue per minute constraint
            if 20 * x + 15 * y > 135:
                continue

            obj = x + y  # minimize total number of machines
            if best_obj is None or obj < best_obj:
                best_obj = obj
                best_solution = (x, y)

    if best_solution is None:
        print("No feasible solution found.")
        print("FINAL_OBJ: None")
    else:
        x_opt, y_opt = best_solution
        # Optional: print the solution details
        print(f"Optimal number of dual machines (x): {x_opt}")
        print(f"Optimal number of single machines (y): {y_opt}")
        print(f"Minimum total number of machines: {best_obj}")
        print(f"Letters per minute: {50 * x_opt + 30 * y_opt}")
        print(f"Glue units per minute: {20 * x_opt + 15 * y_opt}")
        print(f"FINAL_OBJ: {best_obj}")

if __name__ == "__main__":
    solve()
