import pulp

# Create the LP problem: maximize profit
model = pulp.LpProblem("Flooring_Production", pulp.LpMaximize)

# Decision variables: square feet of laminate (x) and carpet (y)
x = pulp.LpVariable('laminate', lowBound=0)
y = pulp.LpVariable('carpet', lowBound=0)

# Objective function: maximize profit = 2.1x + 3.3y
model += 2.1 * x + 3.3 * y

# Constraints
model += x >= 15000          # Demand for laminate
model += y >= 5000           # Demand for carpet
model += x + y >= 50000      # Shipping contract total requirement
model += x <= 40000          # Laminate production capacity
model += y <= 20000          # Carpet production capacity

# Solve the model
model.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract results
laminate_opt = pulp.value(x)
carpet_opt = pulp.value(y)
final_obj = pulp.value(model.objective)

# Print solution
print(f"Optimal laminate production (sq ft): {laminate_opt}")
print(f"Optimal carpet production (sq ft): {carpet_opt}")
print(f"Maximum weekly profit: {final_obj}")

# Required final objective print
print(f"FINAL_OBJ: {final_obj}")
