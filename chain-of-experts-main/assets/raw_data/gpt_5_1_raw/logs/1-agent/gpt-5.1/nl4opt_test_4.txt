import pulp as pl

# Initialize the problem: minimize cost
prob = pl.LpProblem("Feed_Mix_Problem", pl.LpMinimize)

# Decision variables: kilograms of Feed A and Feed B (non-negative)
x_A = pl.LpVariable('Feed_A_kg', lowBound=0)
x_B = pl.LpVariable('Feed_B_kg', lowBound=0)

# Objective function: Minimize total cost
# Feed A costs $100/kg, Feed B costs $80/kg
prob += 100 * x_A + 80 * x_B, "Total_Cost"

# Constraints:
# Protein: 10 units/kg from A, 7 units/kg from B; need at least 30 units
prob += 10 * x_A + 7 * x_B >= 30, "Protein_Requirement"

# Fat: 8 units/kg from A, 15 units/kg from B; need at least 50 units
prob += 8 * x_A + 15 * x_B >= 50, "Fat_Requirement"

# Solve the problem
prob.solve(pl.PULP_CBC_CMD(msg=False))

# Extract the objective value
final_obj = pl.value(prob.objective)

# Print the final objective value in the required format
print(f"FINAL_OBJ: {final_obj}")
