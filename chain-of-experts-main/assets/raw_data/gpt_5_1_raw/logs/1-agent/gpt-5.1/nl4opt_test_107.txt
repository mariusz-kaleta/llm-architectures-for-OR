import pulp

# Define the problem
prob = pulp.LpProblem("Berry_Farmer_Problem", pulp.LpMinimize)

# Decision variables
# x = number of days operating old farm
# y = number of days operating new farm
x = pulp.LpVariable('x_old_farm_days', lowBound=0, cat='Continuous')
y = pulp.LpVariable('y_new_farm_days', lowBound=0, cat='Continuous')

# Objective function: Minimize total cost
prob += 300 * x + 200 * y, "Total_Operating_Cost"

# Constraints
# Raspberries: 2x + 4y >= 10
prob += 2 * x + 4 * y >= 10, "Raspberry_Requirement"

# Blueberries: 2x + 1y >= 9
prob += 2 * x + 1 * y >= 9, "Blueberry_Requirement"

# Strawberries: 4x + 2y >= 15
prob += 4 * x + 2 * y >= 15, "Strawberry_Requirement"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Get the final objective value
final_obj_value = pulp.value(prob.objective)

# Print variable values (optional, can be commented out if not needed)
# for v in prob.variables():
#     print(v.name, "=", v.varValue)

# Print the final objective value in required format
print(f"FINAL_OBJ: {final_obj_value}")
