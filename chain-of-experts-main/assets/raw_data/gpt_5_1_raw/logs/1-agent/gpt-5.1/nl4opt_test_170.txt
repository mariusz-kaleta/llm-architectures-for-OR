import pulp

# Create the problem
problem = pulp.LpProblem("Bee_Farmer_Bottle_Optimization", pulp.LpMaximize)

# Decision variables
x_s = pulp.LpVariable('small_bottles', lowBound=0, cat='Integer')
x_l = pulp.LpVariable('large_bottles', lowBound=0, cat='Integer')

# Objective: Maximize honey transported
problem += 5 * x_s + 20 * x_l

# Constraints
problem += x_s <= 300               # At most 300 small bottles
problem += x_l <= 100               # At most 100 large bottles
problem += x_s >= 2 * x_l           # At least twice as many small as large
problem += x_s + x_l <= 200         # At most 200 bottles total
problem += x_l >= 50                # At least 50 large bottles

# Solve the problem
problem.solve(pulp.PULP_CBC_CMD(msg=0))

# Retrieve results
small_opt = pulp.value(x_s)
large_opt = pulp.value(x_l)
objective_value = pulp.value(problem.objective)

# Print solution
print(f"Optimal number of small bottles: {small_opt}")
print(f"Optimal number of large bottles: {large_opt}")
print(f"Maximum honey transported: {objective_value} units")

print(f"FINAL_OBJ: {objective_value}")
