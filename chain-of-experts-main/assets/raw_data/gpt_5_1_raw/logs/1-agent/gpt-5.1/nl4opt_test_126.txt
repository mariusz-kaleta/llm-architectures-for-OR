import pulp

# Create the problem
problem = pulp.LpProblem("Florist_Bouquet_Optimization", pulp.LpMaximize)

# Decision variables
s = pulp.LpVariable('small_bouquets', lowBound=0, cat='Integer')
l = pulp.LpVariable('large_bouquets', lowBound=0, cat='Integer')

# Objective: maximize total number of flowers
problem += 5 * s + 10 * l

# Constraints
problem += s <= 80, "Max_small_bouquets"
problem += l <= 50, "Max_large_bouquets"
problem += s + l <= 70, "Max_total_bouquets"
problem += l >= 20, "Min_large_bouquets"
problem += s >= 2 * l, "Small_at_least_twice_large"

# Solve the problem
problem.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract results
small_opt = pulp.value(s)
large_opt = pulp.value(l)
final_obj = pulp.value(problem.objective)

# Print solution details (optional)
print(f"Optimal number of small bouquets: {small_opt}")
print(f"Optimal number of large bouquets: {large_opt}")
print(f"Maximum total number of flowers: {final_obj}")

# Required final objective print
print(f"FINAL_OBJ: {final_obj}")
