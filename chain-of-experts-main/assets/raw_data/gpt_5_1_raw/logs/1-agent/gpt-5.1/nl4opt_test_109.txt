import pulp

# Create the problem: minimize total number of boxes
prob = pulp.LpProblem("Mask_Box_Optimization", pulp.LpMinimize)

# Decision variables: number of small and large boxes
s = pulp.LpVariable('small_boxes', lowBound=0, cat='Integer')
l = pulp.LpVariable('large_boxes', lowBound=0, cat='Integer')

# Objective: minimize total number of boxes
prob += s + l

# Constraints:
# At least three times as many small boxes as large boxes: s >= 3l
prob += s >= 3 * l

# At least 5 large boxes
prob += l >= 5

# At least 750 masks: 25s + 45l >= 750
prob += 25 * s + 45 * l >= 750

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract solution
small_boxes_opt = int(pulp.value(s))
large_boxes_opt = int(pulp.value(l))
final_obj = pulp.value(prob.objective)

# Print results
print("Optimal number of small boxes:", small_boxes_opt)
print("Optimal number of large boxes:", large_boxes_opt)
print(f"FINAL_OBJ: {final_obj}")
