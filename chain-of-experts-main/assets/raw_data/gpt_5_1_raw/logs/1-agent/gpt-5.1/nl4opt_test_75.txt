import pulp

# Initialize the problem: minimize number of vans
prob = pulp.LpProblem("Shoe_Transportation", pulp.LpMinimize)

# Decision variables: number of vans (v) and trucks (t), non-negative integers
v = pulp.LpVariable('vans', lowBound=0, cat='Integer')
t = pulp.LpVariable('trucks', lowBound=0, cat='Integer')

# Objective: minimize number of vans
prob += v, "Minimize_vans"

# Constraints:
# 1) Capacity constraint: 50*v + 100*t >= 2000
prob += 50 * v + 100 * t >= 2000, "Capacity"

# 2) Number of trucks cannot exceed number of vans: t <= v
prob += t <= v, "Truck_leq_Van"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract solution
v_opt = pulp.value(v)
t_opt = pulp.value(t)

# Compute objective value (minimum number of vans)
final_obj = v_opt

# Print results (as requested, only final objective in required format)
print(f"FINAL_OBJ: {int(final_obj)}")
