import pulp

# Create the problem
prob = pulp.LpProblem("Advertising_Exposure_Maximization", pulp.LpMaximize)

# Decision variables: number of radio ads (x) and social media ads (y)
x = pulp.LpVariable('radio_ads', lowBound=15, upBound=40, cat='Integer')
y = pulp.LpVariable('social_media_ads', lowBound=35, cat='Integer')

# Parameters
budget = 250000
cost_radio = 5000
cost_social = 9150
exposure_radio = 60500
exposure_social = 50000

# Objective: maximize total exposure
prob += exposure_radio * x + exposure_social * y, "Total_Exposure"

# Budget constraint
prob += cost_radio * x + cost_social * y <= budget, "Budget_Constraint"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Retrieve results
radio_ads_opt = pulp.value(x)
social_media_ads_opt = pulp.value(y)
total_exposure_opt = pulp.value(prob.objective)

# Print solution details (optional prints for understanding; not required by instructions)
print(f"Optimal number of radio ads: {radio_ads_opt}")
print(f"Optimal number of social media ads: {social_media_ads_opt}")
print(f"Total exposure: {total_exposure_opt}")

# Final objective value print as required
print(f"FINAL_OBJ: {total_exposure_opt}")
