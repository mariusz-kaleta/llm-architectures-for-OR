import pulp as pl

# Define the problem
prob = pl.LpProblem("Syrup_Sugar_Minimization", pl.LpMinimize)

# Decision variables: servings of syrup 1 (x1) and syrup 2 (x2)
x1 = pl.LpVariable('x1', lowBound=0)
x2 = pl.LpVariable('x2', lowBound=0)

# Objective: minimize total sugar intake
# Syrup 1: 0.5 units sugar per serving
# Syrup 2: 0.3 units sugar per serving
prob += 0.5 * x1 + 0.3 * x2, "Total_Sugar"

# Constraints:
# Throat medicine: 0.5*x1 + 0.2*x2 <= 5
prob += 0.5 * x1 + 0.2 * x2 <= 5, "Throat_Constraint"

# Lungs medicine: 0.4*x1 + 0.5*x2 >= 4
prob += 0.4 * x1 + 0.5 * x2 >= 4, "Lungs_Constraint"

# Solve the problem
prob.solve(pl.PULP_CBC_CMD(msg=False))

# Extract results
x1_val = pl.value(x1)
x2_val = pl.value(x2)
obj_val = pl.value(prob.objective)

# Print solution
print(f"Servings of syrup 1 (x1): {x1_val}")
print(f"Servings of syrup 2 (x2): {x2_val}")
print(f"Minimum total sugar intake: {obj_val}")

print(f"FINAL_OBJ: {obj_val}")
