import pulp

# Create the problem
prob = pulp.LpProblem("Vaccine_Administration", pulp.LpMaximize)

# Decision variables
P = pulp.LpVariable('Pills', lowBound=30, cat='Integer')   # at least 30 pills
S = pulp.LpVariable('Shots', lowBound=0, cat='Integer')

# Objective: maximize number of patients vaccinated
prob += P + S

# Constraints

# Time constraint: 10 minutes per pill, 20 minutes per shot, total <= 10000
prob += 10 * P + 20 * S <= 10000

# At least 3 times as many shots as pills
prob += S >= 3 * P

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Get results
P_opt = pulp.value(P)
S_opt = pulp.value(S)
obj_value = pulp.value(prob.objective)

# Print solution details (optional)
print(f"Optimal number of pills: {P_opt}")
print(f"Optimal number of shots: {S_opt}")

# Required final objective print
print(f"FINAL_OBJ: {obj_value}")
