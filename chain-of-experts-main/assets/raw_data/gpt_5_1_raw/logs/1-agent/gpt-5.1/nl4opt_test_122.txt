import pulp

# Initialize the problem
prob = pulp.LpProblem("Mrs_Watson_Real_Estate_Investment", pulp.LpMaximize)

# Decision variables:
# x: dollars invested in condos
# y: dollars invested in detached houses
x = pulp.LpVariable('Condos', lowBound=0, cat='Continuous')
y = pulp.LpVariable('Detached_Houses', lowBound=0, cat='Continuous')

# Objective function: maximize total profit
# Profit: $0.50 per dollar in condos, $1 per dollar in detached houses
prob += 0.5 * x + 1.0 * y, "Total_Profit"

# Constraints:

# 1. Total budget constraint: x + y <= 760000
prob += x + y <= 760000, "Budget_Constraint"

# 2. Minimum 20% of all money invested must be in condos
#    x >= 0.2 * (x + y)  ->  0.8x - 0.2y >= 0
prob += 0.8 * x - 0.2 * y >= 0, "Condos_Min_20_Percent"

# 3. At least $20000 must be in detached houses: y >= 20000
prob += y >= 20000, "Detached_Houses_Min_20000"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Get the final objective value
final_obj = pulp.value(prob.objective)

# Print the final objective value in the required format
print(f"FINAL_OBJ: {final_obj}")
