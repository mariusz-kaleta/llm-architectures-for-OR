import pulp

# Define the problem
prob = pulp.LpProblem("Souvenir_Shop_Production", pulp.LpMaximize)

# Decision variables: number of elephants (E) and tigers (T)
E = pulp.LpVariable('Elephants', lowBound=0, cat='Continuous')
T = pulp.LpVariable('Tigers', lowBound=0, cat='Continuous')

# Objective function: Maximize profit
# Profit per elephant = 5, profit per tiger = 4
prob += 5 * E + 4 * T, "Total_Profit"

# Constraints:
# Wood constraint: 50E + 40T <= 5000
prob += 50 * E + 40 * T <= 5000, "Wood_Constraint"

# Plastic constraint: 20E + 30T <= 4000
prob += 20 * E + 30 * T <= 4000, "Plastic_Constraint"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Get the results
optimal_elephants = pulp.value(E)
optimal_tigers = pulp.value(T)
optimal_profit = pulp.value(prob.objective)

# Print results
print(f"Optimal number of elephants to produce: {optimal_elephants}")
print(f"Optimal number of tigers to produce: {optimal_tigers}")
print(f"Maximum profit: {optimal_profit}")

# Final required print
print(f"FINAL_OBJ: {optimal_profit}")
