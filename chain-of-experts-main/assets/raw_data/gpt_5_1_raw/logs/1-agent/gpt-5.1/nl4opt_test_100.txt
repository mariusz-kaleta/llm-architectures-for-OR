import pulp

# Initialize the problem
prob = pulp.LpProblem("Model_Trains_and_Planes", pulp.LpMaximize)

# Decision variables
trains = pulp.LpVariable('trains', lowBound=0, cat='Continuous')
planes = pulp.LpVariable('planes', lowBound=0, cat='Continuous')

# Parameters
wood_available = 120
paint_available = 90
wood_train = 3
paint_train = 3
wood_plane = 4
paint_plane = 2
profit_train = 8
profit_plane = 10

# Objective function: Maximize profit
prob += profit_train * trains + profit_plane * planes

# Constraints
prob += wood_train * trains + wood_plane * planes <= wood_available, "Wood_Constraint"
prob += paint_train * trains + paint_plane * planes <= paint_available, "Paint_Constraint"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=0))

# Get results
optimal_trains = pulp.value(trains)
optimal_planes = pulp.value(planes)
optimal_profit = pulp.value(prob.objective)

# Print detailed results (optional for user understanding)
print("Optimal number of trains to make:", optimal_trains)
print("Optimal number of planes to make:", optimal_planes)
print("Maximum profit:", optimal_profit)

# Required final objective value print
print(f"FINAL_OBJ: {optimal_profit}")
