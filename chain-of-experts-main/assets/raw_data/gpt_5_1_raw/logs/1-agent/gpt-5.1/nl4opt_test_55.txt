import pulp

# Define the LP problem
prob = pulp.LpProblem("Berry_Farm_Profit_Maximization", pulp.LpMaximize)

# Decision variables: acres of blueberries (B) and raspberries (R)
B = pulp.LpVariable('Blueberries', lowBound=0, cat='Continuous')
R = pulp.LpVariable('Raspberries', lowBound=0, cat='Continuous')

# Objective function: Maximize profit
# Profit per acre: Blueberries = 56, Raspberries = 75
prob += 56 * B + 75 * R, "Total_Profit"

# Constraints:

# 1. Land constraint: total acres cannot exceed 300
prob += B + R <= 300, "Land_Constraint"

# 2. Labor constraint: total labor days cannot exceed 575
# Blueberries require 6 days per acre; Raspberries require 3 days per acre
prob += 6 * B + 3 * R <= 575, "Labor_Constraint"

# 3. Watering cost constraint: total watering cost cannot exceed $10000
# Blueberries cost $22 per acre; Raspberries cost $25 per acre
prob += 22 * B + 25 * R <= 10000, "Water_Constraint"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Retrieve the optimal objective value
final_obj_value = pulp.value(prob.objective)

# Print the result in the required format
print(f"FINAL_OBJ: {final_obj_value}")
