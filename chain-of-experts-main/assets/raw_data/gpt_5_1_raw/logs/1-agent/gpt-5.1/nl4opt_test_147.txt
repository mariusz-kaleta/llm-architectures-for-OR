import itertools

def solve_transportation_problem():
    # Decision variables (non-negative integers):
    # c: number of cars
    # b: number of buses
    #
    # Constraints:
    # 1) Capacity: 4c + 20b >= 300
    # 2) Bus limit: b <= 4
    #
    # Objective: Minimize pollution = 10c + 30b

    best_obj = None
    best_solution = None

    # We know at most 4 buses, and in the worst case, if 0 buses, all by car:
    # 4c >= 300 -> c >= 75. So search a reasonable range for cars, say 0..100
    for b in range(0, 5):  # b <= 4
        for c in range(0, 101):
            # Capacity constraint
            if 4 * c + 20 * b >= 300:
                obj = 10 * c + 30 * b
                if best_obj is None or obj < best_obj:
                    best_obj = obj
                    best_solution = (c, b)

    c_opt, b_opt = best_solution

    # Print solution details (optional; not required by the problem statement)
    print(f"Optimal number of cars: {c_opt}")
    print(f"Optimal number of buses: {b_opt}")
    print(f"Minimum total pollution: {best_obj}")

    # Required final output format
    print(f"FINAL_OBJ: {best_obj}")

if __name__ == "__main__":
    solve_transportation_problem()
