import pulp

# Define the problem
prob = pulp.LpProblem("Van_Selection_Problem", pulp.LpMinimize)

# Decision variables: number of regular vans (R) and hybrid vans (H)
R = pulp.LpVariable('Regular_Vans', lowBound=0, cat='Integer')
H = pulp.LpVariable('Hybrid_Vans', lowBound=0, cat='Integer')

# Parameters
packages_regular = 500
packages_hybrid = 300
pollution_regular = 200
pollution_hybrid = 100
max_pollution = 7000
min_packages = 20000

# Objective: minimize total number of vans
prob += R + H, "Total_Number_of_Vans"

# Constraints
# Pollution constraint
prob += pollution_regular * R + pollution_hybrid * H <= max_pollution, "Pollution_Constraint"

# Delivery capacity constraint
prob += packages_regular * R + packages_hybrid * H >= min_packages, "Package_Constraint"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=0))

# Extract solution
R_opt = pulp.value(R)
H_opt = pulp.value(H)
obj_value = pulp.value(prob.objective)

# Print results
print(f"Regular_Vans: {R_opt}")
print(f"Hybrid_Vans: {H_opt}")
print(f"Total_Vans: {obj_value}")
print(f"FINAL_OBJ: {obj_value}")
