import pulp

# Initialize the problem
prob = pulp.LpProblem("Vitamin_Clinic_Production", pulp.LpMaximize)

# Decision variables: number of batches of vitamin shots (s) and pills (p)
s = pulp.LpVariable('s', lowBound=0, cat='Continuous')
p = pulp.LpVariable('p', lowBound=0, cat='Continuous')

# Objective: Maximize number of people supplied
prob += 10 * s + 7 * p

# Constraints
prob += 30 * s + 50 * p <= 1200  # Vitamin C constraint
prob += 40 * s + 30 * p <= 1500  # Vitamin D constraint
prob += p >= s + 1               # Pills more popular: p > s, modeled as p >= s + 1
prob += s <= 10                  # At most 10 batches of vitamin shots

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Retrieve the results
s_val = pulp.value(s)
p_val = pulp.value(p)
final_obj = pulp.value(prob.objective)

# Print solution details
print(f"Optimal number of vitamin shot batches (s): {s_val}")
print(f"Optimal number of vitamin pill batches (p): {p_val}")
print(f"Maximum number of people supplied: {final_obj}")

# Final objective value in required format
print(f"FINAL_OBJ: {final_obj}")
