import pulp

# Initialize the problem: minimize total number of grills
prob = pulp.LpProblem("Burger_Grill_Selection", pulp.LpMinimize)

# Decision variables: number of gas grills (G) and electric grills (E)
G = pulp.LpVariable('G', lowBound=0, cat='Integer')
E = pulp.LpVariable('E', lowBound=0, cat='Integer')

# Objective: minimize total number of grills
prob += G + E, "Total_Grills"

# Constraints:
# Cooking capacity: at least 150 patties per minute
prob += 20 * G + 30 * E >= 150, "Capacity_Constraint"

# Oil usage: at most 140 units per minute
prob += 20 * G + 25 * E <= 140, "Oil_Constraint"

# Preference: number of electric grills must be less than number of gas grills
prob += E <= G - 1, "Preference_Constraint"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract solution
G_opt = pulp.value(G)
E_opt = pulp.value(E)
final_obj = pulp.value(prob.objective)

# Print results
print(f"Optimal number of gas grills (G): {G_opt}")
print(f"Optimal number of electric grills (E): {E_opt}")
print(f"Minimum total number of grills: {final_obj}")
print(f"FINAL_OBJ: {final_obj}")
