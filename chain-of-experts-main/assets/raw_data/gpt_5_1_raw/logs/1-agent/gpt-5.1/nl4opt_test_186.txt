import pulp

# Initialize the problem
prob = pulp.LpProblem("Mine_Wagon_Optimization", pulp.LpMinimize)

# Decision variables
S = pulp.LpVariable('Small_Wagons', lowBound=0, cat='Integer')
L = pulp.LpVariable('Large_Wagons', lowBound=0, cat='Integer')

# Objective: Minimize total number of wagons
prob += S + L

# Constraints
# Capacity constraint: 20*S + 50*L >= 2000
prob += 20 * S + 50 * L >= 2000

# At least 10 large wagons
prob += L >= 10

# Number of small wagons at least twice the number of large wagons
prob += S >= 2 * L

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=0))

# Get the final objective value
final_obj = pulp.value(prob.objective)

print(f"Small Wagons: {pulp.value(S)}")
print(f"Large Wagons: {pulp.value(L)}")
print(f"Total Wagons: {final_obj}")
print(f"FINAL_OBJ: {final_obj}")
