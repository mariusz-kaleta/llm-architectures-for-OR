import pulp

# Create the problem
prob = pulp.LpProblem("Maximize_Sports_Equipment", pulp.LpMaximize)

# Decision variables
B = pulp.LpVariable('Basketballs', lowBound=0, cat='Continuous')
F = pulp.LpVariable('Footballs', lowBound=50, cat='Continuous')  # At least 50 footballs

# Objective: maximize total number of sports equipment
prob += B + F

# Constraints

# Material constraint: 5B + 3F <= 1500
prob += 5 * B + 3 * F <= 1500

# Labor constraint: 1B + 2F <= 750
prob += B + 2 * F <= 750

# At least three times as many basketballs as footballs: B >= 3F
prob += B >= 3 * F

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Retrieve results
B_val = pulp.value(B)
F_val = pulp.value(F)
obj_val = pulp.value(prob.objective)

# Print solution details (optional for user inspection)
print(f"Basketballs: {B_val}")
print(f"Footballs: {F_val}")
print(f"Total sports equipment: {obj_val}")

# Final required output
print(f"FINAL_OBJ: {obj_val}")
