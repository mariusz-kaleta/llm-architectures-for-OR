import pulp

# Initialize the problem
prob = pulp.LpProblem("Diet_Problem", pulp.LpMinimize)

# Decision variables: servings of Rice (r) and Kebabs (k)
r = pulp.LpVariable('Rice_Servings', lowBound=0, cat='Continuous')
k = pulp.LpVariable('Kebab_Servings', lowBound=0, cat='Continuous')

# Objective function: Minimize total cost
# Rice: $3 per serving, Kebab: $2 per serving
prob += 3 * r + 2 * k, "Total_Cost"

# Constraints:
# Calorie constraint: at least 2200 calories
# Rice: 300 calories/serving, Kebab: 200 calories/serving
prob += 300 * r + 200 * k >= 2200, "Calorie_Requirement"

# Protein constraint: at least 30 grams of protein
# Rice: 4.5 g/serving, Kebab: 4 g/serving
prob += 4.5 * r + 4 * k >= 30, "Protein_Requirement"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=0))

# Extract final objective value
final_obj = pulp.value(prob.objective)

# Print solution details (optional for inspection)
print("Status:", pulp.LpStatus[prob.status])
print("Rice Servings:", r.varValue)
print("Kebab Servings:", k.varValue)
print("Minimum Cost:", final_obj)

# Required final objective print
print(f"FINAL_OBJ: {final_obj}")
