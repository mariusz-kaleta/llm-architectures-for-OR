import pulp

# Initialize the problem: maximize total gifts
prob = pulp.LpProblem("ToyStoreVolunteers", pulp.LpMaximize)

# Decision variables: number of seasonal (S) and full-time (F) volunteers
S = pulp.LpVariable('Seasonal', lowBound=0, cat='Integer')
F = pulp.LpVariable('FullTime', lowBound=0, cat='Integer')

# Objective: maximize total gifts delivered
# Seasonal: 5 gifts each, Full-time: 8 gifts each
prob += 5 * S + 8 * F, "TotalGifts"

# Constraints:

# Points constraint: Seasonal: 2 points, Full-time: 5 points, total <= 200
prob += 2 * S + 5 * F <= 200, "PointsLimit"

# At least 10 full-time volunteers
prob += F >= 10, "MinFullTime"

# At most 30% of volunteers can be seasonal: S <= 0.3 * (S + F)
# This can be rearranged to 0.7*S <= 0.3*F -> 7S <= 3F
prob += 7 * S <= 3 * F, "SeasonalLimit"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=0))

# Retrieve values
S_val = int(pulp.value(S))
F_val = int(pulp.value(F))
total_gifts = 5 * S_val + 8 * F_val

# Print the solution and final objective in required format
print(f"Seasonal volunteers: {S_val}")
print(f"Full-time volunteers: {F_val}")
print(f"Total gifts delivered: {total_gifts}")
print(f"FINAL_OBJ: {total_gifts}")
