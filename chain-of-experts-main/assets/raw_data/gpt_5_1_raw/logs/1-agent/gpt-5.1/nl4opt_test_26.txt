import pulp

# Create the problem
prob = pulp.LpProblem("Workbook_Production", pulp.LpMaximize)

# Decision variables
M = pulp.LpVariable('Math_Workbooks', lowBound=0, cat='Continuous')
E = pulp.LpVariable('English_Workbooks', lowBound=0, cat='Continuous')

# Objective function: Maximize profit
prob += 15 * M + 17 * E

# Constraints
prob += M >= 40          # At least 40 math workbooks
prob += M <= 140         # At most 140 math workbooks
prob += E >= 60          # At least 60 English workbooks
prob += E <= 170         # At most 170 English workbooks
prob += M + E >= 200     # At least 200 workbooks total

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract the results
M_opt = pulp.value(M)
E_opt = pulp.value(E)
final_obj = pulp.value(prob.objective)

# Print solution details (optional prints; not required but useful)
print(f"Optimal Math Workbooks: {M_opt}")
print(f"Optimal English Workbooks: {E_opt}")
print(f"Maximum Profit: {final_obj}")

# Required final objective print
print(f"FINAL_OBJ: {final_obj}")
