import pulp

# Define the problem
prob = pulp.LpProblem("Investment_Problem", pulp.LpMaximize)

# Decision variables: amounts invested in telecom (T) and healthcare (H)
T = pulp.LpVariable('Telecom', lowBound=0)
H = pulp.LpVariable('Healthcare', lowBound=0)

# Total available funds
total_funds = 100000

# Objective function: maximize total profit (0.03*T + 0.01*H)
prob += 0.03 * T + 0.01 * H

# Constraints
# 1. Total investment cannot exceed available funds
prob += T + H <= total_funds

# 2. Telecom investment at least three times healthcare: T >= 3H
prob += T - 3 * H >= 0

# 3. Telecom investment at most 70000
prob += T <= 70000

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract the optimal values
T_opt = pulp.value(T)
H_opt = pulp.value(H)
profit_opt = pulp.value(prob.objective)

# Print results
print(f"Optimal investment in Telecom: {T_opt}")
print(f"Optimal investment in Healthcare: {H_opt}")
print(f"Maximum profit: {profit_opt}")

print(f"FINAL_OBJ: {profit_opt}")
