import pulp

# Initialize the problem: minimize total number of vehicles
prob = pulp.LpProblem("Daycare_Transportation", pulp.LpMinimize)

# Decision variables: number of buses (B) and personal cars (C)
B = pulp.LpVariable('B', lowBound=0, cat='Integer')
C = pulp.LpVariable('C', lowBound=0, cat='Integer')

# Objective: minimize total number of vehicles
prob += B + C, "Total_Vehicles"

# Constraints:
# 1) Carry at least 100 children: 9B + 4C >= 100
prob += 9 * B + 4 * C >= 100, "Capacity_Constraint"

# 2) More buses than personal cars: B >= C + 1
prob += B >= C + 1, "More_Buses_Than_Cars"

# 3) At least 5 personal cars: C >= 5
prob += C >= 5, "Min_Cars"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract results
optimal_B = int(pulp.value(B))
optimal_C = int(pulp.value(C))
optimal_obj = pulp.value(prob.objective)

# Print detailed solution
print(f"Optimal number of buses (B): {optimal_B}")
print(f"Optimal number of personal cars (C): {optimal_C}")
print(f"Total vehicles: {optimal_obj}")

# Required final objective print
print(f"FINAL_OBJ: {optimal_obj}")
