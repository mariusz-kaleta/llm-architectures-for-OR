import pulp

# Define the problem
prob = pulp.LpProblem("Vitamin_Mix_Optimization", pulp.LpMinimize)

# Decision variables: cups of drink A (x) and B (y)
x = pulp.LpVariable('x', lowBound=0)
y = pulp.LpVariable('y', lowBound=0)

# Vitamin content per cup:
# Drink A: A=8, D=6, E=10, K=4
# Drink B: A=15, D=2, E=20, K=12

# Objective: Minimize total Vitamin K = 4x + 12y
prob += 4 * x + 12 * y

# Constraints:
# At least 150 units of Vitamin A: 8x + 15y >= 150
prob += 8 * x + 15 * y >= 150, "Vitamin_A_min"

# At least 300 units of Vitamin D: 6x + 2y >= 300
prob += 6 * x + 2 * y >= 300, "Vitamin_D_min"

# At most 400 units of Vitamin E: 10x + 20y <= 400
prob += 10 * x + 20 * y <= 400, "Vitamin_E_max"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=0))

# Retrieve results
x_opt = pulp.value(x)
y_opt = pulp.value(y)
obj_val = pulp.value(prob.objective)

# Print solution and final objective value
print(f"Optimal cups of Drink A (x): {x_opt}")
print(f"Optimal cups of Drink B (y): {y_opt}")
print(f"Minimum total Vitamin K: {obj_val}")
print(f"FINAL_OBJ: {obj_val}")
