import pulp

# Initialize the problem
prob = pulp.LpProblem("Catalyst_Optimization", pulp.LpMaximize)

# Decision variables: number of palladium-heavy (x) and platinum-heavy (y) catalysts
x = pulp.LpVariable('Palladium_Heavy', lowBound=0, cat='Continuous')
y = pulp.LpVariable('Platinum_Heavy', lowBound=0, cat='Continuous')

# Objective function: maximize conversion rate (units per second)
# Palladium-heavy: 5 units/sec, Platinum-heavy: 4 units/sec
prob += 5 * x + 4 * y, "Total_Conversion_Rate"

# Constraints
# Platinum: 15x + 20y <= 450
prob += 15 * x + 20 * y <= 450, "Platinum_Constraint"

# Palladium: 25x + 14y <= 390
prob += 25 * x + 14 * y <= 390, "Palladium_Constraint"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract the results
x_val = pulp.value(x)
y_val = pulp.value(y)
obj_val = pulp.value(prob.objective)

# Print results (decision variables and final objective)
print(f"Palladium-heavy catalysts (x): {x_val}")
print(f"Platinum-heavy catalysts (y): {y_val}")
print(f"Maximum conversion rate (units per second): {obj_val}")

print(f"FINAL_OBJ: {obj_val}")
