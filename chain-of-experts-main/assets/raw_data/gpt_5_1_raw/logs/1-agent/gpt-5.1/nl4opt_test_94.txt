import pulp

# Define the problem
problem = pulp.LpProblem("Minimize_Trips", pulp.LpMinimize)

# Decision variables: number of ferry trips (f) and light rail trips (l)
f = pulp.LpVariable('f', lowBound=0, cat='Integer')
l = pulp.LpVariable('l', lowBound=0, cat='Integer')

# Objective: Minimize total number of trips
problem += f + l, "Total_Trips"

# Constraints:
# Capacity constraint: 20 boxes per ferry, 15 per light rail, at least 500 boxes total
problem += 20 * f + 15 * l >= 500, "Capacity_Constraint"

# Light rail trips must be at least 4 times ferry trips
problem += l >= 4 * f, "LightRail_Ferry_Ratio"

# Solve the problem
problem.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract the objective value
final_obj = pulp.value(problem.objective)

# Print the final objective value in the required format
print(f"FINAL_OBJ: {final_obj}")
