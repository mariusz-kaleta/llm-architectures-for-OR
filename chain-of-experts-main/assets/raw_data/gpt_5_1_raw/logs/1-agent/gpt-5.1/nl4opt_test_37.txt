import pulp

# Define the problem
problem = pulp.LpProblem("Terracotta_Jars_Optimization", pulp.LpMaximize)

# Decision variables: number of thin jars (x) and stubby jars (y)
x = pulp.LpVariable('Thin_Jars', lowBound=0, cat='Continuous')
y = pulp.LpVariable('Stubby_Jars', lowBound=0, cat='Continuous')

# Objective function: Maximize profit
# Profit: $5 per thin jar, $9 per stubby jar
problem += 5 * x + 9 * y

# Constraints
# Shaping time: 50x + 30y <= 3000 minutes
problem += 50 * x + 30 * y <= 3000

# Baking time: 90x + 150y <= 4000 minutes
problem += 90 * x + 150 * y <= 4000

# Solve the problem
problem.solve(pulp.PULP_CBC_CMD(msg=0))

# Get results
thin_jars = pulp.value(x)
stubby_jars = pulp.value(y)
final_profit = pulp.value(problem.objective)

# Print decisions (optional for debugging, but not requested by problem)
# print(f"Thin jars to produce: {thin_jars}")
# print(f"Stubby jars to produce: {stubby_jars}")

# Final required output
print(f"FINAL_OBJ: {final_profit}")
