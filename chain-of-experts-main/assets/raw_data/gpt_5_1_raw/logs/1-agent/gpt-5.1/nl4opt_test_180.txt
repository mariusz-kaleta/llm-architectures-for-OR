import pulp

# Initialize the problem: minimize sugar intake
prob = pulp.LpProblem("Vitamin_Intake_Problem", pulp.LpMinimize)

# Decision variables: number of oranges (x) and grapefruits (y)
x = pulp.LpVariable('oranges', lowBound=0, cat='Continuous')
y = pulp.LpVariable('grapefruits', lowBound=0, cat='Continuous')

# Objective function: minimize total sugar
# Each orange: 5 g sugar, each grapefruit: 6 g sugar
prob += 5 * x + 6 * y, "Total_Sugar"

# Constraints:
# Vitamin C: 5x + 7y >= 80
prob += 5 * x + 7 * y >= 80, "Vitamin_C"

# Vitamin A: 3x + 5y >= 70
prob += 3 * x + 5 * y >= 70, "Vitamin_A"

# Preference constraint: x >= 2y
prob += x >= 2 * y, "Preference_Oranges"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=0))

# Retrieve the optimal values
opt_x = pulp.value(x)
opt_y = pulp.value(y)
opt_obj = pulp.value(prob.objective)

# Print solution details (optional, but useful)
print(f"Optimal number of oranges: {opt_x}")
print(f"Optimal number of grapefruits: {opt_y}")
print(f"Minimum sugar intake: {opt_obj} grams")

# Final required output
print(f"FINAL_OBJ: {opt_obj}")
