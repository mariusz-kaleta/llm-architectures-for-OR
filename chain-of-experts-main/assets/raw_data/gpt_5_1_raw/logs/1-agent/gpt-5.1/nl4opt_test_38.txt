import itertools

def solve():
    # Parameters
    chickens_needed = 1200
    bus_capacity = 100
    car_capacity = 40
    bus_time = 2.0
    car_time = 1.5
    max_bus_trips = 10
    car_trip_ratio = 0.6  # at least 60% of trips must be by car

    best_solution = None
    best_time = float('inf')

    # Enumerate possible trips
    # Bus trips: 0 to 10
    # Car trips: let's upper bound reasonably. In worst case, all by car: 1200/40 = 30, so search 0..30
    for b in range(0, max_bus_trips + 1):
        for c in range(0, 31):
            # Capacity constraint
            if b * bus_capacity + c * car_capacity < chickens_needed:
                continue

            # At least 60% of trips must be by car: c >= 0.6 * (b + c)
            if b + c == 0:
                continue
            if c < car_trip_ratio * (b + c):
                continue

            # Compute total time
            total_time = b * bus_time + c * car_time

            if total_time < best_time:
                best_time = total_time
                best_solution = (b, c)

    if best_solution is None:
        raise ValueError("No feasible solution found.")

    b_opt, c_opt = best_solution
    print(f"Optimal number of bus trips: {b_opt}")
    print(f"Optimal number of car trips: {c_opt}")
    print(f"Minimum total time: {best_time}")
    print(f"FINAL_OBJ: {best_time}")

if __name__ == "__main__":
    solve()
