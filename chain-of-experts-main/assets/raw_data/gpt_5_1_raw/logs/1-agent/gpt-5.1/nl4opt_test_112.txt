import pulp

# Create the LP problem: minimize total time
prob = pulp.LpProblem("Chemical_Mix_Time_Minimization", pulp.LpMinimize)

# Decision variables: units of A and B
A = pulp.LpVariable('A', lowBound=0)
B = pulp.LpVariable('B', lowBound=0)

# Objective: minimize total time = 30*A + 45*B
prob += 30 * A + 45 * B

# Constraints:
# At most a third as much A as B: A <= (1/3) * B
prob += A <= (1.0 / 3.0) * B

# At least 300 units of A
prob += A >= 300

# At least 1500 units total
prob += A + B >= 1500

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract results
A_val = A.value()
B_val = B.value()
objective_value = pulp.value(prob.objective)

# Print solution (for reference)
print(f"Optimal A (units): {A_val}")
print(f"Optimal B (units): {B_val}")
print(f"Minimum total time (seconds): {objective_value}")

# Final required print
print(f"FINAL_OBJ: {objective_value}")
