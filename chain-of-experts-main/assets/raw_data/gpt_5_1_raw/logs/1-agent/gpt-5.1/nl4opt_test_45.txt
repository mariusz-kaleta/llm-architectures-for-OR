import pulp as pl

# Create the problem
prob = pl.LpProblem("Oil_Production_Problem", pl.LpMaximize)

# Decision variables: number of containers of Oil Max (x1) and Oil Max Pro (x2)
x1 = pl.LpVariable('Oil_Max', lowBound=0, cat='Continuous')
x2 = pl.LpVariable('Oil_Max_Pro', lowBound=0, cat='Continuous')

# Objective function: maximize profit
prob += 10 * x1 + 15 * x2

# Constraints based on substance availability
# Substance A: 46x1 + 13x2 <= 1345
prob += 46 * x1 + 13 * x2 <= 1345

# Substance B: 43x1 + 4x2 <= 346
prob += 43 * x1 + 4 * x2 <= 346

# Substance C: 56x1 + 45x2 <= 1643
prob += 56 * x1 + 45 * x2 <= 1643

# Solve the problem
prob.solve(pl.PULP_CBC_CMD(msg=0))

# Retrieve the solution
x1_val = pl.value(x1)
x2_val = pl.value(x2)
obj_val = pl.value(prob.objective)

# Print solution details
print(f"Optimal number of Oil Max containers: {x1_val}")
print(f"Optimal number of Oil Max Pro containers: {x2_val}")
print(f"Maximum Profit: {obj_val}")

# Final required print
print(f"FINAL_OBJ: {obj_val}")
