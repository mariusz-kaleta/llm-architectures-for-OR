import pulp

# Define the problem
prob = pulp.LpProblem("Metal_Extraction", pulp.LpMaximize)

# Decision variables: number of times each process is performed
J = pulp.LpVariable('J', lowBound=0, cat='Continuous')
P = pulp.LpVariable('P', lowBound=0, cat='Continuous')

# Objective function: maximize total metal extracted
prob += 5 * J + 9 * P

# Constraints
prob += 8 * J + 6 * P <= 1500  # Water constraint
prob += 3 * J + 5 * P <= 1350  # Pollution constraint

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Retrieve results
J_opt = pulp.value(J)
P_opt = pulp.value(P)
obj_value = pulp.value(prob.objective)

# Print solution details (optional)
print(f"Optimal number of process J: {J_opt}")
print(f"Optimal number of process P: {P_opt}")
print(f"Maximum metal extracted: {obj_value}")

# Required final objective print
print(f"FINAL_OBJ: {obj_value}")
