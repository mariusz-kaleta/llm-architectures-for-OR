import pulp

# Create the problem: minimization
prob = pulp.LpProblem("Shampoo_Mix_Problem", pulp.LpMinimize)

# Decision variables: x = units of sulfate, y = units of ginger
x = pulp.LpVariable('sulfate_units', lowBound=0)
y = pulp.LpVariable('ginger_units', lowBound=0)

# Objective function: minimize total effectiveness time
# Each unit of sulfate: 0.5 minutes, ginger: 0.75 minutes
prob += 0.5 * x + 0.75 * y

# Constraints:
# 1) At least 100 units of sulfate
prob += x >= 100

# 2) Total of both ingredients must be 400 units
prob += x + y == 400

# 3) At most twice the amount of sulfate as ginger: x <= 2y
prob += x <= 2 * y

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract results
x_val = pulp.value(x)
y_val = pulp.value(y)
obj_val = pulp.value(prob.objective)

# Print solution details (optional for understanding; not required by instructions)
print(f"Sulfate units (x): {x_val}")
print(f"Ginger units (y): {y_val}")
print(f"Minimum total effective time: {obj_val}")

# Required final objective value print
print(f"FINAL_OBJ: {obj_val}")
