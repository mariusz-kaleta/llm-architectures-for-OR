import pulp

# Initialize the problem: maximize slime
prob = pulp.LpProblem("SummerCampSlimeMaximization", pulp.LpMaximize)

# Decision variables: number of times each beaker is used
x1 = pulp.LpVariable('Beaker1', lowBound=0)
x2 = pulp.LpVariable('Beaker2', lowBound=0)

# Objective function: maximize total slime
# Beaker 1 -> 5 units slime, Beaker 2 -> 3 units slime
prob += 5 * x1 + 3 * x2, "Total_Slime"

# Constraints:
# Flour: 4 units for Beaker 1, 6 units for Beaker 2; total available 150
prob += 4 * x1 + 6 * x2 <= 150, "Flour_Constraint"

# Special liquid: 6 units for Beaker 1, 3 units for Beaker 2; total available 100
prob += 6 * x1 + 3 * x2 <= 100, "Liquid_Constraint"

# Waste: 4 units from Beaker 1, 2 units from Beaker 2; at most 30 units
prob += 4 * x1 + 2 * x2 <= 30, "Waste_Constraint"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Retrieve solution
beaker1_used = pulp.value(x1)
beaker2_used = pulp.value(x2)
final_obj = pulp.value(prob.objective)

# Print solution details
print(f"Beaker 1 used: {beaker1_used}")
print(f"Beaker 2 used: {beaker2_used}")
print(f"Maximum slime produced: {final_obj}")

# Final required output
print(f"FINAL_OBJ: {final_obj}")
