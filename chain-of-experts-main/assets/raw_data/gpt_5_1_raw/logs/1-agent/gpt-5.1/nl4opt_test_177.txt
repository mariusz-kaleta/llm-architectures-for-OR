import pulp

# Initialize the problem: minimize number of boats
prob = pulp.LpProblem("Fishing_Transport_Problem", pulp.LpMinimize)

# Decision variables: number of canoes (C) and diesel boats (D)
C = pulp.LpVariable('C', lowBound=0, cat='Integer')
D = pulp.LpVariable('D', lowBound=0, cat='Integer')

# Objective: minimize total number of boats
prob += C + D, "Total_Boats"

# Constraints:
# Fish capacity: 10*C + 15*D >= 1000
prob += 10 * C + 15 * D >= 1000, "Fish_Requirement"

# Environmental rule: C >= 3*D
prob += C >= 3 * D, "Environmental_Rule"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract results
C_val = pulp.value(C)
D_val = pulp.value(D)
obj_val = pulp.value(prob.objective)

# Print detailed solution (optional)
print(f"Optimal number of canoes: {C_val}")
print(f"Optimal number of diesel boats: {D_val}")
print(f"Minimum total number of boats: {obj_val}")

# Final required output
print(f"FINAL_OBJ: {obj_val}")
