import pulp

# Define the problem: maximize total vitamin D intake
prob = pulp.LpProblem("Maximize_Vitamin_D", pulp.LpMaximize)

# Decision variables: number of orange and apple juice boxes
O = pulp.LpVariable('Orange', lowBound=0, cat='Continuous')
A = pulp.LpVariable('Apple', lowBound=0, cat='Continuous')

# Objective function: maximize 10*O + 12*A
prob += 10 * O + 12 * A, "Total_Vitamin_D"

# Constraints:

# At least 3 times as many apple juice boxes as orange juice boxes: A >= 3O
prob += A >= 3 * O, "Apple_at_least_3_times_Orange"

# At least 3 orange juice boxes: O >= 3
prob += O >= 3, "At_least_3_Orange"

# Vitamin C limit: 8O + 6A <= 300
prob += 8 * O + 6 * A <= 300, "Vitamin_C_Limit"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=0))

# Retrieve the results
O_opt = pulp.value(O)
A_opt = pulp.value(A)
obj_value = pulp.value(prob.objective)

# Print solution
print(f"Optimal number of orange juice boxes: {O_opt}")
print(f"Optimal number of apple juice boxes: {A_opt}")
print(f"Maximum total vitamin D intake: {obj_value}")

# Final required print
print(f"FINAL_OBJ: {obj_value}")
