import pulp

# Initialize the problem: minimize total number of workers
prob = pulp.LpProblem("Worker_Scheduling", pulp.LpMinimize)

# Decision variables
F = pulp.LpVariable('FullTime', lowBound=0, cat='Integer')
P = pulp.LpVariable('PartTime', lowBound=0, cat='Integer')

# Parameters
hours_required = 500
budget = 15000
fulltime_hours = 8
parttime_hours = 4
fulltime_cost = 300
parttime_cost = 100

# Objective: minimize number of workers
prob += F + P, "TotalWorkers"

# Constraints
# Hours requirement
prob += fulltime_hours * F + parttime_hours * P >= hours_required, "HoursRequirement"

# Budget constraint
prob += fulltime_cost * F + parttime_cost * P <= budget, "BudgetConstraint"

# Solve
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract solution
F_val = int(pulp.value(F))
P_val = int(pulp.value(P))
obj_val = pulp.value(prob.objective)

# Print solution details (optional, not requested but useful for verification)
print(f"Full-time workers: {F_val}")
print(f"Part-time workers: {P_val}")
print(f"Total workers: {obj_val}")

# Required final output
print(f"FINAL_OBJ: {obj_val}")
