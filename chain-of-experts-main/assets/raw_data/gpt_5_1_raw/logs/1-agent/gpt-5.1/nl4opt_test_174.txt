import pulp

# Initialize the problem
prob = pulp.LpProblem("Maximize_Patients", pulp.LpMaximize)

# Decision variables
x_e = pulp.LpVariable('Electronic_Thermometer_Patients', lowBound=0, cat='Integer')
x_r = pulp.LpVariable('Regular_Thermometer_Patients', lowBound=50, cat='Integer')  # at least 50

# Constraints
prob += 3 * x_e + 2 * x_r <= 15000  # time constraint
prob += x_e >= 2 * x_r              # at least twice as many electronic as regular

# Objective function
prob += x_e + x_r

# Solve
prob.solve(pulp.PULP_CBC_CMD(msg=0))

# Get the objective value
final_obj = pulp.value(prob.objective)

# Print the result
print(f"FINAL_OBJ: {int(final_obj)}")
