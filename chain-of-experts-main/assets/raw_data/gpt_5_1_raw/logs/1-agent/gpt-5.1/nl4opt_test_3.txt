import pulp

# Create the problem
prob = pulp.LpProblem("IceCreamMachines", pulp.LpMinimize)

# Decision variables: number of each machine type (non-negative integers)
x = pulp.LpVariable('counter_top', lowBound=0, cat='Integer')
y = pulp.LpVariable('fridge_size', lowBound=0, cat='Integer')

# Objective: minimize total number of machines
prob += x + y, "Total_Machines"

# Constraints:
# Production constraint: at least 1000 cones
prob += 80 * x + 150 * y >= 1000, "ProductionConstraint"

# Heat constraint: at most 500 units of heat
prob += 50 * x + 70 * y <= 500, "HeatConstraint"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract results
counter_top = pulp.value(x)
fridge_size = pulp.value(y)
final_obj = pulp.value(prob.objective)

# Print results
print(f"Counter-top machines: {counter_top}")
print(f"Fridge-size machines: {fridge_size}")
print(f"Total machines: {final_obj}")
print(f"FINAL_OBJ: {final_obj}")
