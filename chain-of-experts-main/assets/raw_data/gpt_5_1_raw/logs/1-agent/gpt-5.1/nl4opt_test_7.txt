import pulp as pl

# Define the problem
prob = pl.LpProblem("Elm_Furniture_Production", pl.LpMaximize)

# Decision variables: number of chairs (C) and dressers (D)
C = pl.LpVariable('Chairs', lowBound=0)
D = pl.LpVariable('Dressers', lowBound=0)

# Objective function: maximize profit
prob += 43 * C + 52 * D

# Constraints
# Stain constraint: 1.4C + 1.1D <= 17
prob += 1.4 * C + 1.1 * D <= 17

# Wood constraint: 2C + 3D <= 11
prob += 2 * C + 3 * D <= 11

# Solve the problem
prob.solve(pl.PULP_CBC_CMD(msg=False))

# Fetch the maximum profit
final_obj = pl.value(prob.objective)

# Print final objective value in requested format
print(f"FINAL_OBJ: {final_obj}")
