import pulp

# Define the problem: minimize number of lifts
prob = pulp.LpProblem("Ski_Lift_Planning", pulp.LpMinimize)

# Decision variables: number of densely-seated (x) and loosely-seated (y) lifts
x = pulp.LpVariable('dense_lifts', lowBound=0, cat='Integer')
y = pulp.LpVariable('loose_lifts', lowBound=5, cat='Integer')  # at least 5 loosely-seated lifts

# Objective: minimize total number of lifts
prob += x + y, "Total_Lifts"

# Constraints:
# Capacity: 45x + 20y >= 1000 guests per minute
prob += 45 * x + 20 * y >= 1000, "Capacity_Constraint"

# Electricity: 30x + 22y <= 940 units
prob += 30 * x + 22 * y <= 940, "Electricity_Constraint"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Retrieve results
x_val = pulp.value(x)
y_val = pulp.value(y)
obj_val = pulp.value(prob.objective)

# Print detailed solution
print(f"Densely-seated lifts (x): {x_val}")
print(f"Loosely-seated lifts (y): {y_val}")
print(f"Minimum total number of lifts: {obj_val}")

# Final required output
print(f"FINAL_OBJ: {obj_val}")
