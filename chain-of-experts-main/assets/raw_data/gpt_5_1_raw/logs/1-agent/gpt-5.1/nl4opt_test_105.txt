import pulp

# Create the LP problem: maximize profit
prob = pulp.LpProblem("Meat_Processing_Profit_Maximization", pulp.LpMaximize)

# Decision variables: batches of hams (H) and pork ribs (R)
H = pulp.LpVariable('H', lowBound=0, cat='Continuous')
R = pulp.LpVariable('R', lowBound=0, cat='Continuous')

# Objective function: maximize profit
prob += 150 * H + 300 * R

# Constraints:
# Meat slicer time: 4H + 2R <= 4000
prob += 4 * H + 2 * R <= 4000, "Slicer_Time"

# Meat packer time: 2.5H + 3.5R <= 4000
prob += 2.5 * H + 3.5 * R <= 4000, "Packer_Time"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract results
H_opt = pulp.value(H)
R_opt = pulp.value(R)
obj_val = pulp.value(prob.objective)

# Print solution details (optional)
print(f"Optimal batches of hams (H): {H_opt}")
print(f"Optimal batches of pork ribs (R): {R_opt}")
print(f"Maximum profit: {obj_val}")

# Required final objective print
print(f"FINAL_OBJ: {obj_val}")
