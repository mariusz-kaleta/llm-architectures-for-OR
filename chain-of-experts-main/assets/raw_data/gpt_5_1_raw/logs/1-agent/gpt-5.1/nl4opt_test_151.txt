import pulp

# Initialize the problem: maximize zinc intake
prob = pulp.LpProblem("Maximize_Zinc_Intake", pulp.LpMaximize)

# Decision variables: number of gummies (g) and pills (p)
g = pulp.LpVariable('g', lowBound=0, cat='Continuous')
p = pulp.LpVariable('p', lowBound=0, cat='Continuous')

# Objective function: maximize total zinc
# Each gummy: 4 units zinc, each pill: 5 units zinc
prob += 4 * g + 5 * p, "Total_Zinc"

# Constraints:
# 1. At least 10 pills
prob += p >= 10, "Min_Pills"

# 2. At least 3 times the amount of gummies as pills: g >= 3p
prob += g >= 3 * p, "Gummies_At_Least_Three_Times_Pills"

# 3. Magnesium constraint: at most 200 units
# Each gummy: 3 units Mg, each pill: 2 units Mg
prob += 3 * g + 2 * p <= 200, "Magnesium_Limit"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=0))

g_value = pulp.value(g)
p_value = pulp.value(p)
obj_value = pulp.value(prob.objective)

print(f"Optimal number of gummies: {g_value}")
print(f"Optimal number of pills: {p_value}")
print(f"Maximum zinc intake: {obj_value}")
print(f"FINAL_OBJ: {obj_value}")
