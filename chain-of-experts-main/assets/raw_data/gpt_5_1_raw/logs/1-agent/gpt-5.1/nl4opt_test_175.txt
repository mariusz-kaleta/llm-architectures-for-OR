import pulp

# Initialize the problem: minimize total number of trips
prob = pulp.LpProblem("Chocolate_Transport", pulp.LpMinimize)

# Decision variables: number of van trips (v) and truck trips (t)
v = pulp.LpVariable('v', lowBound=0, cat='Integer')
t = pulp.LpVariable('t', lowBound=0, cat='Integer')

# Objective: minimize total number of trips
prob += v + t, "TotalTrips"

# Constraints

# Capacity: each van trip carries 50 boxes, each truck trip 80 boxes
prob += 50 * v + 80 * t >= 1500, "CapacityConstraint"

# Budget: van trip costs 30, truck trip costs 50, total budget 1000
prob += 30 * v + 50 * t <= 1000, "BudgetConstraint"

# Advertising requirement: number of van trips must be larger than number of truck trips
prob += v >= t + 1, "AdvertisingConstraint"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract solution
v_opt = pulp.value(v)
t_opt = pulp.value(t)
obj_value = pulp.value(prob.objective)

# Print results
print(f"Optimal van trips: {v_opt}")
print(f"Optimal truck trips: {t_opt}")
print(f"Minimum total trips: {obj_value}")

print(f"FINAL_OBJ: {obj_value}")
