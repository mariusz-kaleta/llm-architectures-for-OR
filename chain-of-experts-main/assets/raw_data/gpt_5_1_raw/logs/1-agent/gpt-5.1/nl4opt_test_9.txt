import pulp

# Create the problem
prob = pulp.LpProblem("Maximize_Paste", pulp.LpMaximize)

# Decision variables: number of small and large containers
x_small = pulp.LpVariable('x_small', lowBound=0, cat='Continuous')
x_large = pulp.LpVariable('x_large', lowBound=0, cat='Continuous')

# Parameters from the problem
water_small = 10
pill_small = 15
paste_small = 20

water_large = 20
pill_large = 20
paste_large = 30

water_available = 500
pill_available = 700

# Objective: maximize total paste
prob += paste_small * x_small + paste_large * x_large, "Total_Paste"

# Constraints
prob += water_small * x_small + water_large * x_large <= water_available, "Water_Constraint"
prob += pill_small * x_small + pill_large * x_large <= pill_available, "Pill_Constraint"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=0))

# Extract results
x_small_val = pulp.value(x_small)
x_large_val = pulp.value(x_large)
final_obj = pulp.value(prob.objective)

# Print solution details
print(f"Number of small containers: {x_small_val}")
print(f"Number of large containers: {x_large_val}")
print(f"Maximum amount of paste: {final_obj}")

# Required final objective print
print(f"FINAL_OBJ: {final_obj}")
