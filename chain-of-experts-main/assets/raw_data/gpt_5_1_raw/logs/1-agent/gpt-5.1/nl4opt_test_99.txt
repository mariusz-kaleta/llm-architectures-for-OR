import pulp

# Initialize the problem
prob = pulp.LpProblem("Lawn_Mowing_Teams", pulp.LpMaximize)

# Decision variables: number of small teams (S) and large teams (L)
S = pulp.LpVariable('Small_Teams', lowBound=0, cat='Continuous')
L = pulp.LpVariable('Large_Teams', lowBound=0, cat='Continuous')

# Objective function: maximize lawn area mowed
prob += 50 * S + 80 * L, "Total_Lawn_Area"

# Constraints
# Employee availability
prob += 3 * S + 5 * L <= 150, "Employee_Constraint"

# Small teams at least three times large teams
prob += S >= 3 * L, "Small_at_least_3_times_Large"

# Minimum numbers of teams
prob += L >= 6, "Min_Large_Teams"
prob += S >= 10, "Min_Small_Teams"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract results
small_teams = pulp.value(S)
large_teams = pulp.value(L)
objective_value = pulp.value(prob.objective)

# Print solution details (optional)
print(f"Small teams: {small_teams}")
print(f"Large teams: {large_teams}")
print(f"Max lawn area mowed: {objective_value}")

# Required final objective print
print(f"FINAL_OBJ: {objective_value}")
