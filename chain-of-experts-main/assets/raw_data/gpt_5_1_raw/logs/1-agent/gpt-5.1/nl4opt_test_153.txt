import pulp

# Define the problem
prob = pulp.LpProblem("Zeta_Bakery_Profit_Maximization", pulp.LpMaximize)

# Decision variables
x1 = pulp.LpVariable('strawberry_cookies', lowBound=0)  # number of strawberry cookies
x2 = pulp.LpVariable('sugar_cookies', lowBound=0)       # number of sugar cookies

# Objective function: Maximize profit
prob += 5.5 * x1 + 12 * x2

# Constraints
# Demand constraints
prob += x1 <= 100  # at most 100 strawberry cookies
prob += x2 <= 80   # at most 80 sugar cookies

# Capacity constraint (short staffed: total cookies <= 100)
prob += x1 + x2 <= 100

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Get results
x1_opt = pulp.value(x1)
x2_opt = pulp.value(x2)
optimal_profit = pulp.value(prob.objective)

# Print results
print(f"Optimal strawberry cookies (x1): {x1_opt}")
print(f"Optimal sugar cookies (x2): {x2_opt}")
print(f"Maximum Profit: {optimal_profit}")
print(f"FINAL_OBJ: {optimal_profit}")
