import pulp

# Define the problem
prob = pulp.LpProblem("Minimize_Pollution", pulp.LpMinimize)

# Decision variables: number of hot-air balloon rides (x) and gondola rides (y)
x = pulp.LpVariable('hot_air_balloons', lowBound=0, cat='Integer')
y = pulp.LpVariable('gondola_lifts', lowBound=0, cat='Integer')

# Objective function: minimize total pollution
# Each hot-air balloon produces 10 units, each gondola 15 units
prob += 10 * x + 15 * y

# Constraints
# Capacity constraint: at least 70 visitors need to be transported
prob += 4 * x + 6 * y >= 70

# At most 10 hot-air balloon rides
prob += x <= 10

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=0))

# Extract the objective value
final_obj = pulp.value(prob.objective)

# Print the final objective value in the required format
print(f"FINAL_OBJ: {final_obj}")
