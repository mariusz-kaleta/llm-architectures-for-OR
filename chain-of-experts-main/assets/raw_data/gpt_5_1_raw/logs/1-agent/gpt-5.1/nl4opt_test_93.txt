import pulp

# Create the problem
prob = pulp.LpProblem("Science_Fair_Tables", pulp.LpMaximize)

# Decision variables
c = pulp.LpVariable('circular_tables', lowBound=0, cat='Integer')
r = pulp.LpVariable('rectangular_tables', lowBound=0, cat='Integer')

# Parameters
participants_c = 5   # per circular table
posters_c = 4        # per circular table
guests_c = 8         # per circular table
space_c = 15         # per circular table

participants_r = 4   # per rectangular table
posters_r = 4        # per rectangular table
guests_r = 12        # per rectangular table
space_r = 20         # per rectangular table

min_participants = 500
min_posters = 300
max_space = 1900

# Objective: maximize number of catered guests
prob += guests_c * c + guests_r * r, "Total_Guests"

# Constraints
prob += participants_c * c + participants_r * r >= min_participants, "Participants_Constraint"
prob += posters_c * c + posters_r * r >= min_posters, "Posters_Constraint"
prob += space_c * c + space_r * r <= max_space, "Space_Constraint"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=0))

# Retrieve solution
c_val = int(pulp.value(c))
r_val = int(pulp.value(r))
final_obj = pulp.value(prob.objective)

# Print solution details (optional, not required by instructions)
print("circular_tables:", c_val)
print("rectangular_tables:", r_val)
print("total_guests:", final_obj)

# Required final objective print
print(f"FINAL_OBJ: {final_obj}")
