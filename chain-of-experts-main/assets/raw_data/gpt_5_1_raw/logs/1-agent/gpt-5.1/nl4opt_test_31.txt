import pulp

# Initialize the problem
prob = pulp.LpProblem("CO2_Production_Maximization", pulp.LpMaximize)

# Decision variables
x = pulp.LpVariable('Process_with_catalyst', lowBound=0, cat='Continuous')
y = pulp.LpVariable('Process_without_catalyst', lowBound=0, cat='Continuous')

# Objective function: Maximize CO2 production
prob += 15 * x + 18 * y

# Constraints
prob += 10 * x + 15 * y <= 300  # Wood constraint
prob += 20 * x + 12 * y <= 300  # Oxygen constraint

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract values
x_val = pulp.value(x)
y_val = pulp.value(y)
final_obj = pulp.value(prob.objective)

# Print solution details (optional, but not requested explicitly)
print(f"Process with catalyst (x): {x_val}")
print(f"Process without catalyst (y): {y_val}")

# Final objective value in required format
print(f"FINAL_OBJ: {final_obj}")
