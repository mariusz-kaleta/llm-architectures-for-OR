import pulp

# Define the problem
prob = pulp.LpProblem("Taxi_Company_Optimization", pulp.LpMaximize)

# Decision variables: number of motorcycles (m) and sedans (s)
m = pulp.LpVariable('motorcycles', lowBound=0, cat='Continuous')
s = pulp.LpVariable('sedans', lowBound=0, cat='Continuous')

# Objective function: maximize earnings
# Motorcycle: $100 per shift, Sedan: $225 per shift
prob += 100 * m + 225 * s

# Constraints:

# 1. At most 25% of vehicles can be motorcycles: m <= (1/3) * s
prob += m - (1.0/3.0) * s <= 0

# 2. Pollution constraint: 4 units for motorcycle, 15 for sedan, total < 200
# In LP, we use <= 200 for feasibility
prob += 4 * m + 15 * s <= 200

# 3. Transportation capacity: at least 1200 people
# Motorcycle: 30 people, Sedan: 70 people
prob += 30 * m + 70 * s >= 1200

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Retrieve results
m_val = pulp.value(m)
s_val = pulp.value(s)
obj_val = pulp.value(prob.objective)

# Print decision variables and objective
print(f"Motorcycles: {m_val}")
print(f"Sedans: {s_val}")
print(f"FINAL_OBJ: {obj_val}")
