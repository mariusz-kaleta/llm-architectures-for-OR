import pulp

# Define the problem
prob = pulp.LpProblem("Car_Production_Problem", pulp.LpMaximize)

# Decision variables
x1 = pulp.LpVariable('x1', lowBound=0, cat='Continuous')  # regular models
x2 = pulp.LpVariable('x2', lowBound=0, cat='Continuous')  # premium models

# Objective function: Maximize profit
prob += 5000 * x1 + 8500 * x2

# Constraints
prob += x1 <= 8          # demand limit for regular models
prob += x2 <= 6          # demand limit for premium models
prob += x1 + x2 <= 12    # total production capacity

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=0))

# Extract results
x1_opt = pulp.value(x1)
x2_opt = pulp.value(x2)
final_obj = pulp.value(prob.objective)

# Print solution details (optional for understanding)
print(f"Optimal number of regular models (x1): {x1_opt}")
print(f"Optimal number of premium models (x2): {x2_opt}")
print(f"Maximum profit: {final_obj}")

# Required final objective print
print(f"FINAL_OBJ: {final_obj}")
