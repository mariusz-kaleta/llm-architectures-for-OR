import pulp

# Define the problem
prob = pulp.LpProblem("Teacher_Staffing_Problem", pulp.LpMinimize)

# Decision variables: number of staff teachers (s) and substitute teachers (t)
s = pulp.LpVariable('staff_teachers', lowBound=0, cat='Integer')
t = pulp.LpVariable('substitute_teachers', lowBound=0, cat='Integer')

# Parameters
hours_staff = 6
hours_sub = 3
pay_staff = 300
pay_sub = 100
required_hours = 1000
budget = 40000

# Objective: Minimize total number of teachers
prob += s + t, "Minimize_number_of_teachers"

# Constraints
# Teaching hours requirement
prob += hours_staff * s + hours_sub * t >= required_hours, "Hours_requirement"

# Budget constraint
prob += pay_staff * s + pay_sub * t <= budget, "Budget_constraint"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract results
staff_teachers_opt = int(pulp.value(s))
substitute_teachers_opt = int(pulp.value(t))
final_obj = pulp.value(prob.objective)

# Print solution details
print(f"Optimal number of staff teachers: {staff_teachers_opt}")
print(f"Optimal number of substitute teachers: {substitute_teachers_opt}")
print(f"Minimum total number of teachers: {final_obj}")

# Required final print
print(f"FINAL_OBJ: {final_obj}")
