import pulp

# Define the problem
prob = pulp.LpProblem("Farm_Optimization", pulp.LpMaximize)

# Decision variables: acres of turnips (T) and pumpkins (P)
T = pulp.LpVariable('Turnips', lowBound=0, cat='Continuous')
P = pulp.LpVariable('Pumpkins', lowBound=0, cat='Continuous')

# Objective function: maximize revenue
prob += 300 * T + 450 * P

# Constraints
# Watering time: 50 minutes per acre of turnips, 90 minutes per acre of pumpkins
prob += 50 * T + 90 * P <= 40000

# Pesticide budget: $80 per acre of turnips, $50 per acre of pumpkins
prob += 80 * T + 50 * P <= 34000

# Land constraint: total acres cannot exceed 500
prob += T + P <= 500

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=0))

# Extract the optimal values
T_opt = pulp.value(T)
P_opt = pulp.value(P)
final_obj = pulp.value(prob.objective)

# Print the solution
print(f"Optimal acres of turnips: {T_opt}")
print(f"Optimal acres of pumpkins: {P_opt}")
print(f"Maximum revenue: {final_obj}")

# Required final objective print
print(f"FINAL_OBJ: {final_obj}")
