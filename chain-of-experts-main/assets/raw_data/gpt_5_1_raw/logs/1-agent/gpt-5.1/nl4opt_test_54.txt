import pulp as pl

# Define the problem
prob = pl.LpProblem("Diet_Problem", pl.LpMinimize)

# Decision variables: number of glasses of milk (m) and plates of vegetables (v)
m = pl.LpVariable('milk', lowBound=0, cat='Continuous')
v = pl.LpVariable('vegetables', lowBound=0, cat='Continuous')

# Objective function: Minimize cost
prob += 1 * m + 2 * v

# Constraints
# Calcium: 40m + 15v >= 100
prob += 40 * m + 15 * v >= 100, "CalciumRequirement"

# Iron: 25m + 30v >= 50
prob += 25 * m + 30 * v >= 50, "IronRequirement"

# Solve the problem
prob.solve(pl.PULP_CBC_CMD(msg=False))

# Extract the results
milk_opt = pl.value(m)
veg_opt = pl.value(v)
final_obj = pl.value(prob.objective)

# Print solution details (optional for user visibility)
print(f"Optimal number of milk glasses: {milk_opt}")
print(f"Optimal number of vegetable plates: {veg_opt}")
print(f"Minimum cost: {final_obj}")

# Required final objective print
print(f"FINAL_OBJ: {final_obj}")
