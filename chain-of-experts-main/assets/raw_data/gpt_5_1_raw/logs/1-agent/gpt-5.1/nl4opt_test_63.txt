import pulp

# Create the problem
prob = pulp.LpProblem("Bakery_Profit_Maximization", pulp.LpMaximize)

# Decision variables
bread = pulp.LpVariable('bread_loaves', lowBound=0, cat='Continuous')
cookies = pulp.LpVariable('cookie_batches', lowBound=0, cat='Continuous')

# Parameters
mixer_hours = 3000
oven_hours = 3000

# Constraints:
# Mixer time: 1 hour per bread, 0.5 hours per cookie batch
prob += 1 * bread + 0.5 * cookies <= mixer_hours, "Mixer_Time"

# Oven time: 3 hours per bread, 1 hour per cookie batch
prob += 3 * bread + 1 * cookies <= oven_hours, "Oven_Time"

# Objective function: Maximize profit
# Profit: $5 per loaf of bread, $3 per batch of cookies
prob += 5 * bread + 3 * cookies, "Total_Profit"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=0))

# Extract results
bread_opt = pulp.value(bread)
cookies_opt = pulp.value(cookies)
final_obj = pulp.value(prob.objective)

# Print solution details (optional for understanding)
print(f"Optimal number of bread loaves: {bread_opt}")
print(f"Optimal number of cookie batches: {cookies_opt}")
print(f"Maximum profit: {final_obj}")

# Required final objective print
print(f"FINAL_OBJ: {final_obj}")
