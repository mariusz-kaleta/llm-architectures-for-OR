import pulp

# Define the problem: minimize number of cars
prob = pulp.LpProblem("Voter_Transportation", pulp.LpMinimize)

# Decision variables: number of vans (v) and cars (c)
v = pulp.LpVariable('v', lowBound=0, cat='Integer')
c = pulp.LpVariable('c', lowBound=0, cat='Integer')

# Objective: minimize number of cars
prob += c, "Minimize_cars"

# Constraints:
# 1) Capacity constraint: 6v + 3c >= 200
prob += 6 * v + 3 * c >= 200, "Capacity"

# 2) At most 30% of vehicles are vans: v <= 0.3 * (v + c)
#    This rearranges to: 0.7v <= 0.3c  ->  7v <= 3c
prob += 7 * v <= 3 * c, "Van_percentage"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract solution
vans_used = pulp.value(v)
cars_used = pulp.value(c)
final_obj = pulp.value(prob.objective)

# Print solution details (not required, but useful for understanding)
print(f"Vans used: {vans_used}")
print(f"Cars used: {cars_used}")

# Required final objective print
print(f"FINAL_OBJ: {final_obj}")
