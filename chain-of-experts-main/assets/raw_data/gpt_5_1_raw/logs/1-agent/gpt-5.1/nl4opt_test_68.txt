import pulp

# Create the LP problem: minimize total time
prob = pulp.LpProblem("Drug_Production_Time_Minimization", pulp.LpMinimize)

# Decision variables: hours to run factory 1 (x1) and factory 2 (x2)
x1 = pulp.LpVariable('x1', lowBound=0)
x2 = pulp.LpVariable('x2', lowBound=0)

# Objective: minimize total time x1 + x2
prob += x1 + x2, "Total_Time"

# Constraints:

# Allergy pills: 20*x1 + 10*x2 >= 700
prob += 20 * x1 + 10 * x2 >= 700, "Allergy_Requirement"

# Fever reducing pills: 15*x1 + 30*x2 >= 600
prob += 15 * x1 + 30 * x2 >= 600, "Fever_Requirement"

# Rare compound constraint: 20*x1 + 30*x2 <= 1000
prob += 20 * x1 + 30 * x2 <= 1000, "Compound_Limit"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=0))

# Extract optimal values
x1_opt = pulp.value(x1)
x2_opt = pulp.value(x2)
obj_val = pulp.value(prob.objective)

# Print solution details (not required but useful if running the script)
print(f"Optimal hours for Factory 1 (x1): {x1_opt}")
print(f"Optimal hours for Factory 2 (x2): {x2_opt}")
print(f"Minimum total time: {obj_val}")

# Required final objective print
print(f"FINAL_OBJ: {obj_val}")
