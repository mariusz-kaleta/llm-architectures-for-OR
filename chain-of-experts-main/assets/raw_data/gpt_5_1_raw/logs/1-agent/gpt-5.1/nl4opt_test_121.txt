import pulp

# Initialize the problem
prob = pulp.LpProblem("Carrots_Cucumbers_Profit_Max", pulp.LpMaximize)

# Decision variables: number of carrots (c) and cucumbers (k)
c = pulp.LpVariable('carrots', lowBound=0, cat='Continuous')
k = pulp.LpVariable('cucumbers', lowBound=0, cat='Continuous')

# Parameters
budget = 500.0
cost_carrot = 0.30
cost_cucumber = 0.50
profit_carrot = 0.75
profit_cucumber = 0.80

# Objective function: maximize profit
prob += profit_carrot * c + profit_cucumber * k

# Constraints
# Budget constraint
prob += cost_carrot * c + cost_cucumber * k <= budget

# Number of cucumbers at most one third of carrots
prob += k <= (1.0 / 3.0) * c

# Carrot bounds
prob += c >= 300
prob += c <= 500

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Retrieve results
opt_c = pulp.value(c)
opt_k = pulp.value(k)
opt_profit = pulp.value(prob.objective)

# Print solution details
print(f"Optimal number of carrots: {opt_c}")
print(f"Optimal number of cucumbers: {opt_k}")
print(f"Maximum profit: {opt_profit}")

# Final required output
print(f"FINAL_OBJ: {opt_profit}")
