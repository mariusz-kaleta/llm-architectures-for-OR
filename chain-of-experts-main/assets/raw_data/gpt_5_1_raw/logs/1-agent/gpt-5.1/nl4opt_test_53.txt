import pulp as pl

# Create the LP problem: minimize cost
prob = pl.LpProblem("Plant_Growth_Compound", pl.LpMinimize)

# Decision variables: kg of fertilizer C and Y
C = pl.LpVariable('C', lowBound=0)
Y = pl.LpVariable('Y', lowBound=0)

# Objective function: Minimize cost
prob += 2 * C + 3 * Y

# Constraints
# Nitrous oxide: 1.5C + 5Y >= 5
prob += 1.5 * C + 5 * Y >= 5

# Vitamin mix: 3C + 1Y >= 8
prob += 3 * C + Y >= 8

# Solve the problem
prob.solve(pl.PULP_CBC_CMD(msg=False))

# Get the final objective value
final_obj = pl.value(prob.objective)

# Print the final objective value in the required format
print(f"FINAL_OBJ: {final_obj}")
