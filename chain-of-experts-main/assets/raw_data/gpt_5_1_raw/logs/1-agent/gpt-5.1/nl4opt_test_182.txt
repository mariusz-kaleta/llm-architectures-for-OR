import pulp

# Define the problem
prob = pulp.LpProblem("Transport_Optimization", pulp.LpMinimize)

# Decision variables: number of ship trips (x) and plane trips (y)
x = pulp.LpVariable('ship_trips', lowBound=0, cat='Continuous')
y = pulp.LpVariable('plane_trips', lowBound=0, cat='Continuous')

# Objective function: minimize total fuel consumption
# Ship: 500 liters/trip, Plane: 300 liters/trip
prob += 500 * x + 300 * y, "Total_Fuel"

# Constraints
# Capacity: ship 40 containers/trip, plane 20 containers/trip
prob += 40 * x + 20 * y >= 500, "Capacity"

# Plane trip upper bound
prob += y <= 10, "Max_Plane_Trips"

# At least 50% of trips must be by ship:
# x >= y  (since total trips = x + y, x >= 0.5*(x + y) simplifies to x >= y)
prob += x >= y, "Min_50pct_Ship_Trips"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract solution
ship_trips = pulp.value(x)
plane_trips = pulp.value(y)
final_obj = pulp.value(prob.objective)

# Print solution details
print(f"Ship trips: {ship_trips}")
print(f"Plane trips: {plane_trips}")
print(f"Total fuel consumed: {final_obj}")

# Required final objective print
print(f"FINAL_OBJ: {final_obj}")
