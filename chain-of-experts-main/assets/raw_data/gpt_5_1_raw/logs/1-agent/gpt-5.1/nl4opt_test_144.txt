import pulp

# Initialize the problem
prob = pulp.LpProblem("DietProblem", pulp.LpMinimize)

# Decision variables: servings of vegetables (v) and fruits (f)
v = pulp.LpVariable('v', lowBound=0, cat='Continuous')
f = pulp.LpVariable('f', lowBound=0, cat='Continuous')

# Objective function: Minimize cost = 3v + 5f
prob += 3 * v + 5 * f

# Constraints
# Vitamins: 2v + 4f >= 20
prob += 2 * v + 4 * f >= 20

# Minerals: 3v + f >= 30
prob += 3 * v + f >= 30

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Retrieve results
v_opt = pulp.value(v)
f_opt = pulp.value(f)
obj_val = pulp.value(prob.objective)

# Print detailed solution
print(f"Optimal servings of vegetables (v): {v_opt}")
print(f"Optimal servings of fruits (f): {f_opt}")
print(f"Minimum cost: {obj_val}")

# Final required output
print(f"FINAL_OBJ: {obj_val}")
