import pulp

# Define the problem
problem = pulp.LpProblem("Soda_Van_Optimization", pulp.LpMinimize)

# Decision variables
old_vans = pulp.LpVariable('old_vans', lowBound=0, cat='Integer')
new_vans = pulp.LpVariable('new_vans', lowBound=0, upBound=30, cat='Integer')

# Objective function: Minimize pollution
problem += 50 * old_vans + 30 * new_vans

# Constraints
problem += 100 * old_vans + 80 * new_vans >= 5000  # Bottle requirement

# Solve the problem
problem.solve(pulp.PULP_CBC_CMD(msg=0))

# Retrieve results
old_vans_opt = pulp.value(old_vans)
new_vans_opt = pulp.value(new_vans)
final_obj = pulp.value(problem.objective)

# Print detailed solution
print(f"Old vans used: {old_vans_opt}")
print(f"New vans used: {new_vans_opt}")
print(f"Total pollution: {final_obj}")

# Required final objective print
print(f"FINAL_OBJ: {final_obj}")
