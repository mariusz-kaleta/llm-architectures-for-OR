import pulp as pl

# Create the problem: maximize hay processed
prob = pl.LpProblem("Hay_Processing", pl.LpMaximize)

# Decision variables: acres for windrower (x) and hay harvester (y)
x = pl.LpVariable('Windrower_acres', lowBound=0)
y = pl.LpVariable('Harvester_acres', lowBound=0)

# Parameters
total_acres = 200
fuel_available = 300
methane_max = 800

# Objective: maximize hay processed
# Windrower: 10 kg hay/acre, Harvester: 8 kg hay/acre
prob += 10 * x + 8 * y

# Constraints

# Total land
prob += x + y <= total_acres, "Land_Constraint"

# Fuel constraint
# Windrower: 2 kg fuel/acre, Harvester: 1 kg fuel/acre
prob += 2 * x + 1 * y <= fuel_available, "Fuel_Constraint"

# Methane constraint
# Windrower: 5 kg methane/acre, Harvester: 3 kg methane/acre
prob += 5 * x + 3 * y <= methane_max, "Methane_Constraint"

# Solve the problem
prob.solve(pl.PULP_CBC_CMD(msg=False))

# Retrieve results
x_opt = pl.value(x)
y_opt = pl.value(y)
obj_value = pl.value(prob.objective)

# Print solution details
print(f"Optimal acres for windrower: {x_opt}")
print(f"Optimal acres for hay harvester: {y_opt}")
print(f"Maximum hay processed (kg): {obj_value}")

# Required final objective print
print(f"FINAL_OBJ: {obj_value}")
