import pulp

# Define the problem
prob = pulp.LpProblem("Painkiller_Optimization", pulp.LpMaximize)

# Decision variables: number of doses of pain killer 1 (x1) and pain killer 2 (x2)
x1 = pulp.LpVariable('x1', lowBound=0)
x2 = pulp.LpVariable('x2', lowBound=0)

# Objective: maximize medicine delivered to the back
# Back: 0.8 units per dose of pain killer 1, 0.4 units per dose of pain killer 2
prob += 0.8 * x1 + 0.4 * x2

# Constraints:

# Sleeping medicine constraint: at most 8 units
# Sleep: 0.3 units for pain killer 1, 0.6 units for pain killer 2
prob += 0.3 * x1 + 0.6 * x2 <= 8

# Legs medicine constraint: at least 4 units
# Legs: 0.5 units for pain killer 1, 0.7 units for pain killer 2
prob += 0.5 * x1 + 0.7 * x2 >= 4

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Retrieve the results
x1_val = pulp.value(x1)
x2_val = pulp.value(x2)
objective_value = pulp.value(prob.objective)

# Print solution details (optional)
print(f"Optimal doses of pain killer 1 (x1): {x1_val}")
print(f"Optimal doses of pain killer 2 (x2): {x2_val}")
print(f"Maximum back medicine delivered: {objective_value}")

# Final required print
print(f"FINAL_OBJ: {objective_value}")
