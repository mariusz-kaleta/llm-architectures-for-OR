[CODER]:
import pulp

def main():
    # Create the problem: maximize profit
    prob = pulp.LpProblem("Desk_Drawer_Production", pulp.LpMaximize)

    # Decision variables (non-negative, integer or continuous as needed)
    x1 = pulp.LpVariable('x1', lowBound=0, cat='Integer')  # desks
    x2 = pulp.LpVariable('x2', lowBound=0, cat='Integer')  # drawers

    # Objective function
    prob += 100 * x1 + 90 * x2, "Total_Profit"

    # Constraints
    prob += 40 * x1 + 30 * x2 <= 4000, "Assembly_Time"
    prob += 20 * x1 + 10 * x2 <= 3500, "Sanding_Time"

    # Solve
    prob.solve(pulp.PULP_CBC_CMD(msg=False))

    # Get final objective value
    final_obj = pulp.value(prob.objective)

    # Print strictly in required format
    print(f"FINAL_OBJ: {final_obj}")

if __name__ == "__main__":
    main()
