import pulp

def main():
    # Create the problem: maximization
    prob = pulp.LpProblem("Sandwich_Production", pulp.LpMaximize)

    # Decision variables: non-negative integers
    x = pulp.LpVariable('x', lowBound=0, cat='Integer')  # regular sandwiches
    y = pulp.LpVariable('y', lowBound=0, cat='Integer')  # special sandwiches

    # Objective function: Maximize profit 3x + 4y
    prob += 3 * x + 4 * y

    # Constraints
    prob += 2 * x + 3 * y <= 40  # egg constraint
    prob += 3 * x + 5 * y <= 70  # bacon constraint

    # Solve the problem
    prob.solve(pulp.PULP_CBC_CMD(msg=False))

    # Get final objective value
    final_obj = pulp.value(prob.objective)

    # Print result in required format
    print(f"FINAL_OBJ: {final_obj}")

if __name__ == "__main__":
    main()
