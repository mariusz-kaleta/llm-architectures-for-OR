import pulp

def main():
    # Create the problem: maximize zinc intake
    prob = pulp.LpProblem("Zinc_Maximization", pulp.LpMaximize)

    # Decision variables: G (gummies), P (pills), both non-negative integers
    G = pulp.LpVariable('G', lowBound=0, cat='Integer')
    P = pulp.LpVariable('P', lowBound=0, cat='Integer')

    # Objective function: Maximize 4G + 5P
    prob += 4 * G + 5 * P

    # Constraints
    # 1. Minimum pills requirement: P >= 10
    prob += P >= 10

    # 2. Preference for gummies: G >= 3P
    prob += G >= 3 * P

    # 3. Magnesium intake limit: 3G + 2P <= 200
    prob += 3 * G + 2 * P <= 200

    # Solve the problem
    prob.solve(pulp.PULP_CBC_CMD(msg=False))

    # Get the objective value
    final_obj = pulp.value(prob.objective)

    # Print final objective value in required format
    print(f"FINAL_OBJ: {final_obj}")

if __name__ == "__main__":
    main()
