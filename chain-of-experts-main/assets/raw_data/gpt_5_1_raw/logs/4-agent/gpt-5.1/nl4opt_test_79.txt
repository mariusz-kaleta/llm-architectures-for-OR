## 6. Example PuLP formulation

```python
import pulp as pl

# Problem
prob = pl.LpProblem("Painkiller_Back_Max", pl.LpMaximize)

# Decision variables (continuous, non-negative)
x1 = pl.LpVariable("x1", lowBound=0)  # doses of pain killer 1
x2 = pl.LpVariable("x2", lowBound=0)  # doses of pain killer 2

# Objective: maximize back medicine
prob += 0.8 * x1 + 0.4 * x2, "Total_Back_Medicine"

# Constraints
prob += 0.5 * x1 + 0.7 * x2 >= 4, "Leg_Minimum"
prob += 0.3 * x1 + 0.6 * x2 <= 8, "Sleep_Maximum"

# Solve
prob.solve(pl.PULP_CBC_CMD(msg=False))

print("Status:", pl.LpStatus[prob.status])
print("x1 (doses of pain killer 1) =", x1.value())
print("x2 (doses of pain killer 2) =", x2.value())
print("Total back medicine =", pl.value(prob.objective))
```

import pulp as pl

def main():
    # Define the problem
    prob = pl.LpProblem("Painkiller_Back_Max", pl.LpMaximize)

    # Decision variables (continuous, non-negative)
    x1 = pl.LpVariable("x1", lowBound=0)  # doses of pain killer 1
    x2 = pl.LpVariable("x2", lowBound=0)  # doses of pain killer 2

    # Objective: maximize back medicine
    prob += 0.8 * x1 + 0.4 * x2, "Total_Back_Medicine"

    # Constraints
    prob += 0.5 * x1 + 0.7 * x2 >= 4, "Leg_Minimum"
    prob += 0.3 * x1 + 0.6 * x2 <= 8, "Sleep_Maximum"

    # Solve the problem
    prob.solve(pl.PULP_CBC_CMD(msg=False))

    # Retrieve objective value
    obj_value = pl.value(prob.objective)

    # Print final objective value in required format
    print(f"FINAL_OBJ: {obj_value}")

if __name__ == "__main__":
    main()
