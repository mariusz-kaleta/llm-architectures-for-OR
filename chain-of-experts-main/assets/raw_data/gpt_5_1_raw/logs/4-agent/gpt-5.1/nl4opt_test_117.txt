## 5. Example PuLP formulation (continuous LP)

```python
import pulp as pl

# Create problem: maximize
prob = pl.LpProblem("Slime_Production", pl.LpMaximize)

# Decision variables (continuous, >= 0)
x1 = pl.LpVariable('x1', lowBound=0)
x2 = pl.LpVariable('x2', lowBound=0)

# Objective: maximize 5x1 + 3x2
prob += 5 * x1 + 3 * x2, "Total_Slime"

# Constraints
prob += 4 * x1 + 6 * x2 <= 150, "Flour"
prob += 6 * x1 + 3 * x2 <= 100, "Special_Liquid"
prob += 4 * x1 + 2 * x2 <= 30,  "Waste"

# Solve
prob.solve()

print("Status:", pl.LpStatus[prob.status])
print("x1 (Beaker 1 batches) =", x1.value())
print("x2 (Beaker 2 batches) =", x2.value())
print("Max slime =", pl.value(prob.objective))
```

x1 = pl.LpVariable('x1', lowBound=0, cat='Integer')

import pulp as pl

def main():
    # Create problem: maximize
    prob = pl.LpProblem("Slime_Production", pl.LpMaximize)

    # Decision variables (integer, >= 0)
    x1 = pl.LpVariable('x1', lowBound=0, cat='Integer')
    x2 = pl.LpVariable('x2', lowBound=0, cat='Integer')

    # Objective: maximize 5x1 + 3x2
    prob += 5 * x1 + 3 * x2, "Total_Slime"

    # Constraints
    prob += 4 * x1 + 6 * x2 <= 150, "Flour"
    prob += 6 * x1 + 3 * x2 <= 100, "Special_Liquid"
    prob += 4 * x1 + 2 * x2 <= 30,  "Waste"

    # Solve
    prob.solve(pl.PULP_CBC_CMD(msg=False))

    final_obj = pl.value(prob.objective)
    print(f"FINAL_OBJ: {final_obj}")

if __name__ == "__main__":
    main()
