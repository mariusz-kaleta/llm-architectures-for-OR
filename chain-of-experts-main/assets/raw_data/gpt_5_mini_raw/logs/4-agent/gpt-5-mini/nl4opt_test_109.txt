```python
import math

# Problem data
demand = 750
cap_small = 25
cap_large = 45

# Feasible range for l
l_min = 5
l_max = math.ceil(demand / cap_large)

best = None  # tuple for best (total_boxes, tie_breaker..., s, l, capacity)

for l in range(l_min, l_max + 1):
    # s required by capacity (ceil of remaining demand divided by small box capacity)
    remaining = max(0, demand - cap_large * l)
    s_from_capacity = math.ceil(remaining / cap_small)
    # s required by ratio s >= 3*l
    s = max(3 * l, s_from_capacity)
    capacity = cap_small * s + cap_large * l
    total_boxes = s + l

    # tie-breaker:
    # prefer exact capacity (capacity == demand) over over-capacity
    exact_flag = 0 if capacity == demand else 1
    # prefer fewer large boxes
    tie = (total_boxes, exact_flag, l, s)

    if best is None or tie < best[0]:
        best = (tie, s, l, capacity, total_boxes)

# Unpack best solution
_, best_s, best_l, best_capacity, best_total = best

# Print solution details (optional)
print(f"s = {best_s}, l = {best_l}")
print(f"Total masks delivered = {best_capacity}")
print(f"Total boxes used = {best_total}")

# Final required line with objective value
print(f"FINAL_OBJ: {best_total}")
```
