LB_L = 15000.0
LB_C = 5000.0
MIN_TOTAL = 50000.0
UB_L = 40000.0
UB_C = 20000.0

TOL = 1e-6

# Represent the constraint lines we will consider for intersections:
# We'll consider equalities from: L = LB_L, L = UB_L, C = LB_C, C = UB_C, L + C = MIN_TOTAL
lines = [
    ("L_eq_LB", lambda: ("L", LB_L)),
    ("L_eq_UB", lambda: ("L", UB_L)),
    ("C_eq_LB", lambda: ("C", LB_C)),
    ("C_eq_UB", lambda: ("C", UB_C)),
    ("LplusC_eq_MIN", lambda: ("LplusC", MIN_TOTAL)),
]

def solve_from_lines(line1, line2):
    # Solve the intersection of two lines among the set above.
    # Return (L, C) or None if no unique solution.
    name1, func1 = line1
    name2, func2 = line2
    t1, v1 = func1()
    t2, v2 = func2()
    # Cases:
    # t can be "L", "C", or "LplusC"
    # Solve linear system
    # Represent as a1*L + b1*C = v1 and a2*L + b2*C = v2
    def coeffs(t, v):
        if t == "L":
            return (1.0, 0.0, v)
        if t == "C":
            return (0.0, 1.0, v)
        if t == "LplusC":
            return (1.0, 1.0, v)
        raise ValueError("unknown line type")
    a1, b1, r1 = coeffs(t1, v1)
    a2, b2, r2 = coeffs(t2, v2)
    det = a1 * b2 - a2 * b1
    if abs(det) < 1e-12:
        return None  # parallel / no unique intersection

    if L > ub_L + 1e-8: return False
    if C > ub_C + 1e-8: return False
    return True

    loc_lines = [
        ("L_eq_LB", lambda: ("L", lb_L)),
        ("L_eq_UB", lambda: ("L", ub_L)),
        ("C_eq_LB", lambda: ("C", lb_C)),
        ("C_eq_UB", lambda: ("C", ub_C)),
        ("LplusC_eq_MIN", lambda: ("LplusC", min_total)),
    ]
    pts = []
    for l1, l2 in itertools.combinations(loc_lines, 2):
        # solve intersection similar to earlier
        name1, func1 = l1
        name2, func2 = l2
        t1, v1 = func1()
        t2, v2 = func2()
        def coeffs(t, v):
            if t == "L":
                return (1.0, 0.0, v)
            if t == "C":
                return (0.0, 1.0, v)
            if t == "LplusC":
                return (1.0, 1.0, v)
            raise ValueError("unknown line type")
        a1, b1, r1 = coeffs(t1, v1)
        a2, b2, r2 = coeffs(t2, v2)
        det = a1 * b2 - a2 * b1
        if abs(det) < 1e-12:
            continue
        L = (r1 * b2 - r2 * b1) / det
        C = (a1 * r2 - a2 * r1) / det
        if is_feasible(L, C, lb_L=lb_L, lb_C=lb_C, min_total=min_total, ub_L=ub_L, ub_C=ub_C):
            pts.append((L, C, objective(L, C)))
    # add explicit corner clamps
    for L in [lb_L, ub_L]:
        for C in [lb_C, ub_C]:
            if is_feasible(L, C, lb_L=lb_L, lb_C=lb_C, min_total=min_total, ub_L=ub_L, ub_C=ub_C):
                pts.append((L, C, objective(L, C)))
    # If none found, try constructing feasible points by meeting the total constraint minimally
    if not pts:
        for L in [lb_L, ub_L]:
            C_needed = max(lb_C, min(ub_C, min_total - L))
            if is_feasible(L, C_needed, lb_L=lb_L, lb_C=lb_C, min_total=min_total, ub_L=ub_L, ub_C=ub_C):
                pts.append((L, C_needed, objective(L, C_needed)))
        for C in [lb_C, ub_C]:
            L_needed = max(lb_L, min(ub_L, min_total - C))
            if is_feasible(L_needed, C, lb_L=lb_L, lb_C=lb_C, min_total=min_total, ub_L=ub_L, ub_C=ub_C):
                pts.append((L_needed, C, objective(L_needed, C)))
    if not pts:
        return None  # infeasible
    best_local = max(pts, key=lambda x: x[2])
    return best_local  # (L, C, Z)
