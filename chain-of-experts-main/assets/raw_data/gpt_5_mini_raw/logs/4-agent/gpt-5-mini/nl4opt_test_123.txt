from pulp import LpProblem, LpMinimize, LpVariable, LpInteger, lpSum, value

# Problem
prob = LpProblem("Vaccine_Production", LpMinimize)

# Variables
C = LpVariable("Children", lowBound=50, cat=LpInteger)  # C >= 50
A = LpVariable("Adults", lowBound=0, cat=LpInteger)

# Objective
prob += 50*C + 75*A, "Total_Fever_Suppressant"

# Constraints
prob += 50*C + 75*A <= 20000, "mRNA_limit"
prob += 3*A - 7*C >= 0, "Adult_share"

# Solve

import math

# Parameters
mRNA_limit = 20000
coef_C = 50  # fever suppressant per child vaccine
coef_A = 75  # fever suppressant per adult vaccine

best_obj = None
best_C = None
best_A = None

# Search C from its minimum 50 up to the absolute maximum allowed by mRNA when A=0
for C in range(50, mRNA_limit // coef_C + 1):
    # Adult-share constraint: A >= ceil(7/3 * C)
    A_min = math.ceil(7 * C / 3)
    # Check mRNA feasibility
    if coef_C * C + coef_A * A_min <= mRNA_limit:
        obj = coef_C * C + coef_A * A_min
        if best_obj is None or obj < best_obj:
            best_obj = obj
            best_C = C
            best_A = A_min

# If no feasible solution found (should not happen here), handle gracefully
if best_obj is None:
    value = 0
else:
    value = int(best_obj)

print(f"FINAL_OBJ: {value}")
