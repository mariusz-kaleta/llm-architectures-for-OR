#!/usr/bin/env python3
# Self-contained solver for the "bones" integer optimisation problem.
# Decision variables:
#   s >= 0 integer: number of small bones
#   l >= 0 integer: number of large bones
# Objective:
#   minimize M = 12*s + 15*l
# Constraints:
#   10*s + 15*l <= 2000   (medication)
#   s >= l                (at least 50% small)
#   l >= 30               (minimum large bones)
#
# The script performs an exhaustive integer search over feasible ranges.

def main():
    med_limit = 2000
    l_min = 30
    # Upper bound for l: medication constraint with s = l => 25*l <= med_limit
    l_max = med_limit // 25  # floor(2000/25) = 80
    if l_max < l_min:
        raise SystemExit("No feasible solution because l_min > l_max under medication limit.")
    best_obj = None
    best_s = None
    best_l = None
    best_med = None

    for l in range(l_min, l_max + 1):
        s_min = l  # s >= l
        # medication: 10*s + 15*l <= med_limit  => s <= (med_limit - 15*l)/10
        s_max = (med_limit - 15 * l) // 10
        if s_max < s_min:
            continue
        for s in range(s_min, s_max + 1):
            med = 10 * s + 15 * l
            obj = 12 * s + 15 * l
            if med <= med_limit:
                if (best_obj is None) or (obj < best_obj):
                    best_obj = obj
                    best_s = s
                    best_l = l
                    best_med = med

    if best_obj is None:
        raise SystemExit("No feasible integer solution found.")

    # Print solution details
    print(f"Optimal small = {best_s}")
    print(f"Optimal large = {best_l}")
    print(f"Total meat used = {best_obj}")
    print(f"Total medication used = {best_med}")
    print(f"Integers used = yes")

    # Final required single-line output (must be the last line)
    print(f"FINAL_OBJ: {best_obj}")

if __name__ == "__main__":
    main()
