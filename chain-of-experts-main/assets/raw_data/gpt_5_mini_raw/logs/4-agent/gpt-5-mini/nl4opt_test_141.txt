import math

# Problem data
profit_m = 3    # profit per mango
profit_g = 4    # profit per guava
cost_m = 5
cost_g = 3
budget = 20000
m_lb = 100
m_ub = 150
ratio = 1.0/3.0

final_obj = None

# First, attempt to solve with PuLP (MIP). If PuLP isn't available, fall back to analytical solution.
try:
    import pulp

    prob = pulp.LpProblem("FruitProfit", pulp.LpMaximize)
    m = pulp.LpVariable('m', lowBound=m_lb, upBound=m_ub, cat=pulp.LpInteger)
    g = pulp.LpVariable('g', lowBound=0, cat=pulp.LpInteger)

    prob += profit_m * m + profit_g * g
    prob += cost_m * m + cost_g * g <= budget
    # g <= m/3  -> represent as 3*g <= m to avoid fractional coefficient issues with integer vars
    prob += 3 * g <= m

    solver = pulp.PULP_CBC_CMD(msg=False)
    prob.solve(solver)

    m_val = int(pulp.value(m))
    g_val = int(pulp.value(g))
    obj_val = profit_m * m_val + profit_g * g_val

    final_obj = obj_val

except Exception:
    # Fallback: analytical solution described in the problem statement.
    # Because profit coefficients positive and g is upper-bounded, optimal g = m/3.
    # Objective Z = (13/3) m is increasing in m, so pick m = m_ub.
    m_val = m_ub
    g_val = int(m_val * ratio)
    # ensure g respects integer rounding and constraint 3*g <= m
    if 3 * g_val > m_val:
        g_val = m_val // 3
    # verify budget; if budget smaller than needed, adjust (not needed here since budget is huge)
    if cost_m * m_val + cost_g * g_val > budget:
        # If infeasible with current budget, reduce m until feasible
        while m_val >= m_lb and cost_m * m_val + cost_g * g_val > budget:
            m_val -= 1
            g_val = m_val // 3
        if m_val < m_lb:
            raise RuntimeError("No feasible solution under given budget and bounds.")
    obj_val = profit_m * m_val + profit_g * g_val
    final_obj = obj_val

# Final required print (must be the very last output line)
