#!/usr/bin/env python3
# Self-contained solver for the LP:
# minimize T = 30*A + 45*B
# s.t.
#   A - (1/3)B <= 0
#   A >= 300
#   A + B >= 1500
#   A, B >= 0
# The script computes all pairwise intersections of the bounding lines,
# filters feasible points, evaluates the objective and prints the final objective.

import math

# Define the lines (as equalities) to consider for vertices:
# Represent each line as (coef_A, coef_B, rhs) for coef_A * A + coef_B * B = rhs
lines = [
    (1.0, 0.0, 300.0),         # A = 300
    (3.0, -1.0, 0.0),          # 3A - B = 0  (i.e., A - (1/3)B = 0)
    (1.0, 1.0, 1500.0),        # A + B = 1500
    (1.0, 0.0, 0.0),           # A = 0
    (0.0, 1.0, 0.0),           # B = 0
]

def intersect(l1, l2):
    a1, b1, c1 = l1
    a2, b2, c2 = l2
    det = a1 * b2 - a2 * b1
    if abs(det) < 1e-12:
        return None
    A = (c1 * b2 - c2 * b1) / det
    B = (a1 * c2 - a2 * c1) / det
    return (A, B)

# Generate intersections
pts = set()
n = len(lines)
for i in range(n):
    for j in range(i+1, n):
        pt = intersect(lines[i], lines[j])
        if pt is None:
            continue
        # round small numerical noise
        A, B = pt
        # Keep reasonable precision
        A = 0.0 if abs(A) < 1e-12 else A
        B = 0.0 if abs(B) < 1e-12 else B
        pts.add((A, B))

# Also consider some natural corner/edge points (just in case):
# e.g., the intersection when A is at its lower bound and B at minimum implied by constraints
# But those are covered by the lines above.

# Feasibility check
def feasible(A, B, tol=1e-9):
    if A < -tol or B < -tol:
        return False
    if A - (1.0/3.0)*B > tol:
        return False
    if A + B + tol < 1500.0:
        return False
    if A + tol < 300.0:
        return False
    return True

# Evaluate objective
def obj(A, B):
    return 30.0*A + 45.0*B

feasible_pts = []
for A, B in pts:
    if feasible(A, B):
        feasible_pts.append((A, B, obj(A,B)))

# It's possible the true optimum lies along an edge where one of the inequality constraints
# is not represented by a line in our set (but we included all relevant boundaries).
# If no feasible vertex found due to numerical issues, fall back to analytic solution.
if not feasible_pts:
    # Analytic derivation: with A + B = 1500 and A <= (1/3)B -> 4A <= 1500 -> A <= 375
    # Also A >= 300, so choose A = 375 (max feasible A) to minimize cost because A cheaper.
    A_opt = 375.0
    B_opt = 1500.0 - A_opt
    T_opt = obj(A_opt, B_opt)
else:
    # Choose the feasible point with minimum objective
    A_opt, B_opt, T_opt = min(feasible_pts, key=lambda x: x[2])

# For safety, round to a small tolerance
T_opt = float(round(T_opt, 10))

print(f"FINAL_OBJ: {T_opt}")
