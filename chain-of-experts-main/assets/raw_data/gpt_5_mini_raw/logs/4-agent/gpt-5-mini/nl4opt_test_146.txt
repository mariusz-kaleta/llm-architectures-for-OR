3) MILP formulation
- Objective: minimize total baking time
  minimize Z = 12 L + 8 M

- Subject to:
  12 L + 8 M ≥ 10000        (dough requirement)
  5 L + 4 M ≥ 4400          (toppings requirement)
  M ≥ 200
  L ≥ 2 M
  L, M ∈ Z≥0

4) Key modelling observation
- The objective coefficients (12,8) equal the dough coefficients (12,8). Thus any feasible solution with 12L+8M = 10000 would achieve the theoretical lower bound Z = 10000. However toppings, minimum-medium and ratio constraints may prevent achieving that lower bound; solver will therefore seek the smallest feasible 12L+8M ≥ 10000.

5) Solving / analytical reduction (short)
Let L = 2M + x with x ∈ Z≥0 (enforces L ≥ 2M). Substitute:
  Z = 12L + 8M = 12(2M+x)+8M = 32M + 12x
  Toppings constraint → 5L+4M = 5(2M+x)+4M = 14M + 5x ≥ 4400
For each integer M ≥ 200 compute minimal x ≥ 0 satisfying 14M + 5x ≥ 4400:
  x_min(M) = max(0, ceil((4400 − 14M)/5))
Then Z(M) = 32M + 12 x_min(M). Minimize Z(M) over integer M ≥ 200.

#!/usr/bin/env python3
import math

# Parameters
dough_per_L = 12
dough_per_M = 8
toppings_per_L = 5
toppings_per_M = 4
bake_L = 12
bake_M = 8

req_dough = 10000
req_toppings = 4400
min_M = 200

# helper: integer ceiling division
def ceil_div(a, b):
    return -(-a // b)

best_obj = None
best_solutions = []

# We'll search over a reasonable range for M.
# Upper bound: if L = 2M (smallest L allowed by ratio), objective Z = 32M,
# so any M > req_dough/32 will have Z>req_dough. We can search a bit above that.
M_upper = max(min_M + 2000, ceil_div(req_dough, 8) + 1000)  # generous upper bound
M_upper = min(M_upper, 5000)  # cap to keep loop small; still plenty for this problem

for M in range(min_M, M_upper + 1):
    # constraints give lower bounds on L
    # 1) ratio: L >= 2M
    req_L_ratio = 2 * M
    # 2) dough: 12L + 8M >= 10000 -> L >= ceil((10000 - 8M)/12)
    req_L_dough = ceil_div(req_dough - dough_per_M * M, dough_per_L)
    # 3) toppings: 5L + 4M >= 4400 -> L >= ceil((4400 - 4M)/5)
    req_L_toppings = ceil_div(req_toppings - toppings_per_M * M, toppings_per_L)
    L_min = max(req_L_ratio, req_L_dough, req_L_toppings, 0)

    # L must be integer >=0
    L = L_min

    # compute objective and check feasibility
    dough_used = dough_per_L * L + dough_per_M * M
    toppings_used = toppings_per_L * L + toppings_per_M * M
    feasible = (dough_used >= req_dough and toppings_used >= req_toppings and M >= min_M and L >= 2 * M)
    if not feasible:
        continue

    Z = bake_L * L + bake_M * M

    if (best_obj is None) or (Z < best_obj):
        best_obj = Z
        best_solutions = [(L, M, Z, dough_used, toppings_used)]
    elif Z == best_obj:
        best_solutions.append((L, M, Z, dough_used, toppings_used))

# If no solution found in range (shouldn't happen), expand search
if best_obj is None:
    # brute force with larger bounds (fallback)
    for M in range(min_M, 20000):
        req_L_ratio = 2 * M
        req_L_dough = ceil_div(req_dough - dough_per_M * M, dough_per_L)
        req_L_toppings = ceil_div(req_toppings - toppings_per_M * M, toppings_per_L)
        L_min = max(req_L_ratio, req_L_dough, req_L_toppings, 0)
        L = L_min
        dough_used = dough_per_L * L + dough_per_M * M
        toppings_used = toppings_per_L * L + toppings_per_M * M
        feasible = (dough_used >= req_dough and toppings_used >= req_toppings and M >= min_M and L >= 2 * M)
        if not feasible:
            continue
        Z = bake_L * L + bake_M * M
        if (best_obj is None) or (Z < best_obj):
            best_obj = Z
            best_solutions = [(L, M, Z, dough_used, toppings_used)]
        elif Z == best_obj:
            best_solutions.append((L, M, Z, dough_used, toppings_used))

# Report results (diagnostics)
print("Found optimal objective (primary):", best_obj)
print("Number of optimal integer solutions found:", len(best_solutions))
print("All optimal solutions (L, M, Z, dough_used, toppings_used):")
for sol in best_solutions:
    print(sol)

# Apply lexicographic secondary objective: minimize total pizzas (L+M) among optimal-Z solutions
best_secondary = None
best_secondary_sol = None
for (L, M, Z, dough_used, toppings_used) in best_solutions:
    total_pizzas = L + M
    if (best_secondary is None) or (total_pizzas < best_secondary):
        best_secondary = total_pizzas
        best_secondary_sol = (L, M, Z, dough_used, toppings_used)

print("Secondary objective (minimize total pizzas) solution:")
print(best_secondary_sol)

# Detailed diagnostics for chosen solution
L_opt, M_opt, Z_opt, dough_used, toppings_used = best_secondary_sol
print("--- Detailed diagnostics ---")
print(f"L = {L_opt}, M = {M_opt}")
print(f"Objective Z = {Z_opt} (baking minutes)")
print(f"Dough used = {dough_used} (requirement {req_dough}), slack = {dough_used - req_dough}")
print(f"Toppings used = {toppings_used} (requirement {req_toppings}), slack = {toppings_used - req_toppings}")
print(f"Ratio constraint L >= 2M: {L_opt} >= {2*M_opt} (slack = {L_opt - 2*M_opt})")
print(f"Minimum M constraint M >= {min_M}: {M_opt} >= {min_M} (slack = {M_opt - min_M})")

# Final required print (exact format)
print(f"FINAL_OBJ: {best_obj}")
