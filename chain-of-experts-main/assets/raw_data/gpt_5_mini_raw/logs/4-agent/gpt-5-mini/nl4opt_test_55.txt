import pulp

# Define the LP problem (maximize)
prob = pulp.LpProblem("berry_farm", pulp.LpMaximize)

# Decision variables: acres of blueberries (B) and raspberries (R)
B = pulp.LpVariable("B", lowBound=0, cat="Continuous")
R = pulp.LpVariable("R", lowBound=0, cat="Continuous")

# Objective: maximize profit 56*B + 75*R
prob += 56 * B + 75 * R

# Constraints
prob += B + R <= 300            # land availability
prob += 22 * B + 25 * R <= 10000  # watering budget
prob += 6 * B + 3 * R <= 575    # labor availability

# Solve using the default CBC solver (quiet)
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract objective value into variable named `value` as required
value = pulp.value(prob.objective)

# Final required output line
print(f"FINAL_OBJ: {value}")
