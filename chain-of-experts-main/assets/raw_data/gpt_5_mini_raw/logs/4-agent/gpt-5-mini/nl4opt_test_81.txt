Dual / shadow-price remarks
The dual problem has three nonnegative multipliers (u1 for land, u2 for fuel, u3 for methane) satisfying:
u1 + 2u2 + 5u3 = 10
u1 + 1u2 + 3u3 = 8

Subtracting gives u2 + 2u3 = 2, so u2 = 2 - 2u3 and u1 = 6 - u3.
Because u2, u3 >= 0, u3 ∈ [0,1]. Consequently the shadow prices are not unique (degeneracy) but lie in ranges:
- u1 (land) ∈ [5, 6] (kg hay per extra acre)
- u2 (fuel) ∈ [0, 2] (kg hay per extra kg fuel)
- u3 (methane) ∈ [0, 1] (kg hay per extra kg methane)

For a concrete interpretation, one valid dual solution is u1 = 6, u2 = 2, u3 = 0: then each extra acre would allow +6 kg hay (if methane/fuel unchanged), and each extra kg of fuel would allow +2 kg hay. But because of degeneracy, these marginal values are not unique without further tie-breaking.

#!/usr/bin/env python3
import numpy as np

# Define constraint lines in form a*x + b*y = c
# Constraints:
# 1) x + y <= 200    -> L1: 1*x + 1*y = 200
# 2) 2x + y <= 300   -> L2: 2*x + 1*y = 300
# 3) 5x + 3y <= 800  -> L3: 5*x + 3*y = 800
# plus nonnegativity boundaries x=0 and y=0 as lines
lines = [
    (1.0, 1.0, 200.0),   # L1
    (2.0, 1.0, 300.0),   # L2
    (5.0, 3.0, 800.0),   # L3
    (1.0, 0.0, 0.0),     # x = 0
    (0.0, 1.0, 0.0)      # y = 0
]

# Tolerance for numerical comparisons
tol = 1e-9

# Compute all pairwise intersections
points = []
n = len(lines)
for i in range(n):
    a1, b1, c1 = lines[i]
    for j in range(i+1, n):
        a2, b2, c2 = lines[j]
        A = np.array([[a1, b1], [a2, b2]], dtype=float)
        det = np.linalg.det(A)
        if abs(det) < 1e-12:
            # Lines are parallel or nearly so; skip
            continue
        C = np.array([c1, c2], dtype=float)
        sol = np.linalg.solve(A, C)
        x, y = float(sol[0]), float(sol[1])
        # Clean near-zero values
        if abs(x) < 1e-12:
            x = 0.0
        if abs(y) < 1e-12:
            y = 0.0
        points.append((x, y))

# Also include the origin explicitly (intersection of x=0 and y=0)
points.append((0.0, 0.0))

# Remove duplicates (within tolerance)
unique_points = []
for p in points:
    x, y = p
    found = False
    for q in unique_points:
        if abs(x - q[0]) < 1e-8 and abs(y - q[1]) < 1e-8:
            found = True
            break
    if not found:
        unique_points.append((x, y))

# Feasible region constraints (inequalities)
def feasible(pt):
    x, y = pt
    if x < -tol or y < -tol:
        return False
    if x + y - 200.0 > tol:
        return False
    if 2.0*x + 1.0*y - 300.0 > tol:
        return False
    if 5.0*x + 3.0*y - 800.0 > tol:
        return False
    return True

feasible_points = [p for p in unique_points if feasible(p)]

# Evaluate objective Z = 10x + 8y
def objective(pt):
    x, y = pt
    return 10.0*x + 8.0*y

best_pt = None
best_obj = -1e100
for p in feasible_points:
    z = objective(p)
    if z > best_obj + 1e-9:
        best_obj = z
        best_pt = p

# As a fallback (in case of numerical issues), check corners formed by bounding each constraint at axes
# (This is redundant here but ensures robustness)
extra_candidates = [
    (0.0, 200.0),  # land boundary with x=0
    (150.0, 0.0),  # fuel boundary with y=0 (2x+y=300 -> x=150)
]
for p in extra_candidates:
    if feasible(p):
        z = objective(p)
        if z > best_obj + 1e-9:
            best_obj = z
            best_pt = p

# Post-process best_obj to print clean integer if very close
if abs(best_obj - round(best_obj)) < 1e-8:
    best_obj = int(round(best_obj))

# Final required print line (must be the very last printed line)
print(f"FINAL_OBJ: {best_obj}")
