H(t) is strictly decreasing in t, so (without any camel-preference) choose the largest feasible t (t = 10).

#!/usr/bin/env python3
# Self-contained script to solve the camel/truck delivery scheduling problem by enumeration.
# It computes three variants:
#  - No preference (pure time minimisation)
#  - Hard camel-preference (c >= t + 1)
#  - Soft camel-preference (tiny penalty on trucks to break ties)
#
# Data are initialised directly in the script. The final printed line is exactly:

from math import isclose

# Problem data
total_load = 1500          # total load to deliver
load_per_camel = 50        # load delivered by one camel caravan
load_per_truck = 150       # load delivered by one truck
hours_per_camel = 12       # hours per camel caravan
hours_per_truck = 5        # hours per truck

# Derived simple bound on trucks and camels
max_trucks = total_load // load_per_truck  # 1500/150 = 10
max_camels = total_load // load_per_camel  # 1500/50 = 30

# Feasible solutions enumeration function
def enumerate_solutions(apply_hard_pref=False, epsilon=0.0):
    best = None
    best_obj = None
    best_details = None
    # As derived, c + 3t = 30  (since divide load equation by 50)
    # Enumerate t from 0..max_trucks
    for t in range(max_trucks + 1):
        c = (total_load - load_per_truck * t) // load_per_camel
        # Check feasibility: exact delivery required
        if load_per_camel * c + load_per_truck * t != total_load:
            continue
        if c < 0:
            continue
        if apply_hard_pref and not (c >= t + 1):
            continue
        # Objective with optional tiny penalty on trucks for soft preference
        obj = hours_per_camel * c + hours_per_truck * t + epsilon * t
        # Use strict comparison for floats (epsilon might make obj float)
        if best is None or obj < best_obj - 1e-9:
            best = (c, t)
            best_obj = obj
            best_details = {"c": c, "t": t, "hours": hours_per_camel * c + hours_per_truck * t}
    return best, best_obj, best_details

# 1) No preference (pure time minimisation)
best_np, best_obj_np, details_np = enumerate_solutions(apply_hard_pref=False, epsilon=0.0)
# details_np['hours'] is integer hours (without epsilon)
