#!/usr/bin/env python3
import math

# Problem data
TOTAL_SEATS = 200
SEATS_VAN = 6
SEATS_CAR = 3
MAX_V = 33  # floor(200/6)
MAX_C = 67  # ceil(200/3)

best = None  # will store tuple (V, C)

for V in range(0, MAX_V + 1):
    # Capacity requirement: 6V + 3C >= 200  =>  C >= ceil((200 - 6V)/3)
    required_seats = TOTAL_SEATS - SEATS_VAN * V
    if required_seats <= 0:
        C_capacity = 0
    else:
        C_capacity = math.ceil(required_seats / SEATS_CAR)
    # Vans-percentage requirement: 7V <= 3C  =>  C >= ceil(7V/3)
    C_percent = math.ceil((7 * V) / 3)
    # Feasible C is the max of the two requirements
    C = max(C_capacity, C_percent)
    if C > MAX_C:
        continue
    # Choose solution minimizing C; tie-breaker: minimize total vehicles (V + C), then minimize V
    if best is None:
        best = (V, C)
    else:
        best_C = best[1]
        best_sum = best[0] + best[1]
        if (C < best_C) or (C == best_C and (V + C) < best_sum) or (C == best_C and (V + C) == best_sum and V < best[0]):
            best = (V, C)

if best is None:
    raise RuntimeError("No feasible solution found within bounds.")

V_opt, C_opt = best
value = C_opt

# Final required print line (must be the last output line)
print(f"FINAL_OBJ: {value}")
