#!/usr/bin/env python3
# Self-contained script to solve the table planning ILP and report results.
# Uses PuLP (CBC) as solver.

from pulp import LpProblem, LpMaximize, LpVariable, LpInteger, PULP_CBC_CMD, value, LpStatus

# Problem data (from the model)
# Decision variables:
# c : number of circular tables (integer, >=0)
# r : number of rectangular tables (integer, >=0)
# Objective: maximize 8*c + 12*r
# Constraints:
# 5c + 4r >= 500       (participant-seat requirement)
# c + r >= 75          (poster-board requirement, since each table gives 4 boards)
# 15c + 20r <= 1900    (space limit)

# Build the problem
prob = LpProblem("Table_Planning", LpMaximize)

c = LpVariable("c", lowBound=0, cat=LpInteger)
r = LpVariable("r", lowBound=0, cat=LpInteger)

# Objective
prob += 8 * c + 12 * r, "Total_Catered_Guests"

# Constraints
prob += 5 * c + 4 * r >= 500, "Participant_Seat_Requirement"
prob += c + r >= 75, "Poster_Board_Requirement"
prob += 15 * c + 20 * r <= 1900, "Space_Limit"

# Solve with CBC (silent)
solver = PULP_CBC_CMD(msg=False)
prob.solve(solver)

# Retrieve solution values (cast to int for reporting)
c_val = int(round(value(c)))
r_val = int(round(value(r)))
total_guests = 8 * c_val + 12 * r_val
participants_supplied = 5 * c_val + 4 * r_val
poster_boards = 4 * (c_val + r_val)
space_used = 15 * c_val + 20 * r_val

# Print a short report
print(f"Status: {LpStatus[prob.status]}")
print(f"c (circular tables): {c_val}")
print(f"r (rectangular tables): {r_val}")
print(f"Total catered guests (objective): {total_guests}")
print(f"Participants supplied (seats): {participants_supplied}")
print(f"Poster boards available: {poster_boards}")
print(f"Space used: {space_used} / 1900")

# Final required single-line output
print(f"FINAL_OBJ: {total_guests}")
