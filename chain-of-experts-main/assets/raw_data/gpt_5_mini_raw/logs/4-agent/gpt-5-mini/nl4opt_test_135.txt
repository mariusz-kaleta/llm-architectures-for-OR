#!/usr/bin/env python3
import math

# Problem data (initialized directly)
hours_required = 300
budget = 6500
wage_S = 120        # seasonal wage
wage_P = 250        # permanent wage
hours_S = 6         # hours provided by one seasonal
hours_P = 10        # hours provided by one permanent

# Practical bounds from budget
S_max = 6500 // wage_S  # 54
P_max = 6500 // wage_P  # 26

best_solution = None  # tuple (total_workers, S, P, cost, hours)
feasible_solutions = []

for P in range(P_max + 1):
    # hours still required after P permanents
    remaining_hours = max(0, hours_required - hours_P * P)
    # minimal seasonal workers needed to cover remaining hours
    S_min = (remaining_hours + hours_S - 1) // hours_S  # ceil division
    if S_min > S_max:
        continue
    cost = wage_S * S_min + wage_P * P
    if cost <= budget:
        total_workers = S_min + P
        hours_total = hours_S * S_min + hours_P * P
        feasible_solutions.append((total_workers, S_min, P, cost, hours_total))
        if (best_solution is None) or (total_workers < best_solution[0]) or (
            total_workers == best_solution[0] and cost < best_solution[3]
        ):
            best_solution = (total_workers, S_min, P, cost, hours_total)

# Ensure we found a feasible solution
if best_solution is None:
    raise SystemExit("No feasible solution found.")

best_total, best_S, best_P, best_cost, best_hours = best_solution

# Optional: show best solution details
print(f"Best solution: S = {best_S}, P = {best_P}, total workers = {best_total}")
print(f"Hours provided = {best_hours}, Cost = ${best_cost}")

# Final required output line (exact format)
print(f"FINAL_OBJ: {best_total}")
