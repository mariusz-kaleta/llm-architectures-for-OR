```python
import numpy as np
from scipy.optimize import linprog

# Parameters
cost_milk = 1.0       # $ per glass
cost_veg  = 2.0       # $ per plate
calcium_milk = 40.0   # units per glass
calcium_veg  = 15.0   # units per plate
iron_milk = 25.0      # units per glass
iron_veg  = 30.0      # units per plate
calcium_req = 100.0   # required units
iron_req = 50.0       # required units

# Continuous LP using scipy.optimize.linprog
# Decision variables: x = [M, V]
c = np.array([cost_milk, cost_veg])
# Constraints are:
# 40 M + 15 V >= 100  -->  -40 M -15 V <= -100
# 25 M + 30 V >= 50   -->  -25 M -30 V <= -50
A_ub = np.array([
    [-calcium_milk, -calcium_veg],
    [-iron_milk,    -iron_veg]
])
b_ub = np.array([-calcium_req, -iron_req])
bounds = [(0, None), (0, None)]

res = linprog(c, A_ub=A_ub, b_ub=b_ub, bounds=bounds, method='highs')

if res.success:
    M_cont, V_cont = res.x
    cost_cont = res.fun
else:
    # Fallback to analytic solution if solver fails
    # As reasoned, milk dominates vegetables in nutrients per dollar:
    M_cont = max(calcium_req / calcium_milk, iron_req / iron_milk)
    V_cont = 0.0
    cost_cont = cost_milk * M_cont + cost_veg * V_cont

# Integer solution by exhaustive search over a reasonable grid
max_M = int(np.ceil(calcium_req / min(calcium_milk, 1))) + 10  # generous upper bound
max_V = int(np.ceil(calcium_req / min(calcium_veg, 1))) + 10
max_M = max(0, min(max_M, 100))  # keep limits reasonable
max_V = max(0, min(max_V, 100))

best_cost_int = float('inf')
best_M_int = None
best_V_int = None

for M in range(0, max_M + 1):
    for V in range(0, max_V + 1):
        if (calcium_milk * M + calcium_veg * V) >= calcium_req and (iron_milk * M + iron_veg * V) >= iron_req:
            cost = cost_milk * M + cost_veg * V
            if cost < best_cost_int - 1e-9:
                best_cost_int = cost
                best_M_int = M
                best_V_int = V

# Print results
print("Continuous LP solution:")
print(f"  M (milk glasses) = {M_cont:.6g}")
print(f"  V (veg plates)   = {V_cont:.6g}")
print(f"  Cost             = ${cost_cont:.6g}")

if best_M_int is not None:
    print("Integer (MILP) solution (by exhaustive search):")
    print(f"  M (milk glasses) = {best_M_int}")
    print(f"  V (veg plates)   = {best_V_int}")
    print(f"  Cost             = ${best_cost_int:.2f}")
else:
    print("No integer feasible solution found within search bounds.")

# Per instructions: final printed line must be exactly this format.
# We choose the integer optimal cost as the final objective value if available,
# otherwise fall back to the continuous optimum cost.
final_obj = best_cost_int if best_M_int is not None else cost_cont
print(f"FINAL_OBJ: {final_obj}")
```
