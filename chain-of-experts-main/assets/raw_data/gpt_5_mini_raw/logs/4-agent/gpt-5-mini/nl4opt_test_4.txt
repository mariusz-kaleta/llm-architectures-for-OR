import numpy as np

# Data (costs and nutrient coefficients)
c = np.array([100.0, 80.0])   # costs for x (Feed A) and y (Feed B)

# Nutrient constraints (coefficients for x and y)
# Protein: 10 x + 7 y >= 30
# Fat:     8 x +15 y >= 50
A = np.array([[10.0, 7.0],
              [8.0, 15.0]])
b = np.array([30.0, 50.0])

# Candidate 1: only A (y = 0)
x_onlyA = max(b[0] / A[0, 0], b[1] / A[1, 0])  # max of protein and fat requirements when y=0
y_onlyA = 0.0
cost_onlyA = c[0] * x_onlyA + c[1] * y_onlyA

# Candidate 2: only B (x = 0)
y_onlyB = max(b[0] / A[0, 1], b[1] / A[1, 1])  # max of protein and fat requirements when x=0
x_onlyB = 0.0
cost_onlyB = c[0] * x_onlyB + c[1] * y_onlyB

# Candidate 3: intersection of equalities (solve A [x;y] = b)
solution = np.linalg.solve(A, b)
x_inter, y_inter = solution.tolist()
cost_inter = c.dot(solution)

# Collect candidates and choose the best feasible one
candidates = [
    {"x": x_onlyA, "y": y_onlyA, "cost": cost_onlyA, "name": "onlyA"},
    {"x": x_onlyB, "y": y_onlyB, "cost": cost_onlyB, "name": "onlyB"},
    {"x": x_inter, "y": y_inter, "cost": cost_inter, "name": "intersection"},
]

# Filter feasible (nonnegative) and choose minimum cost
feasible = [cand for cand in candidates if cand["x"] >= -1e-9 and cand["y"] >= -1e-9]
optimal = min(feasible, key=lambda d: d["cost"])

# Optional: print solution details
print(f"Optimal candidate: {optimal['name']}")
print(f"x* = {optimal['x']}")
print(f"y* = {optimal['y']}")
print(f"Minimum cost = {optimal['cost']}")

# Final required output line (exact format)
print(f"FINAL_OBJ: {optimal['cost']}")
