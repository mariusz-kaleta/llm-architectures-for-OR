[CODER]:
import math

# Parameters
N_target = 10000                # target total images
r_s = 60.0                      # specialized rate (images/hour)
r_c = 40.0                      # common rate (images/hour)
p_s = 100.0                     # specialized hourly cost ($/hour)
p_c = 72.0                      # common hourly cost ($/hour)

# Derived per-image costs
c_s = p_s / r_s
c_c = p_c / r_c

# Optional capacity limits (set to None for unlimited)
# Example: Ns_max = 8000  # maximum images specialized can do
Ns_max = None
Nc_max = None

# Tolerance for feasibility checks
EPS = 1e-9

# Build list of constraint equalities to consider for candidate corner solutions.
# Constraints (equalities we may combine):
# 1) Ns + Nc = N_target
# 2) 2*Ns = Nc  (from 2 Ns >= Nc binding)
# 3) Ns = 0
# 4) Nc = 0
# 5) Ns = Ns_max (if given)
# 6) Nc = Nc_max (if given)

# Represent a constraint as a callable that returns (a,b,c) for equation a*Ns + b*Nc = c
def cons_sum_target():
    return (1.0, 1.0, float(N_target))

def cons_share_binding():
    return (2.0, -1.0, 0.0)  # 2*Ns - Nc = 0

def cons_Ns_zero():
    return (1.0, 0.0, 0.0)

def cons_Nc_zero():
    return (0.0, 1.0, 0.0)

def cons_Ns_max():
    return (1.0, 0.0, float(Ns_max)) if Ns_max is not None else None

def cons_Nc_max():
    return (0.0, 1.0, float(Nc_max)) if Nc_max is not None else None

constraints = [cons_sum_target, cons_share_binding, cons_Ns_zero, cons_Nc_zero]
if Ns_max is not None:
    constraints.append(cons_Ns_max)
if Nc_max is not None:
    constraints.append(cons_Nc_max)

# Function to solve two linear equations: a1*Ns + b1*Nc = c1 ; a2*Ns + b2*Nc = c2
def solve_two(a1,b1,c1,a2,b2,c2):
    det = a1*b2 - a2*b1
    if abs(det) < EPS:
        return None
    Ns = (c1*b2 - c2*b1) / det
    Nc = (a1*c2 - a2*c1) / det
    return (Ns, Nc)

# Feasibility check for a candidate (Ns,Nc)
def feasible(Ns, Nc):
    if Ns is None or Nc is None:
        return False
    if math.isnan(Ns) or math.isnan(Nc):
        return False
    if Ns < -1e-8 or Nc < -1e-8:
        return False
    # inequality constraints:
    if Ns + Nc + 1e-8 < N_target:
        return False
    if 2.0*Ns + 1e-8 < Nc:
        return False
    if Ns_max is not None and Ns - 1e-8 > Ns_max:
        return False
    if Nc_max is not None and Nc - 1e-8 > Nc_max:
        return False
    return True

# Generate candidate corner points by intersecting every pair of constraints
cand_points = []
cons_funcs = constraints
ncons = len(cons_funcs)
for i in range(ncons):
    for j in range(i+1, ncons):
        c1 = cons_funcs[i]()
        c2 = cons_funcs[j]()
        if c1 is None or c2 is None:
            continue
        sol = solve_two(c1[0], c1[1], c1[2], c2[0], c2[1], c2[2])
        if sol is not None:
            Ns_sol, Nc_sol = sol
            if feasible(Ns_sol, Nc_sol):
                cand_points.append((Ns_sol, Nc_sol))

# Also consider intersections of single-constraint boundary with implied minimal other variable
# For example, if Ns + Nc = N_target is not binding at optimum, boundary might be Ns minimal for share constraint.
# To be safe, also sample a few boundary points derived from single constraints:
# - Ns set to minimum allowed by share constraint given Nc=0: from 2*Ns >= 0 -> Ns>=0 (already covered)
# - Ns = N_target (all on Ns) and Nc = 0
cand_points.append((float(N_target), 0.0))
# - Ns = N_target/3 (minimum specialized share) and Nc = N_target - Ns
Ns_min_share = float(N_target) / 3.0
cand_points.append((Ns_min_share, float(N_target) - Ns_min_share))
# - Ns = 0 (if feasible), Nc = N_target (but check share constraint)
cand_points.append((0.0, float(N_target)))
# - If capacities exist, consider maxes with remaining filled by other vendor
if Ns_max is not None:
    rem = max(0.0, float(N_target) - float(Ns_max))
    cand_points.append((float(Ns_max), rem))
if Nc_max is not None:
    rem = max(0.0, float(N_target) - float(Nc_max))
    cand_points.append((rem, float(Nc_max)))

# Filter feasible and unique (within tolerance)
unique_points = []
for Ns, Nc in cand_points:
    if not feasible(Ns, Nc):
        continue
    # Normalize tiny negatives to zero
    Ns = max(0.0, Ns)
    Nc = max(0.0, Nc)
    # Check uniqueness
    found = False
    for a,b in unique_points:
        if abs(a-Ns) < 1e-6 and abs(b-Nc) < 1e-6:
            found = True
            break
    if not found:
        unique_points.append((Ns, Nc))

# Evaluate objective at candidate points
best = None
best_point = None
for Ns, Nc in unique_points:
    cost = c_s * Ns + c_c * Nc
    if best is None or cost < best - 1e-9:
        best = cost
        best_point = (Ns, Nc)

# As a fallback (shouldn't be necessary in this simple problem), if no candidate found,
# use analytic reasoning: if specialized per-image cost <= common, assign all to specialized,
# else assign minimal Ns to satisfy share and remainder to cheaper vendor.
if best is None:
    if c_s <= c_c:
        Ns_opt = float(N_target)
        Nc_opt = 0.0
    else:
        # We need Ns >= N_target/3, so set Ns = N_target/3 and rest to common
        Ns_opt = float(N_target) / 3.0
        Nc_opt = float(N_target) - Ns_opt
    best_point = (Ns_opt, Nc_opt)
    best = c_s * Ns_opt + c_c * Nc_opt

Ns_opt, Nc_opt = best_point
Hs = Ns_opt / r_s
Hc = Nc_opt / r_c
Z = best

# Print some solution details (not the final required line)
print("Solution:")
print(f"  Ns (specialized images) = {Ns_opt:.6f}")
print(f"  Nc (common images)      = {Nc_opt:.6f}")
print(f"  Hs (hours specialized)  = {Hs:.6f}")
print(f"  Hc (hours common)       = {Hc:.6f}")
print(f"  Total cost Z            = {Z:.6f}")

# Final required output line (must be the very last line of output)
value = Z
print(f"FINAL_OBJ: {value}")
