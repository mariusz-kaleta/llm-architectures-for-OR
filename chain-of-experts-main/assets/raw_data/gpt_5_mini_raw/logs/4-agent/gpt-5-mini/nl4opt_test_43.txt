#!/usr/bin/env python3
# Self-contained solver for the ILP described by the user.
# Enumerates integer (D,S) within given practical bounds and finds the minimum D+S.

# Parameters
rate_D = 50  # letters/minute per dual machine
rate_S = 30  # letters/minute per single machine
glue_D = 20  # glue units/minute per dual machine
glue_S = 15  # glue units/minute per single machine
required_rate = 300
glue_limit = 135

# Practical bounds
D_min, D_max = 0, 6  # inclusive bounds for D
S_min, S_max = 0, 9  # inclusive bounds for S

best_obj = None
best_solution = None

for D in range(D_min, D_max + 1):
    for S in range(S_min, S_max + 1):
        # Constraints
        if rate_D * D + rate_S * S < required_rate:
            continue
        if glue_D * D + glue_S * S > glue_limit:
            continue
        if S < D + 1:
            continue
        obj = D + S
        if (best_obj is None) or (obj < best_obj):
            best_obj = obj
            best_solution = (D, S)

# At this point best_obj should hold the minimal number of machines
# Print final objective value in required format
print(f"FINAL_OBJ: {best_obj}")
