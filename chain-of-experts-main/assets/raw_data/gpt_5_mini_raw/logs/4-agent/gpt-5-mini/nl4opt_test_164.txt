import math

# Problem data
profit_R = 400.0  # profit per refrigerator
profit_S = 260.0  # profit per stove

# Resource capacities (minutes)
mover_cap = 20000.0
setup_cap = 13000.0

# Resource usage per appliance (minutes)
mover_R = 60.0
mover_S = 45.0
setup_R = 20.0
setup_S = 25.0

# 1) Continuous LP: evaluate corner points (axes intercepts and intersection)
candidates = []

# Origin
candidates.append((0.0, 0.0))

# R-axis intercepts (S=0) limited by both constraints
R_by_mover = mover_cap / mover_R
R_by_setup = setup_cap / setup_R
R_axis = min(R_by_mover, R_by_setup)
candidates.append((R_axis, 0.0))

# S-axis intercepts (R=0)
S_by_mover = mover_cap / mover_S
S_by_setup = setup_cap / setup_S
S_axis = min(S_by_mover, S_by_setup)
candidates.append((0.0, S_axis))

# Intersection of the two constraint lines:
# Solve:
# 60 R + 45 S = 20000
# 20 R + 25 S = 13000
# Solve using linear algebra
det = mover_R * setup_S - mover_S * setup_R
if abs(det) > 1e-12:
    R_inter = (mover_cap * setup_S - mover_S * setup_cap) / det
    S_inter = (mover_R * setup_cap - mover_cap * setup_R) / det
    candidates.append((R_inter, S_inter))

# Evaluate feasible candidates and pick best continuous solution
def feasible(R, S, tol=1e-9):
    if R < -tol or S < -tol:
        return False
    if mover_R * R + mover_S * S - mover_cap > tol:
        return False
    if setup_R * R + setup_S * S - setup_cap > tol:
        return False
    return True

best_cont = (0.0, 0.0)
best_cont_obj = -1e18
for R, S in candidates:
    if feasible(R, S):
        obj = profit_R * R + profit_S * S
        if obj > best_cont_obj:
            best_cont_obj = obj
            best_cont = (R, S)

# 2) Integer (practical) solution: brute-force search over feasible integer (R,S)
# Upper bounds for loops can be derived from mover capacity alone (tighter generally)
R_max_int = int(math.floor(mover_cap / mover_R))  # 333
S_max_int = int(math.floor(mover_cap / mover_S))  # 444

best_int_obj = -1e18
best_int_sol = (0, 0)

for R_int in range(0, R_max_int + 1):
    # For given R_int, compute max feasible S due to each constraint
    rem_mover = mover_cap - mover_R * R_int
    rem_setup = setup_cap - setup_R * R_int
    if rem_mover < 0 or rem_setup < 0:
        # R_int alone violates a constraint
        continue
    S_max_by_mover = int(math.floor(rem_mover / mover_S))
    S_max_by_setup = int(math.floor(rem_setup / setup_S))
    S_max_for_R = min(S_max_by_mover, S_max_by_setup)
    if S_max_for_R < 0:
        continue
    # Instead of only checking S_max_for_R, scan all possible S up to that to be safe
    for S_int in range(0, S_max_for_R + 1):
        obj = profit_R * R_int + profit_S * S_int
        if obj > best_int_obj:
            best_int_obj = obj
            best_int_sol = (R_int, S_int)

# Prepare results (not printed except final required line)
continuous_solution = {
    "R": best_cont[0],
    "S": best_cont[1],
    "objective": best_cont_obj
}
integer_solution = {
    "R": best_int_sol[0],
    "S": best_int_sol[1],
    "objective": best_int_obj
}

# Final required output line (print only this line in this exact format)
print(f"FINAL_OBJ: {int(integer_solution['objective'])}")
