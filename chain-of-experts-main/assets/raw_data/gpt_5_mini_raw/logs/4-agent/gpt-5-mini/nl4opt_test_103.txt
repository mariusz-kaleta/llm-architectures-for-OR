#!/usr/bin/env python3
# Self-contained script to enumerate integer solutions for the balloon/gondola problem
from math import ceil

# Parameters (initialized directly)
balloon_capacity = 4
gondola_capacity = 6
balloon_pollution = 10
gondola_pollution = 15
required_visitors = 70
max_balloons = 10

# Enumerate feasible integer solutions
solutions = []
for b in range(0, max_balloons + 1):
    needed = max(0, required_visitors - balloon_capacity * b)
    g = ceil(needed / gondola_capacity)
    # Ensure non-negative integer g
    if g < 0:
        g = 0
    seats = balloon_capacity * b + gondola_capacity * g
    pollution = balloon_pollution * b + gondola_pollution * g
    if seats >= required_visitors:
        solutions.append((b, g, seats, pollution))

# Find minimum pollution and all optimal solutions
if not solutions:
    raise RuntimeError("No feasible solutions found.")
min_pollution = min(s[3] for s in solutions)
optimal_solutions = [s for s in solutions if s[3] == min_pollution]

# Print results (details)
print("All feasible solutions (b, g, seats, pollution):")
for s in solutions:
    print(s)
print()
print("Optimal solutions (b, g, seats, pollution):")
for s in optimal_solutions:
    print(s)

# Final required line with exact format
print(f"FINAL_OBJ: {min_pollution}")
