```python
import math

# Problem data
total_children = 100
bus_capacity = 9
car_capacity = 4
min_c = 5

best_obj = None
best_solution = None

# Search over reasonable range for c
for c in range(min_c, 26):  # 5..25 inclusive
    # capacity requirement: 9*b + 4*c >= 100
    remaining = max(0, total_children - car_capacity * c)
    b_capacity = math.ceil(remaining / bus_capacity) if remaining > 0 else 0
    b_min = max(c + 1, b_capacity)  # also require b >= c+1
    obj = b_min + c
    if best_obj is None or obj < best_obj:
        best_obj = obj
        best_solution = (b_min, c)

# Final objective value
value = best_obj

print(f"FINAL_OBJ: {value}")
```
