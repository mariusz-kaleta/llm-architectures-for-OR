import math

# Problem data
iron_A = 5
iron_B = 4
calcium_A = 10
calcium_B = 15
cost_A = 2
cost_B = 3

req_iron = 40
req_calcium = 50

# Search bounds:
# A-only required x gives an upper bound for reasonable search range
x_bound = max(0, math.ceil(req_iron / iron_A), math.ceil(req_calcium / calcium_A)) + 10
y_bound = max(0, math.ceil(req_iron / iron_B), math.ceil(req_calcium / calcium_B)) + 10

# Keep them reasonably small
x_bound = min(x_bound, 100)
y_bound = min(y_bound, 100)

best_cost = float('inf')
best_solutions = []

for x in range(0, x_bound + 1):
    for y in range(0, y_bound + 1):
        if iron_A * x + iron_B * y >= req_iron and calcium_A * x + calcium_B * y >= req_calcium:
            cost = cost_A * x + cost_B * y
            if cost < best_cost:
                best_cost = cost
                best_solutions = [(x, y)]
            elif cost == best_cost:
                best_solutions.append((x, y))

# If no feasible solution found (shouldn't happen here), handle gracefully
