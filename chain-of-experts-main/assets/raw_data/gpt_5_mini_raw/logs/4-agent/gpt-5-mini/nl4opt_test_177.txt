import math

# Problem data
capacity_needed = 1000
canoe_cap = 10
diesel_cap = 15

# Reformulation parameters
# c = 3*d + s, s >= 0
# capacity: 45*d + 10*s >= 1000
# search d in 0..23 (as per problem reasoning)
best = None  # tuple (Z, d, s, c)

for d in range(0, 24):  # 0..23 inclusive
    remaining = capacity_needed - 45 * d
    if remaining <= 0:
        s = 0
    else:
        s = math.ceil(remaining / 10)
    if s < 0:
        s = 0
    c = 3 * d + s
    Z = c + d  # or 4*d + s
    # Check constraints
    if 10 * c + 15 * d < capacity_needed:
        continue
    if c < 3 * d:
        continue
    if best is None or Z < best[0]:
        best = (Z, d, s, c)

if best is None:
    raise RuntimeError("No feasible solution found by enumeration.")

best_Z, best_d, best_s, best_c = best

# Verify by brute force over feasible ranges (c up to 100, d up to 23)
bf_best = None
for c in range(0, 101):
    for d in range(0, 24):
        if 10 * c + 15 * d >= capacity_needed and c >= 3 * d:
            Z = c + d
            if bf_best is None or Z < bf_best[0]:
                bf_best = (Z, c, d)
# Ensure enumeration matches brute force
if bf_best is None:
    raise RuntimeError("No feasible solution found by brute force.")
if bf_best[0] != best_Z or bf_best[1] != best_c or bf_best[2] != best_d:
    raise AssertionError(f"Mismatch between methods: enumeration {best} vs brute-force {bf_best}")

# Print results (summary)
print(f"Optimal solution: c (canoes) = {best_c}, d (diesel) = {best_d}")
print(f"Extra canoes s = {best_s}")
print(f"Total capacity = {10*best_c + 15*best_d}")

# Final required line
value = best_Z
