So T is increasing in G; we minimize by taking G as small as possible subject to the other constraints.

import math

# Data (initialized directly)
TOTAL = 400
MIN_S = 100

# Analytic continuous solution
# From S + G = TOTAL and S <= 2G => TOTAL - G <= 2G => G >= TOTAL/3
G_cont = TOTAL / 3.0  # 400/3
S_cont = TOTAL - G_cont
obj_cont = 0.5 * S_cont + 0.75 * G_cont  # should be 700/3

# Compute integer optimum by enumeration (if integrality required)
best_int_obj = None
best_int_sol = None
for G in range(TOTAL + 1):
    S = TOTAL - G
    if S < 0 or G < 0:
        continue
    if S < MIN_S:
        continue
    if S > 2 * G:
        continue
    obj = 0.5 * S + 0.75 * G
    if (best_int_obj is None) or (obj < best_int_obj - 1e-12):
        best_int_obj = obj
        best_int_sol = (S, G)

# Print results (informational)
print(f"Continuous solution: S = {S_cont:.12f}, G = {G_cont:.12f}, Objective = {obj_cont:.12f}")
if best_int_sol is not None:
    S_int, G_int = best_int_sol
    print(f"Integer solution:    S = {S_int}, G = {G_int}, Objective = {best_int_obj:.12f}")
else:
    print("No feasible integer solution found under given constraints.")

# FINAL required print line with final objective value (continuous optimum chosen here)
value = obj_cont
print(f"FINAL_OBJ: {value}")
