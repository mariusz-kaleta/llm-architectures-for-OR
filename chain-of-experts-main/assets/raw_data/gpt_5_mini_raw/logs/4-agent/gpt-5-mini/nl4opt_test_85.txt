#!/usr/bin/env python3
# Self-contained script to solve the continuous LP and the integer (whole-tank) MILP variant
# Problem data encoded directly below. No external solver required; LP solved by enumerating vertices,
# MILP solved by brute-force enumeration within tight bounds.

import itertools
import math

# Problem data
r_L, r_N, r_H = 550.0, 750.0, 950.0
a_L, a_N, a_H = 3.0, 6.0, 9.0
b_L, b_N, b_H = 3.0, 2.0, 3.0
A_budget = 250.0
B_budget = 150.0

# Helper: evaluate feasibility of a candidate solution (xL,xN,xH)
def feasible(x):
    xL, xN, xH = x
    if xL < -1e-9 or xN < -1e-9 or xH < -1e-9:
        return False
    if a_L * xL + a_N * xN + a_H * xH - A_budget > 1e-8:
        return False
    if b_L * xL + b_N * xN + b_H * xH - B_budget > 1e-8:
        return False
    return True

# Objective
def objective(x):
    xL, xN, xH = x
    return r_L * xL + r_N * xN + r_H * xH

# Solve the continuous LP by enumerating all vertices formed by choosing 3 active constraints
# Active constraints set: A_eq, B_eq, xL=0, xN=0, xH=0  (total 5). Choose 3 at a time, solve linear system.
# We'll solve small 3x3 systems using plain linear algebra (no external libs).
def solve_linear_system(mat, rhs):
    # Solve 3x3 linear system mat * x = rhs using Cramer's rule to avoid numpy dependency
    # mat is list of 3 lists, rhs is list of 3 numbers
    def det3(m):
        return (
            m[0][0] * (m[1][1]*m[2][2] - m[1][2]*m[2][1])
            - m[0][1] * (m[1][0]*m[2][2] - m[1][2]*m[2][0])
            + m[0][2] * (m[1][0]*m[2][1] - m[1][1]*m[2][0])
        )
    D = det3(mat)
    if abs(D) < 1e-12:
        return None  # singular
    # build Dx, Dy, Dz
    mats = []
    for i in range(3):
        m_copy = [row[:] for row in mat]
        for r in range(3):
            m_copy[r][i] = rhs[r]
        mats.append(m_copy)
    Dx = det3(mats[0])
    Dy = det3(mats[1])
    Dz = det3(mats[2])
    return [Dx / D, Dy / D, Dz / D]

# Define the 5 possible active constraints as linear equations in form coeffs = [cL, cN, cH], rhs
constraints = [
    ([a_L, a_N, a_H], A_budget),  # A_eq
    ([b_L, b_N, b_H], B_budget),  # B_eq
    ([1.0, 0.0, 0.0], 0.0),       # xL = 0
    ([0.0, 1.0, 0.0], 0.0),       # xN = 0
    ([0.0, 0.0, 1.0], 0.0),       # xH = 0
]

best_continuous = None
best_continuous_obj = -1e100
# iterate all combinations of 3 constraints chosen out of 5
for combo in itertools.combinations(range(len(constraints)), 3):
    mat = [constraints[i][0][:] for i in combo]
    rhs = [constraints[i][1] for i in combo]
    sol = solve_linear_system(mat, rhs)
    if sol is None:
        continue
    # Numerical tolerances: treat tiny negative as zero
    sol = [0.0 if abs(v) < 1e-10 else v for v in sol]
    if feasible(sol):
        obj = objective(sol)
        if obj > best_continuous_obj + 1e-8:
            best_continuous_obj = obj
            best_continuous = sol

# As a fallback (shouldn't be needed), also consider corner points where two variables = 0 and third from constraints
if best_continuous is None:
    # try xL,xN=0 solve for xH from min of budgets
    # but in normal LP there should be a vertex found above
    pass

# Print continuous solution found
print("Continuous LP solution (by vertex enumeration):")
if best_continuous is not None:
    xL_c, xN_c, xH_c = best_continuous
    print(f" x_L = {xL_c:.9f}")
    print(f" x_N = {xN_c:.9f}")
    print(f" x_H = {xH_c:.9f}")
    print(f" Objective Z = {best_continuous_obj:.9f}")
else:
    print(" No feasible continuous vertex found (unexpected).")

# Now solve integer (MILP) by brute-force enumeration of integer tanks within tight bounds
# Compute individual upper bounds for loops
ub_L = int(min(math.floor(A_budget / a_L) if a_L > 0 else float('inf'),
               math.floor(B_budget / b_L) if b_L > 0 else float('inf')))
ub_N = int(min(math.floor(A_budget / a_N) if a_N > 0 else float('inf'),
               math.floor(B_budget / b_N) if b_N > 0 else float('inf')))
ub_H = int(min(math.floor(A_budget / a_H) if a_H > 0 else float('inf'),
               math.floor(B_budget / b_H) if b_H > 0 else float('inf')))

# Slightly tighten upper bounds by global resource check (though above already does)
ub_L = max(0, ub_L)
ub_N = max(0, ub_N)
ub_H = max(0, ub_H)

best_int_obj = -1e100
best_int_solution = None

# Triple nested loops; ranges are small (approx 0..83, 0..41, 0..27 => ~100k combos)
for xL in range(ub_L + 1):
    # Precompute resource usage by xL
    A_after_L = A_budget - a_L * xL
    B_after_L = B_budget - b_L * xL
    if A_after_L < -1e-9 or B_after_L < -1e-9:
        continue
    # bounds for xN given xL
    max_xN_by_A = int(math.floor(A_after_L / a_N)) if a_N > 0 else ub_N
    max_xN_by_B = int(math.floor(B_after_L / b_N)) if b_N > 0 else ub_N
    max_xN = min(ub_N, max_xN_by_A, max_xN_by_B)
    if max_xN < 0:
        continue
    for xN in range(max_xN + 1):
        A_rem = A_after_L - a_N * xN
        B_rem = B_after_L - b_N * xN
        if A_rem < -1e-9 or B_rem < -1e-9:
            continue
        # max feasible xH given remaining resources
        max_xH_by_A = int(math.floor(A_rem / a_H)) if a_H > 0 else ub_H
        max_xH_by_B = int(math.floor(B_rem / b_H)) if b_H > 0 else ub_H
        max_xH = min(ub_H, max_xH_by_A, max_xH_by_B)
        if max_xH < 0:
            continue
        # For integer optimization, often best xH will be either 0..max_xH. We enumerate all.
        # But to speed slightly, check objective monotonicity in xH: r_H per resource might affect; but keep exhaustive for correctness.
        for xH in range(max_xH + 1):
            # All constraints satisfied by construction
            obj = r_L * xL + r_N * xN + r_H * xH
            if obj > best_int_obj + 1e-8:
                best_int_obj = obj
                best_int_solution = (xL, xN, xH)

# Print integer (MILP) solution found
print("\nInteger MILP (whole-tank) optimal solution (by brute-force):")
if best_int_solution is not None:
    xL_i, xN_i, xH_i = best_int_solution
    print(f" x_L = {xL_i}")
    print(f" x_N = {xN_i}")
    print(f" x_H = {xH_i}")
    print(f" Objective Z = {best_int_obj:.9f}")
    # compute resource usage
    A_used = a_L * xL_i + a_N * xN_i + a_H * xH_i
    B_used = b_L * xL_i + b_N * xN_i + b_H * xH_i
    print(f" A used = {A_used} / {A_budget}")
    print(f" B used = {B_used} / {B_budget}")
else:
    print(" No feasible integer solution found (unexpected).")

# As instructed: the very last line must be exactly the following print with the final objective value.
# We choose to report the integer (MILP) optimal objective as the final value (as the integer variant was highlighted).
final_value = int(best_int_obj) if abs(best_int_obj - round(best_int_obj)) < 1e-8 else float(best_int_obj)
print(f"FINAL_OBJ: {final_value}")
