#!/usr/bin/env python3
# Self-contained script to solve the ILP by exhaustive search (small bounds)
# Decision variables:
#   x: number of washing machines to repair (integer >=0)
#   y: number of freezers to repair (integer >=0)
# Objective:
#   maximize Z = 250*x + 375*y
# Constraints:
#   30*x + 20*y <= 5000         (inspection-time)
#   120*x + 145*y <= 20000      (total schedule time)

def solve():
    # Parameters
    rev_x = 250
    rev_y = 375
    # Constraints coefficients
    a1_x, a1_y, b1 = 30, 20, 5000
    a2_x, a2_y, b2 = 120, 145, 20000

    # Optional solver bounds (tight integer bounds)
    x_max = min(b1 // a1_x, b2 // a2_x)  # floor(5000/30), floor(20000/120)
    y_max = min(b1 // a1_y, b2 // a2_y)  # floor(5000/20), floor(20000/145)

    best_obj = None
    best_x = None
    best_y = None

    for x in range(x_max + 1):
        # optional prune: compute max y given constraints
        # from constraint1: y <= (b1 - a1_x*x) // a1_y
        # from constraint2: y <= (b2 - a2_x*x) // a2_y
        if a1_x * x > b1 or a2_x * x > b2:
            continue
        y_upper1 = (b1 - a1_x * x) // a1_y
        y_upper2 = (b2 - a2_x * x) // a2_y
        y_upper = min(y_max, y_upper1, y_upper2)
        if y_upper < 0:
            continue
        for y in range(y_upper + 1):
            # constraints already enforced by y_upper, but check to be safe
            if a1_x * x + a1_y * y > b1:
                continue
            if a2_x * x + a2_y * y > b2:
                continue
            obj = rev_x * x + rev_y * y
            if (best_obj is None) or (obj > best_obj):
                best_obj = obj
                best_x = x
                best_y = y

    # Compute resource usages and slacks for the best solution
    inspection_used = a1_x * best_x + a1_y * best_y
    total_used = a2_x * best_x + a2_y * best_y
    inspection_slack = b1 - inspection_used
    total_slack = b2 - total_used

    # Print solution summary
    print(f"Optimal x (washers): {best_x}")
    print(f"Optimal y (freezers): {best_y}")
    print(f"Objective (revenue): {best_obj}")
    print(f"Inspection minutes used: {inspection_used}  Slack: {inspection_slack}")
    print(f"Total minutes used: {total_used}  Slack: {total_slack}")

    # Final required output line (must be the last line)
    print(f"FINAL_OBJ: {best_obj}")

if __name__ == "__main__":
    solve()
