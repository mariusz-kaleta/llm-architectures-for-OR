#!/usr/bin/env python3
# Self-contained script to solve the described LP and ILP (no external solvers required).
# Data is initialised directly in code. The script:
#  - Solves the continuous LP analytically (corner points + intersection)
#  - Computes dual prices for the continuous optimum (by solving the dual system)
#  - Solves the integer variant by brute-force enumeration (small finite domain)
#  - Prints summaries and at the very end prints the final objective value in the required format.

from math import isclose

# Problem data
# Decision variables: A >= 0, B >= 0
# Constraints:
#   10A + 15B <= 300   (wood)
#   20A + 12B <= 300   (oxygen)
# Objective: maximize Z = 15A + 18B

# Capacities
wood_cap = 300.0
oxy_cap = 300.0

# Coefficients
a_wood = 10.0
b_wood = 15.0
a_oxy = 20.0
b_oxy = 12.0

cA = 15.0
cB = 18.0

tol = 1e-9

# Helper to evaluate feasibility and objective
def feasible(A, B):
    return (A >= -tol) and (B >= -tol) and (a_wood * A + b_wood * B <= wood_cap + tol) and (a_oxy * A + b_oxy * B <= oxy_cap + tol)

def obj(A, B):
    return cA * A + cB * B

# 1) Continuous LP: check corner points and intersection
candidates = []

# Corner 1: A = 0, B from wood or oxy constraints -> B <= min(wood_cap/b_bwood, oxy_cap/b_oxy)
B_max_by_wood = wood_cap / b_wood
B_max_by_oxy = oxy_cap / b_oxy
B_corner1 = min(B_max_by_wood, B_max_by_oxy)
candidates.append((0.0, B_corner1))

# Corner 2: B = 0, A <= min(wood_cap/a_wood, oxy_cap/a_oxy)
A_max_by_wood = wood_cap / a_wood
A_max_by_oxy = oxy_cap / a_oxy
A_corner2 = min(A_max_by_wood, A_max_by_oxy)
candidates.append((A_corner2, 0.0))

# Intersection of the two binding constraints (if any)
# Solve:
# a_wood * A + b_wood * B = wood_cap
# a_oxy  * A + b_oxy  * B = oxy_cap
det = a_wood * b_oxy - a_oxy * b_wood
if abs(det) > tol:
    A_inter = (wood_cap * b_oxy - oxy_cap * b_wood) / det
    B_inter = (a_wood * oxy_cap - a_oxy * wood_cap) / det
    # If the intersection gives nonnegative A,B, include it
    if A_inter >= -tol and B_inter >= -tol:
        # clamp tiny negatives to zero
        A_inter = max(A_inter, 0.0)
        B_inter = max(B_inter, 0.0)
        candidates.append((A_inter, B_inter))

# Evaluate candidates and pick the best feasible continuous solution
best_cont = None
best_cont_obj = -1e99
for A_c, B_c in candidates:
    if feasible(A_c, B_c):
        z = obj(A_c, B_c)
        if z > best_cont_obj + tol:
            best_cont = (A_c, B_c)
            best_cont_obj = z

# If none of the above found (shouldn't happen), fallback to checking boundary lines
if best_cont is None:
    # try sampling A in [0, A_corner2] and solve for B on binding constraints
    best_cont = (0.0, 0.0)
    best_cont_obj = 0.0
    for A_s in [i * 0.5 for i in range(int(A_corner2 * 2) + 1)]:
        # compute max B allowed by both constraints
        B1 = (wood_cap - a_wood * A_s) / b_wood if a_wood * A_s <= wood_cap + tol else -1.0
        B2 = (oxy_cap - a_oxy * A_s) / b_oxy if a_oxy * A_s <= oxy_cap + tol else -1.0
        B_s_max = min(B1, B2)
        if B_s_max >= -tol:
            B_s_max = max(B_s_max, 0.0)
            z = obj(A_s, B_s_max)
            if z > best_cont_obj + tol:
                best_cont = (A_s, B_s_max)
                best_cont_obj = z

A_cont, B_cont = best_cont
Z_cont = best_cont_obj

# Determine which constraints are binding at continuous optimum
wood_usage_cont = a_wood * A_cont + b_wood * B_cont
oxy_usage_cont = a_oxy * A_cont + b_oxy * B_cont
wood_binding_cont = isclose(wood_usage_cont, wood_cap, rel_tol=0.0, abs_tol=1e-6)
oxy_binding_cont = isclose(oxy_usage_cont, oxy_cap, rel_tol=0.0, abs_tol=1e-6)

# Compute dual (shadow) prices for continuous LP.
# If both constraints binding, solve:
# a_wood * u1 + a_oxy * u2 = cA   (coeffs for A: col of constraint matrix)
# b_wood * u1 + b_oxy * u2 = cB   (coeffs for B)
u1 = None
u2 = None
if wood_binding_cont and oxy_binding_cont:
    # Solve 2x2 system:
    # [a_wood  a_oxy]   [u1]   = [cA]
    # [b_wood  b_oxy]   [u2]     [cB]
    # Note: careful with mapping; original dual eqs are:
    # a_wood*u1 + a_oxy*u2 = cA  where a_oxy should be coefficient of A in oxygen constraint (that's a_oxy)
    # b_wood*u1 + b_oxy*u2 = cB
    det_dual = a_wood * b_oxy - b_wood * a_oxy
    if abs(det_dual) > tol:
        u1 = (cA * b_oxy - cB * a_oxy) / det_dual
        u2 = (a_wood * cB - b_wood * cA) / det_dual
    else:
        # degenerate; fall back to zeros
        u1 = 0.0
        u2 = 0.0
else:
    # If only one constraint binding, the dual for the nonbinding constraint is zero.
    # We can attempt to compute dual(s) from binding constraints only.
    # For simplicity handle the typical cases:
    if wood_binding_cont and not oxy_binding_cont:
        # Then u2 = 0, solve a_wood * u1 = cA and b_wood * u1 = cB (overdetermined but consistent if one variable is zero)
        # Try least squares style or pick u1 = cA / a_wood if consistent
        if abs(a_wood) > tol:
            u1 = cA / a_wood
        else:
            u1 = 0.0
        u2 = 0.0
    elif oxy_binding_cont and not wood_binding_cont:
        if abs(a_oxy) > tol:
            u2 = cA / a_oxy
        else:
            u2 = 0.0
        u1 = 0.0
    else:
        # No binding constraints: duals zero
        u1 = 0.0
        u2 = 0.0

# Normalize tiny numerical noise
u1 = 0.0 if abs(u1) < 1e-12 else u1
u2 = 0.0 if abs(u2) < 1e-12 else u2

# 2) Integer ILP: brute-force enumeration (small domain)
# Determine integer bounds
A_int_max = int(min(wood_cap / a_wood, oxy_cap / a_oxy))  # floor
B_int_max = int(min(wood_cap / b_wood, oxy_cap / b_oxy))

best_ilp_obj = -1e99
best_ilp_sol = (0, 0)
feasible_count = 0
for A_i in range(0, A_int_max + 1):
    for B_i in range(0, B_int_max + 1):
        if a_wood * A_i + b_wood * B_i <= wood_cap + tol and a_oxy * A_i + b_oxy * B_i <= oxy_cap + tol:
            feasible_count += 1
            z = obj(A_i, B_i)
            if z > best_ilp_obj + tol:
                best_ilp_obj = z
                best_ilp_sol = (A_i, B_i)

A_ilp, B_ilp = best_ilp_sol
Z_ilp = best_ilp_obj
wood_usage_ilp = a_wood * A_ilp + b_wood * B_ilp
oxy_usage_ilp = a_oxy * A_ilp + b_oxy * B_ilp
wood_binding_ilp = isclose(wood_usage_ilp, wood_cap, rel_tol=0.0, abs_tol=1e-6)
oxy_binding_ilp = isclose(oxy_usage_ilp, oxy_cap, rel_tol=0.0, abs_tol=1e-6)

# Print results
print("Continuous LP solution (analytic/corner-point check):")
print(f"  A* = {A_cont:.6f}")
print(f"  B* = {B_cont:.6f}")
print(f"  Max CO2 (continuous) = {Z_cont:.6f}")
print("  Resource usage (continuous):")
print(f"    Wood used = {wood_usage_cont:.6f} / {wood_cap}")
print(f"    Oxygen used = {oxy_usage_cont:.6f} / {oxy_cap}")
print(f"  Binding constraints (continuous): wood={wood_binding_cont}, oxygen={oxy_binding_cont}")
print(f"  Dual (shadow) prices: u_wood = {u1:.6f}, u_oxygen = {u2:.6f}")
print()
print("Integer ILP solution (enumeration):")
print(f"  A_int = {A_ilp}")
print(f"  B_int = {B_ilp}")
print(f"  Max CO2 (integer) = {Z_ilp}")
print("  Resource usage (integer):")
print(f"    Wood used = {wood_usage_ilp} / {wood_cap}")
print(f"    Oxygen used = {oxy_usage_ilp} / {oxy_cap}")
print(f"  Binding constraints (integer): wood={wood_binding_ilp}, oxygen={oxy_binding_ilp}")
print(f"  Number of feasible integer combinations checked = {feasible_count}")

# Final required print (must be the very last printed line and in exact format)
print(f"FINAL_OBJ: {int(Z_ilp)}")
