# Define model
m = pulp.LpProblem("Lawn", pulp.LpMaximize)

import math

# Problem data
employee_capacity = 150
small_team_size = 3
large_team_size = 5
area_small = 50
area_large = 80

S_min_global = 10
L_min_global = 6
S_max_bound = employee_capacity // small_team_size  # 50
L_max_bound = employee_capacity // large_team_size  # 30

best_obj = None
best_S = None
best_L = None

for L in range(L_min_global, L_max_bound + 1):
    S_min = max(S_min_global, 3 * L)
    # compute maximum S allowed by employee capacity
    remaining_employees = employee_capacity - large_team_size * L
    if remaining_employees < 0:
        continue
    S_max = remaining_employees // small_team_size
    if S_max < S_min:
        continue
    # since objective increases with S, pick S_max
    S = S_max
    obj = area_small * S + area_large * L
    if best_obj is None or obj > best_obj:
        best_obj = obj
        best_S = S
        best_L = L

# Print solution summary
print(f"Best solution found: S = {best_S}, L = {best_L}, Area = {best_obj}")
# Final required line (must be the last printed line)
print(f"FINAL_OBJ: {best_obj}")
