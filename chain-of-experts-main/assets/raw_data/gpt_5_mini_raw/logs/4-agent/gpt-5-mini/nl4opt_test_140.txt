import numpy as np

# Data
# Objective: maximize Z = 3*x + 5*y
c = np.array([3.0, 5.0])

# Constraints:
# 0.8 x + 0.1 y <= 80
# 0.2 x + 0.9 y <= 60
A = np.array([[0.8, 0.1],
              [0.2, 0.9]])
b = np.array([80.0, 60.0])

# Candidate corner points of feasible region:
candidates = []

# Origin
candidates.append(np.array([0.0, 0.0]))

# x-axis intercept: set y=0, take min enforced by constraints
# From first constraint: 0.8 x <= 80 -> x = 80/0.8 = 100
# From second constraint: 0.2 x <= 60 -> x = 60/0.2 = 300
# Feasible x-axis intercept is min -> 100
candidates.append(np.array([80.0/0.8, 0.0]))

# y-axis intercept: set x=0, take min enforced by constraints
# From first constraint: 0.1 y <= 80 -> y = 80/0.1 = 800
# From second constraint: 0.9 y <= 60 -> y = 60/0.9 = 66.666...
# Feasible y-axis intercept is min -> 60/0.9
candidates.append(np.array([0.0, 60.0/0.9]))

# Intersection of both constraints (solve A * [x,y] = b)
intersection = np.linalg.solve(A, b)
candidates.append(intersection)

# Evaluate feasibility (allow small tolerance) and objective
feasible_points = []
tol = 1e-9
for pt in candidates:
    if np.all(pt >= -tol) and np.all(A.dot(pt) <= b + 1e-8):
        feasible_points.append(pt)

# Compute objective values and find maximum
best_obj = -np.inf
best_point = None
for pt in feasible_points:
    obj = c.dot(pt)
    if obj > best_obj:
        best_obj = obj
        best_point = pt

# Print results
x_opt, y_opt = best_point
print(f"Optimal x = {x_opt:.6f} kg")
print(f"Optimal y = {y_opt:.6f} kg")
print(f"Maximum profit = {best_obj:.6f} dollars")

# Final required line
print(f"FINAL_OBJ: {best_obj}")
