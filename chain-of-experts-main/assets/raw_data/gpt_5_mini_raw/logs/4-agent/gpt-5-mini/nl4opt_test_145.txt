#!/usr/bin/env python3
# Self-contained solver for the crab cakes (C) and lobster rolls (L) LP and MILP problem.
# Computes the continuous LP optimum analytically and finds the integer MILP optimum by brute force search.
# Prints results and ends with the required final objective line.

from math import isclose

# Analytical LP solution (solve the two vitamin equalities):
# 5C + 8L = 80
# 7C + 4L = 100
# Solve using elimination:
# Multiply first by 7: 35C + 56L = 560
# Multiply second by 5: 35C + 20L = 500
# Subtract: 36L = 60 -> L = 60/36 = 5/3
C_lp = 40.0 / 3.0
L_lp = 5.0 / 3.0
obj_lp = 4.0 * C_lp + 6.0 * L_lp  # 190/3

# Verify constraints
vitA_lp = 5.0 * C_lp + 8.0 * L_lp
vitC_lp = 7.0 * C_lp + 4.0 * L_lp
prop_ok_lp = (3.0 * L_lp) <= (2.0 * C_lp) + 1e-9  # 3L ≤ 2C

print(f"Continuous LP solution (may be fractional): C = {C_lp:.6f}, L = {L_lp:.6f}")
print(f"  Objective (unsaturated fat) = {obj_lp:.6f}")
print(f"  Vitamin A = {vitA_lp:.6f} (require ≥ 80)")
print(f"  Vitamin C = {vitC_lp:.6f} (require ≥ 100)")
print(f"  Lobster proportion constraint (3L ≤ 2C) satisfied? {prop_ok_lp}")

# Integer MILP: brute force search over reasonable bounds
# Determine reasonable search bounds:
# If L=0 then C must be at least 16 (from vitamins). If C=0 then L at least 25.
# We'll search C in [0, 200], L in [0, 200] which is small and safe.
min_obj_int = None
best_pairs = []

for C in range(0, 201):
    for L in range(0, 201):
        # Constraints:
        if 5 * C + 8 * L < 80:
            continue
        if 7 * C + 4 * L < 100:
            continue
        if 3 * L > 2 * C:
            continue
        obj = 4 * C + 6 * L
        if (min_obj_int is None) or (obj < min_obj_int):
            min_obj_int = obj
            best_pairs = [(C, L)]
        elif obj == min_obj_int:
            best_pairs.append((C, L))

if min_obj_int is None:
    print("No integer feasible solution found in search bounds.")
else:
    print()
    print("Integer MILP optimal solutions (C, L) with minimal unsaturated fat:")
    for pair in best_pairs:
        Cb, Lb = pair
        print(f"  C = {Cb}, L = {Lb}, objective = {4*Cb + 6*Lb}")
    print(f"Best integer objective = {min_obj_int}")

# The required final print line (must be the last line):
# Print the final objective value. We'll print the integer MILP optimum if available,
# otherwise fall back to the continuous LP objective.
final_value = min_obj_int if min_obj_int is not None else obj_lp
print(f"FINAL_OBJ: {final_value}")
