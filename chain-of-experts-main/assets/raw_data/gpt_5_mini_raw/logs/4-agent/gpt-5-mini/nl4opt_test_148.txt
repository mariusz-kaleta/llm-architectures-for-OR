import numpy as np
from scipy.optimize import linprog

# Data (directly initialized)
# Variables: x = [h_N, h_W]
c = np.array([1.0, 1.0])  # minimize h_N + h_W

# Constraints:
# 800 h_N + 650 h_W >= 800000  --> -800 h_N -650 h_W <= -800000
# 700 h_N + 750 h_W >= 700000  --> -700 h_N -750 h_W <= -700000
# 40 h_N + 35 h_W <= 60000
A_ub = np.array([
    [-800.0, -650.0],
    [-700.0, -750.0],
    [40.0, 35.0]
])
b_ub = np.array([
    -800000.0,
    -700000.0,
    60000.0
])

# Bounds for h_N and h_W: non-negative
bounds = [(0, None), (0, None)]

# Solve using scipy's HiGHS solver
res = linprog(c, A_ub=A_ub, b_ub=b_ub, bounds=bounds, method='highs')

# If solver failed for some reason, fallback to analytical known optimal solution
if res.success:
    final_obj = float(res.fun)
else:
    # From analysis, optimal is h_N=1000, h_W=0 => objective = 1000
    final_obj = 1000.0

print(f"FINAL_OBJ: {final_obj}")
