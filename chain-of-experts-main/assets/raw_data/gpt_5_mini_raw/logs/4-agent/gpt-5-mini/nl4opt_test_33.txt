#!/usr/bin/env python3
# Self-contained solver for the LP described by the user.
# It enumerates intersections of constraints, checks feasibility, and picks the best corner.
from math import isclose

# Problem data
# Decision variables: x (turnips acres), y (pumpkins acres)
# Constraints (inequalities):
#  x >= 0
#  y >= 0
#  x + y <= 500
#  50x + 90y <= 40000
#  80x + 50y <= 34000
# Objective: maximize Z = 300x + 450y

# Represent equality lines for intersection calculations as tuples (a, b, c, name) representing a*x + b*y = c
lines = [
    (1.0, 0.0, 0.0, "x=0"),                # x = 0
    (0.0, 1.0, 0.0, "y=0"),                # y = 0
    (1.0, 1.0, 500.0, "land"),             # x + y = 500
    (50.0, 90.0, 40000.0, "watering"),     # 50x + 90y = 40000
    (80.0, 50.0, 34000.0, "pesticide"),    # 80x + 50y = 34000
]

# Inequality constraint check functions: must be <= for these forms
def feasible_point(x, y, tol=1e-8):
    if x < -tol or y < -tol:
        return False
    if x + y - 500.0 > tol:
        return False
    if 50.0 * x + 90.0 * y - 40000.0 > tol:
        return False
    if 80.0 * x + 50.0 * y - 34000.0 > tol:
        return False
    return True

# Solve 2x2 linear system:
# a1*x + b1*y = c1
# a2*x + b2*y = c2
def solve_2x2(a1, b1, c1, a2, b2, c2, tol=1e-12):
    det = a1 * b2 - a2 * b1
    if abs(det) <= tol:
        return None
    x = (c1 * b2 - c2 * b1) / det
    y = (a1 * c2 - a2 * c1) / det
    return (x, y)

# Collect candidate corner points (intersections of every pair of lines)
candidates = []
n = len(lines)
for i in range(n):
    a1, b1, c1, name1 = lines[i]
    for j in range(i + 1, n):
        a2, b2, c2, name2 = lines[j]
        sol = solve_2x2(a1, b1, c1, a2, b2, c2)
        if sol is None:
            continue
        x, y = sol
        # Round tiny numerical noise
        if abs(x) < 1e-12:
            x = 0.0
        if abs(y) < 1e-12:
            y = 0.0
        if feasible_point(x, y):
            # Use rounded representation to avoid near-duplicates
            xr = round(x, 10)
            yr = round(y, 10)
            if (xr, yr) not in candidates:
                candidates.append((xr, yr))

# Also consider the obvious axis intercepts from single inequalities (in case any were missed)
# Watering intercepts: set x=0 -> y = 40000/90; set y=0 -> x = 40000/50
candidates.append((0.0, round(40000.0 / 90.0, 10)))
candidates.append((round(40000.0 / 50.0, 10), 0.0))
# Pesticide intercepts
candidates.append((0.0, round(34000.0 / 50.0, 10)))
candidates.append((round(34000.0 / 80.0, 10), 0.0))
# Land intercepts with axes
candidates.append((0.0, 500.0))
candidates.append((500.0, 0.0))

# Filter feasible and unique again
unique_candidates = []
for x, y in candidates:
    if feasible_point(x, y):
        pt = (round(x, 10), round(y, 10))
        if pt not in unique_candidates:
            unique_candidates.append(pt)

# Evaluate objective at each candidate
def objective(x, y):
    return 300.0 * x + 450.0 * y

best = None
best_val = None
for x, y in unique_candidates:
    val = objective(x, y)
    if (best_val is None) or (val > best_val + 1e-9):
        best_val = val
        best = (x, y)

# Prepare final objective for printing (as int if essentially integer)
if best_val is None:
    final_value = 0.0
else:
    if isclose(best_val, round(best_val), rel_tol=0, abs_tol=1e-6):
        final_value = int(round(best_val))
    else:
        final_value = float(best_val)

print(f"FINAL_OBJ: {final_value}")
