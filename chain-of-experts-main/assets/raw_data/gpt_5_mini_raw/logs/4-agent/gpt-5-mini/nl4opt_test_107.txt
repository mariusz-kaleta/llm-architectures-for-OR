import numpy as np

# Data
c = np.array([300.0, 200.0])  # objective coefficients [d_o, d_n]
A = np.array([[2.0, 4.0],
              [2.0, 1.0],
              [4.0, 2.0]])
b = np.array([10.0, 9.0, 15.0])

tol = 1e-9

candidates = []

# 1) Intersections of pairs of constraint lines (A_i x = b_i)
m = A.shape[0]
for i in range(m):
    for j in range(i+1, m):
        M = np.vstack([A[i], A[j]])
        try:
            x = np.linalg.solve(M, np.array([b[i], b[j]]))
        except np.linalg.LinAlgError:
            continue
        if np.all(x >= -tol):
            if np.all(A.dot(x) >= b - tol):
                candidates.append(x.clip(min=0.0))

# 2) Intersections with axes: set d_o = 0, solve for d_n from each constraint; and d_n = 0, solve for d_o
# d_o = 0
for i in range(m):
    a_dn = A[i,1]
    if abs(a_dn) > tol:
        dn = b[i] / a_dn
        x = np.array([0.0, dn])
        if x[1] >= -tol and np.all(A.dot(x) >= b - tol):
            candidates.append(x.clip(min=0.0))
# d_n = 0
for i in range(m):
    a_do = A[i,0]
    if abs(a_do) > tol:
        do = b[i] / a_do
        x = np.array([do, 0.0])
        if x[0] >= -tol and np.all(A.dot(x) >= b - tol):
            candidates.append(x.clip(min=0.0))

# If no candidates found (should not happen), try a simple grid search as fallback
if not candidates:
    best_obj = float('inf')
    best_x = None
    # grid up to 20 days
    for do in np.linspace(0, 20, 201):
        for dn in np.linspace(0, 20, 201):
            x = np.array([do, dn])
            if np.all(A.dot(x) >= b - tol):
                obj = c.dot(x)
                if obj < best_obj:
                    best_obj = obj
                    best_x = x
    if best_x is None:
        raise RuntimeError("No feasible solution found.")
    final_obj = best_obj
else:
