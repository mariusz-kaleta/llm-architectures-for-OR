from pulp import LpProblem, LpMinimize, LpVariable, lpSum, LpInteger, PULP_CBC_CMD

prob = LpProblem("machines", LpMinimize)
A = LpVariable("A", lowBound=5, upBound=30, cat=LpInteger)
B = LpVariable("B", lowBound=0, upBound=25, cat=LpInteger)

prob += A + B                           # objective
prob += 30*A + 50*B >= 1000             # production
prob += 100*A + 120*B <= 3000           # electricity
prob += 7*B <= 3*A                      # 30% B linearised

prob.solve(PULP_CBC_CMD(msg=False))
print("A =", int(A.value()), "B =", int(B.value()),

import math

# Brute-force enumeration to solve the MILP described
# Decision variables:
#   A integer in [5,30]
#   B integer in [0,25]
# Constraints:
#   30*A + 50*B >= 1000       (production)
#   100*A + 120*B <= 3000     (electricity)
#   7*B <= 3*A                (B at most 30% of machines linearised)
#
# Objective: minimize Z = A + B
# Tie-breaker: minimize energy E = 100*A + 120*B

best = None  # will store tuple (Z, E, A, B)

for A in range(5, 31):     # 5..30
    for B in range(0, 26): # 0..25
        prod = 30*A + 50*B
        energy = 100*A + 120*B
        if prod < 1000:
            continue
        if energy > 3000:
            continue
        if 7*B > 3*A:
            continue
        Z = A + B
        E = energy
        candidate = (Z, E, A, B)
        if best is None or candidate < best:
            best = candidate

if best is None:
    raise RuntimeError("No feasible solution found.")

best_Z, best_E, best_A, best_B = best
best_prod = 30*best_A + 50*best_B
best_energy = best_E
b_prop = best_B / (best_A + best_B) if (best_A + best_B) > 0 else 0.0

# Print results (other prints allowed before the final required line)
print(f"A = {best_A}")
print(f"B = {best_B}")
print(f"Total machines = {best_Z}")
print(f"Production = {best_prod}")
print(f"Electricity = {best_energy}")
print(f"B proportion = {b_prop:.5f} ({best_B}/{best_A+best_B})")

# Final required output line (must be exactly this format)
print(f"FINAL_OBJ: {best_Z}")
