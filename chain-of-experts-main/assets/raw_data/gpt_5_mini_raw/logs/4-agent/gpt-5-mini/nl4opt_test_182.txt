import math

# Problem data
capacity = 500
ship_cap = 40
plane_cap = 20
ship_fuel = 500
plane_fuel = 300
p_max = 10

best = None  # will store tuple (fuel, s, p)

for p in range(0, p_max + 1):
    remaining = capacity - plane_cap * p
    s_required = 0 if remaining <= 0 else math.ceil(remaining / ship_cap)
    s = max(s_required, p)  # enforce s >= p (50% share by ship)
    # Optional safe upper bound for s
    if s > 13:
        s = min(s, 13)
    if ship_cap * s + plane_cap * p >= capacity:
        fuel = ship_fuel * s + plane_fuel * p
        if best is None or fuel < best[0]:
            best = (fuel, s, p)

if best is None:
    final_obj = float('inf')
else:
    final_obj = best[0]

print(f"FINAL_OBJ: {int(final_obj)}")
