# Self-contained solution for the ILP described:
# Maximize E + R
# s.t. 3E + 2R <= 15000, E >= 2R, R >= 50, E,R integers >= 0

# Initialize data
t_e = 3            # minutes per electronic reading
t_r = 2            # minutes per regular reading
T = 15000          # total minutes available
k = 2              # ratio requirement E >= k * R
R_min = 50         # minimum regular uses

# Analytical solution:
# For fixed R, set E = k * R (minimum allowed). Time constraint:
# k*t_e*R + t_r*R <= T  => R <= T / (k*t_e + t_r)
R_max = T // (k * t_e + t_r)

if R_max < R_min:
    # Infeasible: cannot satisfy minimum regular uses with available time
    R_opt = None
    E_opt = None
    total_patients = 0
else:
    R_opt = R_max
    E_opt = k * R_opt
    total_patients = E_opt + R_opt

# Final required print
print(f"FINAL_OBJ: {total_patients}")
