import math

# Problem data
total_students = 500
cap_small = 20
cap_large = 50

# Brute-force search over possible numbers of large buses
min_total = None
best_S = None
best_L = None

# Upper bound for L: reasonably total_students (very loose) or ceil(total_students/cap_large)
max_L = math.ceil(total_students / cap_large) + 50  # extra margin

for L in range(0, max_L + 1):
    # Number of small buses required to meet capacity (if any)
    remaining = max(0, total_students - cap_large * L)
    S_capacity = math.ceil(remaining / cap_small)
    # Parking constraint: S >= 4L
    S_proportion = 4 * L
    S = max(S_capacity, S_proportion)

    # Ensure integrality and nonnegativity (S is integer by construction)
    if S < 0:
        continue

    # Check feasibility (should hold by construction)
    if cap_small * S + cap_large * L >= total_students and S >= 4 * L:
        total = S + L
        if (min_total is None) or (total < min_total) or (total == min_total and L < best_L):
            min_total = total
            best_S = S
            best_L = L

# Sanity check (expected solution: S=16, L=4 -> total 20)
assert min_total is not None, "No feasible solution found."

# Final required print (exact format)
print(f"FINAL_OBJ: {min_total}")
