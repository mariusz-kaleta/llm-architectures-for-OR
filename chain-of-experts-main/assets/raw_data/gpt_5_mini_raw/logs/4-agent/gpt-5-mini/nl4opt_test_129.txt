# Self-contained script to solve the ILP by brute-force search
# Decision variables:
# x: number of 4-wheelers (integer >=0)
# y: number of 3-wheelers (integer >=0)
# Parameters initialized directly:
capacity_4 = 60   # luggage/day per 4-wheeler
capacity_3 = 40   # luggage/day per 3-wheeler
pollut_4 = 30     # pollutant/day per 4-wheeler
pollut_3 = 15     # pollutant/day per 3-wheeler
demand = 1000     # luggage/day required
pollut_limit = 430  # max pollutant/day

# Safe upper bounds for search (more than enough for this problem)
max_x = 100
max_y = 100

best_obj = None
best_solutions = []

for x in range(max_x + 1):
    for y in range(max_y + 1):
        luggage = capacity_4 * x + capacity_3 * y
        pollution = pollut_4 * x + pollut_3 * y
        if luggage >= demand and pollution <= pollut_limit:
            obj = x + y
            if best_obj is None or obj < best_obj:
                best_obj = obj
                best_solutions = [(x, y, luggage, pollution)]
            elif obj == best_obj:
                best_solutions.append((x, y, luggage, pollution))

# If multiple optima, pick the one with minimal pollution as tie-break, then minimal x
if best_obj is None:
    # No feasible solution found
    print("No feasible solution within search bounds.")
    value = None
else:
    # choose tie-breaker
    best_solutions.sort(key=lambda t: (t[3], t[0]))  # sort by pollution, then x
    x_opt, y_opt, luggage_opt, pollution_opt = best_solutions[0]
    value = best_obj
    # Print results (informational)
    print(f"Optimal solution(s) count: {len(best_solutions)}")
    print(f"Chosen solution: x (4-wheelers) = {x_opt}, y (3-wheelers) = {y_opt}")
    print(f"Total vehicles (objective) = {value}")
    print(f"Total luggage capacity = {luggage_opt}")
    print(f"Total pollution = {pollution_opt}")

# Final required output line
print(f"FINAL_OBJ: {value}")
