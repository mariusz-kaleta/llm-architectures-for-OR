import sys

# Brute-force enumeration as recommended in the problem statement.

best_L = None
best_P = None
best_O = None

# P ranges from 20 to floor(1000/3) = 333
for P in range(20, 1000 // 3 + 1):
    # Owl fraction constraint: 3O <= 2P  => O <= floor(2P/3)
    O_max_fraction = (2 * P) // 3
    # Treats budget: 3P + 5O <= 1000 => O <= floor((1000 - 3P)/5)
    remaining_treats = 1000 - 3 * P
    if remaining_treats < 0:
        continue
    O_max_budget = remaining_treats // 5
    O_max = min(O_max_fraction, O_max_budget)
    if O_max < 0:
        continue
    # Since objective increases with O, choose O = O_max for this P
    O = O_max
    L = 2 * P + 5 * O
    if best_L is None or L > best_L:
        best_L = L
        best_P = P
        best_O = O

# As a precaution, ensure we found a feasible solution
if best_L is None:
    # Should not happen given problem feasibility check, but handle anyway
    print("No feasible solution found.", file=sys.stderr)
    best_L = 0

# Final required print (must be the very last printed line in this exact format)
print(f"FINAL_OBJ: {best_L}")
