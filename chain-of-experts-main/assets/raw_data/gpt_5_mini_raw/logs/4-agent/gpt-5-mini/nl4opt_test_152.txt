#!/usr/bin/env python3
# Self-contained script to (a) check infeasibility of the original LP and
# (b) find minimal relaxations to the E (upper) and D (lower) constraints
#     and, under that minimal relaxation, minimize Vitamin K = 4x + 12y.

import math
from pulp import LpProblem, LpVariable, LpMinimize, LpStatus, value, PULP_CBC_CMD

# Problem data (from the user's model)
# Decision vars: x >=0 (cups of A), y >= 0 (cups of B)
# Objective: minimize K = 4x + 12y
# Constraints:
#  (A) 8x + 15y >= 150
#  (D) 6x + 2y >= 300
#  (E) 10x + 20y <= 400

# Solver configuration
solver = PULP_CBC_CMD(msg=False)

# 1) Try to solve original LP (should be infeasible)
prob_orig = LpProblem("original_LP", LpMinimize)
x = LpVariable("x", lowBound=0)
y = LpVariable("y", lowBound=0)

# Objective: minimize Vitamin K
prob_orig += 4 * x + 12 * y

# Constraints as given
prob_orig += 8 * x + 15 * y >= 150, "VitA"
prob_orig += 6 * x + 2 * y >= 300, "VitD"
prob_orig += 10 * x + 20 * y <= 400, "VitE"

status_orig = prob_orig.solve(solver)
status_str_orig = LpStatus[prob_orig.status]

print("Original LP status:", status_str_orig)
if status_str_orig == "Optimal":
    # If unexpectedly feasible, print solution and K
    x_val = value(x)
    y_val = value(y)
    K_val = 4 * x_val + 12 * y_val
    print(f"Original feasible solution: x={x_val}, y={y_val}, K={K_val}")
else:

# 2) Minimal-relaxation LP:
# Introduce sE >= 0 to relax E upward: 10x + 20y <= 400 + sE
# Introduce sD >= 0 to relax D downward: 6x + 2y >= 300 - sD

# minimize Vitamin K. We'll do a two-stage approach.
