import numpy as np
from scipy.optimize import linprog

# Problem data
# Decision variables: x = [R, K]
c = np.array([3.0, 2.0])  # objective: minimize 3R + 2K

# Constraints (converted to form A_ub x <= b_ub)
# 300R + 200K >= 2200  -->  -300R -200K <= -2200
# 4.5R + 4K >= 30      -->  -4.5R -4K <= -30
A_ub = np.array([[-300.0, -200.0],
                 [-4.5,   -4.0]])
b_ub = np.array([-2200.0, -30.0])

# Bounds: R >= 0, K >= 0
bounds = [(0, None), (0, None)]

# Solve continuous LP
res = linprog(c, A_ub=A_ub, b_ub=b_ub, bounds=bounds, method='highs')

if not res.success:
    # If solver fails for some reason, raise an error
    raise RuntimeError(f"LP solver failed: {res.message}")

# Objective value
obj_val = float(res.fun)

# Final required output line (only this text on the line)
print(f"FINAL_OBJ: {obj_val}")
