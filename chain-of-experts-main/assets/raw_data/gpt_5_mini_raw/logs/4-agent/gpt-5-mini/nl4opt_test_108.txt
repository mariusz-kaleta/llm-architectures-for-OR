import itertools

# Problem data / bounds
M_max = 11  # from grease-only bound floor(35/3)
A_max = 5   # from grease-only bound floor(35/6)
required_slices = 50
grease_limit_original = 35
grease_limit_relaxed = 36

def feasible(M, A, grease_limit):
    if M < 0 or A < 0:
        return False
    # integrality ensured by enumeration
    if 5*M + 8*A < required_slices:
        return False
    if 3*M + 6*A > grease_limit:
        return False
    if M > A - 1:
        return False
    return True

# Enumerate all integer candidates within bounds
candidates = [(M, A) for M, A in itertools.product(range(M_max + 1), range(A_max + 1))]

# Check original problem (grease limit = 35)
feasible_original = [(M, A) for (M, A) in candidates if feasible(M, A, grease_limit_original)]

if not feasible_original:
    print("Original MILP (grease limit = 35) is infeasible: no (M,A) in the bounded domain satisfies all constraints.")
else:
    print("Feasible solutions for original MILP (grease limit = 35):")
    for M, A in feasible_original:
        print(f"  M={M}, A={A}, slices={5*M+8*A}, grease={3*M+6*A}, total={M+A}")

# Check relaxed problem (grease limit = 36)
feasible_relaxed = [(M, A) for (M, A) in candidates if feasible(M, A, grease_limit_relaxed)]

if not feasible_relaxed:
    print("Relaxed MILP (grease limit = 36) is infeasible (unexpected).")
    value = None
else:
    # find minimal objective (M + A)
    obj_values = [(M + A, M, A) for (M, A) in feasible_relaxed]
    obj_values.sort()
    min_obj, min_M, min_A = obj_values[0]
    print("Relaxed MILP (grease limit = 36) feasible solutions (showing all):")
    for M, A in feasible_relaxed:
        print(f"  M={M}, A={A}, slices={5*M+8*A}, grease={3*M+6*A}, total={M+A}")
    print(f"Optimal for relaxed problem: M={min_M}, A={min_A}, total slicers={min_obj}")
    value = min_obj

# Final required single-line output
print(f"FINAL_OBJ: {value}")
