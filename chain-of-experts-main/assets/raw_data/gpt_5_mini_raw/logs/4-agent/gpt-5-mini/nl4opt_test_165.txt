```python
import pulp

# Define the LP problem: maximize profit
prob = pulp.LpProblem("Taco_Profit_Maximization", pulp.LpMaximize)

# Decision variables
x1 = pulp.LpVariable("x1_regular", lowBound=0)  # regular tacos
x2 = pulp.LpVariable("x2_deluxe", lowBound=0)   # deluxe tacos

# Objective: maximize 2.50*x1 + 3.55*x2
prob += 2.50 * x1 + 3.55 * x2, "Total_Profit"

# Constraints
prob += x1 <= 50, "Demand_limit_regular"
prob += x2 <= 40, "Demand_limit_deluxe"
prob += x1 + x2 <= 70, "Capacity_limit"

# Solve quietly
solver = pulp.PULP_CBC_CMD(msg=False)
prob.solve(solver)

# Get objective value
value = pulp.value(prob.objective)

# Print final objective value in required format
print(f"FINAL_OBJ: {value:.2f}")
```
