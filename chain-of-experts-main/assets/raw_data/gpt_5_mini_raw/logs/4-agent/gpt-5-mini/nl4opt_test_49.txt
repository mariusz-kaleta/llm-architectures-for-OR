#!/usr/bin/env python3
# Self-contained script to check feasibility of the ILP and evaluate relaxations/options.
# Problem:
#   decision vars K (kids, 100 ml each), A (adults, 300 ml each)
#   Maximize K + A
#   s.t. 100 K + 300 A <= supply
#        A >= r * K
#        K >= K_min
#        K,A integers >= 0
#
# This script evaluates:
#   - Original model (supply=25000, r=3, K_min=50) -> infeasible
#   - Option A: drop K_min (supply=25000, r=3, K_min=0)
#   - Option B: keep K_min=50 but relax ratio to r=1 (supply=25000, r=1, K_min=50)
#   - Option C: increase supply to 50000 (supply=50000, r=3, K_min=50)
#
# It brute-forces integer feasible solutions (small search space), reports best solution per scenario,
# and finally prints the best objective found among these scenarios in the required format.

import math

def solve_bounded_ilp(supply_ml, ratio_r, K_min):
    """
    Brute-force solve:
      max K + A
      s.t. 100 K + 300 A <= supply_ml
           A >= ratio_r * K
           K >= K_min
           K,A integers >= 0
    ratio_r may be non-integer; we interpret A >= ceil(ratio_r * K) for integer feasibility.
    Returns (best_obj, best_K, best_A, syrup_used) or (None, None, None, None) if infeasible.
    """
    best_obj = None
    best_K = None
    best_A = None
    best_syrup = None

    # Upper bounds for loops:
    K_max = supply_ml // 100  # if A=0
    A_max = supply_ml // 300  # if K=0

    # Ensure K_min <= K_max else infeasible quickly
    if K_min > K_max:
        return (None, None, None, None)

    for K in range(max(0, K_min), K_max + 1):
        # minimal A to satisfy ratio
        A_min = math.ceil(ratio_r * K)
        if A_min < 0:
            A_min = 0
        # syrup used by K and minimal A; if that already exceeds supply, skip
        syrup_min = 100 * K + 300 * A_min
        if syrup_min > supply_ml:
            continue
        # maximum A allowed by supply
        A_upper = min(A_max, (supply_ml - 100 * K) // 300)
        if A_upper < A_min:
            continue
        # For maximizing K+A, under fixed K the best is A = A_min (because each extra A uses 300 ml for +1 bottle,
        # while we could perhaps increase K instead; but we'll still iterate to be safe).
        # We'll iterate A only from A_min to A_upper to be exhaustive (space is small).
        for A in range(A_min, A_upper + 1):
            syrup = 100 * K + 300 * A
            if syrup > supply_ml:
                break
            obj = K + A
            if (best_obj is None) or (obj > best_obj) or (obj == best_obj and syrup < best_syrup):
                best_obj = obj
                best_K = K
                best_A = A
                best_syrup = syrup

    if best_obj is None:
        return (None, None, None, None)
    return (best_obj, best_K, best_A, best_syrup)


def report_scenario(name, supply, ratio, K_min):
    print(f"Scenario: {name}")
    print(f"  supply_ml = {supply}, ratio A >= {ratio} * K, K_min = {K_min}")
    best_obj, best_K, best_A, best_syrup = solve_bounded_ilp(supply, ratio, K_min)
    if best_obj is None:
        print("  Feasible: NO (no integer solution satisfies the constraints)")
    else:
        print("  Feasible: YES")
        print(f"  Optimal integer solution: K = {best_K}, A = {best_A}")
        print(f"  Total bottles (objective) = {best_obj}")
        print(f"  Syrup used = {best_syrup} ml (supply {supply} ml)")
    print("")
    return (best_obj, best_K, best_A, best_syrup)


def main():
    scenarios = []

    # Original model
    scenarios.append(("Original (infeasible check)", 25000, 3.0, 50))

    # Option A: drop the minimum kids constraint (keep ratio r=3)
    scenarios.append(("Option A (drop K>=50, keep A>=3K)", 25000, 3.0, 0))

    # Option B: keep K>=50 but relax ratio to A>=1*K
    scenarios.append(("Option B (keep K>=50, relax ratio to A>=K)", 25000, 1.0, 50))

    # Option C: increase syrup supply to 50000 to satisfy original constraints
    scenarios.append(("Option C (increase supply to 50000, keep original constraints)", 50000, 3.0, 50))

    results = []
    for name, supply, ratio, K_min in scenarios:
        res = report_scenario(name, supply, ratio, K_min)
        results.append((name, supply, ratio, K_min, res))

    # Summarize best objective among the evaluated (feasible) scenarios
    best_overall = None
    best_scenario = None
    for name, supply, ratio, K_min, (best_obj, best_K, best_A, best_syrup) in results:
        if best_obj is None:
            continue
        if (best_overall is None) or (best_obj > best_overall):
            best_overall = best_obj
            best_scenario = (name, supply, ratio, K_min, best_K, best_A, best_syrup)

    if best_overall is None:
        print("No feasible solution found in any scenario.")
        final_value = 0
    else:
        name, supply, ratio, K_min, best_K, best_A, best_syrup = best_scenario
        print("Best scenario found:")
        print(f"  {name}")
        print(f"  supply_ml = {supply}, ratio = {ratio}, K_min = {K_min}")
        print(f"  Solution: K = {best_K}, A = {best_A}, total bottles = {best_overall}, syrup_used = {best_syrup} ml")
        final_value = int(best_overall)

    # CRITICAL final line: print only this format and nothing else on the line
    print(f"FINAL_OBJ: {final_value}")


if __name__ == "__main__":
    main()
