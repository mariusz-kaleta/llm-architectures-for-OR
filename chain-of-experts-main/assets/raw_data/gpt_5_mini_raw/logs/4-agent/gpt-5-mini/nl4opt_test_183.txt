[CODER]:
import math

# Data
cap_large = 2000
cap_small = 800
pollution_large = 20
pollution_small = 15
demand = 20000
max_L = 7

bestZ = None
best_solution = None

for L in range(0, max_L + 1):
    # capacity requirement for S
    remaining = demand - cap_large * L
    if remaining <= 0:
        s_cap = 0
    else:
        s_cap = math.ceil(remaining / cap_small)
    # proportion requirement: S >= (2/3) L -> S >= ceil(2L/3)
    s_prop = math.ceil((2 * L) / 3)
    S = max(s_cap, s_prop)
    # compute objective
    Z = pollution_large * L + pollution_small * S
    # record best
    if bestZ is None or Z < bestZ:
        bestZ = Z
        best_solution = (L, S, Z)

# Final objective value
value = bestZ
print(f"FINAL_OBJ: {value}")
