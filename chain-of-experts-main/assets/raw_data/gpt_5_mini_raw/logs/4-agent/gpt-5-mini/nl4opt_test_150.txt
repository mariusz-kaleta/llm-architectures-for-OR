import math

# Parameters
rH1 = 20.0  # heart pills per hour at lab1
rH2 = 30.0  # heart pills per hour at lab2
rL1 = 30.0  # lung pills per hour at lab1
rL2 = 40.0  # lung pills per hour at lab2

s1 = 3.0    # hours per lab1 session
s2 = 5.0    # hours per lab2 session

Wmax = 1500.0
Hreq = 20000.0
Lreq = 30000.0

# Continuous-hours (LP) analytic solution
# Since lab2 dominates lab1 on both products, set h1 = 0 and compute h2 needed
h1_lp = 0.0
h2_needed_hearts = Hreq / rH2
h2_needed_lungs = Lreq / rL2
h2_lp = max(h2_needed_hearts, h2_needed_lungs)
total_hours_lp = h1_lp + h2_lp

# Check feasibility with worker-hours cap
feasible_lp = total_hours_lp <= Wmax

# Determine which constraint is binding at LP solution
hearts_produced_lp = rH1 * h1_lp + rH2 * h2_lp
lungs_produced_lp = rL1 * h1_lp + rL2 * h2_lp
heart_binding = abs(hearts_produced_lp - Hreq) < 1e-8
lung_binding = abs(lungs_produced_lp - Lreq) < 1e-8

# Integer-session (ILP) analytic solution
# Use only lab2 sessions (lab2 dominates)
# Each lab2 session produces:
hearts_per_x2 = rH2 * s2  # 30 * 5 = 150
lungs_per_x2 = rL2 * s2   # 40 * 5 = 200

x1_ilp = 0
x2_needed_hearts = math.ceil(Hreq / hearts_per_x2)
x2_needed_lungs = math.ceil(Lreq / lungs_per_x2)
x2_ilp = max(x2_needed_hearts, x2_needed_lungs)

# Respect worker-hours cap: check feasibility
total_hours_ilp = s1 * x1_ilp + s2 * x2_ilp
feasible_ilp = total_hours_ilp <= Wmax

# Compute productions for ILP solution
hearts_produced_ilp = hearts_per_x2 * x2_ilp
lungs_produced_ilp = lungs_per_x2 * x2_ilp

# Print results (informational)
print("CONTINUOUS LP SOLUTION")
print(f"h1 = {h1_lp:.6f} hours")
print(f"h2 = {h2_lp:.6f} hours")
print(f"Total hours (LP) = {total_hours_lp:.6f}")
print(f"Feasible under Wmax? {feasible_lp}")
print(f"Hearts produced (LP) = {hearts_produced_lp:.6f}")
print(f"Lungs produced (LP) = {lungs_produced_lp:.6f}")
print(f"Heart constraint binding? {heart_binding}")
print(f"Lung constraint binding? {lung_binding}")

print(f"x1 = {x1_ilp} sessions (lab1)")
print(f"x2 = {x2_ilp} sessions (lab2)")
print(f"Total hours (ILP) = {total_hours_ilp}")
print(f"Feasible under Wmax? {feasible_ilp}")
