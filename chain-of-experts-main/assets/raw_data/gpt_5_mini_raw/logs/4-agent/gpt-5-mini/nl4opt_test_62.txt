#!/usr/bin/env python3
# Self-contained script to solve the handbag production LP/MIP for two interpretations.
# No external solvers required: continuous solutions found by enumerating corner intersections;
# integer solutions found by brute-force enumeration (small ranges).

from typing import Tuple, List, Dict
import math

# Problem data
profit_R = 30.0      # profit per regular handbag
profit_P = 180.0     # profit per premium handbag
cost_R = 200.0       # manufacturing cost per regular
cost_P = 447.0       # manufacturing cost per premium
budget = 250000.0    # manufacturing budget

# Common capacity constants
cap_total = 475      # interpretation A: R + P <= 475
cap_each = 475       # interpretation B: R <= 475, P <= 475

EPS = 1e-7

def feasible_A(R: float, P: float) -> bool:
    if R < -EPS or P < -EPS: 
        return False
    if (R + P) - cap_total > EPS:
        return False
    if (cost_R * R + cost_P * P) - budget > EPS:
        return False
    return True

def feasible_B(R: float, P: float) -> bool:
    if R < -EPS or P < -EPS:
        return False
    if R - cap_each > EPS or P - cap_each > EPS:
        return False
    if (cost_R * R + cost_P * P) - budget > EPS:
        return False
    return True

def objective(R: float, P: float) -> float:
    return profit_R * R + profit_P * P

def solve_continuous_interpretation_A() -> Dict:
    # Constraints for active sets: R=0, P=0, R+P=cap_total, cost_R*R + cost_P*P = budget
    solutions = []
    # List of pairs of active constraints as functions returning (R,P) or None if no unique intersection
    # We'll handle each pair explicitly
    # Pair: R=0 and P=0
    solutions.append((0.0, 0.0))
    # R=0 and R+P=cap_total => P = cap_total
    solutions.append((0.0, float(cap_total)))
    # P=0 and R+P=cap_total => R = cap_total
    solutions.append((float(cap_total), 0.0))
    # R=0 and budget => P = budget / cost_P
    solutions.append((0.0, budget / cost_P))
    # P=0 and budget => R = budget / cost_R
    solutions.append((budget / cost_R, 0.0))
    # R+P=cap_total and budget -> solve 200R + 447P = budget and R+P = cap_total
    # Solve linear system:
    # R + P = cap_total
    # cost_R*R + cost_P*P = budget
    # => R = cap_total - P
    # cost_R*(cap_total - P) + cost_P*P = budget
    # => cost_R*cap_total + (cost_P - cost_R)*P = budget
    denom = (cost_P - cost_R)
    if abs(denom) > EPS:
        P_sol = (budget - cost_R * cap_total) / denom
        R_sol = cap_total - P_sol
        solutions.append((R_sol, P_sol))
    # Evaluate feasible corners and pick best
    best = None
    for (R, P) in solutions:
        if feasible_A(R, P):
            val = objective(R, P)
            if (best is None) or (val > best['obj'] + EPS):
                best = {'R': R, 'P': P, 'obj': val}
    # Also ensure we didn't miss intersection of budget with axes already included
    if best is None:
        # fallback to zero solution
        best = {'R': 0.0, 'P': 0.0, 'obj': 0.0}
    return best

def solve_continuous_interpretation_B() -> Dict:
    # Constraints: R=0, P=0, R=cap_each, P=cap_each, budget line
    solutions = []
    solutions.append((0.0, 0.0))
    solutions.append((0.0, float(cap_each)))
    solutions.append((float(cap_each), 0.0))
    # budget intersections with R=0 and P=0
    solutions.append((0.0, budget / cost_P))
    solutions.append((budget / cost_R, 0.0))
    # budget intersection with R=cap_each => solve for P
    # cost_R*cap_each + cost_P*P = budget => P = (budget - cost_R*cap_each)/cost_P
    P_from_Rcap = (budget - cost_R * cap_each) / cost_P
    solutions.append((float(cap_each), P_from_Rcap))
    # budget intersection with P=cap_each => solve for R
    R_from_Pcap = (budget - cost_P * cap_each) / cost_R
    solutions.append((R_from_Pcap, float(cap_each)))
    # Evaluate feasible corners and pick best
    best = None
    for (R, P) in solutions:
        if feasible_B(R, P):
            val = objective(R, P)
            if (best is None) or (val > best['obj'] + EPS):
                best = {'R': R, 'P': P, 'obj': val}
    if best is None:
        best = {'R': 0.0, 'P': 0.0, 'obj': 0.0}
    return best

def solve_integer_interpretation_A() -> Dict:
    # R,P are integers, constraints: R+P <= cap_total and cost_R*R + cost_P*P <= budget
    best = {'R': 0, 'P': 0, 'obj': 0.0}
    # R and P each naturally bounded between 0 and cap_total
    for P in range(0, cap_total + 1):
        # For given P, maximum R is min(cap_total - P, floor((budget - cost_P*P)/cost_R))
        remaining_budget = budget - cost_P * P
        if remaining_budget < 0:
            continue
        max_R_by_budget = int(math.floor(remaining_budget / cost_R + 1e-12))
        max_R_by_capacity = cap_total - P
        R = min(max_R_by_budget, max_R_by_capacity)
        if R < 0:
            continue
        val = objective(R, P)
        if val > best['obj'] + EPS:
            best = {'R': R, 'P': P, 'obj': val}
    return best

def solve_integer_interpretation_B() -> Dict:
    # R,P integers, R <= cap_each, P <= cap_each, budget constraint
    best = {'R': 0, 'P': 0, 'obj': 0.0}
    # iterate over P (0..cap_each) and compute best R
    for P in range(0, cap_each + 1):
        remaining_budget = budget - cost_P * P
        if remaining_budget < 0:
            continue
        max_R_by_budget = int(math.floor(remaining_budget / cost_R + 1e-12))
        if max_R_by_budget > cap_each:
            max_R_by_budget = cap_each
        if max_R_by_budget < 0:
            continue
        R = max_R_by_budget
        val = objective(R, P)
        if val > best['obj'] + EPS:
            best = {'R': R, 'P': P, 'obj': val}
    return best

def pretty_print_solution(label: str, sol: Dict, continuous: bool):
    R = sol['R']
    P = sol['P']
    obj = sol['obj']
    if continuous:
        print(f"{label} (continuous): R = {R:.6f}, P = {P:.6f}, Obj = {obj:.6f}")
    else:
        print(f"{label} (integer): R = {int(R)}, P = {int(P)}, Obj = {obj:.6f}")
