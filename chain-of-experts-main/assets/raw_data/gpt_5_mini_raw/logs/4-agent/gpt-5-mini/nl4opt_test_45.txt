#!/usr/bin/env python3
import math

# Coefficients and RHS for constraints
# Variables: x (Oil Max), y (Oil Max Pro)
profit_x = 10
profit_y = 15

# Constraints:
# 46 x + 13 y <= 1345   (A)
# 43 x + 4  y <= 346    (B)
# 56 x + 45 y <= 1643   (C)
A = (46.0, 13.0, 1345.0)
B = (43.0, 4.0, 346.0)
C = (56.0, 45.0, 1643.0)

constraints = {
    'A': A,
    'B': B,
    'C': C,
    'X0': (1.0, 0.0, 0.0),  # x = 0
    'Y0': (0.0, 1.0, 0.0),  # y = 0
}

def intersect(line1, line2):
    # Each line is (a, b, r) representing a*x + b*y = r
    a1, b1, r1 = line1
    a2, b2, r2 = line2
    det = a1 * b2 - a2 * b1
    tol = 1e-9
    if abs(det) < tol:
        return None  # parallel or coincident
    x = (r1 * b2 - r2 * b1) / det
    y = (a1 * r2 - a2 * r1) / det
    return (x, y)

# Generate all intersection points of pairs of boundary lines
points = set()
keys = list(constraints.keys())
for i in range(len(keys)):
    for j in range(i+1, len(keys)):
        k1 = keys[i]; k2 = keys[j]
        p = intersect(constraints[k1], constraints[k2])
        if p is not None:
            # round a bit to avoid floating point duplicates
            x, y = p
            xr = round(x, 10)
            yr = round(y, 10)
            points.add((xr, yr))

# Also include the origin explicitly
points.add((0.0, 0.0))

# Feasibility check and evaluation
def feasible(x, y, tol=1e-8):
    if x < -tol or y < -tol:
        return False
    if 46.0*x + 13.0*y > 1345.0 + tol:
        return False
    if 43.0*x + 4.0*y  > 346.0  + tol:
        return False
    if 56.0*x + 45.0*y > 1643.0 + tol:
        return False
    return True

best_cont = {'x': 0.0, 'y': 0.0, 'Z': 0.0}
for (x, y) in points:
    if feasible(x, y):
        Z = profit_x * x + profit_y * y
        if Z > best_cont['Z'] + 1e-9:
            best_cont = {'x': x, 'y': y, 'Z': Z}

# As a safeguard, also check some boundary points along each constraint where it meets axes
# (these are already included via intersections, but keep for robustness)
# Now solve the integer problem by enumeration within sensible bounds
# Compute upper bounds for enumeration from each constraint
max_x = int(math.floor(min(1345.0/46.0, 346.0/43.0, 1643.0/56.0))) + 1
max_y = int(math.floor(min(1345.0/13.0, 346.0/4.0, 1643.0/45.0))) + 1

# Ensure bounds are non-negative and not too large
max_x = max(0, max_x)
max_y = max(0, max_y)

best_int = {'x': 0, 'y': 0, 'Z': 0}
for xi in range(0, max_x + 1):
    for yi in range(0, max_y + 1):
        if feasible(xi, yi):
            Z = profit_x * xi + profit_y * yi
            if Z > best_int['Z']:
                best_int = {'x': xi, 'y': yi, 'Z': Z}

# Print results (informational)
print(f"Continuous LP optimum (relax integrality): x = {best_cont['x']}, y = {best_cont['y']}, Z = {best_cont['Z']}")
print(f"Integer (ILP) optimum found by enumeration: x = {best_int['x']}, y = {best_int['y']}, Z = {best_int['Z']}")

# Final required single-line output (must be the last printed line)
print(f"FINAL_OBJ: {best_int['Z']}")
