# Self-contained solver for the ILP described in the prompt.
# Decision variables:
#   G (glass jars, integer >= 0)

# Initialize parameters
capacity_glass = 250   # ml per glass jar
capacity_plastic = 300 # ml per plastic jar
H = 20000              # total honey in ml

# Bounds and requirements
G_min = 20
G_max = H // capacity_glass  # practical upper bound

records = []  # to store feasible (G, P, total, leftover)

for G in range(G_min, G_max + 1):
    remaining_after_glass = H - capacity_glass * G
    if remaining_after_glass < 0:
        continue
    P_max = remaining_after_glass // capacity_plastic
    # Ratio constraint: P >= 2 * G
    if P_max >= 2 * G:
        P = P_max
        total = G + P
        leftover = H - capacity_glass * G - capacity_plastic * P
        records.append((G, P, total, leftover))

# If no feasible records found (shouldn't happen here), set objective to 0
if not records:
    value = 0
else:
    # Find maximum total jars
    max_total = max(r[2] for r in records)
    # Filter candidates achieving max total
    candidates = [r for r in records if r[2] == max_total]
    # Tie-breaker: minimize leftover honey, then prefer more glass jars
    best = min(candidates, key=lambda r: (r[3], -r[0]))
    # Prepare final objective value
    value = max_total

print(f"FINAL_OBJ: {value}")
