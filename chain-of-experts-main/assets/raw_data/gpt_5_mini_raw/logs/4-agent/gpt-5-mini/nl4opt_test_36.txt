# Self-contained script to solve the ILP by enumeration (using substitution S = 3L)
best_Z = None
best_L = None
best_S = None

# Problem parameters (from model)
min_S = 10
min_L = 4
# Worker availability: 2*S + 5*L <= 100

# We'll search over a reasonable range for L (0..100)
for L in range(0, 101):
    S = 3 * L  # ratio constraint S = 3L
    if L < min_L:
        continue
    if S < min_S:
        continue
    if 2 * S + 5 * L > 100:
        continue
    Z = 25 * S + 60 * L
    if best_Z is None or Z > best_Z:
        best_Z = Z
        best_L = L
        best_S = S

# best_Z should be found
if best_Z is None:
    best_Z = 0

print(f"FINAL_OBJ: {best_Z}")
