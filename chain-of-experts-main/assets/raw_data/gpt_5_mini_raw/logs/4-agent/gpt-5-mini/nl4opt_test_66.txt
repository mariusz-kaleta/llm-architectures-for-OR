```python
import math

# Problem data (directly initialized)
# Vehicle capacities (kg)
cap_tractor = 40
cap_car = 20
required_capacity = 500

# Practical bounds
T_max = 13  # 13 tractors give 520 kg
C_max = 25  # 25 cars give 500 kg

best_obj = None
best_T = None
best_C = None

for T in range(0, T_max + 1):
    # Minimum cars needed to satisfy capacity: 40*T + 20*C >= 500 -> C >= (500 - 40*T)/20
    cap_deficit = required_capacity - cap_tractor * T
    if cap_deficit <= 0:
        C_by_cap = 0
    else:
        C_by_cap = int(math.ceil(cap_deficit / cap_car))
    # Minimum cars needed by ratio constraint: C >= 2T
    C_by_ratio = 2 * T
    # Feasible minimum C for this T
    C_needed = max(C_by_cap, C_by_ratio)
    if C_needed > C_max:
        continue  # infeasible within practical bounds
    obj = T + C_needed
    # Tie-breaker: prefer smaller T if same objective (optional)
    if best_obj is None or obj < best_obj or (obj == best_obj and T < best_T):
        best_obj = obj
        best_T = T
        best_C = C_needed

# At this point best_obj, best_T, best_C hold the optimal solution found by brute force
# Print final objective value in required format
print(f"FINAL_OBJ: {best_obj}")
```
