import numpy as np
from scipy.optimize import linprog

# Data
# Per-hour outputs
A_allergy = np.array([20.0, 10.0])  # coefficients for h1, h2
A_fever   = np.array([15.0, 30.0])
A_compound = np.array([20.0, 30.0])

# RHS
b_allergy = 700.0
b_fever = 600.0
compound_available = 1000.0

# 1) Solve primary LP: minimize h1 + h2
# Variables: x = [h1, h2]
c = np.array([1.0, 1.0])

# Constraints:
# 20 h1 + 10 h2 >= 700  ->  -20 h1 -10 h2 <= -700
# 15 h1 + 30 h2 >= 600  ->  -15 h1 -30 h2 <= -600
# 20 h1 + 30 h2 <= 1000 ->   20 h1 +30 h2 <= 1000
A_ub = np.vstack([
    -A_allergy,
    -A_fever,
    A_compound
])
b_ub = np.array([-b_allergy, -b_fever, compound_available])

bounds = [(0, None), (0, None)]

res = linprog(c, A_ub=A_ub, b_ub=b_ub, bounds=bounds, method="highs")

if not res.success:
    raise RuntimeError("LP solver failed: " + res.message)

h1, h2 = res.x
total_hours = float(h1 + h2)

# Compute constraint left-hand-sides for diagnosis
allergy_val = A_allergy.dot([h1, h2])
fever_val = A_fever.dot([h1, h2])
compound_val = A_compound.dot([h1, h2])

tol = 1e-8
allergy_binding = abs(allergy_val - b_allergy) <= tol
fever_binding = abs(fever_val - b_fever) <= tol
compound_binding = abs(compound_val - compound_available) <= tol
compound_slack = compound_available - compound_val

# Sensitivity: derivative of total time w.r.t RHS b = [b1,b2]
A_mat = np.array([[20.0, 10.0],
                  [15.0, 30.0]])
invA = np.linalg.inv(A_mat)
# s = 1^T * A^{-1}
s = np.array([1.0, 1.0]).dot(invA)  # length-2 vector; s[0]=dT/db1, s[1]=dT/db2

# Compound as linear function of b: coeffs such that compound = coeffs @ b
# compound = [20 30] * h = [20 30] * A^{-1} * b  -> coeffs = [20 30] @ invA
compound_coeffs = A_compound.dot(invA)  # coefficients for [b1, b2]
# Solve thresholds where compound becomes binding if varying only one demand
# With b = [b1, fixed_b2], need compound_coeffs[0]*b1 + compound_coeffs[1]*fixed_b2 = compound_available
fixed_b2 = b_fever
if abs(compound_coeffs[0]) > 1e-12:
    b1_threshold = (compound_available - compound_coeffs[1]*fixed_b2) / compound_coeffs[0]
else:
    b1_threshold = np.nan
fixed_b1 = b_allergy
if abs(compound_coeffs[1]) > 1e-12:
    b2_threshold = (compound_available - compound_coeffs[0]*fixed_b1) / compound_coeffs[1]
else:
    b2_threshold = np.nan

# Makespan alternative: minimize T subject to same production constraints and h1 <= T, h2 <= T
# Variables x = [h1, h2, T], objective minimize T
c_m = np.array([0.0, 0.0, 1.0])
# Constraints as A_ub_m x <= b_ub_m
# -allergy >= -> -20 -10 0 <= -700
# -fever  -> -15 -30 0 <= -600
# compound -> 20 30 0 <= 1000
# h1 - T <= 0 -> 1 0 -1 <= 0
# h2 - T <= 0 -> 0 1 -1 <= 0
A_ub_m = np.array([
    [-A_allergy[0], -A_allergy[1], 0.0],
    [-A_fever[0], -A_fever[1], 0.0],
    [ A_compound[0],  A_compound[1], 0.0],
    [1.0, 0.0, -1.0],
    [0.0, 1.0, -1.0]
])
b_ub_m = np.array([-b_allergy, -b_fever, compound_available, 0.0, 0.0])
bounds_m = [(0, None), (0, None), (0, None)]

res_m = linprog(c_m, A_ub=A_ub_m, b_ub=b_ub_m, bounds=bounds_m, method="highs")
if not res_m.success:
    raise RuntimeError("Makespan LP solver failed: " + res_m.message)

h1_m, h2_m, T_opt = res_m.x

# Print diagnostic info (these prints are informational; final line below must match required format)
print(f"Primary LP solution: h1 = {h1:.12f}, h2 = {h2:.12f}, total_hours = {total_hours:.12f}")
print(f"Constraint LHS values: allergy = {allergy_val:.12f} (req {b_allergy}), fever = {fever_val:.12f} (req {b_fever}), compound = {compound_val:.12f} (avail {compound_available})")
print(f"Binding status (within tol={tol}): allergy={allergy_binding}, fever={fever_binding}, compound={compound_binding}")
print(f"Compound slack = {compound_slack:.12f}")
print(f"Sensitivity dT/db (dT/d(allergy), dT/d(fever)) = ({s[0]:.12f}, {s[1]:.12f})  (hours per extra unit)")
print(f"Compound coefficients w.r.t b = {compound_coeffs}  (compound = coeffs[0]*b1 + coeffs[1]*b2)")
print(f"Compound becomes binding if b1 (with b2={fixed_b2}) >= {b1_threshold:.12f}")
print(f"Compound becomes binding if b2 (with b1={fixed_b1}) >= {b2_threshold:.12f}")
print(f"Makespan-optimal solution: h1 = {h1_m:.12f}, h2 = {h2_m:.12f}, T = {T_opt:.12f}")
