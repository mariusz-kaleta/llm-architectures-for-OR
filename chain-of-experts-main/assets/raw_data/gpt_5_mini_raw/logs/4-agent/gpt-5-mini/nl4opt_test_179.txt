# Investment optimization problem solved analytically in-code

# Data / parameters
total_capital = 600_000         # total dollars available
cap_apartments = 200_000        # maximum dollars allowed in apartments
r_A = 0.10                      # return rate for apartments
r_T = 0.15                      # return rate for townhouses

# Reduce problem using A + T = total_capital => T = total_capital - A
# Relative-investment constraint: A >= 0.5 * T = 0.5 * (total_capital - A)
# Solve: A >= 0.5*total_capital - 0.5*A -> 1.5*A >= 0.5*total_capital -> A >= total_capital / 3
A_lower_from_relative = total_capital / 3.0

# Other bounds: 0 <= A <= min(cap_apartments, total_capital)
A_lower = max(0.0, A_lower_from_relative)
A_upper = min(cap_apartments, total_capital)

# Check feasibility
if A_lower > A_upper + 1e-9:
    raise ValueError("No feasible solution for the given constraints.")

# Objective as function of A: R(A) = r_A*A + r_T*(total_capital - A) = (r_A - r_T)*A + r_T*total_capital
# Since r_A - r_T < 0 here, objective decreases with A, so choose smallest feasible A.
A_opt = A_lower
T_opt = total_capital - A_opt
R_opt = r_A * A_opt + r_T * T_opt

# If R_opt is essentially integer, present as integer
if abs(R_opt - round(R_opt)) < 1e-8:
    R_print = int(round(R_opt))
else:
    R_print = R_opt

print(f"FINAL_OBJ: {R_print}")
