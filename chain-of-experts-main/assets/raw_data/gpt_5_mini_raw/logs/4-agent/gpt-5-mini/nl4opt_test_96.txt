#!/usr/bin/env python3
import itertools
import math

# Problem data
# Decision variables: b (burgers), p (pizza slices)
# Objective: minimize Z = 12 b + 10 p
c_b = 12.0
c_p = 10.0

# Constraints (left-hand side coefficients and RHS)
# 1) 10 b + 8 p >= 130   (fat)
# 2) 300 b + 250 p >= 3000 (calories)
# 3) p - 2 b >= 0         (ratio p >= 2b)
# plus b >= 0, p >= 0

# We'll find all intersection points formed by pairs of constraint boundaries (equalities),
# plus intersections with axes, then test feasibility and pick the minimum objective.

tol = 1e-9

def is_feasible(b, p):
    if b is None or p is None:
        return False
    if b < -tol or p < -tol:
        return False
    if 10*b + 8*p + tol < 130:   # fat
        return False
    if 300*b + 250*p + tol < 3000:  # calories
        return False
    if p + tol < 2*b:           # ratio
        return False
    return True

# Solve pairs of equalities. Represent equalities as functions returning (A, rhs)
# eq1: 10 b + 8 p = 130
# eq2: 300 b + 250 p = 3000
# eq3: -2 b + 1 p = 0   (p = 2 b)
eqs = [
    ( (10.0, 8.0), 130.0 ),
    ( (300.0, 250.0), 3000.0 ),
    ( (-2.0, 1.0), 0.0 )
]

candidates = []

# Solve all pairs of eqs (2x2 linear systems)
for (i, eqA), (j, eqB) in itertools.combinations(enumerate(eqs), 2):
    (a1, b1), r1 = eqA
    (a2, b2), r2 = eqB
    det = a1*b2 - a2*b1
    if abs(det) < tol:
        continue
    # solve for (x,y) = (b,p)
    b_sol = (r1*b2 - r2*b1) / det
    p_sol = (a1*r2 - a2*r1) / det
    candidates.append((b_sol, p_sol, f"eq{i+1}=,eq{j+1}="))

# Intersections with axes (b=0 or p=0) together with one equality
# b = 0
b0 = 0.0
for (coeffs, rhs) in eqs:
    a, bcoef = coeffs
    # a*b + bcoef*p = rhs => p = (rhs - a*b)/bcoef
    if abs(bcoef) < tol:
        continue
    p_sol = (rhs - a * b0) / bcoef
    candidates.append((b0, p_sol, "b=0 & eq"))
# p = 0
p0 = 0.0
for (coeffs, rhs) in eqs:
    a, bcoef = coeffs
    if abs(a) < tol:
        continue
    b_sol = (rhs - bcoef * p0) / a
    candidates.append((b_sol, p0, "p=0 & eq"))

# Also consider corner b=0,p=0 (though likely infeasible)
candidates.append((0.0, 0.0, "origin"))

# Evaluate candidates for feasibility and objective
best = None
best_point = None
for b_val, p_val, src in candidates:
    # Clean up tiny numerical noise
    if abs(b_val) < tol:
        b_val = 0.0
    if abs(p_val) < tol:
        p_val = 0.0
    if math.isinf(b_val) or math.isinf(p_val) or math.isnan(b_val) or math.isnan(p_val):
        continue
    if is_feasible(b_val, p_val):
        obj = c_b * b_val + c_p * p_val
        if (best is None) or (obj < best - tol):
            best = obj
            best_point = (b_val, p_val, src)

# As a fallback (shouldn't be needed), we can also check some boundary extreme points:
# For example, minimize objective along ratio p=2b and along b=0
# Solve along p=2b: substitute into fat constraint and calories to find minimal b satisfying both.
# But the pairwise intersections already capture that. Still include analytic solution for safety.
if best is None:
    # analytic: p = 2b, need 10b + 8(2b) >= 130 => 26b >= 130 => b >= 5
    b_a = 5.0
    p_a = 2.0 * b_a
    if is_feasible(b_a, p_a):
        best = c_b*b_a + c_p*p_a
        best_point = (b_a, p_a, "analytic_ratio_fat")
# Final safeguard: brute-force grid search over a reasonable range
if best is None:
    best = float('inf')
    best_point = None
    for b_try in [i*0.25 for i in range(0, 201)]:  # up to 50 burgers
        # minimal p given constraints: p >= max( (130-10b)/8, (3000-300b)/250, 2b, 0 )
        p_req1 = (130.0 - 10.0*b_try) / 8.0 if b_try*10.0 < 130.0 else 0.0
        p_req2 = (3000.0 - 300.0*b_try) / 250.0 if b_try*300.0 < 3000.0 else 0.0
        p_req = max(p_req1, p_req2, 2.0*b_try, 0.0)
        p_try = max(0.0, p_req)
        if is_feasible(b_try, p_try):
            obj = c_b*b_try + c_p*p_try
            if obj < best - tol:
                best = obj
                best_point = (b_try, p_try, "grid")

# At this point best should hold the minimal objective for the continuous LP
if best is None:
    raise RuntimeError("No feasible solution found.")

# Clean numerical result for printing
value = best
if abs(value - round(value)) < 1e-8:
    value = int(round(value))

# Print final objective in the exact required format as last line
print(f"FINAL_OBJ: {value}")
