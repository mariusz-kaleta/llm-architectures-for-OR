#!/usr/bin/env python3
import math

# Problem data
capacity_needed = 900
cap_w = 25
cap_n = 15

W_min = 5
W_max = 36
N_min = 0
N_max = 60

feasible_solutions = []

for W in range(W_min, W_max + 1):
    # Minimum N from capacity constraint
    residual = capacity_needed - cap_w * W
    if residual <= 0:
        n_from_capacity = 0
    else:
        n_from_capacity = math.ceil(residual / cap_n)
    # Minimum N from ratio constraint
    n_from_ratio = 3 * W
    N_req = max(N_min, n_from_capacity, n_from_ratio)
    # Check upper bound on N
    if N_req <= N_max:
        total_pipes = W + N_req
        total_flow = cap_w * W + cap_n * N_req
        feasible_solutions.append((W, N_req, total_pipes, total_flow))

# Find optimal solutions (minimal total pipes)
if not feasible_solutions:
    raise RuntimeError("No feasible solutions found within bounds.")

min_total = min(sol[2] for sol in feasible_solutions)
optimal_solutions = [sol for sol in feasible_solutions if sol[2] == min_total]

# Print optimal solutions (informational)
for W, N, total, flow in optimal_solutions:
    print(f"Optimal solution found: W={W}, N={N}, TotalPipes={total}, Flow={flow}")

# Final required print (must be the last line)
value = min_total
print(f"FINAL_OBJ: {value}")
