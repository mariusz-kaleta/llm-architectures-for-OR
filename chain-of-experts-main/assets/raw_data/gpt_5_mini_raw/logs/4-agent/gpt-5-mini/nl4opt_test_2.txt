# Problem data
hours_needed = 500
ft_hours = 8      # full-time hours per shift
pt_hours = 4      # part-time hours per shift
ft_cost = 300     # full-time cost per shift
pt_cost = 100     # part-time cost per shift
budget = 15000

# Derived constraint in simpler form: 2x + y >= 125 and 3x + y <= 150
# From derivation x is bounded by 0..25

best_obj = None
best_solutions = []

for x in range(0, max_x + 1):
    # minimal y to meet hours: 2x + y >= 125  => y >= 125 - 2x
    y_min = max(0, 125 - 2 * x)  # integer
    cost = ft_cost * x + pt_cost * y_min
    hours = ft_hours * x + pt_hours * y_min

    # check budget feasibility
    if cost <= budget and hours >= hours_needed:
        obj = x + y_min
        if best_obj is None or obj < best_obj:
            best_obj = obj
            best_solutions = [(x, y_min, cost, hours)]
        elif obj == best_obj:
            best_solutions.append((x, y_min, cost, hours))

# If no solution found in tightened range, fallback to broader search
if best_obj is None:
