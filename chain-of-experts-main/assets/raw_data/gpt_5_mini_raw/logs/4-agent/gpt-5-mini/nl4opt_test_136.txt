9) Quick sensitivity notes
- Because seasonal volunteers deliver more gifts per point (2.5 gifts/point) than full-time volunteers (1.6 gifts/point), the model tries to use as many seasonals as allowed by the 30% rule. #!/usr/bin/env python3
import math

# Problem data / bounds
POINTS_BUDGET = 200
MIN_F = 10
MAX_F = 40
MAX_S = 100

best_obj = None
best_s = None
best_f = None

for f in range(MIN_F, MAX_F + 1):
    # From 7*s <= 3*f  -> s <= floor(3*f / 7)
    s_max_proportion = (3 * f) // 7
    # From 2*s + 5*f <= 200 -> s <= floor((200 - 5*f) / 2)
    s_max_points = (POINTS_BUDGET - 5 * f) // 2
    s_candidate = min(s_max_proportion, s_max_points, MAX_S)
    if s_candidate < 0:
        continue
    s = s_candidate
    obj = 5 * s + 8 * f
    if (best_obj is None) or (obj > best_obj):
        best_obj = obj
        best_s = s
        best_f = f

# Final objective value (required output line)
print(f"FINAL_OBJ: {int(best_obj)}")
