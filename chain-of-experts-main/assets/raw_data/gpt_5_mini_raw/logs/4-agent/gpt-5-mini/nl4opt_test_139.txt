#!/usr/bin/env python3
import math

# Problem data
# Maximize Z = 3R + 4S
# s.t.
# 2R + 3S <= 40
# 3R + 5S <= 70
# R >= 0, S >= 0

constraints = [
    (2.0, 3.0, 40.0),  # 2R + 3S <= 40
    (3.0, 5.0, 70.0),  # 3R + 5S <= 70
]

# Include the non-negativity boundary lines as equalities for corner computation:
# R = 0 -> 1*R + 0*S = 0
# S = 0 -> 0*R + 1*S = 0
lines = constraints + [
    (1.0, 0.0, 0.0),  # R = 0
    (0.0, 1.0, 0.0),  # S = 0
]

tol = 1e-9
points = []

# Compute intersections of every pair of lines (as equalities)
n = len(lines)
for i in range(n):
    a1, b1, c1 = lines[i]
    for j in range(i + 1, n):
        a2, b2, c2 = lines[j]
        det = a1 * b2 - a2 * b1
        if abs(det) < tol:
            continue  # parallel or coincident
        R = (c1 * b2 - c2 * b1) / det
        S = (a1 * c2 - a2 * c1) / det
        # Check feasibility: nonnegativity and all inequality constraints
        if R >= -tol and S >= -tol:
            feasible = True
            for (a, b, c) in constraints:
                if a * R + b * S > c + 1e-7:
                    feasible = False
                    break
            if feasible:
                # Round tiny negatives to zero
                R = 0.0 if abs(R) < 1e-9 else R
                S = 0.0 if abs(S) < 1e-9 else S
                points.append((R, S))

# Also include origin explicitly (in case not captured)
points.append((0.0, 0.0))

# Remove duplicates (within tolerance)
unique_points = []
for p in points:
    found = False
    for q in unique_points:
        if abs(p[0] - q[0]) < 1e-7 and abs(p[1] - q[1]) < 1e-7:
            found = True
            break
    if not found:
        unique_points.append(p)

# Evaluate objective at feasible corner points
best_Z = -math.inf
best_point = None
for (R, S) in unique_points:
    Z = 3.0 * R + 4.0 * S
    if Z > best_Z + 1e-9:
        best_Z = Z
        best_point = (R, S)

# Ensure numeric neatness: if very close to integer, snap
def neat(x):
    if abs(x - round(x)) < 1e-9:
        return int(round(x))
    return float(x)

best_Z = neat(best_Z)

# Final required print
print(f"FINAL_OBJ: {best_Z}")
