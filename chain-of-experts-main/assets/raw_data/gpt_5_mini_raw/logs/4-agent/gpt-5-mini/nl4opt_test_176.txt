6) Brute‑force Python (simple, no external solver) — finds optimal pair and tie‑breaks by lower cost:

# brute-force search
best = None  # tuple (total_trips, total_cost, f, a, tons)
for f in range(5, 41):              # 5..40
    # minimal a required by air-share: 7a >= 5f  => a >= ceil(5f/7)
    a_min_share = -(-5*f // 7)     # integer ceiling
    # minimal a required by tonnage: 3a >= 200 - 5f  => a >= ceil(max(0, 200-5f)/3)
    remain = max(0, 200 - 5*f)
    a_min_tonnage = -(-remain // 3)
    a_min = max(a_min_share, a_min_tonnage, 0)
    # practical upper bound for a from budget and global bound
    a_max = min(36, (20000 - 300*f) // 550) if (20000 - 300*f) >= 0 else -1
    if a_max < a_min:
        continue
    # since objective is minimize f+a, minimal a for this f is a_min
    a = a_min
    cost = 300*f + 550*a
    tons = 5*f + 3*a
    if cost <= 20000 and tons >= 200:
        total_trips = f + a
        candidate = (total_trips, cost, f, a, tons)
        if (best is None) or (candidate[0] < best[0]) or (candidate[0] == best[0] and candidate[1] < best[1]):
            best = candidate

print("best =", best)  # prints (total_trips, total_cost, f, a, tons)
