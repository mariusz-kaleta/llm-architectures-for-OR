import math

# Problem data
capacity_needed = 1500
budget_limit = 1000

van_cap = 50
truck_cap = 80

van_cost = 30
truck_cost = 50

# Bounds from description
t_max = min(20, budget_limit // truck_cost)  # 20
v_max = budget_limit // van_cost  # 33

best_trips = None
best_cost = None
best_solution = None

for t in range(0, t_max + 1):
    # Minimum vans to satisfy advertising constraint
    v_ad = t + 1

    # Minimum vans to satisfy capacity constraint
    remaining = capacity_needed - truck_cap * t
    if remaining <= 0:
        v_cap = 0
    else:
        v_cap = (remaining + van_cap - 1) // van_cap  # ceil(remaining / van_cap)

    v_min = max(v_ad, v_cap)

    if v_min > v_max:
        continue

    # Check budget for this minimal v
    total_cost = van_cost * v_min + truck_cost * t
    if total_cost > budget_limit:
        continue

    trips = v_min + t

    if (best_trips is None) or (trips < best_trips) or (trips == best_trips and total_cost < best_cost):
        best_trips = trips
        best_cost = total_cost
        best_solution = (v_min, t)

# If no feasible solution found, set best_trips to None (but problem statement has feasible solutions)
if best_trips is None:
    # No feasible solution; print an indicator (as numeric -1)
    print(f"FINAL_OBJ: {-1}")
