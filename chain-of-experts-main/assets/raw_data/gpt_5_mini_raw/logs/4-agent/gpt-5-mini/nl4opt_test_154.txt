import math

# Problem data
cap_reg = 500        # packages per regular van
cap_hyb = 300        # packages per hybrid van
poll_reg = 200       # pollution units per regular van
poll_hyb = 100       # pollution units per hybrid van
req_packages = 20000
poll_cap = 7000

# Derived finite bounds (from pollution cap)
max_x = poll_cap // poll_reg  # = 7000 // 200 = 35

best = None  # tuple (total_vans, x, y)

for x in range(0, max_x + 1):
    # minimal y to meet delivery requirement
    remaining = req_packages - cap_reg * x
    if remaining <= 0:
        y_min = 0
    else:
        y_min = math.ceil(remaining / cap_hyb)
    # enforce non-negativity
    if y_min < 0:
        y_min = 0
    # check pollution constraint
    if poll_reg * x + poll_hyb * y_min <= poll_cap:
        total = x + y_min
        if (best is None) or (total < best[0]) or (total == best[0] and x < best[1]):
            best = (total, x, y_min)

# If not found by this direct method (should not happen), do full enumeration up to some safe bound
if best is None:
    best = (10**9, 0, 0)
    for x in range(0, 71):
        for y in range(0, 71):
            if cap_reg * x + cap_hyb * y >= req_packages and poll_reg * x + poll_hyb * y <= poll_cap:
                total = x + y
                if total < best[0] or (total == best[0] and x < best[1]):
                    best = (total, x, y)

if best is None or best[0] >= 10**9:
    print("No feasible solution found.")
else:
    total_vans, x_opt, y_opt = best
    delivered = cap_reg * x_opt + cap_hyb * y_opt
    pollution = poll_reg * x_opt + poll_hyb * y_opt
    print(f"Optimal regular vans x = {x_opt}")
    print(f"Optimal hybrid vans  y = {y_opt}")
    print(f"Total vans (objective) = {total_vans}")
    print(f"Total packages delivered = {delivered}")
    print(f"Total pollution = {pollution}")

# Final required line (must be exactly this format)
value = total_vans
print(f"FINAL_OBJ: {value}")
