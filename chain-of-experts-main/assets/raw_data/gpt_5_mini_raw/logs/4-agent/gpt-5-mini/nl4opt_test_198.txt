#!/usr/bin/env python3
import math

# Parameters
train_capacity = 120  # people/hour per train
tram_capacity = 30    # people/hour per tram
demand = 600          # people/hour

# Search bounds for x (number of trains)
max_x = math.ceil(demand / train_capacity)

best_solution = None  # tuple (z, x, y)

for x in range(0, max_x + 1):
    # capacity remaining after trains
    remaining = demand - train_capacity * x
    if remaining <= 0:
        needed_trams_by_capacity = 0
    else:
        needed_trams_by_capacity = math.ceil(remaining / tram_capacity)
    # build-ratio constraint y >= 2x
    y = max(2 * x, needed_trams_by_capacity)
    # ensure nonnegative integer
    if y < 0:
        y = 0
    z = x + y
    # check feasibility explicitly (should hold)
    if 120 * x + 30 * y < demand:
        continue
    if y < 2 * x:
        continue
    cand = (z, x, y)
    if best_solution is None or cand < best_solution:
        best_solution = cand

if best_solution is None:
    raise RuntimeError("No feasible solution found")

best_z, best_x, best_y = best_solution

# Final required print (only this line must be in this exact format)
print(f"FINAL_OBJ: {best_z}")
