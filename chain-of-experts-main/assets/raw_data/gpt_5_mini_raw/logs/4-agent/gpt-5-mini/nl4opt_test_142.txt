#!/usr/bin/env python3
# Self-contained solver for the described LP/ILP problem.
# Decision variables: L (large pieces), S (small pieces)
# Objective: maximize 30*L + 15*S
# Constraints:
#   4L + 2S <= 100   (paint)
#   3L + 1S <= 50    (glitter)
#   5L + 2S <= 70    (glue)
#   L >= 5, S >= 5
# We compute both the continuous (LP) optimum by enumerating vertices and
# the integer optimum by brute-force search over integer L,S >= 5.

import math

# Problem data
c_L = 30.0
c_S = 15.0

# Constraint coefficients (left-hand side coefficients and RHS)
# Each constraint represented as a*L + b*S <= rhs
constraints = [
    (4.0, 2.0, 100.0),  # paint
    (3.0, 1.0, 50.0),   # glitter
    (5.0, 2.0, 70.0)    # glue
]

# Lower bounds
L_min = 5.0
S_min = 5.0

# Helper: solve 2x2 linear system:
# a1*L + b1*S = c1
# a2*L + b2*S = c2
def solve_2x2(a1, b1, c1, a2, b2, c2, eps=1e-9):
    det = a1 * b2 - a2 * b1
    if abs(det) < eps:
        return None  # parallel or nearly so
    L = (c1 * b2 - c2 * b1) / det
    S = (a1 * c2 - a2 * c1) / det
    return (L, S)

# Build list of equality-lines to intersect:
# include the three constraint lines as equalities and the bounds L=5 and S=5
lines = []
# constraint lines: a*L + b*S = rhs
for (a, b, rhs) in constraints:
    lines.append((a, b, rhs))
# L = 5  -> 1*L + 0*S = 5
lines.append((1.0, 0.0, L_min))
# S = 5  -> 0*L + 1*S = 5
lines.append((0.0, 1.0, S_min))

# Generate all intersection points from pairs of lines
vertices = []
n_lines = len(lines)
for i in range(n_lines):
    for j in range(i + 1, n_lines):
        a1, b1, c1 = lines[i]
        a2, b2, c2 = lines[j]
        sol = solve_2x2(a1, b1, c1, a2, b2, c2)
        if sol is None:
            continue
        L, S = sol
        # check feasibility: all constraints <= rhs and lower bounds
        feasible = True
        # allow a tiny tolerance for numerical precision
        tol = 1e-9
        if L + tol < L_min or S + tol < S_min:
            feasible = False
        for (a, b, rhs) in constraints:
            if a * L + b * S - rhs > 1e-7:  # violate constraint
                feasible = False
                break
        if feasible:
            vertices.append((L, S))

# Also consider points at (L_min, S_min) explicitly (corner of bounds)
if all(a * L_min + b * S_min <= rhs + 1e-9 for (a, b, rhs) in constraints):
    vertices.append((L_min, S_min))

# Remove duplicates (within tolerance)
unique_vertices = []
seen = []
for (L, S) in vertices:
    key = (round(L, 9), round(S, 9))
    if key not in seen:
        seen.append(key)
        unique_vertices.append((L, S))

# Evaluate objective at each vertex to get continuous optimum
best_cont = None
best_cont_val = -1e18
for (L, S) in unique_vertices:
    val = c_L * L + c_S * S
    if val > best_cont_val + 1e-9:
        best_cont_val = val
        best_cont = (L, S)

# If no vertices found (shouldn't happen), default to L_min,S_min if feasible
if best_cont is None:
    if all(a * L_min + b * S_min <= rhs + 1e-9 for (a, b, rhs) in constraints):
        best_cont = (L_min, S_min)
        best_cont_val = c_L * L_min + c_S * S_min
    else:
        raise RuntimeError("No feasible continuous solution found.")

# Integer search (brute force)
# Determine reasonable upper bounds for L to iterate
# Use each constraint to bound L: a*L + b*S_min <= rhs -> L <= (rhs - b*S_min)/a
L_upper_bounds = []
for (a, b, rhs) in constraints:
    if a > 0:
        ub = (rhs - b * S_min) / a
        L_upper_bounds.append(ub)
# Also include a coarse absolute upper bound
L_upper_bounds.append(100.0)
L_max = int(math.floor(max(5.0, min(L_upper_bounds))))  # ensure at least 5 and integer
# But to be safe, cap L_max to 100
L_max = min(L_max, 100)

best_int_val = -1e18
best_int_solutions = []
for L_int in range(5, L_max + 1):
    # compute maximum feasible S for this L_int
    S_max_limits = []
    for (a, b, rhs) in constraints:
        if b > 0:
            s_lim = (rhs - a * L_int) / b
            S_max_limits.append(s_lim)
        else:
            # if b == 0, constraint doesn't bound S
            S_max_limits.append(float('inf'))
    S_max = math.floor(min(S_max_limits))
    # ensure we respect S_min
    if S_max < 5:
        continue
    for S_int in range(5, S_max + 1):
        # verify all constraints (integer values)
        feasible = True
        for (a, b, rhs) in constraints:
            if a * L_int + b * S_int - rhs > 1e-9:
                feasible = False
                break
        if not feasible:
            continue
        val = c_L * L_int + c_S * S_int
        if val > best_int_val + 1e-9:
            best_int_val = val
            best_int_solutions = [(L_int, S_int)]
        elif abs(val - best_int_val) <= 1e-9:
            best_int_solutions.append((L_int, S_int))

# Print results (continuous and integer), final required print at the end.
print("Continuous (LP) optimum (vertex enumeration):")
print(f"  L = {best_cont[0]:.6f}, S = {best_cont[1]:.6f}, Profit = {best_cont_val:.6f}")
print()
print("Integer optimum (brute-force search over integers >= 5):")
if best_int_val < -1e17:
    print("  No feasible integer solution found.")
else:
    print(f"  Best integer profit = {best_int_val:.0f}")
    print("  Optimal integer solutions (L, S):")
    for (L_i, S_i) in best_int_solutions:
        print(f"    ({L_i}, {S_i})  -> Profit = {int(c_L*L_i + c_S*S_i)}")
print()

# According to the problem remarks, integer counts are typically required.
# We therefore choose the integer optimal profit as the final reported objective.
final_value = int(round(best_int_val))

# Final required single-line print (must be the very last line)
print(f"FINAL_OBJ: {final_value}")
