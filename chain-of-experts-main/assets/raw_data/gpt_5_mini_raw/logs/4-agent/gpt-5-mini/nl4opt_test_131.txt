import math

# Problem data
# Decision variables: s1, s2 >= 0
# Objective: minimize Z = 0.5*s1 + 0.3*s2
# Constraints:
#   (A) 0.5*s1 + 0.2*s2 <= 5
#   (B) 0.4*s1 + 0.5*s2 >= 4  --> keep as >= in checks
#   s1 >= 0, s2 >= 0

# Define the equality forms for potential active constraints
# Each equation is (a, b, c) representing a*s1 + b*s2 = c
equations = [
    (0.5, 0.2, 5.0),  # A
    (0.4, 0.5, 4.0),  # B (as equality when active)
    (1.0, 0.0, 0.0),  # s1 = 0
    (0.0, 1.0, 0.0),  # s2 = 0
]

tol = 1e-9

def intersect(eq1, eq2):
    a1, b1, c1 = eq1
    a2, b2, c2 = eq2
    det = a1 * b2 - a2 * b1
    if abs(det) < tol:
        return None
    s1 = (c1 * b2 - c2 * b1) / det
    s2 = (a1 * c2 - a2 * c1) / det
    return (s1, s2)

def feasible(point):
    if point is None:
        return False
    s1, s2 = point
    if s1 < -1e-8 or s2 < -1e-8:
        return False
    # Constraint A: 0.5*s1 + 0.2*s2 <= 5
    if 0.5 * s1 + 0.2 * s2 > 5 + 1e-8:
        return False
    # Constraint B: 0.4*s1 + 0.5*s2 >= 4
    if 0.4 * s1 + 0.5 * s2 < 4 - 1e-8:
        return False
    return True

# Collect candidate vertices by intersecting pairs of equalities
candidates = []
n = len(equations)
for i in range(n):
    for j in range(i + 1, n):
        pt = intersect(equations[i], equations[j])
        if pt is not None:
            # Normalize tiny negatives to zero
            s1 = 0.0 if abs(pt[0]) < 1e-12 else pt[0]
            s2 = 0.0 if abs(pt[1]) < 1e-12 else pt[1]
            candidates.append((s1, s2))

# Also consider possible boundary points where one constraint is at an extreme:
# (Though intersections above include s1=0 or s2=0 with A or B.)
# Filter feasible candidates
feasible_pts = [pt for pt in candidates if feasible(pt)]

# It's possible the feasible region is not empty but none of the computed intersections passed due to numerical issues.
# For robustness, include the pure-syrup candidates (use only s1 or only s2 to meet B)
# Pure s1 to meet B: 0.4*s1 >= 4 -> s1 = 4/0.4 = 10
pt_pure_s1 = (10.0, 0.0)
if feasible(pt_pure_s1):
    feasible_pts.append(pt_pure_s1)
# Pure s2 to meet B: 0.5*s2 >= 4 -> s2 = 8
pt_pure_s2 = (0.0, 8.0)
if feasible(pt_pure_s2):
    feasible_pts.append(pt_pure_s2)

# Remove duplicates (within tolerance)
unique_pts = []
for p in feasible_pts:
    found = False
    for q in unique_pts:
        if abs(p[0]-q[0]) < 1e-8 and abs(p[1]-q[1]) < 1e-8:
            found = True
            break
    if not found:
        unique_pts.append(p)

# Evaluate objective at feasible vertices
def objective(pt):
    s1, s2 = pt
    return 0.5 * s1 + 0.3 * s2

if not unique_pts:
    # No feasible solution
    best_obj = float('nan')
else:
    best_pt = None
    best_obj = None
    for p in unique_pts:
        val = objective(p)
        if best_obj is None or val < best_obj - 1e-12:
            best_obj = val
            best_pt = p

# For this problem the known optimal continuous solution is s1=0, s2=8 with objective 2.4
# Ensure numeric cleanliness
if isinstance(best_obj, float) and not math.isnan(best_obj):
    # Round small floating error
    best_obj = round(best_obj, 12)

print(f"FINAL_OBJ: {best_obj}")
