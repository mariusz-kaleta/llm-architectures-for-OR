```python
# Self-contained solver for the described ILP (minimize fat subject to nutrition and turkey-share constraints).
# Uses simple enumeration since there are only two integer decision variables.

from math import inf, ceil

# Parameters
protein_t, carbs_t, fat_t = 20, 30, 12   # turkey per meal
protein_s, carbs_s, fat_s = 18, 25, 8    # tuna per meal

req_protein = 150
req_carbs = 200

# Turkey-share constraint linearised: 3 t <= 2 s  (i.e., s >= 1.5 t)

# Search bounds (reasonable upper bound)
MAX_MEALS = 50

best_obj = inf
best_t = None
best_s = None

for t in range(0, MAX_MEALS + 1):
    # We can derive a lower bound for s from constraints to reduce checks
    # From protein: 18 s >= 150 - 20 t  => s >= max(0, ceil((150-20t)/18))
    need_s_from_protein = 0
    rem_protein = req_protein - protein_t * t
    if rem_protein > 0:
        need_s_from_protein = ceil(rem_protein / protein_s)
    # From carbs: 25 s >= 200 - 30 t => s >= max(0, ceil((200-30t)/25))
    need_s_from_carbs = 0
    rem_carbs = req_carbs - carbs_t * t
    if rem_carbs > 0:
        need_s_from_carbs = ceil(rem_carbs / carbs_s)
    # From turkey-share: 2 s >= 3 t => s >= ceil(3t/2)
    need_s_from_share = ceil(3 * t / 2)

    s_start = max(0, need_s_from_protein, need_s_from_carbs, need_s_from_share)
    # iterate s from s_start up to MAX_MEALS
    for s in range(s_start, MAX_MEALS + 1):
        # Check constraints precisely (redundant given s_start but safe)
        if protein_t * t + protein_s * s < req_protein:
            continue
        if carbs_t * t + carbs_s * s < req_carbs:
            continue
        if 3 * t > 2 * s:
            continue
        obj = fat_t * t + fat_s * s
        if obj < best_obj:
            best_obj = obj
            best_t = t
            best_s = s

# If no feasible integer solution found within bounds, raise an error
if best_t is None:
    raise RuntimeError("No feasible integer solution found within search bounds.")

# Print final objective value as required (final printed line)
print(f"FINAL_OBJ: {best_obj}")
```
