Because only two integer variables, iterate S over its feasible integer range and compute the best feasible B for each S:

For S from S_min to floor(C / c_S) (i.e., 20..100):
  B_max_charge = floor((C - c_S * S) / c_B)  # from 3B + 2S ≤ 200

  B_max = min(B_max_charge, B_max_share)

    keep (B_max, S) if meals is largest seen

This is O(100) checks and trivial.

#!/usr/bin/env python3
# Self-contained script to solve the ILP by exact enumeration (two integer variables)

# Initialize parameters
m_B = 8    # meals per bike
m_S = 5    # meals per scooter
c_B = 3    # charge units per bike
c_S = 2    # charge units per scooter
C   = 200  # total available charge units

bike_share_max = 0.30  # bikes at most 30% of total vehicles (used to derive 7B <= 3S)
S_min = 20             # minimum scooters

# Derived integer constraint coefficients: 7B <= 3S
# Search bounds
S_upper = C // c_S  # 100

best_obj = None
best_B = None
best_S = None

for S in range(S_min, S_upper + 1):
    # Charge-based bound for B
    remaining_charge = C - c_S * S
    if remaining_charge < 0:
        continue
    B_max_charge = remaining_charge // c_B

    # Share-based bound for B: 7B <= 3S  => B <= floor(3S/7)
    B_max_share = (3 * S) // 7

    B_max = min(B_max_charge, B_max_share)
    if B_max < 0:
        continue

    B = B_max
    obj = m_B * B + m_S * S

    if (best_obj is None) or (obj > best_obj):
        best_obj = obj
        best_B = B
        best_S = S

# Final result
value = best_obj

# (Optional internal check - will raise if solution invalid)
assert best_B is not None and best_S is not None
assert 3*best_B + 2*best_S <= C
assert 7*best_B <= 3*best_S
assert best_S >= S_min

# Print final objective in required format
print(f"FINAL_OBJ: {value}")
