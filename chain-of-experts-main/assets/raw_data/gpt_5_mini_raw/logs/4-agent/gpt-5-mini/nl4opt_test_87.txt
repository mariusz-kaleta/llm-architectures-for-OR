#!/usr/bin/env python3
# Self-contained solver for the noodle / protein-bar minimisation problem.
# Solves both the continuous LP (analytic) and the integer (MILP) via brute force search.
# Prints some summary information and finally prints the required FINAL_OBJ line
# (FINAL_OBJ will contain the continuous LP optimal cost as a float).

from fractions import Fraction
import math

# Problem data
cost_x = Fraction(5)       # cost per noodle serving
cost_y = Fraction(5,2)     # cost per protein bar serving = 2.5

# Constraints coefficients (left-hand side)
cal_x = Fraction(600)
cal_y = Fraction(250)
cal_req = Fraction(2000)

prot_x = Fraction(3,2)     # 1.5
prot_y = Fraction(5)
prot_req = Fraction(16)

# Solve continuous LP by solving the two-binding constraints analytically:
# 600 x + 250 y = 2000
# 1.5 x + 5 y = 16
# Represent as exact rationals to avoid solver dependency.

# Build 2x2 system A * [x,y]^T = b
a11 = cal_x
a12 = cal_y
a21 = prot_x
a22 = prot_y
b1 = cal_req
b2 = prot_req

# Compute determinant
det = a11 * a22 - a12 * a21
if det == 0:
    raise RuntimeError("Constraint matrix singular; cannot compute analytic intersection.")

# Cramer's rule
x_star = (b1 * a22 - a12 * b2) / det
y_star = (a11 * b2 - b1 * a21) / det

cont_cost = cost_x * x_star + cost_y * y_star

# Convert to floats for readable printing
x_star_f = float(x_star)
y_star_f = float(y_star)
cont_cost_f = float(cont_cost)

print("Continuous LP solution (analytic intersection):")
print(f" x = {x_star_f:.12f} servings/day")
print(f" y = {y_star_f:.12f} servings/day")
print(f" Minimum cost = ${cont_cost_f:.12f} per day")
print("")

# Now integer search (MILP) by brute force over reasonable bounds.
# Compute safe upper bounds for integer search based on each variable individually.
x_max_cal = math.ceil(float(cal_req / cal_x))       # if only calories considered
x_max_prot = math.ceil(float(prot_req / prot_x))    # if only protein considered
x_max = max(x_max_cal, x_max_prot, 0) + 5            # add some slack

y_max_cal = math.ceil(float(cal_req / cal_y))
y_max_prot = math.ceil(float(prot_req / prot_y))
y_max = max(y_max_cal, y_max_prot, 0) + 5

# Limit bounds to reasonable sizes
x_max = min(x_max, 100)
y_max = min(y_max, 100)

best_int_cost = None
best_int_solutions = []

for xi in range(0, x_max + 1):
    for yi in range(0, y_max + 1):
        # Check feasibility
        if (cal_x * xi + cal_y * yi) >= cal_req and (prot_x * xi + prot_y * yi) >= prot_req:
            cost_i = cost_x * xi + cost_y * yi
            if (best_int_cost is None) or (cost_i < best_int_cost):
                best_int_cost = cost_i
                best_int_solutions = [(xi, yi)]
            elif cost_i == best_int_cost:
                best_int_solutions.append((xi, yi))

if best_int_cost is None:
    print("No integer feasible solution found in the search bounds.")
else:
    best_int_cost_f = float(best_int_cost)
    print("Integer (MILP) best solutions found by brute force search:")
    print(f" Minimum integer cost = ${best_int_cost_f:.12f} per day")
    print(f" Example integer-optimal (x,y) pairs: {best_int_solutions}")

# Final required print line: print final objective value.
# According to the model, the primary (continuous) LP was the main objective.
# Set 'value' to the continuous LP optimal cost (float).
value = cont_cost_f
print(f"FINAL_OBJ: {value}")
