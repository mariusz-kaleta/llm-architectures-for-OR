#!/usr/bin/env python3
# Self-contained script to solve the described pill-mix LP (continuous) and the integer variant (MIP by enumeration).
# Uses scipy.optimize.linprog for the continuous LP and a small brute-force search for the integer solution.

import math
import itertools
import numpy as np

try:
    from scipy.optimize import linprog
except Exception as e:
    raise ImportError("scipy is required to run this script. Please install scipy (e.g. pip install scipy).") from e

# Problem data (directly initialized)
# Decision variables: x_Z (Zodiac pills), x_S (Sunny pills)
c = np.array([1.0, 3.0])            # costs: [cost_Z, cost_S]
# Constraints (A_pos x >= b_pos)
A_pos = np.array([[1.3, 1.2],
                  [1.5, 5.0]])
b_pos = np.array([5.0, 10.0])

# Convert to linprog form: A_ub x <= b_ub  by multiplying both sides by -1
A_ub = -A_pos
b_ub = -b_pos
bounds = [(0, None), (0, None)]  # x_Z >=0, x_S >=0

# Solve continuous LP
res = linprog(c=c, A_ub=A_ub, b_ub=b_ub, bounds=bounds, method="highs")

if not res.success:
    raise RuntimeError(f"Linear program failed: {res.message}")

x_lp = res.x
obj_lp = float(res.fun)

# Check which constraints are binding (within tolerance)
tol = 1e-8
lhs_vals = A_pos.dot(x_lp)
binding = [abs(lhs_vals[i] - b_pos[i]) <= tol for i in range(len(b_pos))]

print("Continuous LP (relaxation) solution:")
print(f"  x_Z (Zodiac) = {x_lp[0]:.9f}")
print(f"  x_S (Sunny)  = {x_lp[1]:.9f}")
print(f"  Objective (cost) = {obj_lp:.9f}")
print(f"  Constraint LHS values = {lhs_vals.tolist()}")
print(f"  Constraints binding (Z1, D3) = {binding}")

# Integer (MIP) version: brute-force search over reasonable integer bounds
# Compute simple upper bounds based on single-pill feasibility, then add margin
xz_only_needed = max(b_pos[0] / A_pos[0,0], b_pos[1] / A_pos[1,0])  # if x_S = 0
xs_only_needed = max(b_pos[0] / A_pos[0,1], b_pos[1] / A_pos[1,1])  # if x_Z = 0
upper_z = int(math.ceil(xz_only_needed)) + 20  # margin
upper_s = int(math.ceil(xs_only_needed)) + 20  # margin

# Keep bounds reasonable
upper_z = min(upper_z, 200)
upper_s = min(upper_s, 200)

best_int_cost = None
best_int_solutions = []

for xz in range(0, upper_z + 1):
    # Quickly check if even with max xs we can satisfy, but we just brute force small space
    for xs in range(0, upper_s + 1):
        if (A_pos[0,0] * xz + A_pos[0,1] * xs) + 1e-12 >= b_pos[0] and (A_pos[1,0] * xz + A_pos[1,1] * xs) + 1e-12 >= b_pos[1]:
            cost = c[0] * xz + c[1] * xs
            if (best_int_cost is None) or (cost < best_int_cost - 1e-12):
                best_int_cost = cost
                best_int_solutions = [(xz, xs)]
            elif abs(cost - best_int_cost) <= 1e-12:
                best_int_solutions.append((xz, xs))

if best_int_cost is None:
    print("No integer feasible solution found within search bounds.")
else:
    print("\nInteger MIP solution(s) (searched by enumeration):")
    for sol in best_int_solutions:
        xz, xs = sol
        print(f"  x_Z = {xz}, x_S = {xs}, cost = {int(c[0]*xz + c[1]*xs)}")
    # also print as float for clarity
    print(f"  Best integer objective (cost) = {float(best_int_cost):.9f}")

# Decide final objective value to report in the required final print line.
# We will treat the integer solution as the "final" objective if any integer feasible solution was found,
# otherwise fall back to the LP objective.
if best_int_cost is not None:
    final_value = int(best_int_cost) if abs(best_int_cost - round(best_int_cost)) < 1e-12 else float(best_int_cost)
else:
    final_value = float(obj_lp)

# CRITICAL final print line with exact required format
print(f"FINAL_OBJ: {final_value}")
