#!/usr/bin/env python3
# Self-contained script to solve the medication allocation problem (continuous LP and integer MILP).
# Uses PuLP (CBC) to solve both relaxations and prints the final integer objective as required.

import pulp

# Problem data
time_A = 3  # minutes per unit of anxiety medication
time_D = 5  # minutes per unit of anti-depressant
min_total = 100
min_A = 30

# 1) Continuous LP relaxation
prob_lp = pulp.LpProblem("Medication_Continuous", pulp.LpMinimize)
A_cont = pulp.LpVariable("A_cont", lowBound=0, cat="Continuous")
D_cont = pulp.LpVariable("D_cont", lowBound=0, cat="Continuous")

prob_lp += time_A * A_cont + time_D * D_cont  # objective
prob_lp += A_cont + D_cont >= min_total
prob_lp += A_cont >= min_A
prob_lp += A_cont <= 2 * D_cont

prob_lp.solve(pulp.PULP_CBC_CMD(msg=False))
A_cont_val = pulp.value(A_cont)
D_cont_val = pulp.value(D_cont)
obj_cont = pulp.value(prob_lp.objective)

print("Continuous LP relaxation solution:")
print(f"  A = {A_cont_val:.6f}")
print(f"  D = {D_cont_val:.6f}")
print(f"  Objective (total time) = {obj_cont:.6f}")
print()

# 2) Integer MILP
prob_milp = pulp.LpProblem("Medication_Integer", pulp.LpMinimize)
A_int = pulp.LpVariable("A_int", lowBound=0, cat="Integer")
D_int = pulp.LpVariable("D_int", lowBound=0, cat="Integer")

prob_milp += time_A * A_int + time_D * D_int  # objective
prob_milp += A_int + D_int >= min_total
prob_milp += A_int >= min_A
prob_milp += A_int <= 2 * D_int

prob_milp.solve(pulp.PULP_CBC_CMD(msg=False))
A_int_val = pulp.value(A_int)
D_int_val = pulp.value(D_int)
obj_int = pulp.value(prob_milp.objective)

print("Integer MILP solution:")
print(f"  A = {int(A_int_val)}")
print(f"  D = {int(D_int_val)}")
print(f"  Objective (total time) = {int(obj_int)}")
print()

# Final required single-line output (must be the very last line)
print(f"FINAL_OBJ: {int(round(obj_int))}")
