import math

# Problem data
hour_req = 1000
budget = 40000
cost_s = 300  # cost per staff teacher (6 hours)
cost_t = 100  # cost per substitute teacher (3 hours)
hours_s = 6
hours_t = 3

# Search bounds
max_s = budget // cost_s  # upper bound on s by budget
best = None  # will store tuple (H, total_cost, -s, s, t, total_hours)

for s in range(0, max_s + 1):
    # minimal t to meet hours requirement
    t_req = max(0, math.ceil((hour_req - hours_s * s) / hours_t))
    total_cost = cost_s * s + cost_t * t_req
    total_hours = hours_s * s + hours_t * t_req
    if total_cost <= budget and total_hours >= hour_req:
        H = s + t_req
        # tie-breaker: prefer lower headcount, then lower cost, then more staff (so use -s)
        candidate = (H, total_cost, -s, s, t_req, total_hours)
        if best is None or candidate < best:
            best = candidate

if best is None:
    # No feasible solution
    print(f"FINAL_OBJ: infeasible")
else:
