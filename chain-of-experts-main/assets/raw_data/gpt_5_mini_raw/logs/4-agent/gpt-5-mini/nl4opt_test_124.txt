#!/usr/bin/env python3
# Self-contained script to solve the integer problem by exhaustive search (small problem).
# Finds integer H,L >= 0 minimizing H+L subject to:
# 50H + 30L >= 800
# 50H + 20L <= 700
# 3H - 2L <= 0
# L >= 10

def main():
    # Parameters / bounds
    # From water constraint: 50H + 20L <= 700 -> with L=0, H <= 14
    max_H = 14
    min_L = 10
    # From water constraint with H=0: 20L <=700 -> L <=35
    max_L = 35

    feasible_solutions = []
    best_obj = None

    for H in range(0, max_H + 1):
        for L in range(min_L, max_L + 1):
            proc = 50 * H + 30 * L
            water = 50 * H + 20 * L
            prop = 3 * H - 2 * L

            if proc >= 800 and water <= 700 and prop <= 0:
                obj = H + L
                feasible_solutions.append((H, L, obj, proc, water, -prop))  # store proportion slack as -prop = 2L-3H
                if best_obj is None or obj < best_obj:
                    best_obj = obj

    if not feasible_solutions:
        print("No feasible integer solutions found.")
        # Final print still required (but problem is feasible so this branch won't execute here)
        print(f"FINAL_OBJ: None")
        return

    # Filter optimal solutions
    optimal_sols = [s for s in feasible_solutions if s[2] == best_obj]

    # Print summary (optional informational output)
    print(f"Found {len(feasible_solutions)} feasible integer solutions.")
    print(f"Best objective (minimal total drills) = {best_obj}")
    print("Optimal integer solution(s):")
    for H, L, obj, proc, water, prop_slack in optimal_sols:
        proc_slack = proc - 800
        water_slack = 700 - water
        print(f"  H = {H}, L = {L}  -> total drills = {obj}")
        print(f"    processing: {proc} (slack {proc_slack})")
        print(f"    water:      {water} (slack {water_slack})")
        print(f"    proportion slack (2L-3H): {prop_slack}")

    # Final required single-line output (must be last line)
    print(f"FINAL_OBJ: {best_obj}")

if __name__ == "__main__":
    main()
