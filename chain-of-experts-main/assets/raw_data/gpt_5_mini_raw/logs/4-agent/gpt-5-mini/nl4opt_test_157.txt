[CODER]:
import math

# Problem data (original scale)
demand = 50000         # patties required
cap_truck = 1000       # patties per truck trip
cap_van = 500          # patties per van trip
cost_truck = 300       # cost per truck trip
cost_van = 100         # cost per van trip
budget = 12500         # available budget

# Search bounds for T: trucks can't exceed budget alone
max_T = budget // cost_truck

best_obj = None
best_solution = None

for T in range(0, max_T + 1):
    # Minimum vans to meet capacity (if any)
    remaining_demand = demand - cap_truck * T
    if remaining_demand <= 0:
        vans_needed_for_capacity = 0
    else:
        vans_needed_for_capacity = (remaining_demand + cap_van - 1) // cap_van  # ceil division

    # City/operational constraint: V >= T
    V_min = max(vans_needed_for_capacity, T, 0)

    # Budget constraint:  cost_truck*T + cost_van*V <= budget  => V <= floor((budget - cost_truck*T)/cost_van)
    remaining_budget = budget - cost_truck * T
    if remaining_budget < 0:
        continue  # no feasible V for this T
    V_max = remaining_budget // cost_van

    if V_min <= V_max:
        # To minimize T+V for fixed T, choose smallest feasible V (V_min)
        V = V_min
        obj = T + V
        if (best_obj is None) or (obj < best_obj):
            best_obj = obj
            best_solution = (T, V)

# If no feasible solution found (shouldn't happen), set to None
if best_solution is None:
    raise RuntimeError("No feasible solution found")

T_opt, V_opt = best_solution
total_trips = best_obj
total_patties = cap_truck * T_opt + cap_van * V_opt
total_cost = cost_truck * T_opt + cost_van * V_opt

# Print solution details
print(f"Optimal trucks (T): {T_opt}")
print(f"Optimal vans   (V): {V_opt}")
print(f"Total patties shipped: {total_patties}")
print(f"Total cost: {total_cost}")
print(f"Total trips (objective): {total_trips}")

# Final required line (must be the last line)
print(f"FINAL_OBJ: {total_trips}")
