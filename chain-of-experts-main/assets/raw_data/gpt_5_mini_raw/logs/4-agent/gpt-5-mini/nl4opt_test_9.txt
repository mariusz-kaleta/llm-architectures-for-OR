import numpy as np

# Problem data (initialized directly)
# Constraints:
# 10x + 20y <= 500
# 15x + 20y <= 700
# Objective: maximize Z = 20x + 30y
A = np.array([[10.0, 20.0],
              [15.0, 20.0]])
b = np.array([500.0, 700.0])
c = np.array([20.0, 30.0])  # objective coefficients (to maximize)

# Solve LP relaxation by solving the two binding constraints (if non-singular)
try:
    sol = np.linalg.solve(A, b)
    x_lp, y_lp = sol[0], sol[1]
    # Check feasibility (non-negativity)
    feasible_lp = (x_lp >= -1e-9) and (y_lp >= -1e-9)
    Z_lp = c.dot(sol)
except np.linalg.LinAlgError:
    # Fallback (should not happen for this 2x2 A)
    x_lp = y_lp = 0.0
    Z_lp = 0.0
    feasible_lp = False

# Prepare integer (ILP) search: enumerate feasible integer (x,y)
# Determine safe upper bounds from individual constraints
x_max = int(np.floor(min(500.0 / 10.0, 700.0 / 15.0)))  # floor(min(50, 46.666...)) -> 46
y_max = int(np.floor(min(500.0 / 20.0, 700.0 / 20.0)))  # floor(min(25, 35)) -> 25

best_obj = -np.inf
best_x = None
best_y = None

for xi in range(0, x_max + 1):
    # For given xi, we can compute max possible y bound to speed up
    # but simple enumeration over y_max is small so ok
    for yi in range(0, y_max + 1):
        if (10 * xi + 20 * yi) <= 500 + 1e-9 and (15 * xi + 20 * yi) <= 700 + 1e-9:
            obj = 20 * xi + 30 * yi
            if obj > best_obj + 1e-9:
                best_obj = obj
                best_x = xi
                best_y = yi

# Print results (LP and ILP) and final objective line as required.
print(f"LP solution (relaxation): x = {x_lp:.6f}, y = {y_lp:.6f}, Z = {Z_lp:.6f}")
print(f"ILP optimal integer solution: x = {best_x}, y = {best_y}, Z = {int(best_obj)}")

# Final required print line (must be exactly this format)
print(f"FINAL_OBJ: {int(best_obj)}")
