#!/usr/bin/env python3
import math

# Data / parameters
W = 250000  # total water in ml
water_glass = 500
water_plastic = 750

# Decision variable bounds (tightened)
G_min = 20
G_max = 90  # from (250000 / 2750) floor as given

best_T = -1
best_solutions = []

for G in range(G_min, G_max + 1):
    remaining = W - water_glass * G
    if remaining < 0:
        continue
    P_max = remaining // water_plastic  # floor
    if P_max < 0:
        continue
    # Preference ratio constraint: P >= 3G
    if P_max < 3 * G:
        continue
    T = G + P_max
    water_used = water_glass * G + water_plastic * P_max
    if T > best_T:
        best_T = T
        best_solutions = [(G, P_max, water_used)]
    elif T == best_T:
        best_solutions.append((G, P_max, water_used))

# Choose a unique solution from best_solutions.
# Prefer solution that uses all water exactly; if none, pick one with largest G.
chosen = None
for sol in best_solutions:
    if sol[2] == W:
        chosen = sol
        break
if chosen is None:
    # pick solution with maximum G
    chosen = max(best_solutions, key=lambda s: s[0])

G_opt, P_opt, water_used_opt = chosen

# Optional verification (asserts)
assert G_opt >= G_min and isinstance(G_opt, int)
assert P_opt >= 0 and isinstance(P_opt, int)
assert water_glass * G_opt + water_plastic * P_opt <= W
assert P_opt >= 3 * G_opt

# Print chosen solution (optional)
print(f"Chosen solution: G={G_opt}, P={P_opt}, water_used={water_used_opt}, total={best_T}")

# Final required line (must be the last line in this exact format)
print(f"FINAL_OBJ: {best_T}")
