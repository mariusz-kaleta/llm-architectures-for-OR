```python
import math

# Parameters
C_h = 3   # cows per helicopter trip
p_h = 5   # pollution per helicopter trip
C_t = 7   # cows per truck trip
p_t = 10  # pollution per truck trip
N = 80    # total cows required
T_max = 8 # maximum truck trips available

best_obj = None
best_T = None
best_H = None
best_cows = None

for T in range(0, T_max + 1):
    remaining = N - C_t * T
    if remaining <= 0:
        H = 0
    else:
        H = math.ceil(remaining / C_h)
    Z = p_h * H + p_t * T
    cows_transported = C_t * T + C_h * H
    if (best_obj is None) or (Z < best_obj):
        best_obj = Z
        best_T = T
        best_H = H
        best_cows = cows_transported

# Optional: print solution details (not required by the strict final line format)
print(f"Optimal truck trips T = {best_T}")
print(f"Optimal helicopter trips H = {best_H}")
print(f"Total cows transported = {best_cows}")
print(f"Total pollution = {best_obj}")

value = best_obj
print(f"FINAL_OBJ: {value}")
```
