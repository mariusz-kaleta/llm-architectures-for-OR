   20 Z ≥ T = Z + S + W

#!/usr/bin/env python3
# Self-contained script to solve the ILP by enumeration (small bounded ranges).
# Decision variables: Z (z-tube), S (soorchle), W (wassa)
# Objective: maximize 400000 Z + 5000 S + 3000 W
# Constraints implemented as described in the problem statement.

def main():
    # Parameters
    cZ, vZ = 1000, 400000
    cS, vS = 200, 5000
    cW, vW = 100, 3000
    B = 10000
    S_max = 15

    # Bounds for brute-force search
    Z_min, Z_max = 0, 10   # budget-limited
    S_min, S_max = 0, 15   # platform cap
    W_min, W_max = 0, 100  # budget-limited

    best_value = None
    best_solutions = []

    for Z in range(Z_min, Z_max + 1):
        for S in range(S_min, S_max + 1):
            for W in range(W_min, W_max + 1):
                # Budget constraint
                cost = cZ * Z + cS * S + cW * W
                if cost > B:
                    continue
                # Share constraints (linearized)
                # 2 W <= Z + S  (wassa <= 1/3 of total)
                if 2 * W > Z + S:
                    continue
                # 19 Z >= S + W  (z-tube >= 5% of total)
                if 19 * Z < S + W:
                    continue
                # All constraints satisfied; compute objective
                viewers = vZ * Z + vS * S + vW * W

                if (best_value is None) or (viewers > best_value):
                    best_value = viewers
                    best_solutions = [(Z, S, W, cost, viewers)]
                elif viewers == best_value:
                    best_solutions.append((Z, S, W, cost, viewers))

    if best_value is None:
        # No feasible solution found (shouldn't happen given zero is feasible)
        best_value = 0
        chosen = (0, 0, 0, 0, 0)
    else:
        # If multiple optimal solutions, choose the one with minimal cost
        chosen = min(best_solutions, key=lambda x: x[3])  # minimize cost

    Z_opt, S_opt, W_opt, cost_opt, viewers_opt = chosen
    T_opt = Z_opt + S_opt + W_opt
    Z_share = (Z_opt / T_opt) if T_opt > 0 else 0.0
    W_share = (W_opt / T_opt) if T_opt > 0 else 0.0

    # Report results
    print("Optimal integer solution found:")
    print(f"Z (z-tube)   = {Z_opt}")
    print(f"S (soorchle) = {S_opt}")
    print(f"W (wassa)    = {W_opt}")
    print(f"Total ads T  = {T_opt}")
    print(f"Total viewers V = {viewers_opt}")
    print(f"Total cost C     = {cost_opt}")
    print(f"Z share = {Z_share:.4f}, W share = {W_share:.4f}")
    # Final required line (must be the very last printed line)
    print(f"FINAL_OBJ: {viewers_opt}")

if __name__ == "__main__":
    main()
