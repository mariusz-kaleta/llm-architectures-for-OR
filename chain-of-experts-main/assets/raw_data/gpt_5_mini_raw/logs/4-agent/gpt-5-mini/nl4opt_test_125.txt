M ≤ (33/100) (B + M)
=> (67/100) M ≤ (33/100) B
=> 67 M ≤ 33 B  (integer-linear inequality)

subject to
  80 B + 100 M ≥ 4000

import pulp

def main():
    # Define the problem
    prob = pulp.LpProblem("Minimize_Sugar", pulp.LpMinimize)

    # Decision variables: integers >= 0
    B = pulp.LpVariable("B", lowBound=0, cat=pulp.LpInteger)  # bananas
    M = pulp.LpVariable("M", lowBound=0, cat=pulp.LpInteger)  # mangoes

    # Objective: minimize total sugar (10 g per banana, 8 g per mango)
    prob += 10 * B + 8 * M, "Total_Sugar"

    # Constraints
    prob += 80 * B + 100 * M >= 4000, "Calories"
    prob += 20 * B + 15 * M >= 150, "Potassium"
    # Mango-share: 33/100 linearized -> 67 M <= 33 B
    prob += 67 * M <= 33 * B, "Mango_Share_33pct"

    # Solve with the default CBC solver
    solver = pulp.PULP_CBC_CMD(msg=False)
    prob.solve(solver)

    # Extract solution values
    B_val = int(pulp.value(B))
    M_val = int(pulp.value(M))
    total_sugar = 10 * B_val + 8 * M_val
    calories = 80 * B_val + 100 * M_val
    potassium = 20 * B_val + 15 * M_val

    # Constraint slacks / activities
    calories_slack = calories - 4000  # >= constraint
    potassium_slack = potassium - 150  # >= constraint
    mango_share_slack = 33 * B_val - 67 * M_val  # <= constraint: RHS - LHS

    # Print solution summary
    print(f"B (bananas) = {B_val}")
    print(f"M (mangoes) = {M_val}")
    print(f"Total sugar (g) = {total_sugar}")
    print(f"Calories = {calories}")
    print(f"Potassium = {potassium}")
    print(f"Calories slack = {calories_slack}")
    print(f"Potassium slack = {potassium_slack}")
    print(f"Mango-share slack (33% constraint RHS - LHS) = {mango_share_slack}")

    # Final required output line (must be the last line and exactly in this format)
    # Ensure an integer-like formatting if the value is integral.
    obj_val = total_sugar
    if abs(obj_val - round(obj_val)) < 1e-9:
        obj_val = int(round(obj_val))
    print(f"FINAL_OBJ: {obj_val}")

if __name__ == "__main__":
    main()
