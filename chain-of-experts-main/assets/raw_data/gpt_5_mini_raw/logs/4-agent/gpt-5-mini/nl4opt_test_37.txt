```python
#!/usr/bin/env python3
# Self-contained script to solve the jar production LP and ILP described.

# Problem data
# Decision variables:
# T >= 0 : thin jars
# S >= 0 : stubby jars
# Objective: maximize Z = 5 T + 9 S
# Constraints:
#   50 T + 30 S <= 3000   (shaping capacity, minutes)
#   90 T + 150 S <= 4000  (baking capacity, minutes)

from math import floor

# Coefficients
profit_T = 5.0
profit_S = 9.0

# Capacities
shaping_cap = 3000.0
baking_cap = 4000.0

# Constraint coefficients
a_shaping_T = 50.0
a_shaping_S = 30.0

a_baking_T = 90.0
a_baking_S = 150.0

# 1) Solve LP (continuous) by checking relevant corner points analytically.
# Candidate corner points: intersections with axes and intersection of the two constraint lines (if feasible).
candidates = []

# Origin
candidates.append((0.0, 0.0))

# Baking intercepts
# T = baking_cap / a_baking_T when S = 0
T_bake_only = baking_cap / a_baking_T
candidates.append((T_bake_only, 0.0))
# S = baking_cap / a_baking_S when T = 0
S_bake_only = baking_cap / a_baking_S
candidates.append((0.0, S_bake_only))

# Shaping intercepts
T_shape_only = shaping_cap / a_shaping_T
candidates.append((T_shape_only, 0.0))
S_shape_only = shaping_cap / a_shaping_S
candidates.append((0.0, S_shape_only))

# Intersection of baking and shaping (solve linear system)
# a_shaping_T * T + a_shaping_S * S = shaping_cap
# a_baking_T * T + a_baking_S * S = baking_cap
det = a_shaping_T * a_baking_S - a_shaping_S * a_baking_T
if abs(det) > 1e-12:
    T_inter = (shaping_cap * a_baking_S - a_shaping_S * baking_cap) / det
    S_inter = (a_shaping_T * baking_cap - shaping_cap * a_baking_T) / det
    candidates.append((T_inter, S_inter))
# Filter feasible points (satisfy both constraints and nonnegativity)
feasible_lp = []
for T, S in candidates:
    if T >= -1e-9 and S >= -1e-9:
        if a_shaping_T * T + a_shaping_S * S <= shaping_cap + 1e-6 and a_baking_T * T + a_baking_S * S <= baking_cap + 1e-6:
            feasible_lp.append((max(0.0, T), max(0.0, S)))

# Evaluate objective at feasible LP corners
best_lp = None
best_lp_profit = -1e12
for T, S in feasible_lp:
    profit = profit_T * T + profit_S * S
    if profit > best_lp_profit + 1e-9:
        best_lp_profit = profit
        best_lp = (T, S)

# If no feasible corners found (shouldn't happen), fallback to zero
if best_lp is None:
    best_lp = (0.0, 0.0)
    best_lp_profit = 0.0

lp_T, lp_S = best_lp
lp_profit = best_lp_profit

# 2) Solve ILP (integer) by brute force enumeration (small search space).
# Determine reasonable integer bounds
max_T_by_shaping = int(floor(shaping_cap / a_shaping_T))
max_T_by_baking = int(floor(baking_cap / a_baking_T))
max_T = min(max_T_by_shaping, max_T_by_baking, 1000)  # safe cap

max_S_by_shaping = int(floor(shaping_cap / a_shaping_S))
max_S_by_baking = int(floor(baking_cap / a_baking_S))
max_S = min(max_S_by_shaping, max_S_by_baking, 1000)

best_ilp = None
best_ilp_profit = -10**12

for T_int in range(0, max_T + 1):
    for S_int in range(0, max_S + 1):
        if a_shaping_T * T_int + a_shaping_S * S_int <= shaping_cap + 1e-9 and \
           a_baking_T * T_int + a_baking_S * S_int <= baking_cap + 1e-9:
            profit = profit_T * T_int + profit_S * S_int
            if profit > best_ilp_profit:
                best_ilp_profit = profit
                best_ilp = (T_int, S_int)

if best_ilp is None:
    best_ilp = (0, 0)
    best_ilp_profit = 0

ilp_T, ilp_S = best_ilp
ilp_profit = int(best_ilp_profit) if abs(best_ilp_profit - round(best_ilp_profit)) < 1e-9 else best_ilp_profit

# Print results (informational), final required print line must be last.
print(f"LP (continuous) solution: T = {lp_T:.6f}, S = {lp_S:.6f}, profit = {lp_profit:.6f}")
print(f" - Shaping used: {a_shaping_T*lp_T + a_shaping_S*lp_S:.6f} / {shaping_cap}")
print(f" - Baking used:  {a_baking_T*lp_T + a_baking_S*lp_S:.6f} / {baking_cap}")
print(f"ILP (integer) solution: T = {ilp_T}, S = {ilp_S}, profit = {ilp_profit}")
print(f" - Shaping used: {a_shaping_T*ilp_T + a_shaping_S*ilp_S:.6f} / {shaping_cap}")
print(f" - Baking used:  {a_baking_T*ilp_T + a_baking_S*ilp_S:.6f} / {baking_cap}")

# Final required line: print only this format on the last line.
print(f"FINAL_OBJ: {ilp_profit}")
```
