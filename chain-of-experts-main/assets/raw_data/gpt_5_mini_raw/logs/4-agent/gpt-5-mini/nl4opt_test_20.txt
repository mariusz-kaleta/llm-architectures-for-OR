#!/usr/bin/env python3
# MILP to minimize sodium from ramen (r) and fries (f)
# Using PuLP with CBC solver (open-source)

import pulp

# Define problem
prob = pulp.LpProblem("Ramen_Fries_Min_Sodium", pulp.LpMinimize)

# Decision variables (integers, non-negative) with recommended upper bounds
r = pulp.LpVariable('r', lowBound=0, upBound=8, cat='Integer')   # ramen packs
f = pulp.LpVariable('f', lowBound=0, upBound=10, cat='Integer')  # fries packs

# Objective: minimize sodium (mg)
prob += 100 * r + 75 * f

# Constraints
prob += 400 * r + 300 * f >= 3000       # calorie requirement
prob += 20 * r + 10 * f >= 80           # protein requirement
prob += 7 * r <= 3 * f                  # ramen share at most 30%

# Solve
solver = pulp.PULP_CBC_CMD(msg=False)
prob.solve(solver)

# Obtain objective value
obj = pulp.value(prob.objective)
# Ensure integer display (objective is integer here)
value = int(round(obj))

# Final required print (exact format)
print(f"FINAL_OBJ: {value}")
