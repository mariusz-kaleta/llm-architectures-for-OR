#!/usr/bin/env python3
import itertools
import math
import numpy as np

# Problem data
# Decision vars: x1, x2 >= 0
# Objective: maximize Z = 25 x1 + 18 x2
c = np.array([25.0, 18.0])

# Constraints (A x <= b)
# 10 x1 + 12 x2 <= 120   (mint)
# 20 x1 + 15 x2 <= 100   (active)
# 5  x1 + 3  x2 <= 50    (tar)
A = np.array([
    [10.0, 12.0],
    [20.0, 15.0],
    [5.0,  3.0]
])
b = np.array([120.0, 100.0, 50.0])

tolerance = 1e-9

# Helper: check feasibility (A x <= b + tol and x >= -tol)
def is_feasible(x):
    if (x < -tolerance).any():
        return False
    if (A.dot(x) - b > tolerance).any():
        return False
    return True

# 1) Enumerate candidate vertices of the feasible polygon (intersection of pairs of constraints and axes)
candidates = []

# Intersections of pairs of constraint equalities
num_constraints = A.shape[0]
for i, j in itertools.combinations(range(num_constraints), 2):
    Ai = A[[i, j], :]
    bi = b[[i, j]]
    try:
        sol = np.linalg.solve(Ai, bi)
    except np.linalg.LinAlgError:
        continue
    if is_feasible(sol):
        candidates.append(sol)

# Intersections with axes: x1 = 0 solve A[:,1]*x2 <= b ; x2 = 0 solve A[:,0]*x1 <= b
# For each constraint, compute x when the other var = 0 (if non-negative)
# x1 = 0 => for each constraint equality compute x2 = b_i / A[i,1] (if A[i,1] > 0)
for i in range(num_constraints):
    a21 = A[i, 1]
    if abs(a21) > tolerance:
        x2 = b[i] / a21
        x = np.array([0.0, x2])
        if is_feasible(x):
            candidates.append(x)

# x2 = 0 => for each constraint equality compute x1 = b_i / A[i,0] (if A[i,0] > 0)
for i in range(num_constraints):
    a10 = A[i, 0]
    if abs(a10) > tolerance:
        x1 = b[i] / a10
        x = np.array([x1, 0.0])
        if is_feasible(x):
            candidates.append(x)

# Include origin
if is_feasible(np.array([0.0, 0.0])):
    candidates.append(np.array([0.0, 0.0]))

# Remove duplicates (within tolerance)
unique_candidates = []
for cand in candidates:
    good = True
    for uc in unique_candidates:
        if np.linalg.norm(cand - uc) < 1e-7:
            good = False
            break
    if good:
        unique_candidates.append(cand)

# Evaluate objective at each candidate
best_continuous = None
best_continuous_obj = -math.inf
for x in unique_candidates:
    val = c.dot(x)
    if val > best_continuous_obj + 1e-9:
        best_continuous_obj = val
        best_continuous = x.copy()

# 2) Integer (brute force small search)
# Determine conservative upper bounds for search
# For x1 alone (x2=0): ub1 = floor(min(b_i / A[i,0] for i where A[i,0]>0))
# For x2 alone (x1=0): ub2 = floor(min(b_i / A[i,1] for i where A[i,1]>0))
ub1 = math.floor(min((b[i] / A[i,0]) for i in range(num_constraints) if A[i,0] > tolerance))
ub2 = math.floor(min((b[i] / A[i,1]) for i in range(num_constraints) if A[i,1] > tolerance))
# To be safe, search a bit beyond (but keep small)
ub1 = max(ub1, 0)
ub2 = max(ub2, 0)

best_int = None
best_int_obj = -math.inf
for i1 in range(0, ub1 + 1):
    for i2 in range(0, ub2 + 1):
        x = np.array([float(i1), float(i2)])
        if is_feasible(x):
            val = c.dot(x)
            if val > best_int_obj + 1e-9:
                best_int_obj = val
                best_int = (i1, i2)

# 3) Identify binding constraints at continuous optimum
binding = []
if best_continuous is not None:
    residuals = A.dot(best_continuous) - b
    for idx, res in enumerate(residuals):
        if abs(res) <= 1e-7:
            binding.append(idx)  # index of binding constraint

# 4) Numerical shadow price for 'active' resource (constraint index 1) via one-unit perturbation
def continuous_opt_with_rhs(bvec):
    # recompute vertices given new RHS bvec
    cand = []
    for i, j in itertools.combinations(range(num_constraints), 2):
        Ai = A[[i, j], :]
        bi = bvec[[i, j]]
        try:
            sol = np.linalg.solve(Ai, bi)
        except np.linalg.LinAlgError:
            continue
        if is_feasible(sol):
            cand.append(sol)
    for i in range(num_constraints):
        a21 = A[i, 1]
        if abs(a21) > tolerance:
            x2 = bvec[i] / a21
            x = np.array([0.0, x2])
            if is_feasible(x):
                cand.append(x)
    for i in range(num_constraints):
        a10 = A[i, 0]
        if abs(a10) > tolerance:
            x1 = bvec[i] / a10
            x = np.array([x1, 0.0])
            if is_feasible(x):
                cand.append(x)
    if is_feasible(np.array([0.0, 0.0])):
        cand.append(np.array([0.0, 0.0]))
    unique = []
    for cc in cand:
        good = True
        for uc in unique:
            if np.linalg.norm(cc - uc) < 1e-7:
                good = False
                break
        if good:
            unique.append(cc)
    best_val = -math.inf
    best_x = None
    for x in unique:
        val = c.dot(x)
        if val > best_val + 1e-9:
            best_val = val
            best_x = x.copy()
    return best_x, best_val

# Compute shadow price for active constraint (index 1)
delta = 1.0
b_pert = b.copy()
b_pert[1] += delta
x_pert, val_pert = continuous_opt_with_rhs(b_pert)
# If perturbation leads to different feasible set but we can compute approximate shadow
if best_continuous is not None:
    shadow_active_numeric = val_pert - best_continuous_obj
else:
    shadow_active_numeric = None

# Print results (informational)
print("Continuous optimal solution (vertices search):")
print(f" x1 = {best_continuous[0]:.6f}, x2 = {best_continuous[1]:.6f}, objective = {best_continuous_obj:.6f}")
print("Integer optimal solution (brute force):")
print(f" x1 = {best_int[0]}, x2 = {best_int[1]}, objective = {best_int_obj:.6f}")
constraint_names = ["mint (10 x1 + 12 x2 <= 120)", "active (20 x1 + 15 x2 <= 100)", "tar (5 x1 + 3 x2 <= 50)"]
if binding:
    print("Binding constraints at continuous optimum:")
    for idx in binding:
        print(f" - {constraint_names[idx]}")
else:
    print("No binding constraints detected (within tolerance).")

print(f"Numeric shadow price for active resource (one-unit increase): {shadow_active_numeric:.6f}")
print("Analytical foam per active unit for demo 1:", 25.0 / 20.0)
print("Analytical foam per active unit for demo 2:", 18.0 / 15.0)

# Final required single-line output (must be the very last printed line)
# As integer and continuous optima coincide here, print integer optimum objective as final value.
final_value = int(round(best_int_obj))
print(f"FINAL_OBJ: {final_value}")
