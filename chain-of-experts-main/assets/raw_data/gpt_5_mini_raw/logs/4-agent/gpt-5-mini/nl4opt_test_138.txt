import math

# Problem data
p_S = 2     # protein per smoothie
p_B = 7     # protein per bar
c_S = 300   # calories per smoothie
c_B = 250   # calories per bar
C_max = 2000

# Ratio interpretations
interpretations = {
    "A (B = 2·S)": 2,
    "B (B = 3·S)": 3
}

results = {}

for name, k in interpretations.items():
    denom = c_S + c_B * k  # calories per one "bundle" of S with k*B
    # Best integer S is floor(C_max / denom)
    S_star = C_max // denom
    B_star = k * S_star
    protein = (p_S + p_B * k) * S_star
    calories = denom * S_star
    results[name] = {
        "k": k,
        "S": S_star,
        "B": B_star,
        "protein": protein,
        "calories": calories,
        "denom": denom
    }

# Also compute continuous (fractional) solution for interpretation A (k=2) for reference
k_cont = 2
denom_cont = c_S + c_B * k_cont
S_cont = C_max / denom_cont
B_cont = k_cont * S_cont
protein_cont = (p_S + p_B * k_cont) * S_cont

# Print detailed results
for name, res in results.items():
    print(f"{name}: k={res['k']}, S={res['S']}, B={res['B']}, protein={res['protein']}, calories={res['calories']}")

print(f"Continuous (fractional) for interpretation A (B=2·S): S={S_cont:.6f}, B={B_cont:.6f}, protein={protein_cont:.6f}")

# Choose the best integer solution among interpretations (max protein)
best_name = max(results.keys(), key=lambda n: results[n]["protein"])
best_result = results[best_name]
value = best_result["protein"]

# Final required output line (must be the last line)
print(f"FINAL_OBJ: {value}")
