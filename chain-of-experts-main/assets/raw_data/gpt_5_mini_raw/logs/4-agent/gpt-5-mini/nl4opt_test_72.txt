#!/usr/bin/env python3
# Self-contained solver for the described LP:
# Maximize Z = 4 x1 + 5 x2
# s.t.
#   3 x1 + 8 x2 <= 100
#   5 x1 + 6 x2 <= 90
#   2 x1 + 4 x2 <= 30
#   x1 >= 0, x2 >= 0
#
# This script finds all intersection points (vertices) of the constraint lines
# (including the axes), tests feasibility, and returns the maximum objective.

import math

# Define the three resource constraints as lines: a*x1 + b*x2 = c
constraints_eq = [
    (3.0, 8.0, 100.0),  # Powder
    (5.0, 6.0, 90.0),   # Glue
    (2.0, 4.0, 30.0)    # Mess
]

# Add axis lines x1=0 and x2=0 for vertex generation
axis_eq = [
    (1.0, 0.0, 0.0),  # x1 = 0
    (0.0, 1.0, 0.0)   # x2 = 0
]

all_lines = constraints_eq + axis_eq

# Tolerance for numerical checks
tol = 1e-9

def intersect(line1, line2):
    a1, b1, c1 = line1
    a2, b2, c2 = line2
    det = a1 * b2 - a2 * b1
    if abs(det) < tol:
        return None  # parallel or coincident
    x1 = (c1 * b2 - c2 * b1) / det
    x2 = (a1 * c2 - a2 * c1) / det
    return (x1, x2)

def is_feasible(point):
    x1, x2 = point
    # Nonnegativity
    if x1 < -tol or x2 < -tol:
        return False
    # Resource constraints (<=)
    for a, b, c in constraints_eq:
        if a * x1 + b * x2 - c > tol:
            return False
    return True

# Generate candidate vertices from all pairs of lines
candidates = set()
n = len(all_lines)
for i in range(n):
    for j in range(i+1, n):
        pt = intersect(all_lines[i], all_lines[j])
        if pt is None:
            continue
        # Snap tiny negatives to zero
        x1, x2 = pt
        if abs(x1) < 1e-12:
            x1 = 0.0
        if abs(x2) < 1e-12:
            x2 = 0.0
        candidates.add((x1, x2))

# Also ensure origin is considered
candidates.add((0.0, 0.0))

# Evaluate feasible candidates and select best
best_obj = -math.inf
best_pt = None
for pt in candidates:
    if is_feasible(pt):
        x1, x2 = pt
        obj = 4.0 * x1 + 5.0 * x2
        if obj > best_obj + tol:
            best_obj = obj
            best_pt = pt

# If no feasible point found (should not happen), set to 0
if best_pt is None:
    best_obj = 0.0
    best_pt = (0.0, 0.0)

# If objective is very close to an integer, present it as integer
if abs(best_obj - round(best_obj)) < 1e-8:
    best_obj = int(round(best_obj))
else:
    # Round to a reasonable number of decimals for stable output
    best_obj = float(round(best_obj, 12))

print(f"FINAL_OBJ: {best_obj}")
