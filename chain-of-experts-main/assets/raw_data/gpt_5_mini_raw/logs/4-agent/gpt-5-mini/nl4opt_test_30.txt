#!/usr/bin/env python3
import math

# Problem data (initialized directly)
CAPACITY_REQUIRED = 1000
COW_CAPACITY = 20
ELEPHANT_CAPACITY = 50

# Practical search bounds
C_MAX = 50
E_MAX = 20

best_N = None
best_C = None
best_E = None

for E in range(0, E_MAX + 1):
    # minimal cows needed for capacity: ceil((1000 - 50*E) / 20)
    remaining = CAPACITY_REQUIRED - ELEPHANT_CAPACITY * E
    if remaining <= 0:
        c_needed_capacity = 0
    else:
        c_needed_capacity = math.ceil(remaining / COW_CAPACITY)
    # respect elephant ≤ cow
    c_min = max(E, c_needed_capacity)
    # check upper bound C ≤ 2E and C ≤ C_MAX
    if c_min <= 2 * E and c_min <= C_MAX:
        N = c_min + E
        if (best_N is None) or (N < best_N):
            best_N = N
            best_C = c_min
            best_E = E

if best_N is None:
    raise ValueError("No feasible solution found within given bounds.")

# Final objective value output (required exact format)
