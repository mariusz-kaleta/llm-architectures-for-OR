import math

# Try to solve using pulp if available, otherwise fall back to integer enumeration
try:
    import pulp

    # Data
    profit_x = 8
    profit_y = 10
    # Constraints coefficients
    # 3x + 4y <= 120 (wood)
    # 3x + 2y <= 90  (paint)

    # Build LP
    prob = pulp.LpProblem("Model_Production", pulp.LpMaximize)
    x = pulp.LpVariable("x", lowBound=0)
    y = pulp.LpVariable("y", lowBound=0)

    prob += profit_x * x + profit_y * y
    prob += 3 * x + 4 * y <= 120
    prob += 3 * x + 2 * y <= 90

    prob.solve(pulp.PULP_CBC_CMD(msg=False))

    obj_val = float(pulp.value(prob.objective))

    # Clean value to integer if close
    if abs(obj_val - round(obj_val)) < 1e-8:
        final_obj = int(round(obj_val))
    else:
        final_obj = obj_val

except Exception:
    # Fallback: integer enumeration (safe because bounds are small)
    best = None
    best_val = -float("inf")
    # reasonable bounds: from paint constraint 3x <= 90 -> x <= 30, from wood 3x <= 120 -> x <= 40
    for xi in range(0, 41):
        for yi in range(0, 31):
            if 3 * xi + 4 * yi <= 120 and 3 * xi + 2 * yi <= 90:
                val = 8 * xi + 10 * yi
                if val > best_val:
                    best_val = val
                    best = (xi, yi)
    final_obj = int(best_val) if best is not None else 0

print(f"FINAL_OBJ: {final_obj}")
