#!/usr/bin/env python3
# Self-contained script to solve the ILP described in the prompt by enumeration.

# Decision variables:
# s in Z, 0 <= s <= 10
# p in Z, p >= 0
# Objective: maximize Z = 10*s + 7*p
# Constraints:
# 30*s + 50*p <= 1200  (Vitamin C)
# 40*s + 30*p <= 1500  (Vitamin D)
# p >= s + 1           (popularity)

best_obj = None
best_s = None
best_p = None

for s in range(0, 11):  # s = 0..10
    # Compute maximum feasible p given constraints
    # From Vitamin C: p <= (1200 - 30*s) // 50
    # From Vitamin D: p <= (1500 - 40*s) // 30
    # From popularity: p >= s + 1
    # Also p >= 0
    max_p_c = (1200 - 30*s) // 50
    max_p_d = (1500 - 40*s) // 30
    p_max = min(max_p_c, max_p_d)
    p_min = max(0, s + 1)
    if p_max < p_min:
        continue
    # Objective coefficient positive for p, so choose p = p_max
    p = p_max
    obj = 10*s + 7*p
    if (best_obj is None) or (obj > best_obj):
        best_obj = obj
        best_s = s
        best_p = p

# Final objective value
value = int(best_obj) if best_obj is not None else 0
print(f"FINAL_OBJ: {value}")
