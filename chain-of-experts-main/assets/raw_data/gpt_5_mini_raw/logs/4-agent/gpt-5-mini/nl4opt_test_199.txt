#!/usr/bin/env python3
# Complete, self-contained script to solve the ILP described using PuLP

from pulp import LpProblem, LpMaximize, LpVariable, LpInteger, PULP_CBC_CMD, value

# Data
cost1 = 1200  # Pi
cost2 = 2000  # Beta
cost3 = 4000  # Gamma

v1 = 2000
v2 = 5000
v3 = 9000

B = 20000

# Upper bounds (tightening)
ub_x1 = 16  # floor(20000/1200)
ub_x2 = 8   # Beta cap
ub_x3 = 5   # floor(20000/4000)

# Build problem
prob = LpProblem("AdScheduling", LpMaximize)

# Decision variables
x1 = LpVariable('x1', lowBound=0, upBound=ub_x1, cat=LpInteger)  # Pi
x2 = LpVariable('x2', lowBound=0, upBound=ub_x2, cat=LpInteger)  # Beta (cap 8)
x3 = LpVariable('x3', lowBound=0, upBound=ub_x3, cat=LpInteger)  # Gamma

# Objective: maximize audience
prob += v1 * x1 + v2 * x2 + v3 * x3

# Constraints
prob += cost1 * x1 + cost2 * x2 + cost3 * x3 <= B           # budget
prob += 2 * x3 <= x1 + x2                                   # Gamma ≤ 1/3 total -> 2x3 ≤ x1 + x2
prob += 4 * x1 >= x2 + x3                                   # Pi ≥ 20% total -> 4x1 ≥ x2 + x3

# Solve
prob.solve(PULP_CBC_CMD(msg=0))

# Extract objective value (integer)
obj_val = int(round(value(v1 * x1 + v2 * x2 + v3 * x3)))

# Final required output line
print(f"FINAL_OBJ: {obj_val}")
