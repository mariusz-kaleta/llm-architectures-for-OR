[CODER]:
import pulp

# Parameters
B = 20000  # batter available (g)
M = 14000  # milk available (g)

b1, b2, b3 = 400, 500, 450  # batter per cake (g)
m1, m2, m3 = 200, 300, 350  # milk per cake (g)
p1, p2, p3 = 12, 10, 15     # profit per cake ($)

# Define problem
prob = pulp.LpProblem("Cake_Production", pulp.LpMaximize)

# Decision variables (integer, >=0)
x1 = pulp.LpVariable("x1_crepe", lowBound=0, cat="Integer")
x2 = pulp.LpVariable("x2_sponge", lowBound=0, cat="Integer")
x3 = pulp.LpVariable("x3_birthday", lowBound=0, cat="Integer")

# Objective
prob += p1 * x1 + p2 * x2 + p3 * x3

# Constraints
prob += b1 * x1 + b2 * x2 + b3 * x3 <= B, "Batter"
prob += m1 * x1 + m2 * x2 + m3 * x3 <= M, "Milk"

# Solve (CBC by default)
solver = pulp.PULP_CBC_CMD(msg=False)
prob.solve(solver)

# Retrieve objective value
obj_val = pulp.value(prob.objective)
# If objective is essentially integer, convert to int for neat output
if obj_val is None:
    value = None
else:
    value = int(round(obj_val)) if abs(obj_val - round(obj_val)) < 1e-6 else obj_val

print(f"FINAL_OBJ: {value}")
