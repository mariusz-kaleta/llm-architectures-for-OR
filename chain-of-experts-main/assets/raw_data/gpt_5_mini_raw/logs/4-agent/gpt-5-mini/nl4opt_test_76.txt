# Self-contained script to solve the MILP by enumeration
# Decision variables: s (small kegs), l (large kegs)
# Objective: maximize Z = 40*s + 100*l
# Constraints:
#   s <= 30
#   l <= 10
#   s >= 2*l
#   s + l <= 25
#   l >= 5
#   s, l are nonnegative integers

best_z = None
best_s = None
best_l = None

for l in range(5, 11):  # l from 5 to 10 inclusive
    s_max = min(30, 25 - l)  # upper bound on s from availability and total kegs
    if s_max < 0:
        continue
    if s_max < 2 * l:
        continue  # infeasible because s must be at least 2*l
    s = s_max  # objective coefficient for s is positive, so take largest feasible s
    z = 40 * s + 100 * l
    if best_z is None or z > best_z:
        best_z = z
        best_s = s
        best_l = l

# Final objective value
print(f"FINAL_OBJ: {best_z}")
