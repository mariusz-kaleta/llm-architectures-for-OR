# Self-contained script to solve the integer optimisation problem by enumeration

# Problem data (initialized directly)
SMALL_CAP = 300   # S ≤ 300
LARGE_CAP = 100   # L ≤ 100
TOTAL_BOTTLES_LIMIT = 200  # S + L ≤ 200
MIN_LARGE = 50    # L ≥ 50

# Coefficients
SMALL_VOLUME = 5
LARGE_VOLUME = 20

best_obj = None
best_S = None
best_L = None

# Enumerate feasible L (0..LARGE_CAP)
for L in range(0, LARGE_CAP + 1):
    if L < MIN_LARGE:
        continue
    # Maximum S allowed by availability and total bottles constraint
    s_max_by_total = TOTAL_BOTTLES_LIMIT - L
    if s_max_by_total < 0:
        continue
    S_max = min(SMALL_CAP, s_max_by_total)
    # Sales preference: S >= 2L
    if S_max < 2 * L:
        continue
    # Since objective coefficient for S is positive, pick S as large as allowed
    S = S_max
    obj = SMALL_VOLUME * S + LARGE_VOLUME * L
    if (best_obj is None) or (obj > best_obj):
        best_obj = obj
        best_S = S
        best_L = L

# Final check that a solution was found
if best_obj is None:
    raise RuntimeError("No feasible solution found.")

# (Do not print any other text on the final line)
print(f"FINAL_OBJ: {best_obj}")
