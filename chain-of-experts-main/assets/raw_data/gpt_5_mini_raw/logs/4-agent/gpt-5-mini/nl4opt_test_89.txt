import math

# Data / parameters
antiox_per_b = 3
antiox_per_s = 1
minerals_per_b = 5
minerals_per_s = 7
sugar_per_b = 5
sugar_per_s = 7

req_antiox = 90
req_minerals = 100

# Safe bounds for enumeration
b_max = math.ceil(req_antiox / antiox_per_b)  # 30

best_b = None
best_s = None
best_sugar = None

for b in range(0, b_max + 1):
    # Minimum s to satisfy antioxidant constraint
    s_min_antiox = max(0, req_antiox - antiox_per_b * b)
    # Minimum s to satisfy mineral constraint (ceiling division)
    numer = req_minerals - minerals_per_b * b
    if numer <= 0:
        s_min_miner = 0
    else:
        s_min_miner = math.ceil(numer / minerals_per_s)
    # Seasonal ratio
    s_min_ratio = 3 * b

    s_min = max(s_min_antiox, s_min_miner, s_min_ratio)

    sugar = sugar_per_b * b + sugar_per_s * s_min

    if (best_sugar is None) or (sugar < best_sugar):
        best_sugar = sugar
        best_b = b
        best_s = s_min

# Optional verification (does not print)
assert best_b is not None
assert 3 * best_b + 1 * best_s >= req_antiox
assert 5 * best_b + 7 * best_s >= req_minerals
assert best_s >= 3 * best_b
assert best_b >= 0 and best_s >= 0
assert best_sugar == sugar_per_b * best_b + sugar_per_s * best_s

# Final required print (exact format)
print(f"FINAL_OBJ: {best_sugar}")
