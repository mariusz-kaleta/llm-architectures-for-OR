#!/usr/bin/env python3
# Self-contained solver for the car production LP described.
# Tries to use PuLP if available; otherwise falls back to direct analytical solution.

try:
    import pulp
except Exception:
    pulp = None

# Problem data
profit_reg = 5000
profit_prem = 8500
cap_reg = 8
cap_prem = 6
total_capacity = 12

final_obj_value = None

if pulp is not None:
    # Build LP
    x1 = pulp.LpVariable("x1", lowBound=0, cat="Continuous")
    x2 = pulp.LpVariable("x2", lowBound=0, cat="Continuous")
    prob = pulp.LpProblem("Car_Profit_Max", pulp.LpMaximize)
    prob += profit_reg * x1 + profit_prem * x2
    prob += x1 <= cap_reg
    prob += x2 <= cap_prem
    prob += x1 + x2 <= total_capacity

    # Solve with default solver
    try:
        prob.solve(pulp.PULP_CBC_CMD(msg=0))
    except Exception:
        # If solver call fails, try generic solve()
        try:
            prob.solve()
        except Exception:
            pass

    # Check solution validity
    try:
        status = pulp.LpStatus[prob.status]
    except Exception:
        status = "Undefined"

    if status == "Optimal":
        x1_val = float(pulp.value(x1))
        x2_val = float(pulp.value(x2))
        obj_val = profit_reg * x1_val + profit_prem * x2_val
        final_obj_value = obj_val
    else:
        # Fallback to analytical solution if solver didn't return optimal
        pulp = None

if pulp is None:
    # Analytical / enumerative fallback
    # Strategy: produce premium up to its demand cap, then fill remaining capacity with regulars (subject to their cap)
    x2_val = min(cap_prem, total_capacity)  # try to produce all premium up to its cap
    remaining_capacity = total_capacity - x2_val
    x1_val = min(cap_reg, remaining_capacity)
    final_obj_value = profit_reg * x1_val + profit_prem * x2_val

# Ensure final_obj_value is a numeric scalar
if final_obj_value is None:
    final_obj_value = 0.0

# Convert to int if effectively integer
final_obj_value = float(final_obj_value)
if abs(final_obj_value - round(final_obj_value)) < 1e-6:
    final_obj_value = int(round(final_obj_value))

print(f"FINAL_OBJ: {final_obj_value}")
