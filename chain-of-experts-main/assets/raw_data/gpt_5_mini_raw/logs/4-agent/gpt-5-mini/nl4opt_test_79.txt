#!/usr/bin/env python3
import math

# Problem data

# x1 = doses of Pain Killer 1

# Constraints:

c1, c2 = 0.8, 0.4
leg1, leg2 = 0.5, 0.7
sleep1, sleep2 = 0.3, 0.6

z_lp = c1 * x1_lp + c2 * x2_lp
leg_lp = leg1 * x1_lp + leg2 * x2_lp
sleep_lp = sleep1 * x1_lp + sleep2 * x2_lp

sleep_bind_lp = abs(sleep_lp - 8.0) <= eps
leg_bind_lp = abs(leg_lp - 4.0) <= eps
