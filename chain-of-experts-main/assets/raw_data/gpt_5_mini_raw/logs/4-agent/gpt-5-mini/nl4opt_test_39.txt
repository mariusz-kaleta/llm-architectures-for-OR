import numpy as np
from scipy.optimize import linprog

# Problem data (primary LP without any worker constraint)
# Decision variables:
#   x[0] = C (chocolate gallons)
#   x[1] = V (vanilla gallons)
#
# Maximize Z = 200*C + 300*V
# Subject to:
#   C + 2V <= 30
#   5 <= C <= 10
#   5 <= V <= 8

# Convert maximization to minimization for linprog: minimize -Z
c = np.array([-200.0, -300.0])

# Inequality constraints A_ub @ x <= b_ub
A_ub = np.array([[1.0, 2.0]])
b_ub = np.array([30.0])

# Bounds for variables: (lower, upper)
bounds = [(5.0, 10.0), (5.0, 8.0)]

res = linprog(c, A_ub=A_ub, b_ub=b_ub, bounds=bounds, method='highs')

if not res.success:
    raise RuntimeError(f"LP solver failed: {res.message}")

C_opt, V_opt = res.x
objective_value = 200.0 * C_opt + 300.0 * V_opt

# Nicely format the objective value: if it's effectively integer, print as int
if abs(objective_value - round(objective_value)) < 1e-9:
    objective_value_print = int(round(objective_value))
else:
    objective_value_print = float(objective_value)

# Optional prints of solution (not the final required line)

print(f"V_opt = {V_opt:.6f}")
