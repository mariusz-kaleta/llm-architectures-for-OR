#!/usr/bin/env python3
# Diet optimization: continuous LP (analytic) and integer MILP (brute-force search)
# Self-contained script initializing data directly and printing final objective value.

from math import isclose

# Problem data
# Decision variables: F (fish meals), C (chicken meals)
# Objective: minimize Z = 7 F + 10 C
# Constraints:
# 10 F + 15 C >= 130   (protein)
# 12 F + 8  C >= 120   (iron)
# C - 2 F >= 0         (preference)
# F, C >= 0
# integrality optional

# 1) Continuous (relaxed) solution (analytically derived)
# According to the model, the optimum occurs at intersection of iron constraint and preference C = 2F:
# 12F + 8(2F) = 120 -> 28F = 120 -> F = 120/28 = 30/7
F_cont = 120.0 / 28.0  # 30/7
C_cont = 2.0 * F_cont
obj_cont = 7.0 * F_cont + 10.0 * C_cont

# Validate constraints
protein_cont = 10.0 * F_cont + 15.0 * C_cont
iron_cont = 12.0 * F_cont + 8.0 * C_cont
pref_cont = C_cont - 2.0 * F_cont  # should be zero

# Print continuous solution (optional informational prints)
print("Continuous (relaxed) solution:")
print(f"  F = {F_cont:.9f}, C = {C_cont:.9f}")
print(f"  Objective (fat) = {obj_cont:.9f}")
print(f"  Protein = {protein_cont:.9f}, Iron = {iron_cont:.9f}, Preference (C-2F) = {pref_cont:.9f}")
print()

# 2) Integer (MILP) solution: brute-force search over reasonable bounds
# Determine reasonable search bounds:
# From protein: if C = 0 -> 10F >= 130 -> F >= 13
# From iron: if C = 0 -> 12F >= 120 -> F >= 10
# From preference C >= 2F, C can be larger; we pick upper bound for F and C conservatively.
F_max = 50
C_max = 50

best_obj = None
best_sol = None

for F in range(0, F_max + 1):
    for C in range(0, C_max + 1):
        # constraints
        if 10 * F + 15 * C < 130:
            continue
        if 12 * F + 8 * C < 120:
            continue
        if C < 2 * F:
            continue
        obj = 7 * F + 10 * C
        if (best_obj is None) or (obj < best_obj):
            best_obj = obj
            best_sol = (F, C)

# If none found (shouldn't happen), raise error
if best_sol is None:
    raise RuntimeError("No integer feasible solution found within the search bounds.")

F_int, C_int = best_sol
obj_int = best_obj
protein_int = 10 * F_int + 15 * C_int
iron_int = 12 * F_int + 8 * C_int

# Print integer solution (optional informational prints)
print("Integer (MILP) optimal solution (by brute-force search):")
print(f"  F = {F_int}, C = {C_int}")
print(f"  Objective (fat) = {obj_int}")
print(f"  Protein = {protein_int}, Iron = {iron_int}")
print()

# Final objective value to print exactly in required format
value = obj_int
print(f"FINAL_OBJ: {value}")
