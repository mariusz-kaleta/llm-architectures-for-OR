#!/usr/bin/env python3
# Self-contained script to solve the meal-cost LP and integer-feasible variant
import numpy as np
from scipy.optimize import linprog

# Data
# Decision variables: x = hamburgers per worker, y = chicken wraps per worker
c = np.array([6.5, 4.0])  # costs
A = np.array([
    [800.0, 450.0],   # calories
    [19.0,  12.0],    # protein
    [20.0,  10.0]     # carbohydrates
])
b = np.array([2200.0, 50.0, 70.0])

# Solve primal LP (continuous)
# Constraints are >=, convert to <= by multiplying by -1: -A x <= -b
A_ub = -A
b_ub = -b
bounds = [(0, None), (0, None)]

primal_res = linprog(c, A_ub=A_ub, b_ub=b_ub, bounds=bounds, method="highs")

if not primal_res.success:
    raise RuntimeError("Primal LP solve failed: " + str(primal_res.message))

x_primal = primal_res.x
z_primal = primal_res.fun

# Identify which primal constraints are binding
residuals = A.dot(x_primal) - b  # should be >= 0; binding when near zero
tol = 1e-7
binding = [abs(r) <= tol for r in residuals]
constraint_names = ["Calories (>=2200)", "Protein (>=50)", "Carbohydrates (>=70)"]

# Solve dual LP to get shadow prices:
# Dual (max): maximize b^T u s.t. A^T u <= c, u >= 0
# Convert to minimization for linprog: minimize -b^T u s.t. A^T u <= c, u >= 0
c_dual = -b
A_ub_dual = A.T.copy()  # A^T u <= c
b_ub_dual = c.copy()
bounds_dual = [(0, None), (0, None), (0, None)]

dual_res = linprog(c_dual, A_ub=A_ub_dual, b_ub=b_ub_dual, bounds=bounds_dual, method="highs")
if not dual_res.success:
    raise RuntimeError("Dual LP solve failed: " + str(dual_res.message))

u_dual = dual_res.x
dual_obj = -dual_res.fun  # maximize b^T u

# Integer-feasible search (small grid search) to find best integer solution
# We pick reasonable upper bounds for enumeration
# Upper bound for hamburgers if only hamburgers meet calories: ceil(2200/800)=3 -> allow up to 10 to be safe
max_search = 20
best_int_cost = float("inf")
best_int_xy = None
for xi in range(0, max_search + 1):
    for yi in range(0, max_search + 1):
        vals = A.dot(np.array([xi, yi]))
        if np.all(vals + 1e-9 >= b):  # feasible
            cost = c.dot(np.array([xi, yi]))
            if cost < best_int_cost - 1e-12:
                best_int_cost = cost
                best_int_xy = (xi, yi)

# Display results (not the final required print line)
print("Continuous LP solution (primal):")
print(f" x = {x_primal[0]:.6f} hamburgers, y = {x_primal[1]:.6f} wraps")
print(f" Minimum cost Z* = {z_primal:.6f}")
print("\nWhich primal constraints bind?")
for name, is_bind, val, rhs in zip(constraint_names, binding, A.dot(x_primal), b):
    status = "Binding" if is_bind else "Slack"
    print(f" - {name}: LHS = {val:.6f}, RHS = {rhs:.6f} -> {status}")

print("\nDual (shadow) prices (u for [calories, protein, carbs]):")
for i, ui in enumerate(u_dual):
    print(f" u_{i+1} = {ui:.6f}")
print(f" Dual objective (max b^T u) = {dual_obj:.6f}")

print("\nInteger-feasible best solution (enumeration):")
if best_int_xy is not None:
    xi, yi = best_int_xy
    print(f" x = {xi} hamburgers, y = {yi} wraps, cost = {best_int_cost:.6f}")
    print(f"  Nutrients: calories = {A[0].dot([xi, yi])}, protein = {A[1].dot([xi, yi])}, carbs = {A[2].dot([xi, yi])}")
else:
    print(" No integer feasible solution found in the search range.")

# Final required print line (must be the very last output)
# We print the continuous LP optimal objective value here.
print(f"FINAL_OBJ: {z_primal}")
