# Self-contained script to solve the desk selection ILP by enumeration

# Parameters
cost_long = 300
area_long = 10
seats_long = 6

cost_short = 100
area_short = 4
seats_short = 2

budget = 2000
area_limit = 200

# Derived bounds
max_L = min(budget // cost_long, area_limit // area_long)  # 6
max_S = min(budget // cost_short, area_limit // area_short)  # 20

feasible_solutions = []

for L in range(max_L + 1):
    for S in range(max_S + 1):
        total_cost = cost_long * L + cost_short * S
        total_area = area_long * L + area_short * S
        if total_cost <= budget and total_area <= area_limit:
            seats = seats_long * L + seats_short * S
            feasible_solutions.append({
                'L': L, 'S': S,
                'cost': total_cost, 'area': total_area,
                'seats': seats, 'units': L + S
            })

# Find maximum seats
max_seats = max(sol['seats'] for sol in feasible_solutions)

# Filter optimal solutions (max seats)
optimal_solutions = [sol for sol in feasible_solutions if sol['seats'] == max_seats]

# Tie-breaker: minimize area, then minimize number of units
optimal_solutions.sort(key=lambda sol: (sol['area'], sol['units']))
best = optimal_solutions[0]

# Print results (informational)
print(f"Best solution: L={best['L']}, S={best['S']}")
print(f"Total seats: {best['seats']}")
print(f"Total cost: ${best['cost']}")
print(f"Total area: {best['area']} sq ft")
print(f"Number of desk units: {best['units']}")

# Final required line (only this exact format on the final line)
value = best['seats']
print(f"FINAL_OBJ: {value}")
