  This is strictly increasing in y, so to minimise Z we should choose the smallest feasible y.

#!/usr/bin/env python3
# Self-contained solver for the MILP described in the prompt.

from math import floor

# Data
ANTIBIOTICS_LIMIT = 35000
ANTIBIOTICS_FIRST = 30
ANTIBIOTICS_SECOND = 65
GEL_FIRST = 20
GEL_SECOND = 60

# Bounds for search
y_min = 40
y_max = ANTIBIOTICS_LIMIT // ANTIBIOTICS_SECOND  # crude upper bound for y

best_obj = None
best_x = None
best_y = None

for y in range(y_min, y_max + 1):
    # x must satisfy precedence x >= y + 1
    x_lower = y + 1
    # antibiotics constraint gives upper bound
    remaining = ANTIBIOTICS_LIMIT - ANTIBIOTICS_SECOND * y
    if remaining < 0:
        continue
    x_upper = remaining // ANTIBIOTICS_FIRST
    if x_lower > x_upper:
        continue
    # To minimize objective 20*x + 60*y, for fixed y choose smallest feasible x (x_lower)
    x = x_lower
    obj = GEL_FIRST * x + GEL_SECOND * y
    if best_obj is None or obj < best_obj:
        best_obj = obj
        best_x = x
        best_y = y

# As a sanity check, ensure we found a feasible solution
if best_obj is None:
    raise RuntimeError("No feasible solution found.")

# (Optional) we could print best_x, best_y but final line must be exactly as required.
# Final required output:
print(f"FINAL_OBJ: {best_obj}")
