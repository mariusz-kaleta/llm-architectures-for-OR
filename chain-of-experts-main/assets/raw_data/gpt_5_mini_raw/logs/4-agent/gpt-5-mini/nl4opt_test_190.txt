#!/usr/bin/env python3
# Self-contained solver for the milk tea LP described.
# Initializes data, evaluates corner points, finds optimum, computes duals if possible,
# and prints final objective value in the required format.

# Data
p_B = 7.5   # profit per black bottle
p_M = 5.0   # profit per matcha bottle

m_B = 600.0  # milk per black bottle (grams)
m_M = 525.0  # milk per matcha bottle (grams)
M_milk = 30000.0  # available milk (grams)

h_B = 10.0   # honey per black bottle (grams)
h_M = 5.0    # honey per matcha bottle (grams)
M_honey = 500.0   # available honey (grams)

# Helper to evaluate objective and check feasibility
def objective(B, M):
    return p_B * B + p_M * M

def feasible(B, M, tol=1e-9):
    if B < -tol or M < -tol:
        return False
    if m_B * B + m_M * M - M_milk > tol:
        return False
    if h_B * B + h_M * M - M_honey > tol:
        return False
    return True

# Corner 1: B = 0
B1 = 0.0
M1 = min(M_milk / m_M, M_honey / h_M)
if not feasible(B1, M1):
    # If numerical issues occur, adjust
    M1 = max(0.0, min(M_milk / m_M, M_honey / h_M))

Z1 = objective(B1, M1)

# Corner 2: M = 0
M2 = 0.0
B2 = min(M_milk / m_B, M_honey / h_B)
if not feasible(B2, M2):
    B2 = max(0.0, min(M_milk / m_B, M_honey / h_B))

Z2 = objective(B2, M2)

# Corner 3: intersection of the two resource equations (solve 2x2)
# Solve:
# m_B * B + m_M * M = M_milk
# h_B * B + h_M * M = M_honey
det = m_B * h_M - h_B * m_M
B3 = None
M3 = None
Z3 = None
if abs(det) > 1e-12:
    B3 = (M_milk * h_M - M_honey * m_M) / det
    M3 = (m_B * M_honey - h_B * M_milk) / det
    if not feasible(B3, M3):
        # If intersection is outside feasible region, ignore
        B3 = None
        M3 = None
        Z3 = None
    else:
        Z3 = objective(B3, M3)

# Collect feasible corners
candidates = [
    (B1, M1, Z1),
    (B2, M2, Z2)
]
if B3 is not None:
    candidates.append((B3, M3, Z3))

# Choose best
best = max(candidates, key=lambda t: t[2])
B_opt, M_opt, Z_opt = best

# Compute duals (shadow prices) if both constraints are binding (within tolerance)
tol = 1e-8
milk_slack = M_milk - (m_B * B_opt + m_M * M_opt)
honey_slack = M_honey - (h_B * B_opt + h_M * M_opt)

y_milk = None
y_honey = None
dual_computed = False
if abs(milk_slack) <= tol and abs(honey_slack) <= tol:
    # Solve A^T y = c where A^T = [[m_B, h_B], [m_M, h_M]]
    # i.e. m_B*y_milk + h_B*y_honey = p_B
    #       m_M*y_milk + h_M*y_honey = p_M
    det_dual = m_B * h_M - m_M * h_B
    if abs(det_dual) > 1e-12:
        y_milk = (p_B * h_M - p_M * h_B) / det_dual
        y_honey = (m_B * p_M - m_M * p_B) / det_dual
        dual_computed = True

# Output results (informational)
print("Optimal (B, M):", (B_opt, M_opt))
print("Maximum profit Z:", Z_opt)
if dual_computed:
    print("Shadow price y_milk  (per gram):", y_milk)
    print("Shadow price y_honey (per gram):", y_honey)
else:
    print("Shadow prices not computed (constraints not both binding or singular).")

# Final required print (must be the very last line in this format)
# If objective is effectively integer, print as integer for cleanliness
if abs(Z_opt - round(Z_opt)) < 1e-9:
    Z_to_print = int(round(Z_opt))
else:
    Z_to_print = float(Z_opt)

print(f"FINAL_OBJ: {Z_to_print}")
