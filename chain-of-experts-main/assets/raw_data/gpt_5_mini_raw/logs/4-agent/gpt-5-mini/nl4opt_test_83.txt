#!/usr/bin/env python3
# Self-contained script to solve the MILP under multiple interpretations of the 33% rule.
# Uses PuLP (CBC solver included with PuLP).
# Initializes data in code, solves 4 variants, prints results, and finally prints FINAL_OBJ: <value>

import pulp

def solve_case(name, linear_constraints):
    # Create MILP
    prob = pulp.LpProblem(name, pulp.LpMaximize)
    # Decision variables
    R = pulp.LpVariable('R', lowBound=4, cat='Integer')   # runner trips, integer, >=4
    C = pulp.LpVariable('C', lowBound=0, cat='Integer')   # canoe trips, integer, >=0

    # Objective: maximize 3R + 10C
    prob += 3 * R + 10 * C, "TotalBags"

    # Hours constraint: 4R + 2C <= 200
    prob += 4 * R + 2 * C <= 200, "Hours"

    # Add any additional linear constraints passed in (each is a tuple coeffs and rhs form)
    # Each constraint is expressed as (a_R, a_C, sense, rhs) where sense is '<=' or '>=' or '=='
    for (aR, aC, sense, rhs, cname) in linear_constraints:
        if sense == "<=":
            prob += aR * R + aC * C <= rhs, cname
        elif sense == ">=":
            prob += aR * R + aC * C >= rhs, cname
        elif sense == "==":
            prob += aR * R + aC * C == rhs, cname
        else:
            raise ValueError("Unsupported sense in constraint")

    # Solve with default CBC solver
    solver = pulp.PULP_CBC_CMD(msg=False)
    result = prob.solve(solver)

    status = pulp.LpStatus.get(prob.status, "Unknown")
    R_val = int(pulp.value(R))
    C_val = int(pulp.value(C))
    obj = int(pulp.value(prob.objective))
    hours_used = 4 * R_val + 2 * C_val
    return {
        "name": name,
        "status": status,
        "R": R_val,
        "C": C_val,
        "hours": hours_used,
        "obj": obj,
        "prob": prob
    }

def main():
    cases = []

    # 1) Trips-share interpretation using decimal 0.33:
    # Constraint: C <= 0.33*(R + C) -> 0.67*C <= 0.33*R -> 0.67*C - 0.33*R <= 0
    cases.append((
        "Trips-share (0.33 decimal)",
        [(-0.33, 0.67, "<=", 0, "TripsShare_0.33")]  # rearranged to: -0.33*R + 0.67*C <= 0
        # Note: stored as (aR, aC, sense, rhs, name)
    ))

    # 1b) Trips-share exact 1/3:
    # Constraint: C <= (1/3)*(R + C) -> 2C <= R -> 2*C - R <= 0
    cases.append((
        "Trips-share (1/3 exact)",
        [(-1.0, 2.0, "<=", 0, "TripsShare_1div3")]  # -1*R + 2*C <= 0  <=> 2C <= R
    ))

    # 2) Mail-share interpretation using decimal 0.33:
    # Constraint: 10C <= 0.33*(3R + 10C) -> 6.7*C <= 0.99*R -> 6.7*C - 0.99*R <= 0
    cases.append((
        "Mail-share (0.33 decimal)",
        [(-0.99, 6.7, "<=", 0, "MailShare_0.33")]  # -0.99*R + 6.7*C <= 0
    ))

    # 2b) Mail-share exact 1/3:
    # Constraint: 10C <= (1/3)*(3R +10C) -> 20C <= 3R -> 20*C - 3*R <= 0
    cases.append((
        "Mail-share (1/3 exact)",
        [(-3.0, 20.0, "<=", 0, "MailShare_1div3")]  # -3*R + 20*C <= 0  <=> 20C <= 3R
    ))

    results = []
    for name, constrs in cases:
        res = solve_case(name, constrs)
        results.append(res)

    # Print summary of each case
    for r in results:
        print(f"Case: {r['name']}")
        print(f" Status: {r['status']}")
        print(f" R (runner trips): {r['R']}")
        print(f" C (canoe trips): {r['C']}")
        print(f" Hours used: {r['hours']} / 200")
        print(f" Total bags delivered (objective): {r['obj']}")
        print("")

    # Determine the maximum objective among all cases
    max_obj = max(r['obj'] for r in results)

    # FINAL required output line (must be the last line)
    print(f"FINAL_OBJ: {max_obj}")

if __name__ == "__main__":
    main()
