import numpy as np

# Problem data
# Decision variables: R, T
c = np.array([8.0, 10.0])  # objective coefficients (maximize)
# Constraints: a*R + b*T <= rhs
constraints = [
    (3.0, 5.0, 300.0),  # heating
    (5.0, 8.0, 300.0),  # cooling
    (1.0, 0.0, 0.0),    # R >= 0  represented as R >= 0 -> -R <= 0 but we'll use axis as equality for corner generation
    (0.0, 1.0, 0.0)     # T >= 0
]

# We'll generate candidate corner points by intersecting pairs of the equality versions of these constraints:
eqs = constraints  # treat each as equality for intersection computation

tol = 1e-9
candidates = []

# Solve pairwise intersections
n = len(eqs)
for i in range(n):
    a1, b1, r1 = eqs[i]
    for j in range(i + 1, n):
        a2, b2, r2 = eqs[j]
        A = np.array([[a1, b1], [a2, b2]])
        rhs = np.array([r1, r2])
        det = np.linalg.det(A)
        if abs(det) < tol:
            continue  # parallel or singular, skip
        sol = np.linalg.solve(A, rhs)
        R, T = sol[0], sol[1]
        # Check nonnegativity with tolerance
        if R < -1e-8 or T < -1e-8:
            continue
        # Check all inequality constraints (a*R + b*T <= rhs + tol)
        feasible = True
        for (a, b, r) in constraints[:2]:  # only the two <= constraints matter; axes already handled
            if a * R + b * T - r > 1e-8:
                feasible = False
                break
        if feasible:
            candidates.append((R, T))

# Also include the obvious axis corner points (0,0), (R-axis bound), (T-axis bound)
candidates.append((0.0, 0.0))

# R-axis bound: T = 0, find max R s.t. constraints hold
T = 0.0
R_bound = min(300.0 / 3.0, 300.0 / 5.0)  # derived from 3R <=300 and 5R <=300
candidates.append((R_bound, 0.0))

# T-axis bound: R = 0
R = 0.0
T_bound = min(300.0 / 5.0, 300.0 / 8.0)  # from 5T <=300 and 8T <=300
candidates.append((0.0, T_bound))

# Evaluate objective for all unique feasible candidates
unique_candidates = []
seen = set()
for R, T in candidates:
    # round small numerical noise
    Rr = round(R, 12)
    Tr = round(T, 12)
    if (Rr, Tr) in seen:
        continue
    seen.add((Rr, Tr))
    unique_candidates.append((Rr, Tr))

best_val = -np.inf
best_pt = None
for R, T in unique_candidates:
    # Verify feasibility strictly
    if 3*R + 5*T - 300 > 1e-8:
        continue
    if 5*R + 8*T - 300 > 1e-8:
        continue
    if R < -1e-8 or T < -1e-8:
        continue
    val = c[0]*R + c[1]*T
    if val > best_val + 1e-12:
        best_val = val
        best_pt = (R, T)

# Prepare final objective value (integer if effectively integer)
if abs(round(best_val) - best_val) < 1e-8:
    final_value = int(round(best_val))
else:
    final_value = float(best_val)

print(f"FINAL_OBJ: {final_value}")
