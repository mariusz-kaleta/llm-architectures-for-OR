import math

# Problem data
M = 3000          # available morphine (mg)
min_P = 50        # minimum painkiller production
# Mix constraint: S >= (7/3) * P
# Morphine constraint: 10P + 6S <= M
# Objective: minimize D = 3P + 5S
best_obj = None
best_P = None
best_S = None

# Search over feasible integer P. Upper bound: morphine alone gives P <= M//10
for P in range(min_P, M // 10 + 1):
    # minimal integer S satisfying S >= 7/3 * P
    S_min = (7 * P + 2) // 3  # ceil(7P/3)
    morphine_used = 10 * P + 6 * S_min
    if morphine_used <= M:
        D = 3 * P + 5 * S_min
        if (best_obj is None) or (D < best_obj):
            best_obj = D
            best_P = P
            best_S = S_min

# Compute derived values
morphine_used = 10 * best_P + 6 * best_S
morphine_remaining = M - morphine_used

# Print results (informational)
print(f"Optimal integer solution found:")
print(f"P (painkillers) = {best_P}")
print(f"S (sleeping pills) = {best_S}")
print(f"Digestive medicine used = {best_obj}")
print(f"Morphine used = {morphine_used}")
print(f"Morphine remaining = {morphine_remaining}")

print(f"FINAL_OBJ: {best_obj}")
