# Parameters
cool_coeff_L = 12
cool_coeff_H = 17
cool_target = 250

elec_coeff_L = 150
elec_coeff_H = 250
elec_budget = 3400

# Proportion constraint L <= 0.3 (L+H) linearised -> 7L <= 3H
# Lower bound on H
H_min = 7

# Search bounds for enumeration (sufficiently large to find feasible/relaxed solutions)
H_upper = 200
L_upper = 500

def feasible(L, H, elec_cap, enforce_proportion=True):
    if L < 0 or H < 0:
        return False
    if H < H_min:
        return False
    if cool_coeff_L * L + cool_coeff_H * H < cool_target:
        return False
    if elec_coeff_L * L + elec_coeff_H * H > elec_cap:
        return False
    if enforce_proportion:
        # 7L <= 3H
        if 7 * L > 3 * H:
            return False
    return True

# 1) Check feasibility of the original ILP
feasible_solutions = []
for H in range(H_min, H_upper + 1):
