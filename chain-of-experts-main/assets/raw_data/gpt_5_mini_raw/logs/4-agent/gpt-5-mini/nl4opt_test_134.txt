import math

# Enumerate integer solutions for s (sedans) and b (buses)
# Decision variables: s, b >= 0 integers
# Constraints:
#   50*s + 250*b >= 4600   (seating)
#   10*s + 40*b <= 800    (pollution)
# Useful bounds: b in [0,20], s in [0,80]

best_obj = None
best_solution = None

for b in range(0, 21):        # b <= 20 from pollution cap
    for s in range(0, 81):    # s <= 80 from pollution cap
        if 50*s + 250*b >= 4600 and 10*s + 40*b <= 800:
            obj = s + b
            if best_obj is None or obj < best_obj:
                best_obj = obj
                best_solution = (s, b)

# At this point best_obj should hold the minimum total vehicles
if best_obj is None:
    # No feasible solution found (shouldn't happen for this model)
    final_value = -1
else:
    final_value = int(best_obj)

print(f"FINAL_OBJ: {final_value}")
