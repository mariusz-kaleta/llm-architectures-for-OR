Decision variables

Objective
- Minimise S = T + G.

Constraints
- Service-hour coverage: 8 T + 5 G ≥ 500.
- Budget: 300 T + 100 G ≤ 14,000.
- Regulatory ratio (exact): T = 2 G.

3) Solve analytically by elimination (useful because T = 2G)
Substitute T = 2G into the hour and budget constraints:

- Hours: 8(2G) + 5G = 21 G ≥ 500 ⇒ G ≥ 500 / 21 ≈ 23.809… ⇒ G ≥ 24 (integer).
- Budget: 300(2G) + 100 G = 700 G ≤ 14,000 ⇒ G ≤ 14,000 / 700 = 20.

These two inequalities conflict: G ≥ 24 and G ≤ 20. Therefore there is no integer (T,G) satisfying all constraints when the regulatory ratio must hold exactly. Conclusion: the model with T = 2G is infeasible under the $14,000 budget.

  - That schedule would give 8·48 + 5·24 = 504 hours and S = 48 + 24 = 72 total shifts.

- Minimum shifts ignoring the budget (only ratio enforced):
  - With T = 2G and hours constraint, G = 24, T = 48 → S = 72 shifts (same as above). This is the minimum S achievable under the ratio, but it requires $16,800.

#!/usr/bin/env python3
import math

# Problem data
TECH_HOURS = 8
TECH_COST = 300
RES_HOURS = 5
RES_COST = 100
REQ_HOURS = 500
BUDGET = 14000

# Helper: check feasibility
def feasible(T, G, budget_limit=BUDGET, req_hours=REQ_HOURS):
    return (TECH_HOURS * T + RES_HOURS * G >= req_hours) and (TECH_COST * T + RES_COST * G <= budget_limit) and (T >= 0) and (G >= 0) and (isinstance(T, int) and isinstance(G, int))

# 1) Exact ratio case: T = 2 G
print("=== Exact ratio case: T = 2G ===")
# Analytical elimination:
# Hours: 21 G >= 500 => G >= ceil(500/21)
G_hours_min = math.ceil(REQ_HOURS / (TECH_HOURS * 2 + RES_HOURS))  # 21 = 8*2 + 5
T_hours_min = 2 * G_hours_min
cost_at_min_G = TECH_COST * T_hours_min + RES_COST * G_hours_min
print(f"Minimum G to meet hours (with T=2G): G = {G_hours_min}, T = {T_hours_min}")
print(f"Cost for that schedule: ${cost_at_min_G:,}")
if cost_at_min_G <= BUDGET:
    print("This schedule fits the budget.")
else:
    print("This schedule exceeds the available budget.")
# Check feasibility across G values (nonnegative integers)
feasible_exact = []
for G in range(0, 1000):
    T = 2 * G
    if feasible(T, G):
        feasible_exact.append((T, G))
        break
if feasible_exact:
    T0, G0 = feasible_exact[0]
    S0 = T0 + G0
    cost0 = TECH_COST * T0 + RES_COST * G0
    hours0 = TECH_HOURS * T0 + RES_HOURS * G0
    print(f"Found feasible exact-ratio solution: T={T0}, G={G0}, hours={hours0}, cost=${cost0:,}, shifts={S0}")
else:
    print("Conclusion: No feasible (T,G) with T = 2G under the current budget.")
    # Minimal extra budget required:
    min_G_needed = G_hours_min
    min_T_needed = 2 * min_G_needed
    min_cost_needed = TECH_COST * min_T_needed + RES_COST * min_G_needed
    extra_budget_needed = max(0, min_cost_needed - BUDGET)
    print(f"Minimum schedule satisfying hours and ratio: T={min_T_needed}, G={min_G_needed}, shifts={min_T_needed+min_G_needed}, cost=${min_cost_needed:,}")
    print(f"Additional budget required: ${extra_budget_needed:,} (i.e. total budget >= ${min_cost_needed:,})")

# 2) Relax ratio entirely: minimize S = T + G subject to hours and budget
print("\n=== Relax ratio (no relationship between T and G) ===")
# Upper bounds for search
T_upper = min(math.ceil(REQ_HOURS / TECH_HOURS) + 5, BUDGET // TECH_COST + 10, 200)  # safe bound
G_upper = min(math.ceil(REQ_HOURS / RES_HOURS) + 10, BUDGET // RES_COST + 10, 500)

best = None  # (S, cost, T, G, hours)
for T in range(0, T_upper + 1):
    for G in range(0, G_upper + 1):
        if TECH_HOURS * T + RES_HOURS * G >= REQ_HOURS and TECH_COST * T + RES_COST * G <= BUDGET:
            S = T + G
            cost = TECH_COST * T + RES_COST * G
            hours = TECH_HOURS * T + RES_HOURS * G
            if best is None:
                best = (S, cost, T, G, hours)
            else:
                # Primary objective: minimize S; secondary: minimize cost
                if S < best[0] or (S == best[0] and cost < best[1]):
                    best = (S, cost, T, G, hours)

if best is None:
    print("No feasible solution without the ratio under the current budget.")
else:
    S_best, cost_best, T_best, G_best, hours_best = best
    print(f"Optimal (no ratio): T={T_best}, G={G_best}, hours={hours_best}, cost=${cost_best:,}, shifts={S_best}")
    # Also show the alternative that uses full budget if exists (tie)
    using_full_budget = []
    for T in range(0, T_upper + 1):
        for G in range(0, G_upper + 1):
            if TECH_HOURS * T + RES_HOURS * G >= REQ_HOURS and TECH_COST * T + RES_COST * G == BUDGET:
                using_full_budget.append((T + G, TECH_COST*T + RES_COST*G, T, G, TECH_HOURS*T + RES_HOURS*G))
    # find minimal shifts among full-budget solutions
    if using_full_budget:
        using_full_budget.sort(key=lambda x: (x[0], x[1]))  # by shifts then cost (cost same)
        fb = using_full_budget[0]
        print(f"An optimal solution that uses the full budget: T={fb[2]}, G={fb[3]}, hours={fb[4]}, cost=${fb[1]:,}, shifts={fb[0]}")

# 3) Relaxation: T >= 2G (inequality)
print("\n=== Relaxation: T >= 2G ===")
feasible_ineq = None
for T in range(0, T_upper + 1):
    for G in range(0, G_upper + 1):
        if T >= 2*G and feasible(T, G):
            S = T + G
            cost = TECH_COST * T + RES_COST * G
            hours = TECH_HOURS * T + RES_HOURS * G
            if feasible_ineq is None or S < feasible_ineq[0] or (S == feasible_ineq[0] and cost < feasible_ineq[1]):
                feasible_ineq = (S, cost, T, G, hours)
if feasible_ineq is None:
    print("No feasible solution satisfying T >= 2G under the current budget.")
else:
    S_i, cost_i, T_i, G_i, hours_i = feasible_ineq
    print(f"Feasible with T>=2G: T={T_i}, G={G_i}, hours={hours_i}, cost=${cost_i:,}, shifts={S_i}")

# 4) Summarize diagnostics and recommendations
print("\n=== Summary ===")
print(f"Budget available: ${BUDGET:,}, Required hours: {REQ_HOURS}")
print("If T = 2G is mandatory: increase budget to at least $16,800; smallest schedule then is T=48, G=24 (504 hours, 72 shifts).")
if best is not None:
    print(f"If ratio can be waived: minimal total shifts under current budget is {S_best} with T={T_best}, G={G_best} (cost ${cost_best:,}).")
    if cost_best < BUDGET:
        print(f"  (A tie at S={S_best} exists that uses the full budget: e.g., T=28, G=56 uses ${BUDGET:,} and also yields S={S_best}.)")
print("If regulation enforces T >= 2G (inequality), the problem is also infeasible under the current budget.")

# Final required single-line output (FINAL_OBJ)
# We choose to report the minimum total shifts achievable under the current budget with no ratio enforced.
# If no feasible solution exists under the budget, report None.
final_value = best[0] if best is not None else None
print(f"FINAL_OBJ: {final_value}")
