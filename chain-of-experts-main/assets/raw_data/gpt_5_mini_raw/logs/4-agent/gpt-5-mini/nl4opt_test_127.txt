[CODER]:
import math

# Parameters
R = 3000          # total resin available
rM = 20           # resin per molar
rC = 15           # resin per canine
pM = 3.0          # pain per molar
pC = 2.3          # pain per canine
M_min = 45        # minimum molars
alpha = 0.60      # minimum fraction of canines

# Derived ratio constraint: C >= r_ratio * M where r_ratio = alpha / (1 - alpha)
r_ratio = alpha / (1.0 - alpha)  # 0.6 / 0.4 = 1.5

best_obj = None
best_M = None
best_C = None

# Upper bound for M: from resin alone M <= R / rM, but proportion may be tighter.
M_upper = R // rM  # coarse integer upper bound

for M in range(M_min, M_upper + 1):
    C_min = math.ceil(r_ratio * M)  # minimal integer C satisfying proportion
    # Check resin constraint
    if rM * M + rC * C_min <= R:
        obj = pM * M + pC * C_min
        if (best_obj is None) or (obj < best_obj - 1e-9):
            best_obj = obj
            best_M = M
            best_C = C_min

# If no feasible solution found (should not happen for given data), raise error
if best_obj is None:
    raise ValueError("No feasible solution found with given parameters.")

# (Optional) internal checks -- not printed as final line
# Validate constraints
assert best_M >= M_min
assert best_C >= math.ceil(r_ratio * best_M)
assert rM * best_M + rC * best_C <= R

# Final required output line
print(f"FINAL_OBJ: {best_obj}")
