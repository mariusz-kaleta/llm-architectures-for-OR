import math

# Data (per serving)
cal_alm = 200
prot_alm = 20
fat_alm = 15

cal_cas = 300
prot_cas = 25
fat_cas = 12

# Requirements
req_cal = 10000
req_prot = 800

# --- LP relaxation analytic solution (intersection of A = 2C and calories binding) ---
# A = 2C
# 200A + 300C = 10000 -> 200*(2C) + 300C = 700C = 10000 -> C = 10000/700
C_lp = 10000.0 / 700.0
A_lp = 2.0 * C_lp
F_lp = fat_alm * A_lp + fat_cas * C_lp

# Print LP relaxation results
print("LP relaxation (analytic):")
print(f"  A = {A_lp:.6f}, C = {C_lp:.6f}, Fat = {F_lp:.6f}")

# --- MILP via enumeration on integer C ---
# For any integer C, minimal feasible A is:
# A_min(C) = max{ ceil((10000 − 300 C) / 200),
#                 ceil((800 − 25 C) / 20),
#                 ceil(2 C),
#                 0 }.

def A_min_for_C(C):
    # compute each lower bound; use float then ceil
    b_cal = math.ceil((req_cal - cal_cas * C) / cal_alm) if (req_cal - cal_cas * C) != 0 else 0
    # handle exact zero numerator: ceil(0)=0, but math.ceil(0) returns 0 anyway; kept for clarity
    b_prot = math.ceil((req_prot - prot_cas * C) / prot_alm) if (req_prot - prot_cas * C) != 0 else 0
    b_ratio = math.ceil(2 * C)
    # If the numerator is negative, math.ceil gives the correct integer (possibly negative), we take max with 0
    return max(b_cal, b_prot, b_ratio, 0)

best_obj = None
best_pair = None
best_A = None

# Reasonable search range for C: 0..1000 (enough to include any practical optimum)
for C in range(0, 1001):
    A_required = A_min_for_C(C)
    # Check feasibility explicitly (though A_required enforces constraints)
    calories = cal_alm * A_required + cal_cas * C
    protein = prot_alm * A_required + prot_cas * C
    if calories >= req_cal and protein >= req_prot and A_required >= 2 * C:
        obj = fat_alm * A_required + fat_cas * C
        if (best_obj is None) or (obj < best_obj):
            best_obj = obj
            best_pair = (A_required, C)
            best_A = A_required

# If nothing found (shouldn't happen), raise error
if best_obj is None:
    raise RuntimeError("No feasible integer solution found in the search range.")

# Compute achieved totals for best solution
A_opt, C_opt = best_pair
cal_total = cal_alm * A_opt + cal_cas * C_opt
prot_total = prot_alm * A_opt + prot_cas * C_opt
fat_total = fat_alm * A_opt + fat_cas * C_opt

# Print MILP results
print("\nMILP (enumeration) optimal integer solution:")
print(f"  Almonds A = {A_opt}, Cashews C = {C_opt}")
print(f"  Calories = {cal_total}, Protein = {prot_total}, Fat = {fat_total}")
