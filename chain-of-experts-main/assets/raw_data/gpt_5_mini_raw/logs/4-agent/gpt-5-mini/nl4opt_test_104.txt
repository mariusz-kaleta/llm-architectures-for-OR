#!/usr/bin/env python3
import math

# Problem data
capacity_required = 400
cap_L = 12
cap_B = 18

# Upper bounds from problem description
B_max = math.ceil(capacity_required / cap_B)  # 23
L_max = math.ceil(capacity_required / cap_L)  # 34

best_T = None
best_L = None
best_B = None

for B in range(0, B_max + 1):
    # Minimum L to satisfy capacity: ceil((400 - 18B) / 12), but not less than 0
    remaining = capacity_required - cap_B * B
    if remaining <= 0:
        L_capacity = 0
    else:
        L_capacity = math.ceil(remaining / cap_L)
    # Minimum L to satisfy proportion: 3L >= 7B -> L >= ceil(7B / 3)
    L_proportion = (7 * B + 2) // 3  # integer ceil
    L = max(L_capacity, L_proportion)
    # Ensure non-negative integer
    if L < 0:
        L = 0

    # Verify constraints (safety)
    if 12 * L + 18 * B < capacity_required:
        continue
    if 3 * L - 7 * B < 0:
        continue

    T = L + B

    if best_T is None:
        update = True
    else:
        # Prefer smaller total T. On ties prefer larger L (more limousines).
        if T < best_T:
            update = True
        elif T == best_T and L > best_L:
            update = True
        else:
            update = False

    if update:
        best_T = T
        best_L = L
        best_B = B

# Final check: there should be a solution
if best_T is None:
    raise RuntimeError("No feasible solution found.")

# Optional informative print (not the required final line)
print(f"Optimal solution found: L={best_L}, B={best_B}, total vehicles={best_T}")

# Required final line exactly in this format
print(f"FINAL_OBJ: {best_T}")
