#!/usr/bin/env python3
# Self-contained script to solve the catalyst deployment LP and integer version.
# Data initialized directly in code. No external solver required (small 2-variable problem).
from math import floor, isclose

# Problem data
# Decision variables: P (palladium-heavy), Q (platinum-heavy)
c = (5.0, 4.0)  # objective coefficients (maximize)
# Constraints A x <= b
# 15P + 20Q <= 450  (platinum supply)
# 25P + 14Q <= 390  (palladium supply)
A = ((15.0, 20.0),
     (25.0, 14.0))
b = (450.0, 390.0)

# Helper: solve 2x2 linear system M x = rhs by Cramer's rule
def solve_2x2(a11, a12, a21, a22, b1, b2):
    det = a11 * a22 - a12 * a21
    if isclose(det, 0.0):
        raise ValueError("Singular matrix in 2x2 solver")
    x = (b1 * a22 - a12 * b2) / det
    y = (a11 * b2 - b1 * a21) / det
    return x, y

# 1) Solve LP relaxation by enumerating vertices (axes intersections + intersection of both constraints)
# Vertices candidates:
vertices = []

# Intersection of two constraints (both binding)
P_int, Q_int = solve_2x2(A[0][0], A[0][1], A[1][0], A[1][1], b[0], b[1])
vertices.append((P_int, Q_int))

# Intersections with axes:
# Constraint1 with P=0 -> Q = b1 / a12
if A[0][1] != 0:
    vertices.append((0.0, b[0] / A[0][1]))
# Constraint1 with Q=0 -> P = b1 / a11
if A[0][0] != 0:
    vertices.append((b[0] / A[0][0], 0.0))
# Constraint2 with P=0 -> Q = b2 / a22
if A[1][1] != 0:
    vertices.append((0.0, b[1] / A[1][1]))
# Constraint2 with Q=0 -> P = b2 / a21
if A[1][0] != 0:
    vertices.append((b[1] / A[1][0], 0.0))

# Also origin
vertices.append((0.0, 0.0))

# Filter feasible vertices (satisfy both constraints and nonnegativity)
feasible_vertices = []
for (P, Q) in vertices:
    if P < -1e-9 or Q < -1e-9:
        continue
    c1 = A[0][0] * P + A[0][1] * Q
    c2 = A[1][0] * P + A[1][1] * Q
    if c1 <= b[0] + 1e-9 and c2 <= b[1] + 1e-9:
        feasible_vertices.append((P, Q))

# Evaluate objective at feasible vertices to get LP optimum
best_lp_val = None
best_lp_point = None
for (P, Q) in feasible_vertices:
    val = c[0] * P + c[1] * Q
    if best_lp_val is None or val > best_lp_val + 1e-12:
        best_lp_val = val
        best_lp_point = (P, Q)

# Compute slacks for LP solution
P_lp, Q_lp = best_lp_point
slack1_lp = b[0] - (A[0][0] * P_lp + A[0][1] * Q_lp)  # platinum slack
slack2_lp = b[1] - (A[1][0] * P_lp + A[1][1] * Q_lp)  # palladium slack

# Compute dual/shadow prices for LP (since both constraints bind at the optimal vertex with positive variables)
# Solve A^T y = c for y (y1: platinum price, y2: palladium price)
y1, y2 = solve_2x2(A[0][0], A[1][0], A[0][1], A[1][1], c[0], c[1])
# Note: solve_2x2 used arguments as a11,a12,a21,a22,b1,b2. We passed A^T and c.

# 2) Solve integer MIP by brute force enumeration (small bounds)
# Compute reasonable bounds for enumeration
P_max = int(min(floor(b[0] / A[0][0]) if A[0][0] > 0 else float('inf'),
                floor(b[1] / A[1][0]) if A[1][0] > 0 else float('inf')))
Q_max = int(min(floor(b[0] / A[0][1]) if A[0][1] > 0 else float('inf'),
                floor(b[1] / A[1][1]) if A[1][1] > 0 else float('inf')))

# Safety bounds if anything odd
P_max = max(P_max, 0)
Q_max = max(Q_max, 0)

best_int_val = None
best_int_point = None

for P in range(P_max + 1):
    # For each P, compute max Q allowed by both constraints
    # From constraint1: 20Q <= 450 - 15P => Q <= (450 - 15P)/20
    maxQ1 = (b[0] - A[0][0] * P) / A[0][1] if A[0][1] > 0 else float('inf')
    # From constraint2: 14Q <= 390 - 25P => Q <= (390 - 25P)/14
    maxQ2 = (b[1] - A[1][0] * P) / A[1][1] if A[1][1] > 0 else float('inf')
    maxQ = int(min(maxQ1, maxQ2, Q_max))
    if maxQ < 0:
        continue
    for Q in range(maxQ + 1):
        c1 = A[0][0] * P + A[0][1] * Q
        c2 = A[1][0] * P + A[1][1] * Q
        if c1 <= b[0] + 1e-9 and c2 <= b[1] + 1e-9:
            val = c[0] * P + c[1] * Q
            if best_int_val is None or val > best_int_val:
                best_int_val = val
                best_int_point = (P, Q)

# Prepare outputs
print("LP relaxation solution (continuous variables):")
print(f" P = {P_lp:.12f}")
print(f" Q = {Q_lp:.12f}")
print(f" Objective (LP) = {best_lp_val:.12f}")
print(" Resource usage (LP):")
used_plat_lp = A[0][0] * P_lp + A[0][1] * Q_lp
used_pall_lp = A[1][0] * P_lp + A[1][1] * Q_lp
print(f"  Platinum used = {used_plat_lp:.12f} (available {b[0]:.0f}), slack = {slack1_lp:.12f}")
print(f"  Palladium used = {used_pall_lp:.12f} (available {b[1]:.0f}), slack = {slack2_lp:.12f}")
print(" Dual / shadow prices (LP):")
print(f"  y_platinum = {y1:.12f} (increase in objective per extra unit of platinum)")
print(f"  y_palladium = {y2:.12f} (increase in objective per extra unit of palladium)")
print()
print("Integer (MIP) optimal solution by enumeration:")
P_int_opt, Q_int_opt = best_int_point
print(f" P = {P_int_opt}")
print(f" Q = {Q_int_opt}")
print(f" Objective (integer) = {best_int_val}")
used_plat_int = A[0][0] * P_int_opt + A[0][1] * Q_int_opt
used_pall_int = A[1][0] * P_int_opt + A[1][1] * Q_int_opt
slack1_int = b[0] - used_plat_int
slack2_int = b[1] - used_pall_int
print(f" Resource usage (integer):")
print(f"  Platinum used = {used_plat_int:.0f} (available {b[0]:.0f}), slack = {slack1_int:.0f}")
print(f"  Palladium used = {used_pall_int:.0f} (available {b[1]:.0f}), slack = {slack2_int:.0f}")

# Final required print line (final objective value). We choose the integer optimal objective as the final reported value.
final_obj = int(best_int_val) if isclose(best_int_val, round(best_int_val)) else float(best_int_val)
print(f"FINAL_OBJ: {final_obj}")
