  prob = LpProblem("DeliveryShifts", LpMaximize)
  B = LpVariable("B", lowBound=0, cat="Integer")
  S = LpVariable("S", lowBound=5, cat="Integer")   # enforce S >= 5 directly
  # Objective
  prob += 50*B + 43*S
  # Constraints
  prob += B + S <= 40
  prob += 5*B + 6*S <= 230
  prob += 10*B + 7*S >= 320

  prob.solve(PULP_CBC_CMD(msg=False))
  print("Status:", LpStatus[prob.status])
  print("B =", int(B.value()), "S =", int(S.value()))
  tips = 50*B.value() + 43*S.value()

#!/usr/bin/env python3
# Brute-force solver for the Delivery Shifts ILP described by the user.
# This script is self-contained, does not require external solvers, and
# enumerates integer feasible solutions to find the optimal schedule.

def main():
    # Problem data (directly initialised)
    SHIFT_LIMIT = 40        # B + S <= 40
    ENERGY_LIMIT = 230      # 5*B + 6*S <= 230
    ORDER_REQUIRE = 320     # 10*B + 7*S >= 320
    MIN_SCOOTER = 5         # S >= 5

    # Coefficients
    TIP_B = 50
    TIP_S = 43
    ENERGY_B = 5
    ENERGY_S = 6
    ORDERS_B = 10
    ORDERS_S = 7

    best_obj = None
    best_solutions = []

    # Enumerate possible integer B and S values
    for B in range(0, SHIFT_LIMIT + 1):
        # S must be at least MIN_SCOOTER and also satisfy shift limit
        s_min = max(MIN_SCOOTER, 0)
        s_max = SHIFT_LIMIT - B
        if s_max < s_min:
            continue
        for S in range(s_min, s_max + 1):
            # Check constraints
            if (B + S) > SHIFT_LIMIT:
                continue
            energy = ENERGY_B * B + ENERGY_S * S
            if energy > ENERGY_LIMIT:
                continue
            orders = ORDERS_B * B + ORDERS_S * S
            if orders < ORDER_REQUIRE:
                continue

            # Objective
            tips = TIP_B * B + TIP_S * S

            # Update best
            if (best_obj is None) or (tips > best_obj):
                best_obj = tips
                best_solutions = [(B, S, tips, orders, energy)]
            elif tips == best_obj:
                best_solutions.append((B, S, tips, orders, energy))

    # If no feasible solution found, report and exit
    if best_obj is None:
        print("No feasible integer solution found.")
        print(f"FINAL_OBJ: {None}")
        return

    # Choose solution (if multiple with same obj, pick one with smallest leftover shifts, then highest bikes)
    best_solutions.sort(key=lambda x: ( (SHIFT_LIMIT - (x[0]+x[1])), -x[0] ))
    B_opt, S_opt, tips_opt, orders_opt, energy_opt = best_solutions[0]

    # Compute slacks and bindingness
    unused_shifts = SHIFT_LIMIT - (B_opt + S_opt)
    leftover_energy = ENERGY_LIMIT - energy_opt
    orders_surplus = orders_opt - ORDER_REQUIRE

    bindings = []
    if (B_opt + S_opt) == SHIFT_LIMIT:
        bindings.append("shift availability (B+S ≤ 40)")
    if energy_opt == ENERGY_LIMIT:
        bindings.append("energy budget (5B+6S ≤ 230)")
    if orders_opt == ORDER_REQUIRE:
        bindings.append("orders requirement (10B+7S ≥ 320)")
    if S_opt == MIN_SCOOTER:
        bindings.append("minimum scooters (S ≥ 5)")

    # Print results
    print("Optimal integer solution found:")
    print(f"  Bike shifts B* = {B_opt}")
    print(f"  Scooter shifts S* = {S_opt}")
    print(f"  Total tips = ${tips_opt}")
    print(f"  Total orders delivered = {orders_opt} (requirement {ORDER_REQUIRE})")
    print(f"  Total energy used = {energy_opt} (limit {ENERGY_LIMIT})")
    print(f"  Unused shifts = {unused_shifts}")
    print(f"  Leftover energy = {leftover_energy}")
    print(f"  Orders surplus = {orders_surplus}")
    if bindings:
        print("  Binding constraints at optimum:")
        for b in bindings:
            print(f"   - {b}")
    else:
        print("  No binding constraints at the optimum.")

    # Final required line (must be the very last printed line)
    print(f"FINAL_OBJ: {tips_opt}")

if __name__ == "__main__":
    main()
