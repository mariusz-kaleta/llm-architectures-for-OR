import pulp

# Data
time_per_C = 4.0
time_per_O = 6.0
min_C = 100.0
min_total = 300.0

# ---------------------
# Model A: additive times (minimize 4*C + 6*O)
# ---------------------
modelA = pulp.LpProblem("ModelA_Additive", pulp.LpMinimize)
C_a = pulp.LpVariable("C_A", lowBound=0, cat="Continuous")
O_a = pulp.LpVariable("O_A", lowBound=0, cat="Continuous")

# Objective
modelA += time_per_C * C_a + time_per_O * O_a

# Constraints
modelA += C_a >= min_C
modelA += C_a + O_a >= min_total
modelA += C_a - 2.0 * O_a <= 0.0

# Solve
modelA.solve(pulp.PULP_CBC_CMD(msg=False))
objA = pulp.value(modelA.objective)
C_a_val = C_a.varValue
O_a_val = O_a.varValue

# ---------------------
# Model B: parallel (minimize T subject to T >= 4C, T >= 6O)
# ---------------------
modelB = pulp.LpProblem("ModelB_Parallel", pulp.LpMinimize)
C_b = pulp.LpVariable("C_B", lowBound=0, cat="Continuous")
O_b = pulp.LpVariable("O_B", lowBound=0, cat="Continuous")
T_b = pulp.LpVariable("T_B", lowBound=0, cat="Continuous")

# Objective
modelB += T_b

# Constraints
modelB += T_b >= time_per_C * C_b
modelB += T_b >= time_per_O * O_b
modelB += C_b >= min_C
modelB += C_b + O_b >= min_total
modelB += C_b - 2.0 * O_b <= 0.0

# Solve
modelB.solve(pulp.PULP_CBC_CMD(msg=False))
objB = pulp.value(modelB.objective)
C_b_val = C_b.varValue
O_b_val = O_b.varValue
T_b_val = T_b.varValue

# Final objective values for both models (Model A, Model B)
final_value = (objA, objB)

print(f"FINAL_OBJ: {final_value}")
