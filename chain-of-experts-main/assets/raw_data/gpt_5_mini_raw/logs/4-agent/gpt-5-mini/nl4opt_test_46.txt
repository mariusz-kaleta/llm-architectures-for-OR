import numpy as np

# Problem data (initialised directly)
# Decision variables: x = Acres_traditional, y = Acres_modern
c = np.array([30.0, 40.0])   # objective coefficients (maximize)
# Constraints A x <= b
A = np.array([
    [1.0, 1.0],    # land: x + y <= 500
    [20.0, 15.0],  # fuel: 20x + 15y <= 9000
    [10.0, 15.0]   # waste: 10x + 15y <= 6000
])
b = np.array([500.0, 9000.0, 6000.0])

# Tolerance for feasibility and binding checks
tol = 1e-8

# Helper: compute objective
def objective(xy):
    return c.dot(xy)

# Generate candidate corner points:
candidates = []

# 1) origin
candidates.append(np.array([0.0, 0.0]))

# 2) axis extremes (x=0 -> y limited by min of constraints; y=0 -> x limited)
# For x = 0, y <= min( b_i / A_i2 ) where A_i2 is coefficient of y in constraint i (if >0)
y_limits = []
for i in range(A.shape[0]):
    a_y = A[i, 1]
    if a_y > 0:
        y_limits.append(b[i] / a_y)
if y_limits:
    candidates.append(np.array([0.0, min(y_limits)]))

# For y = 0, x <= min( b_i / A_i1 ) where A_i1 is coefficient of x
x_limits = []
for i in range(A.shape[0]):
    a_x = A[i, 0]
    if a_x > 0:
        x_limits.append(b[i] / a_x)
if x_limits:
    candidates.append(np.array([min(x_limits), 0.0]))

# 3) intersections of every pair of constraints treated as equalities
m = A.shape[0]
for i in range(m):
    for j in range(i+1, m):
        Ai = A[[i, j], :]
        bi = b[[i, j]]
        # Solve Ai * [x,y] = bi if invertible
        if abs(np.linalg.det(Ai)) > 1e-12:
            xy = np.linalg.solve(Ai, bi)
            candidates.append(xy)
        else:
            # nearly singular (parallel constraints) - skip
            pass

# Filter feasible candidates (satisfy all A x <= b and x,y >= 0)
feasible = []
for xy in candidates:
    if np.any(np.isnan(xy)) or np.any(np.isinf(xy)):
        continue
    # Non-negativity tolerance
    if (xy >= -tol).all():
        # Check all constraints A x <= b + tol
        if (A.dot(xy) <= b + tol).all():
            # Clip tiny negatives to zero
            xy_clipped = np.maximum(xy, 0.0)
            feasible.append(xy_clipped)

# Remove duplicates (within tolerance)
unique_feasible = []
for v in feasible:
    if not any(np.allclose(v, u, atol=1e-7, rtol=0) for u in unique_feasible):
        unique_feasible.append(v)
feasible = unique_feasible

if not feasible:
    raise RuntimeError("No feasible point found (unexpected).")

# Evaluate objective at feasible corners and find best
best_xy = None
best_obj = -np.inf
for xy in feasible:
    val = objective(xy)
    if val > best_obj + 1e-9:
        best_obj = val
        best_xy = xy

# Determine which constraints are binding at the optimum
binding = []
lhs_vals = A.dot(best_xy)
for i in range(A.shape[0]):
    if abs(lhs_vals[i] - b[i]) <= 1e-6:
        binding.append(True)
    else:
        binding.append(False)

constraint_names = ["land (x+y<=500)", "fuel (20x+15y<=9000)", "waste (10x+15y<=6000)"]
binding_constraints = [constraint_names[i] for i, bind in enumerate(binding) if bind]

# Compute a family of dual solutions (u,v,w) satisfying A^T * p = c.
# Let p = [u, v, w]. Equations:
# u + 20 v + 10 w = 30
# u + 15 v + 15 w = 40
# Solve for u and w in terms of v:
# From subtracting: (20-15)v + (10-15)w = 30-40  => 5v -5w = -10 => w = v + 2
# Then u + 30 v = 10 => u = 10 - 30 v
# Dual nonnegativity (since primal is max with <= constraints): u>=0, v>=0, w>=0
# This yields v in [0, 1/3], u = 10 - 30 v, w = v + 2
v_min = 0.0
v_max = 1.0/3.0
# Example dual (choose v=0)
v_example = 0.0
u_example = 10.0 - 30.0 * v_example
w_example = v_example + 2.0

# Verify dual objective b^T p equals primal optimal objective (complementary slackness)
def dual_obj(u, v, w):
    return b.dot(np.array([u, v, w]))

dual_obj_example = dual_obj(u_example, v_example, w_example)

# Print results (informational). The final required print is STRICTLY the last line.
print("Optimal primal solution (acres):")
print(f"  Acres_traditional (x) = {best_xy[0]:.6f}")
print(f"  Acres_modern      (y) = {best_xy[1]:.6f}")
print(f"Objective (maximum tea picked) = {best_obj:.6f} kg")
print()
print("Binding constraints at optimum:")
for bc in binding_constraints:
    print(" ", bc)
print()
print("Example valid shadow prices (dual variables) consistent with A^T p = c and p >= 0:")
print(f"  land (u)  = {u_example:.6f}  kg per extra acre")
print(f"  fuel (v)  = {v_example:.6f}  kg per extra liter")
print(f"  waste (w) = {w_example:.6f}  kg per extra kg waste")
print()
print("Range of equivalent duals (parametrised by v in [0, 1/3]):")
print(f"  v (fuel) in [{v_min:.6f}, {v_max:.6f}]")
print(f"  u (land) = 10 - 30 v  -> in [{10 - 30*v_max:.6f}, {10 - 30*v_min:.6f}]")
print(f"  w (waste)= v + 2      -> in [{v_min + 2:.6f}, {v_max + 2:.6f}]")
print()
print("Dual objective for the example dual (should equal primal optimal objective):")
print(f"  dual b^T p = {dual_obj_example:.6f} kg")

# Final required single-line print (must be the very last line)
# Use integer if nearly integer
final_obj_value = float(best_obj)
if abs(round(final_obj_value) - final_obj_value) < 1e-8:
    final_obj_value = int(round(final_obj_value))
print(f"FINAL_OBJ: {final_obj_value}")
