import math

# Problem data
cJ, cP = 5.0, 9.0  # objective coefficients: maximize 5 J + 9 P
# Constraints: 8 J + 6 P <= 1500  (water)
#              3 J + 5 P <= 1350  (pollution)

# Right-hand sides
b1 = 1500.0
b2 = 1350.0

# Candidate corner points:
candidates = []

# Origin
candidates.append((0.0, 0.0))

# J-only (P=0): 8 J <= 1500 and 3 J <= 1350 -> J <= min(1500/8, 1350/3)
J_only = min(b1 / 8.0, b2 / 3.0)
candidates.append((J_only, 0.0))

# P-only (J=0): 6 P <= 1500 and 5 P <= 1350 -> P <= min(1500/6, 1350/5)
P_only = min(b1 / 6.0, b2 / 5.0)
candidates.append((0.0, P_only))

# Intersection of the two constraint lines:
# 8 J + 6 P = 1500
# 3 J + 5 P = 1350
# Solve using determinants
D = 8.0 * 5.0 - 6.0 * 3.0
if abs(D) > 1e-12:
    J_int = (b1 * 5.0 - 6.0 * b2) / D
    P_int = (8.0 * b2 - b1 * 3.0) / D
    candidates.append((J_int, P_int))

# Feasibility check and objective evaluation
def feasible(J, P, tol=1e-8):
    if J < -tol or P < -tol:
        return False
    if 8.0 * J + 6.0 * P > b1 + tol:
        return False
    if 3.0 * J + 5.0 * P > b2 + tol:
        return False
    return True

best_obj = -math.inf
best_sol = (None, None)
for J, P in candidates:
    if feasible(J, P):
        obj = cJ * J + cP * P
        if obj > best_obj:
            best_obj = obj
            best_sol = (J, P)

# If numeric noise produced a near-integer objective, round sensibly
if best_obj is not None and abs(best_obj - round(best_obj)) < 1e-9:
    best_obj = int(round(best_obj))

value = best_obj
print(f"FINAL_OBJ: {value}")
