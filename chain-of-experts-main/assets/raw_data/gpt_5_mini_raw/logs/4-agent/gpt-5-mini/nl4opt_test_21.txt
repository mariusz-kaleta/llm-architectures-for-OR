- (B=0, C=16) from 0.25·0 + 2·C = 32 → profit = 40·16 = 640.
- (B=35, C=0) from 2·B = 70 → profit = 20·35 = 700.
- Intersection of the two resource constraints (solve equalities):
  Solve 2B + C = 70 and 0.25B + 2C = 32.

#!/usr/bin/env python3
# Self-contained script to solve the bakery production LP (continuous) and the integer variant.
# Decision variables:
#  B >= 0 : number of bagel batches per day
#  C >= 0 : number of croissant batches per day
# Objective: maximize Z = 20*B + 40*C
# Constraints:
#  2*B + 1*C <= 70       (oven-hours)
#  0.25*B + 2*C <= 32    (pastry-chef hours)

# We will:
#  - compute the continuous LP optimum by checking corner points and solving the intersection,
#  - enumerate integer feasible solutions to find the integer optimum (small search space).

# No external libraries required.

def evaluate_objective(B, C):
    return 20.0 * B + 40.0 * C

# Continuous solution:
# Corner points to consider (from constraints and axes):
# 1) (0,0)
# 2) (35,0) from 2B = 70
# 3) (0,16) from 2C = 32 -> C = 16
# 4) intersection of 2B + C = 70 and 0.25B + 2C = 32

# Compute intersection analytically:
# From 2B + C = 70  -> C = 70 - 2B
# Substitute into 0.25B + 2C = 32:
# 0.25B + 2*(70 - 2B) = 32
# 0.25B + 140 - 4B = 32
# -3.75B = 32 - 140 = -108
# B = -108 / -3.75 = 28.8
B_inter = 28.8
C_inter = 70 - 2 * B_inter  # should be 12.4

# Evaluate continuous objective at candidate points
candidates = [
    (0.0, 0.0),
    (35.0, 0.0),
    (0.0, 16.0),
    (B_inter, C_inter),
]

best_cont = None
best_cont_val = -float('inf')
for B, C in candidates:
    # Check feasibility (numeric tolerance)
    if 2*B + C <= 70 + 1e-9 and 0.25*B + 2*C <= 32 + 1e-9 and B >= -1e-9 and C >= -1e-9:
        val = evaluate_objective(B, C)
        if val > best_cont_val:
            best_cont_val = val
            best_cont = (B, C)

# Integer enumeration (small finite search)
# Bounds: from oven 2B <=70 -> B <=35, from pastry-chef 0.25B <=32 -> B <=128 but oven tighter.
# For C: from pastry-chef 2C <=32 -> C <=16, oven 1*C <=70 -> C <=70 but pastry-chef tighter.
best_int = None
best_int_val = -float('inf')
best_int_slacks = None

for B in range(0, 36):   # 0..35
    for C in range(0, 17):  # 0..16
        if 2*B + C <= 70 and 0.25*B + 2*C <= 32:
            val = evaluate_objective(B, C)
            if val > best_int_val:
                best_int_val = val
                best_int = (B, C)
                slack_oven = 70 - (2*B + C)
                slack_pastry = 32 - (0.25*B + 2*C)
                best_int_slacks = (slack_oven, slack_pastry)

# Print results (informational)
print("Continuous LP optimum (by corner check):")
print(f"  B = {best_cont[0]:.6g}, C = {best_cont[1]:.6g}, Objective = {best_cont_val:.6g}")
print("Integer optimum (by enumeration):")
print(f"  B = {best_int[0]}, C = {best_int[1]}, Objective = {best_int_val}, Slacks (oven, pastry) = ({best_int_slacks[0]:.6g}, {best_int_slacks[1]:.6g})")

# Final required print line: print final objective value.
# We choose to report the continuous LP objective as the final objective (as the formal model specifies continuous LP by default).
final_value = best_cont_val

print(f"FINAL_OBJ: {final_value}")
