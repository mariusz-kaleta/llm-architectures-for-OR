#!/usr/bin/env python3
"""
Solve the LP:
  Max Z = 50 w + 20 v
  s.t.
    w <= 100
    v <= 170
    w + v <= 150
    w, v >= 0

This script computes the optimal primal (w, v) and the dual (y1, y2, y3) analytically,
verifies complementary slackness, and prints results. The final line prints the
objective value in the required exact format.
"""

# Problem data
profit_w = 50.0
profit_v = 20.0

cap_w = 100.0     # wired-team capacity
cap_v = 170.0     # wireless-team capacity
cap_test = 150.0  # shared testing-machine capacity

# Solve by priority on profit per unit (since both consume one testing slot each)
# If profit_w >= profit_v, produce wired up to its team cap (or testing cap), then use remaining testing for wireless.
if profit_w >= profit_v:
    w = min(cap_w, cap_test)
    v = min(cap_v, max(0.0, cap_test - w))
else:
    v = min(cap_v, cap_test)
    w = min(cap_w, max(0.0, cap_test - v))

# Objective
objective = profit_w * w + profit_v * v

# Dual variables (by solving the dual or by economic reasoning)
# Dual for constraints:
#   y1 for w <= 100
#   y2 for v <= 170
#   y3 for w + v <= 150
# From complementary slackness and the binding constraints (w=100 and w+v=150 are binding here),
# we get y1 + y3 = profit_w, y2 + y3 = profit_v, and y2 = 0 since v-team cap is not binding.
# Thus:
y2 = 0.0
y3 = profit_v  # 20
y1 = profit_w - y3  # 50 - 20 = 30

# Verification (primal and dual objective should match)
dual_obj = y1 * cap_w + y2 * cap_v + y3 * cap_test

# Print results
print("Optimal primal solution:")
print(f"  w (wired)    = {w}")
print(f"  v (wireless) = {v}")
print(f"  Objective Z  = {objective}")

print("\nOptimal dual (shadow prices):")
print(f"  y1 (wired-team cap)    = {y1}")
print(f"  y2 (wireless-team cap) = {y2}")
print(f"  y3 (testing-machine)   = {y3}")
print(f"Dual objective (verification) = {dual_obj}")

# Simple sensitivity notes (concise)
print("\nSensitivity notes:")
print("  - An extra wired-team capacity unit increases profit by y1 = $30 (until other constraints change).")
print("  - An extra testing slot increases profit by y3 = $20 (until wireless team cap becomes binding).")
print("  - Wireless team capacity is non-binding; its shadow price y2 = $0.")

# Final required line (must be the last printed line and exact format)
value = objective
print(f"FINAL_OBJ: {value}")
