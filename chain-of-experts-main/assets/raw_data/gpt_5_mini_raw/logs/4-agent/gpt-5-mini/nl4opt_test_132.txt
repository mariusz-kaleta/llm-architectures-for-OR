#!/usr/bin/env python3
# Self-contained script to solve the desk/drawer production LP (continuous and integer)

import math

# Problem data
profit_x = 100.0  # profit per desk
profit_y = 90.0   # profit per drawer

# Constraints (minutes)
# 40 x + 30 y <= 4000  (assembly)
# 20 x + 10 y <= 3500  (sanding)
A1_x, A1_y, A1_rhs = 40.0, 30.0, 4000.0
A2_x, A2_y, A2_rhs = 20.0, 10.0, 3500.0

# Evaluate corner points for the continuous LP
corners = []

# (0,0)
corners.append((0.0, 0.0))

# assembly intercepts:
# x-axis intercept (y=0): 40 x = 4000 -> x = 100
corners.append((A1_rhs / A1_x, 0.0))

# y-axis intercept (x=0): 30 y = 4000 -> y = 4000/30 = 400/3
corners.append((0.0, A1_rhs / A1_y))

# intersection of both constraints (if feasible)
# Solve:
# 40x + 30y = 4000
# 20x + 10y = 3500
# Solve linear system:
det = A1_x * A2_y - A1_y * A2_x
if abs(det) > 1e-12:
    ix = (A1_rhs * A2_y - A1_y * A2_rhs) / det
    iy = (A1_x * A2_rhs - A1_rhs * A2_x) / det
    # include only if nonnegative and satisfies both constraints (within tolerance)
    if ix >= -1e-9 and iy >= -1e-9 and A1_x * ix + A1_y * iy <= A1_rhs + 1e-6 and A2_x * ix + A2_y * iy <= A2_rhs + 1e-6:
        corners.append((ix, iy))

# Evaluate objective at corners
best_cont = None
best_cont_val = -float('inf')
for (x, y) in corners:
    val = profit_x * x + profit_y * y
    if val > best_cont_val + 1e-9:
        best_cont_val = val
        best_cont = (x, y)

# Round small near-zero values to zero
best_cont = (0.0 if abs(best_cont[0]) < 1e-12 else best_cont[0],
             0.0 if abs(best_cont[1]) < 1e-12 else best_cont[1])

# Integer (MILP) search: brute force over reasonable bounds
max_x_int = int(math.floor(A1_rhs / A1_x))  # cannot exceed assembly intercept
max_y_int = int(math.floor(A1_rhs / A1_y))
best_int = None
best_int_val = -float('inf')
for xi in range(0, max_x_int + 1):
    # For each xi, find max yi such that constraints hold
    # Bound yi by assembly and sanding
    rem_assembly = A1_rhs - A1_x * xi
    if rem_assembly < 0:
        continue
    bound_y_assembly = int(math.floor(rem_assembly / A1_y))
    rem_sanding = A2_rhs - A2_x * xi
    if rem_sanding < 0:
        continue
    bound_y_sanding = int(math.floor(rem_sanding / A2_y))
    y_upper = min(bound_y_assembly, bound_y_sanding, max_y_int)
    for yi in range(0, y_upper + 1):
        val = profit_x * xi + profit_y * yi
        if val > best_int_val + 1e-9:
            best_int_val = val
            best_int = (xi, yi)

# Compute reported shadow prices analytically (for the continuous optimal corner)
# Since the sanding constraint is non-binding at the optimum found, shadow price for sanding is 0.
# For assembly, compute marginal value: at optimum (x=0, y=400/3), objective increases by 3 per extra minute.
# We'll report these computed values.
shadow_assembly = 3.0
shadow_sanding = 0.0

# Display results (these prints are informational; final required print is below)
print("Continuous LP optimal (analytic corners):")
print(f"  x* = {best_cont[0]:.6f}, y* = {best_cont[1]:.6f}, objective = {best_cont_val:.6f}")
print("Integer optimal (brute force):")
print(f"  x = {best_int[0]}, y = {best_int[1]}, objective = {best_int_val:.6f}")
print("Shadow prices (assembly, sanding):", (shadow_assembly, shadow_sanding))

# Final required print line (must be the very last output)
final_value = int(round(best_cont_val)) if abs(best_cont_val - round(best_cont_val)) < 1e-9 else best_cont_val
print(f"FINAL_OBJ: {final_value}")
