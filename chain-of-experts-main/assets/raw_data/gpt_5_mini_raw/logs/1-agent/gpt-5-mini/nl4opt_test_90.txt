import pulp

# Data initialization (from problem statement)
profit_regular = 5000   # profit per regular model
profit_premium = 8500   # profit per premium model

demand_regular_max = 8  # max daily demand for regular
demand_premium_max = 6  # max daily demand for premium
total_production_max = 12  # max total cars per day

# Define the LP problem (maximization)
prob = pulp.LpProblem("Car_Production", pulp.LpMaximize)

# Decision variables: number of cars of each type (nonnegative)
x1 = pulp.LpVariable("x1_regular", lowBound=0)  # regular
x2 = pulp.LpVariable("x2_premium", lowBound=0)  # premium

# Objective: maximize total profit
prob += profit_regular * x1 + profit_premium * x2

# Constraints
prob += x1 <= demand_regular_max
prob += x2 <= demand_premium_max
prob += x1 + x2 <= total_production_max

# Solve
prob.solve()

# Retrieve results
x1_val = x1.varValue if x1.varValue is not None else 0.0
x2_val = x2.varValue if x2.varValue is not None else 0.0
obj_val = pulp.value(prob.objective)

# Print solution
print(f"x1 (regular) = {x1_val}")
print(f"x2 (premium) = {x2_val}")
print(f"Objective (total profit) = {obj_val}")
print(f"FINAL_OBJ: {obj_val}")
