import pulp

# Define the LP problem (maximize profit)
prob = pulp.LpProblem("Souvenir_Profit_Maximization", pulp.LpMaximize)

# Decision variables: number of elephants and tigers (integer, >=0)
elephants = pulp.LpVariable('elephants', lowBound=0, cat='Integer')
tigers = pulp.LpVariable('tigers', lowBound=0, cat='Integer')

# Data (given)
wood_available = 5000   # grams
plastic_available = 4000  # grams

wood_per_elephant = 50
plastic_per_elephant = 20

wood_per_tiger = 40
plastic_per_tiger = 30

profit_elephant = 5  # dollars
profit_tiger = 4     # dollars

# Constraints
prob += wood_per_elephant * elephants + wood_per_tiger * tigers <= wood_available, "WoodConstraint"
prob += plastic_per_elephant * elephants + plastic_per_tiger * tigers <= plastic_available, "PlasticConstraint"

# Objective
prob += profit_elephant * elephants + profit_tiger * tigers

# Solve
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Retrieve results
elephants_val = int(pulp.value(elephants))
tigers_val = int(pulp.value(tigers))
value = pulp.value(prob.objective)

# Print solution details
print(f"Elephants: {elephants_val}")
print(f"Tigers: {tigers_val}")
print(f"Profit: {value}")

# Final required print (must be the last line)
print(f"FINAL_OBJ: {value}")
