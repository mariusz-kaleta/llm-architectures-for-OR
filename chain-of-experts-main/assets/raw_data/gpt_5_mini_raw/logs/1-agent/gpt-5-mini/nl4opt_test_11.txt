import math

# Problem data
people_needed = 500
bike_capacity = 3
car_capacity = 5

best_bikes = None
best_cars = None

# Search over a reasonable range for cars (0..500 is more than enough)
for c in range(0, 501):
    # minimal bikes to meet capacity (if negative or zero, set to 0)
    if people_needed - car_capacity * c > 0:
        req_b_capacity = math.ceil((people_needed - car_capacity * c) / bike_capacity)
    else:
        req_b_capacity = 0
    # minimal bikes to satisfy car fraction constraint: 3c <= 2b -> b >= 3c/2
    req_b_ratio = math.ceil(3 * c / 2)
    b = max(req_b_capacity, req_b_ratio)
    # Check feasibility explicitly
    if b >= 0 and c >= 0 and (bike_capacity * b + car_capacity * c) >= people_needed and (3 * c) <= (2 * b):
        if best_bikes is None or b < best_bikes or (b == best_bikes and c > best_cars):
            best_bikes = b
            best_cars = c

# Output solution
print(f"Bikes: {best_bikes}")
print(f"Cars: {best_cars}")
print(f"FINAL_OBJ: {best_bikes}")
