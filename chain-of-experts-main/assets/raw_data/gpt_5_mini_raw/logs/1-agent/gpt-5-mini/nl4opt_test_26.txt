import pulp

# Define the LP problem
prob = pulp.LpProblem("Workbook_Production", pulp.LpMaximize)

# Decision variables: number of math (m) and English (e) workbooks
# They can be continuous but bounds are given
m = pulp.LpVariable('math_workbooks', lowBound=40, upBound=140, cat='Continuous')
e = pulp.LpVariable('english_workbooks', lowBound=60, upBound=170, cat='Continuous')

# Objective: maximize profit = 15*m + 17*e
prob += 15 * m + 17 * e, "Total_Profit"

# Constraint: at least 200 workbooks of either type combined
prob += m + e >= 200, "Total_Workbooks_Minimum"

# Solve
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract solution
math_opt = pulp.value(m)
english_opt = pulp.value(e)
final_obj = pulp.value(prob.objective)

# Print solution details
print(f"Optimal math workbooks: {math_opt}")
print(f"Optimal English workbooks: {english_opt}")
print(f"Optimal profit: {final_obj}")

# Final required print (must be the last line)
print(f"FINAL_OBJ: {final_obj}")
