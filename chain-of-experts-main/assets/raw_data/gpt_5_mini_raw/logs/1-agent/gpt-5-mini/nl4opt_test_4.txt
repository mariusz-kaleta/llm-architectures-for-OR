# Feed mixing optimization using PuLP

import pulp

# Data (initialized directly)
cost_A = 100.0  # $ per kg for Feed A
cost_B = 80.0   # $ per kg for Feed B

protein_A = 10.0  # units protein per kg for Feed A
protein_B = 7.0   # units protein per kg for Feed B
min_protein = 30.0

fat_A = 8.0    # units fat per kg for Feed A
fat_B = 15.0   # units fat per kg for Feed B
min_fat = 50.0

# Define the LP problem
prob = pulp.LpProblem("Feed_Mix", pulp.LpMinimize)

# Decision variables: kilograms of Feed A (x) and Feed B (y)
x = pulp.LpVariable('x_A', lowBound=0, cat='Continuous')
y = pulp.LpVariable('y_B', lowBound=0, cat='Continuous')

# Objective: minimize total cost
prob += cost_A * x + cost_B * y

# Constraints
prob += protein_A * x + protein_B * y >= min_protein, "Protein_requirement"
prob += fat_A * x + fat_B * y >= min_fat, "Fat_requirement"

# Solve
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Get final objective value
final_obj = pulp.value(prob.objective)

# Print final objective in required format (must be the last output)
print(f"FINAL_OBJ: {final_obj}")
