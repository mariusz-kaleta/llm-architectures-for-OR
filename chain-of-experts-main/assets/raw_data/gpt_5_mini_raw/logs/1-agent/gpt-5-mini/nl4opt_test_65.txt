import pulp

# Data initialization
profit_laminate = 2.1   # profit per sq ft for laminate
profit_carpet = 3.3     # profit per sq ft for carpet

min_laminate = 15000
min_carpet = 5000
min_total = 50000

max_laminate = 40000
max_carpet = 20000

# Define problem
prob = pulp.LpProblem("Flooring_Production", pulp.LpMaximize)

# Decision variables
laminate = pulp.LpVariable("Laminate_sqft", lowBound=min_laminate, upBound=max_laminate, cat='Continuous')
carpet = pulp.LpVariable("Carpet_sqft", lowBound=min_carpet, upBound=max_carpet, cat='Continuous')

# Objective
prob += profit_laminate * laminate + profit_carpet * carpet

# Constraints
prob += laminate + carpet >= min_total

# Solve
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Print decision variables (optional)
print(f"Laminate_sqft: {pulp.value(laminate)}")
print(f"Carpet_sqft: {pulp.value(carpet)}")

# Final objective value as required
print(f"FINAL_OBJ: {pulp.value(prob.objective)}")
