import pulp

# Data
milk_available = 30000  # grams
honey_available = 500   # grams

milk_black = 600  # grams per black bottle
honey_black = 10  # grams per black bottle
profit_black = 7.5

milk_matcha = 525  # grams per matcha bottle
honey_matcha = 5   # grams per matcha bottle
profit_matcha = 5

# Define problem
prob = pulp.LpProblem("Milk_Tea_Profit_Maximization", pulp.LpMaximize)

# Decision variables (integer number of bottles)
x = pulp.LpVariable('black_bottles', lowBound=0, cat='Integer')
y = pulp.LpVariable('matcha_bottles', lowBound=0, cat='Integer')

# Objective
prob += profit_black * x + profit_matcha * y

# Constraints
prob += milk_black * x + milk_matcha * y <= milk_available
prob += honey_black * x + honey_matcha * y <= honey_available

# Solve
prob.solve(pulp.PULP_CBC_CMD(msg=0))

# Get objective value
obj_value = pulp.value(prob.objective)

print(f"FINAL_OBJ: {obj_value}")
