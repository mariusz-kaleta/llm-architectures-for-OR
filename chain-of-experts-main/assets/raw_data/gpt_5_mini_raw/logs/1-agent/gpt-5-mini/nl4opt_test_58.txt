import pulp

# Define the problem
prob = pulp.LpProblem("Pill_Time_Minimization", pulp.LpMinimize)

# Decision variables: number of calcium pills (c) and vitamin D pills (v)
c = pulp.LpVariable('calcium_pills', lowBound=0, cat='Integer')
v = pulp.LpVariable('vitaminD_pills', lowBound=0, cat='Integer')

# Objective: minimize total effective time (5 minutes per calcium, 6 per vitamin D)
prob += 5 * c + 6 * v, "Total_Effective_Time"

# Constraints:
prob += c + v >= 130, "Minimum_total_pills"
prob += v >= 40, "Minimum_vitaminD"
prob += c - v >= 1, "Calcium_more_than_vitD"

# Solve
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract integer solution values
c_val = int(pulp.value(c))
v_val = int(pulp.value(v))
final_value = 5 * c_val + 6 * v_val

# Print solution details
print(f"Calcium pills: {c_val}")
print(f"Vitamin D pills: {v_val}")

# Final required print (must be the last output)
print(f"FINAL_OBJ: {final_value}")
