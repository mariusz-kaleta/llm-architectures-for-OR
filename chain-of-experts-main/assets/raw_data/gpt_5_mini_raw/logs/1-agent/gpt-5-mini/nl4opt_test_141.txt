import pulp

# Data initialization
cost_mango = 5
cost_guava = 3
profit_mango = 3
profit_guava = 4
budget = 20000
mango_min = 100
mango_max = 150

# Define problem
prob = pulp.LpProblem("Food_Truck_Profit_Maximization", pulp.LpMaximize)

# Decision variables (integer number of fruits)
M = pulp.LpVariable("Mangos", lowBound=mango_min, upBound=mango_max, cat='Integer')
G = pulp.LpVariable("Guavas", lowBound=0, cat='Integer')

# Constraints
prob += cost_mango * M + cost_guava * G <= budget, "BudgetConstraint"
prob += 3 * G <= M, "GuavaToMangoRatio"  # G <= M/3

# Objective: maximize profit
prob += profit_mango * M + profit_guava * G, "TotalProfit"

# Solve
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Results
mango_opt = int(pulp.value(M))
guava_opt = int(pulp.value(G))
objective_value = pulp.value(prob.objective)

print(f"Optimal mangos (M): {mango_opt}")
print(f"Optimal guavas (G): {guava_opt}")
print(f"Total profit: {objective_value}")
print(f"FINAL_OBJ: {objective_value}")
