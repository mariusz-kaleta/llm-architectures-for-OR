import math

# Data (from the problem statement)
capacity_balloon = 4       # capacity per hot-air balloon
capacity_gondola = 6       # capacity per gondola lift
pollution_balloon = 10     # pollution per hot-air balloon
pollution_gondola = 15     # pollution per gondola lift
max_balloons = 10          # at most 10 hot-air balloon rides
required_visitors = 70     # at least 70 visitors must be transported

# Brute-force search over integer numbers of rides
best_obj = None
best_solutions = []

# Upper bound for gondolas: if no balloons used, need ceil(required_visitors / capacity_gondola)
max_gondolas = math.ceil(required_visitors / capacity_gondola) + 10  # a safe margin

for x in range(0, max_balloons + 1):  # x = number of hot-air balloons
    for y in range(0, max_gondolas + 1):  # y = number of gondola lifts
        total_capacity = capacity_balloon * x + capacity_gondola * y
        if total_capacity >= required_visitors:
            total_pollution = pollution_balloon * x + pollution_gondola * y
            if (best_obj is None) or (total_pollution < best_obj):
                best_obj = total_pollution
                best_solutions = [(x, y, total_capacity, total_pollution)]
            elif total_pollution == best_obj:
                best_solutions.append((x, y, total_capacity, total_pollution))

# Print all optimal solutions found
print("Optimal solutions (hot-air balloons x, gondola lifts y) minimizing total pollution:")
for sol in best_solutions:
    x, y, cap, pol = sol
    print(f"  x = {x}, y = {y}, capacity = {cap}, pollution = {pol}")

# FINAL objective print (must be the last line)
print(f"FINAL_OBJ: {best_obj}")
