import math

# Problem data
canoe_capacity = 10
diesel_capacity = 15
required_fish = 1000

best_total = None
best_c = None
best_d = None

# Search over possible numbers of diesel boats (upper bound set reasonably high)
for d in range(0, 1001):
    # Minimum canoes needed to satisfy capacity (may be negative if diesel alone enough)
    remaining_fish = required_fish - diesel_capacity * d
    if remaining_fish <= 0:
        c_by_capacity = 0
    else:
        c_by_capacity = math.ceil(remaining_fish / canoe_capacity)
    # Environmental rule: canoes at least 3 times diesel boats
    c = max(c_by_capacity, 3 * d)
    # Check capacity satisfied
    if canoe_capacity * c + diesel_capacity * d >= required_fish:
        total = c + d
        if best_total is None or total < best_total:
            best_total = total
            best_c = c
            best_d = d

# Output results (other prints allowed before final line)
print(f"Best number of canoes: {best_c}")
print(f"Best number of diesel boats: {best_d}")
# Final required output line
print(f"FINAL_OBJ: {best_total}")
