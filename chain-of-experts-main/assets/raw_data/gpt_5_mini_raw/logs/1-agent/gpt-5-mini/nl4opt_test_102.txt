import pulp

# Data initialization
cost_rice = 3.0      # $ per serving of Rice
cost_kebab = 2.0     # $ per serving of Kebab

cal_rice = 300.0     # calories per serving of Rice
cal_kebab = 200.0    # calories per serving of Kebab
min_calories = 2200.0

prot_rice = 4.5      # grams of protein per serving of Rice
prot_kebab = 4.0     # grams of protein per serving of Kebab
min_protein = 30.0

# Define the problem
prob = pulp.LpProblem("Diet_Minimize_Cost", pulp.LpMinimize)

# Decision variables: number of servings (can be fractional)
rice = pulp.LpVariable("Rice_servings", lowBound=0, cat="Continuous")
kebab = pulp.LpVariable("Kebab_servings", lowBound=0, cat="Continuous")

# Objective: minimize total cost
prob += cost_rice * rice + cost_kebab * kebab, "Total_Cost"

# Constraints
prob += cal_rice * rice + cal_kebab * kebab >= min_calories, "Calories_requirement"
prob += prot_rice * rice + prot_kebab * kebab >= min_protein, "Protein_requirement"

# Solve
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Get final objective value
value = pulp.value(prob.objective)

print(f"FINAL_OBJ: {value}")
