import pulp

# Decision variables: number of regular vans (R) and hybrid vans (H)
R = pulp.LpVariable('RegularVans', lowBound=0, cat='Integer')
H = pulp.LpVariable('HybridVans', lowBound=0, cat='Integer')

# Problem: minimize total number of vans
prob = pulp.LpProblem('MinimizeVans', pulp.LpMinimize)
prob += R + H

# Constraints
# Package delivery requirement: 500R + 300H >= 20000
prob += 500 * R + 300 * H >= 20000

# Pollution limit: 200R + 100H <= 7000
prob += 200 * R + 100 * H <= 7000

# Solve
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract solution
R_val = int(pulp.value(R))
H_val = int(pulp.value(H))
obj_val = int(pulp.value(prob.objective))

# Optional: print solution details
print(f"Regular vans: {R_val}")
print(f"Hybrid vans: {H_val}")
print(f"Total vans (objective): {obj_val}")

# Final required print (must be the last line)
print(f"FINAL_OBJ: {obj_val}")
