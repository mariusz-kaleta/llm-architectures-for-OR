import pulp

# Define the problem
prob = pulp.LpProblem("Maximize_Gifts", pulp.LpMaximize)

# Decision variables: number of seasonal (S) and full-time (F) volunteers
S = pulp.LpVariable("Seasonal", lowBound=0, cat='Integer')
F = pulp.LpVariable("FullTime", lowBound=0, cat='Integer')

# Objective: maximize total gifts delivered (5 per seasonal, 8 per full-time)
prob += 5 * S + 8 * F, "Total_Gifts"

# Constraints:
# 1) Points limit: 2 points per seasonal, 5 points per full-time, max 200 points
prob += 2 * S + 5 * F <= 200, "PointsLimit"

# 2) At most 30% of volunteers can be seasonal:
#    S <= 0.3 * (S + F)  -> 0.7 S <= 0.3 F -> 7 S <= 3 F
prob += 7 * S <= 3 * F, "MaxSeasonalShare"

# 3) At least 10 full-time volunteers
prob += F >= 10, "MinFullTime"

# Solve
solver = pulp.PULP_CBC_CMD(msg=False)
prob.solve(solver)

# Extract solution values
S_val = int(pulp.value(S))
F_val = int(pulp.value(F))
total_gifts = int(5 * S_val + 8 * F_val)

# Print solution details (optional)
print(f"Seasonal volunteers: {S_val}")
print(f"Full-time volunteers: {F_val}")
print(f"Total gifts delivered: {total_gifts}")

# Final required output line
print(f"FINAL_OBJ: {total_gifts}")
