import pulp

# Data from the problem
# Syrup 1 per serving: throat 0.5, lungs 0.4, sugar 0.5
# Syrup 2 per serving: throat 0.2, lungs 0.5, sugar 0.3
# Constraints: throat at most 5 units, lungs at least 4 units

# Create LP problem (minimization)
prob = pulp.LpProblem("Minimize_Sugar", pulp.LpMinimize)

# Decision variables: servings of syrup 1 and syrup 2 (nonnegative)
x1 = pulp.LpVariable("syrup1_servings", lowBound=0)
x2 = pulp.LpVariable("syrup2_servings", lowBound=0)

# Objective: minimize total sugar
prob += 0.5 * x1 + 0.3 * x2, "Total_Sugar"

# Constraints
prob += 0.5 * x1 + 0.2 * x2 <= 5, "Throat_Medicine_Upper"
prob += 0.4 * x1 + 0.5 * x2 >= 4, "Lungs_Medicine_Lower"

# Solve
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract solution
s1 = pulp.value(x1)
s2 = pulp.value(x2)
value = pulp.value(prob.objective)

# Optional: print solution details
print(f"syrup1_servings = {s1}")
print(f"syrup2_servings = {s2}")

# Final required output
print(f"FINAL_OBJ: {value}")
