import pulp

# Data (from problem statement)
water_available = 500     # units of water
powder_available = 700    # units of powdered pill

# Resource usage per container
water_small = 10
powder_small = 15
paste_small = 20

water_large = 20
powder_large = 20
paste_large = 30

# Define the problem
prob = pulp.LpProblem("Maximize_Paste", pulp.LpMaximize)

# Decision variables: number of small and large containers (integer, >=0)
x = pulp.LpVariable("small_containers", lowBound=0, cat="Integer")
y = pulp.LpVariable("large_containers", lowBound=0, cat="Integer")

# Objective: maximize total paste produced
prob += paste_small * x + paste_large * y

# Constraints: resource limits
prob += water_small * x + water_large * y <= water_available, "Water_Limit"
prob += powder_small * x + powder_large * y <= powder_available, "Powder_Limit"

# Solve
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract results
small_used = int(pulp.value(x))
large_used = int(pulp.value(y))
obj_val = int(pulp.value(prob.objective))

# Print solution
print(f"Small containers used: {small_used}")
print(f"Large containers used: {large_used}")
print(f"Total paste produced: {obj_val}")

# Final required output line
print(f"FINAL_OBJ: {obj_val}")
