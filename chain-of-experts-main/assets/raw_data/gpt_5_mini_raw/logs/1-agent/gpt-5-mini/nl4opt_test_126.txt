import sys

# Brute-force integer search for the optimal bouquet counts

# Problem data (from the text)
small_flowers = 5    # flowers per small bouquet
large_flowers = 10   # flowers per large bouquet

max_small = 80       # at most 80 small bouquets
max_large = 50       # at most 50 large bouquets
max_total = 70       # at most 70 bouquets total
min_large = 20       # at least 20 large bouquets
# At least twice as many small bouquets as large bouquets: small >= 2 * large

best_obj = -float('inf')
best_solution = None

# Enumerate feasible integer solutions
for l in range(min_large, max_large + 1):
    for s in range(0, max_small + 1):
        if s + l > max_total:
            continue
        if s < 2 * l:
            continue
        # feasible
        obj = small_flowers * s + large_flowers * l
        if obj > best_obj:
            best_obj = obj
            best_solution = (s, l)

# Output the solution
if best_solution is None:
    print("No feasible solution found.")
    final_value = None
else:
    s_opt, l_opt = best_solution
    print(f"Optimal small bouquets (s): {s_opt}")
    print(f"Optimal large bouquets (l): {l_opt}")
    print(f"Total flowers delivered: {best_obj}")
    final_value = best_obj

# Final required output line
print(f"FINAL_OBJ: {final_value}")
