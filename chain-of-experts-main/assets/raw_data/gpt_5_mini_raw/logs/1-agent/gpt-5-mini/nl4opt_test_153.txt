import pulp

# Problem data
profit_strawberry = 5.5
profit_sugar = 12.0

# Create the LP problem (maximization)
prob = pulp.LpProblem("Zeta_Bakery_Profit_Max", pulp.LpMaximize)

# Decision variables
x1 = pulp.LpVariable("x1_strawberry", lowBound=0, cat="Continuous")  # number of strawberry cookies
x2 = pulp.LpVariable("x2_sugar", lowBound=0, cat="Continuous")       # number of sugar cookies

# Objective: maximize profit
prob += profit_strawberry * x1 + profit_sugar * x2

# Constraints
prob += x1 <= 100          # demand limit for strawberry
prob += x2 <= 80           # demand limit for sugar
prob += x1 + x2 <= 100     # staffing total production limit

# Solve
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Retrieve results
x1_val = x1.varValue if x1.varValue is not None else 0.0
x2_val = x2.varValue if x2.varValue is not None else 0.0
value = pulp.value(prob.objective)

# Print solution (optional details)
print(f"x1 (strawberry) = {x1_val}")
print(f"x2 (sugar) = {x2_val}")

# Final required output line
print(f"FINAL_OBJ: {value}")
