import sys

# Problem data
max_color = 20          # max color printers per day
max_bw = 30             # max black & white printers per day
machine_capacity = 35   # common machine capacity (total printers per day)
profit_color = 200
profit_bw = 70

# Brute-force integer search (small discrete space)
best_profit = -float('inf')
best_x = 0
best_y = 0

for x in range(max_color + 1):
    for y in range(max_bw + 1):
        if x + y <= machine_capacity:
            profit = profit_color * x + profit_bw * y
            if profit > best_profit:
                best_profit = profit
                best_x = x
                best_y = y

# Output results
print(f"Optimal number of color printers: {best_x}")
print(f"Optimal number of black & white printers: {best_y}")
print(f"Total profit: ${best_profit}")

# Final required line (must be the last output)
print(f"FINAL_OBJ: {best_profit}")
