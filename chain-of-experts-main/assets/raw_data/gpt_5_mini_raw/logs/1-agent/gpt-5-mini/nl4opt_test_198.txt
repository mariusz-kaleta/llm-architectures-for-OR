import pulp

# Data (from problem statement)
capacity_train = 120   # people per hour per train
capacity_tram = 30     # people per hour per tram
min_trams_per_train = 2  # number of trams must be at least twice the number of trains
demand = 600           # required people per hour

# Define the problem
prob = pulp.LpProblem("Railway_Planning", pulp.LpMinimize)

# Decision variables: number of trains (T) and trams (R), integer and non-negative
T = pulp.LpVariable("Trains", lowBound=0, cat='Integer')
R = pulp.LpVariable("Trams", lowBound=0, cat='Integer')

# Objective: minimize total number of transportation units
prob += T + R, "Minimize_total_units"

# Constraints
prob += capacity_train * T + capacity_tram * R >= demand, "Capacity_constraint"
prob += R >= min_trams_per_train * T, "Tram_to_train_ratio"

# Solve
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract results
num_trains = int(pulp.value(T))
num_trams = int(pulp.value(R))
objective_value = int(pulp.value(prob.objective))

# Optional prints (kept before the required final line)
print(f"Optimal number of trains: {num_trains}")
print(f"Optimal number of trams: {num_trams}")
print(f"Total units: {objective_value}")
print(f"Total capacity provided: {capacity_train * num_trains + capacity_tram * num_trams}")

# CRITICAL final line in required format
print(f"FINAL_OBJ: {objective_value}")
