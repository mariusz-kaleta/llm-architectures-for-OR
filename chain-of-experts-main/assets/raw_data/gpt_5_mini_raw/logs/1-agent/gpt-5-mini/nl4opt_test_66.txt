import pulp

# Define the problem
prob = pulp.LpProblem("Minimize_Vehicles", pulp.LpMinimize)

# Decision variables: number of tractors (x) and cars (y), integers >= 0
x = pulp.LpVariable('tractors', lowBound=0, cat='Integer')
y = pulp.LpVariable('cars', lowBound=0, cat='Integer')

# Objective: minimize total number of vehicles
prob += x + y, "Total_Vehicles"

# Constraints:
# 1) Capacity: 40 kg per tractor, 20 kg per car, need at least 500 kg
prob += 40 * x + 20 * y >= 500, "Capacity"

# 2) At least twice as many cars as tractors
prob += y >= 2 * x, "Car_to_Tractor_Ratio"

# Solve
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Retrieve solution values
tractors_val = int(pulp.value(x))
cars_val = int(pulp.value(y))
objective_val = int(round(pulp.value(prob.objective)))

# Print solution summary (optional)
print(f"Optimal tractors: {tractors_val}")
print(f"Optimal cars: {cars_val}")
print(f"Total vehicles (objective): {objective_val}")

# CRITICAL final line in required format
print(f"FINAL_OBJ: {objective_val}")
