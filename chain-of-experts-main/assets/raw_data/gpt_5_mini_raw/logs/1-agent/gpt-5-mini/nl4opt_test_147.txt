import pulp

# Data
car_capacity = 4
bus_capacity = 20
car_pollution = 10
bus_pollution = 30
required_employees = 300
max_buses = 4

# Define problem
prob = pulp.LpProblem("Minimize_Pollution", pulp.LpMinimize)

# Decision variables (integer counts)
cars = pulp.LpVariable("cars", lowBound=0, cat='Integer')
buses = pulp.LpVariable("buses", lowBound=0, upBound=max_buses, cat='Integer')

# Objective: minimize total pollution
prob += car_pollution * cars + bus_pollution * buses

# Capacity constraint
prob += car_capacity * cars + bus_capacity * buses >= required_employees

# Solve
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract solution
cars_opt = int(pulp.value(cars))
buses_opt = int(pulp.value(buses))
objective_value = pulp.value(prob.objective)

# Output solution
print(f"Optimal number of cars: {cars_opt}")
print(f"Optimal number of buses: {buses_opt}")
print(f"Total pollution: {objective_value}")

# Final required print (must be the last line)
print(f"FINAL_OBJ: {int(objective_value)}")
