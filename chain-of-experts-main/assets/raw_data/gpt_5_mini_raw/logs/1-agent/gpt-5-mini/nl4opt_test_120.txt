import math

# Data from problem
mint_available = 120
active_available = 100
black_max = 50

# Demo 1 usage and output
mint_d1 = 10
active_d1 = 20
foam_d1 = 25
black_d1 = 5

# Demo 2 usage and output
mint_d2 = 12
active_d2 = 15
foam_d2 = 18
black_d2 = 3

# We'll solve as an integer problem by brute-force enumeration (small search space).
# Determine reasonable upper bounds for enumeration
x1_max = min(mint_available // mint_d1 if mint_d1 else 0,
             active_available // active_d1 if active_d1 else 0,
             black_max // black_d1 if black_d1 else 0) + 5  # +5 safety margin
x2_max = min(mint_available // mint_d2 if mint_d2 else 0,
             active_available // active_d2 if active_d2 else 0,
             black_max // black_d2 if black_d2 else 0) + 5

# Ensure non-negative bounds
x1_max = max(0, x1_max)
x2_max = max(0, x2_max)

best_obj = -math.inf
best_x1 = 0
best_x2 = 0

for x1 in range(x1_max + 1):
    for x2 in range(x2_max + 1):
        # Resource consumption
        mint_used = mint_d1 * x1 + mint_d2 * x2
        active_used = active_d1 * x1 + active_d2 * x2
        black_produced = black_d1 * x1 + black_d2 * x2

        # Check feasibility
        if mint_used <= mint_available and active_used <= active_available and black_produced <= black_max:
            foam_produced = foam_d1 * x1 + foam_d2 * x2
            if foam_produced > best_obj:
                best_obj = foam_produced
                best_x1 = x1
                best_x2 = x2

# Print solution details
print(f"Optimal number of demonstration 1: {best_x1}")
print(f"Optimal number of demonstration 2: {best_x2}")
# Final objective print (must be last line in specified format)
# Ensure integer formatting if objective is integer
if abs(best_obj - round(best_obj)) < 1e-9:
    final_value = int(round(best_obj))
else:
    final_value = best_obj

print(f"FINAL_OBJ: {final_value}")
