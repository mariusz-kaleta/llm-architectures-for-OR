[MONOLITH]:
import pulp

# Data initialization
# Resources per container
A_per_max = 46
B_per_max = 43
C_per_max = 56

A_per_pro = 13
B_per_pro = 4
C_per_pro = 45

# Available resources
A_avail = 1345
B_avail = 346
C_avail = 1643

# Profits
profit_max = 10
profit_pro = 15

# Define the problem: maximize profit
prob = pulp.LpProblem("Oil_Production", pulp.LpMaximize)

# Decision variables: number of containers (integers, non-negative)
x = pulp.LpVariable('OilMax_containers', lowBound=0, cat='Integer')      # Oil Max
y = pulp.LpVariable('OilMaxPro_containers', lowBound=0, cat='Integer')   # Oil Max Pro

# Objective
prob += profit_max * x + profit_pro * y, "Total_Profit"

# Resource constraints
prob += A_per_max * x + A_per_pro * y <= A_avail, "A_constraint"
prob += B_per_max * x + B_per_pro * y <= B_avail, "B_constraint"
prob += C_per_max * x + C_per_pro * y <= C_avail, "C_constraint"

# Solve
solver = pulp.PULP_CBC_CMD(msg=False)
prob.solve(solver)

# Extract solution
x_val = int(pulp.value(x))
y_val = int(pulp.value(y))
obj_val = pulp.value(prob.objective)

# Print results
print(f"Oil Max containers (x): {x_val}")
print(f"Oil Max Pro containers (y): {y_val}")
# Final required print (must be the last line)
print(f"FINAL_OBJ: {int(obj_val)}")
