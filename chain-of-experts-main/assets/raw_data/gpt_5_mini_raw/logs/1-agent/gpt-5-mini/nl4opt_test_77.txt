# Optimization of machine usage to minimize waste
# Machine 1: delivers 0.5 heart, 0.8 brain, waste 0.3 per minute
# Machine 2: delivers 0.3 heart, 1.0 brain, waste 0.5 per minute
# Constraints:
#   Heart: 0.5*x1 + 0.3*x2 <= 8
#   Brain: 0.8*x1 + 1.0*x2 >= 4
# Minimize total waste: 0.3*x1 + 0.5*x2

import pulp

# Define the problem
prob = pulp.LpProblem("Minimize_Waste", pulp.LpMinimize)

# Decision variables: minutes machine 1 and machine 2 are used (nonnegative)
x1 = pulp.LpVariable("x1_minutes_machine1", lowBound=0, cat='Continuous')
x2 = pulp.LpVariable("x2_minutes_machine2", lowBound=0, cat='Continuous')

# Objective: minimize waste
prob += 0.3 * x1 + 0.5 * x2, "Total_Waste"

# Constraints
prob += 0.5 * x1 + 0.3 * x2 <= 8, "Heart_capacity_at_most_8"
prob += 0.8 * x1 + 1.0 * x2 >= 4, "Brain_at_least_4"

# Solve
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract results
x1_val = pulp.value(x1)
x2_val = pulp.value(x2)
obj_val = pulp.value(prob.objective)

# Print solution details
print(f"Minutes on machine 1 (x1): {x1_val}")
print(f"Minutes on machine 2 (x2): {x2_val}")
print(f"Total heart medicine delivered: {0.5*x1_val + 0.3*x2_val}")
print(f"Total brain medicine delivered: {0.8*x1_val + 1.0*x2_val}")

# Final required print format (must be the last line)
print(f"FINAL_OBJ: {obj_val}")
