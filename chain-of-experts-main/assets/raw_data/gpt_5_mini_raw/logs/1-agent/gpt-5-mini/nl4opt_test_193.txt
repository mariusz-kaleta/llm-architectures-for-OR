# Brute-force search (no external solver required)
# Data initialization
capacity_small = 20
capacity_large = 50
required_students = 500
max_large_fraction = 0.20  # at most 20% of buses can be large

# Search bounds (reasonable upper bounds)
max_buses = 500  # safe upper bound for number of each type

best_total = None
best_s = None
best_l = None

for s in range(0, max_buses + 1):
    for l in range(0, max_buses + 1):
        total_buses = s + l
        if total_buses == 0:
            continue
        # capacity constraint
        if capacity_small * s + capacity_large * l < required_students:
            continue
        # parking fraction constraint: l <= max_large_fraction * total_buses
        if l > max_large_fraction * total_buses + 1e-9:
            continue
        # feasible, check objective (minimize total buses)
        if best_total is None or total_buses < best_total:
            best_total = total_buses
            best_s = s
            best_l = l

# Output results
if best_total is None:
    print("No feasible solution found.")
else:
    print(f"Optimal number of small buses: {best_s}")
    print(f"Optimal number of large buses: {best_l}")
    print(f"Total buses: {best_total}")

# Final required print exactly in specified format
value = int(best_total) if best_total is not None else None
print(f"FINAL_OBJ: {value}")
