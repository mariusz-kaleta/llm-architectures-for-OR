import pulp

# Define the problem
prob = pulp.LpProblem("Taco_Profit_Maximization", pulp.LpMaximize)

# Decision variables
x1 = pulp.LpVariable("x1_regular_tacos", lowBound=0)  # regular tacos
x2 = pulp.LpVariable("x2_deluxe_tacos", lowBound=0)   # deluxe tacos

# Parameters (from the problem)
profit_x1 = 2.50
profit_x2 = 3.55
demand_x1_max = 50
demand_x2_max = 40
total_tacos_max = 70

# Objective: maximize profit
prob += profit_x1 * x1 + profit_x2 * x2

# Constraints
prob += x1 <= demand_x1_max
prob += x2 <= demand_x2_max
prob += x1 + x2 <= total_tacos_max

# Solve
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Retrieve solution
x1_opt = x1.varValue
x2_opt = x2.varValue
objective_value = pulp.value(prob.objective)

# Optional: print solution details (not required, but helpful)
print(f"Optimal regular tacos (x1): {x1_opt}")
print(f"Optimal deluxe tacos  (x2): {x2_opt}")

# Final required output line
print(f"FINAL_OBJ: {objective_value}")
