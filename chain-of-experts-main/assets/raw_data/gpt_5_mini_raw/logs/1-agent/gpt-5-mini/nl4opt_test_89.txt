import pulp

# Create the LP problem: minimize sugar
prob = pulp.LpProblem("Minimize_Sugar", pulp.LpMinimize)

# Decision variables: number of packs of blueberries (B) and strawberries (S)
# Assume continuous nonnegative packs (can be fractional)
B = pulp.LpVariable('Blueberries', lowBound=0, cat='Continuous')
S = pulp.LpVariable('Strawberries', lowBound=0, cat='Continuous')

# Objective: minimize total sugar (5 grams per blueberry pack, 7 grams per strawberry pack)
prob += 5 * B + 7 * S, "Total_Sugar"

# Constraints:
# 1) Anti-oxidants: 3B + 1S >= 90
prob += 3 * B + 1 * S >= 90, "Antioxidants"

# 2) Minerals: 5B + 7S >= 100
prob += 5 * B + 7 * S >= 100, "Minerals"

# 3) At least 3 times as many packs of strawberries as blueberries: S >= 3B
prob += S >= 3 * B, "Strawberries_vs_Blueberries"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract results
B_val = pulp.value(B)
S_val = pulp.value(S)
obj_val = pulp.value(prob.objective)

# Print solution details
print(f"Blueberries_packs = {B_val}")
print(f"Strawberries_packs = {S_val}")
print(f"Total_sugar_grams = {obj_val}")

# Final required print statement (must be the last line)
print(f"FINAL_OBJ: {obj_val}")
