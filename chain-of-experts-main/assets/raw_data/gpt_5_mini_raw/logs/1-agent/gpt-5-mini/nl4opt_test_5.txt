# Linear program to maximize profit for flooring production (engineered hardwood and vinyl)
# The problem data is initialized directly below.

import pulp

# Data initialization
profit_hardware = 2.5   # profit per sq ft for hardwood
profit_vinyl = 3.0      # profit per sq ft for vinyl

min_hardware = 20000    # minimum demand per week (sq ft)
min_vinyl = 10000       # minimum demand per week (sq ft)
min_total = 60000       # shipping contract: at least this total must be shipped (sq ft)

max_hardware = 50000    # production capacity limit for hardwood (sq ft)
max_vinyl = 30000       # production capacity limit for vinyl (sq ft)

# Define the problem
prob = pulp.LpProblem("Flooring_Production", pulp.LpMaximize)

# Decision variables (with bounds)
H = pulp.LpVariable("Hardwood_sqft", lowBound=min_hardware, upBound=max_hardware, cat='Continuous')
V = pulp.LpVariable("Vinyl_sqft", lowBound=min_vinyl, upBound=max_vinyl, cat='Continuous')

# Objective: maximize profit
prob += profit_hardware * H + profit_vinyl * V, "Total_Profit"

# Constraints
prob += H + V >= min_total, "Shipping_contract_total"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Retrieve results
hardwood_produced = pulp.value(H)
vinyl_produced = pulp.value(V)
objective_value = pulp.value(prob.objective)

# Print results
print(f"Hardwood produced (sq ft): {hardwood_produced}")
print(f"Vinyl produced (sq ft): {vinyl_produced}")
print(f"Total profit: {objective_value}")

# Final required output line (must be the last line)
print(f"FINAL_OBJ: {objective_value}")
