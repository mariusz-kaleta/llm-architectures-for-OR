import math

# Data initialization
capacity_old = 100   # bottles per old van
capacity_new = 80    # bottles per new van
pollution_old = 50   # pollution units per old van
pollution_new = 30   # pollution units per new van
required_bottles = 5000
max_new_vans = 30

# Brute-force search over feasible integer solutions (y = number of new vans)
best_obj = None
best_x = None
best_y = None

for y in range(0, max_new_vans + 1):
    # minimum old vans needed to satisfy capacity, at least 0
    remaining = required_bottles - capacity_new * y
    if remaining <= 0:
        x = 0
    else:
        x = math.ceil(remaining / capacity_old)
    # compute objective and check feasibility
    total_capacity = capacity_old * x + capacity_new * y
    if total_capacity >= required_bottles:
        obj = pollution_old * x + pollution_new * y
        if best_obj is None or obj < best_obj:
            best_obj = obj
            best_x = x
            best_y = y

# Print solution details
print(f"Number of old vans (x): {best_x}")
print(f"Number of new vans (y): {best_y}")
print(f"Total pollution: {best_obj}")

# Final required output line (must be last)
print(f"FINAL_OBJ: {best_obj}")
