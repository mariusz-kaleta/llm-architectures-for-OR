import pulp

# Data initialization
cost_long = 300     # cost of long desk
cost_short = 100    # cost of short desk
area_long = 10      # area of long desk (sq ft)
area_short = 4      # area of short desk (sq ft)
seats_long = 6      # seats per long desk
seats_short = 2     # seats per short desk

budget = 2000       # maximum spending
area_limit = 200    # maximum area

# Define the problem
prob = pulp.LpProblem("Office_Desk_Placement", pulp.LpMaximize)

# Decision variables: integer numbers of desks
x = pulp.LpVariable("long_desks", lowBound=0, cat='Integer')   # number of long desks
y = pulp.LpVariable("short_desks", lowBound=0, cat='Integer')  # number of short desks

# Objective: maximize total seating
prob += seats_long * x + seats_short * y

# Constraints
prob += cost_long * x + cost_short * y <= budget
prob += area_long * x + area_short * y <= area_limit

# Solve
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract results
x_val = int(pulp.value(x))
y_val = int(pulp.value(y))
final_obj = pulp.value(prob.objective)

# Print solution
print(f"Long desks (x) = {x_val}")
print(f"Short desks (y) = {y_val}")
print(f"Total seats = {int(final_obj)}")

# Final required print (must be the last line)
print(f"FINAL_OBJ: {final_obj}")
