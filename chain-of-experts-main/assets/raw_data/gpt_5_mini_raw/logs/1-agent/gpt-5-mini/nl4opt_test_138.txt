import math

# Data from the problem
protein_per_smoothie = 2
calories_per_smoothie = 300

protein_per_bar = 7
calories_per_bar = 250

max_calories = 2000

# Relation: "must eat 2 times more protein bars than smoothies" -> y = 2*x
# We'll search integer values for x (smoothies) and set y accordingly.
best_protein = -1
best_x = None
best_y = None
best_calories = None

# Reasonable upper bound for x: can't exceed max_calories / min calorie per combination
# but we'll just search a small range safely
for x in range(0, 50):  # 50 is safely large for this small problem
    y = 2 * x
    total_cal = calories_per_smoothie * x + calories_per_bar * y
    if total_cal <= max_calories:
        total_protein = protein_per_smoothie * x + protein_per_bar * y
        if total_protein > best_protein:
            best_protein = total_protein
            best_x = x
            best_y = y
            best_calories = total_cal

# Print solution
print(f"Smoothies (x): {best_x}")
print(f"Protein bars (y): {best_y}")
print(f"Total calories: {best_calories}")
print(f"Total protein: {best_protein}")

# Final required print (must be the last line)
print(f"FINAL_OBJ: {best_protein}")
