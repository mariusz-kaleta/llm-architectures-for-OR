import pulp

# Define the LP problem: maximize profit
prob = pulp.LpProblem("Bakery_Production", pulp.LpMaximize)

# Decision variables
x = pulp.LpVariable("bread_loaves", lowBound=0)    # number of loaves of bread
y = pulp.LpVariable("cookie_batches", lowBound=0)  # number of cookie batches

# Objective function: maximize 5*x + 3*y
prob += 5 * x + 3 * y

# Constraints
prob += 1 * x + 0.5 * y <= 3000    # mixer hours constraint
prob += 3 * x + 1 * y <= 3000      # oven hours constraint

# Solve
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Retrieve results
bread_opt = pulp.value(x)
cookies_opt = pulp.value(y)
objective_value = pulp.value(prob.objective)

# Print solution details
print(f"Optimal bread loaves: {bread_opt}")
print(f"Optimal cookie batches: {cookies_opt}")
print(f"Total profit: {objective_value}")

# Required final line (must be the last printed line)
value = objective_value
print(f"FINAL_OBJ: {value}")
