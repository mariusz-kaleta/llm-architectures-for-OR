import sys

# Brute-force search for integer solution
best_obj = None
best_c = None
best_e = None

# We can bound elephants reasonably: more than 1000/50 = 20 elephants is never helpful
# (each extra elephant forces at least one cow, so increases total animals).
for e in range(1, 101):  # 1..100 safe upper bound
    # cows must satisfy both:
    # 20*c + 50*e >= 1000  -> c >= (1000 - 50*e)/20
    # e <= c <= 2*e
    min_c_capacity = (1000 - 50 * e + 19) // 20  # ceil division for integer c
    if min_c_capacity < 0:
        min_c_capacity = 0
    c_min = max(e, min_c_capacity)
    c_max = 2 * e
    if c_min <= c_max:
        # feasible, choose smallest c to minimize total animals
        c = c_min
        obj = c + e
        if best_obj is None or obj < best_obj:
            best_obj = obj
            best_c = c
            best_e = e

# Print found solution
if best_obj is None:
    print("No feasible solution found.")
    # Still must print FINAL_OBJ line; use None or a sentinel
    print(f"FINAL_OBJ: {None}")
    sys.exit(0)

print(f"Minimum total animals: {best_obj}")
print(f"Cows (c): {best_c}")
print(f"Elephants (e): {best_e}")
# Final required line (must be last)
print(f"FINAL_OBJ: {best_obj}")
