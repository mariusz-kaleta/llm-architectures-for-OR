import pulp

# Data (initialized directly from the problem statement)
max_wired = 100        # max wired headphones per day
max_wireless = 170     # max wireless headphones per day
max_testing = 150      # max headphones that can be tested per day (shared machine)
profit_wired = 50      # profit per wired headphone
profit_wireless = 20   # profit per wireless headphone

# Define the problem
prob = pulp.LpProblem("Headphones_Profit_Maximization", pulp.LpMaximize)

# Decision variables (integer number of headphones)
wired = pulp.LpVariable("wired", lowBound=0, cat="Integer")
wireless = pulp.LpVariable("wireless", lowBound=0, cat="Integer")

# Objective: maximize profit
prob += profit_wired * wired + profit_wireless * wireless

# Constraints
prob += wired <= max_wired
prob += wireless <= max_wireless
prob += wired + wireless <= max_testing

# Solve
prob.solve()

# Extract objective value
value = pulp.value(prob.objective)

# Optional: print solution details
print("Status:", pulp.LpStatus[prob.status])
print(f"Wired produced: {int(pulp.value(wired))}")
print(f"Wireless produced: {int(pulp.value(wireless))}")

print(f"FINAL_OBJ: {value}")
