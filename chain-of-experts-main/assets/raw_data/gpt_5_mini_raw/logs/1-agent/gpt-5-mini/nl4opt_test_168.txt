# Optimization of production mix for basketballs and footballs
# Using PuLP for integer linear programming

import pulp

# Data (from problem statement)
# Resources requirements
materials_per_basketball = 5
materials_per_football = 3
hours_per_basketball = 1
hours_per_football = 2

# Available resources
total_materials = 1500
total_hours = 750

# Demand / policy constraints
min_footballs = 50
# basketballs must be at least three times footballs: B >= 3F

# Define the problem: maximize total number of sports equipment (B + F)
prob = pulp.LpProblem("Maximize_Total_Equipment", pulp.LpMaximize)

# Decision variables: integers
B = pulp.LpVariable("Basketballs", lowBound=0, cat='Integer')
F = pulp.LpVariable("Footballs", lowBound=0, cat='Integer')

# Objective
prob += B + F, "Total_Equipment"

# Constraints
prob += materials_per_basketball * B + materials_per_football * F <= total_materials, "Materials"
prob += hours_per_basketball * B + hours_per_football * F <= total_hours, "Hours"
prob += B >= 3 * F, "Basketballs_at_least_three_times_footballs"
prob += F >= min_footballs, "Minimum_footballs"

# Solve
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract results
B_val = int(pulp.value(B))
F_val = int(pulp.value(F))
obj_val = pulp.value(prob.objective)
# Round objective to integer (should be integer)
obj_int = int(round(obj_val))

# Print solution details and final objective as required
print(f"Basketballs = {B_val}")
print(f"Footballs = {F_val}")
print(f"Objective (total equipment) = {obj_int}")
print(f"FINAL_OBJ: {obj_int}")
