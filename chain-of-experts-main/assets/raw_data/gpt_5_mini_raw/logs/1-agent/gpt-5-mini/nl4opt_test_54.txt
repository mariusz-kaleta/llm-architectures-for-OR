import pulp

# Problem data
cost_milk = 1
cost_veg = 2

calcium_milk = 40
calcium_veg = 15
iron_milk = 25
iron_veg = 30

min_calcium = 100
min_iron = 50

# Define problem
prob = pulp.LpProblem("Minimize_Cost_for_Nutrients", pulp.LpMinimize)

# Decision variables: number of glasses of milk (x) and plates of vegetables (y)
x = pulp.LpVariable('milk_glasses', lowBound=0, cat='Continuous')
y = pulp.LpVariable('veg_plates', lowBound=0, cat='Continuous')

# Objective: minimize cost
prob += cost_milk * x + cost_veg * y

# Constraints
prob += calcium_milk * x + calcium_veg * y >= min_calcium, "CalciumRequirement"
prob += iron_milk * x + iron_veg * y >= min_iron, "IronRequirement"

# Solve
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract results
milk_amount = x.varValue if x.varValue is not None else 0.0
veg_amount = y.varValue if y.varValue is not None else 0.0
objective_value = pulp.value(prob.objective)

# Print solution details
print(f"Milk (glasses): {milk_amount}")
print(f"Vegetables (plates): {veg_amount}")
print(f"Cost: {objective_value}")

# Final required print
print(f"FINAL_OBJ: {objective_value}")
