import math

# Problem data (initialized directly from the problem statement)
total_cows = 80
helicopter_capacity = 3
truck_capacity = 7
pollution_helicopter = 5
pollution_truck = 10
max_truck_trips = 8

# Brute-force search over feasible integer number of truck trips (small domain)
best_obj = float('inf')
best_h = None
best_t = None

for t in range(0, max_truck_trips + 1):
    cows_by_trucks = t * truck_capacity
    remaining_cows = max(0, total_cows - cows_by_trucks)
    # minimal helicopter trips needed (integer)
    h = math.ceil(remaining_cows / helicopter_capacity) if remaining_cows > 0 else 0
    total_cows_transported = cows_by_trucks + h * helicopter_capacity
    # ensure feasibility (should be by construction)
    if total_cows_transported >= total_cows:
        obj = pollution_truck * t + pollution_helicopter * h
        if obj < best_obj:
            best_obj = obj
            best_h = h
            best_t = t

# Output the optimal solution
print(f"Optimal helicopter trips: {best_h}")
print(f"Optimal truck trips: {best_t}")
print(f"Total pollution: {best_obj}")

# Final required line (must be last)
print(f"FINAL_OBJ: {best_obj}")
