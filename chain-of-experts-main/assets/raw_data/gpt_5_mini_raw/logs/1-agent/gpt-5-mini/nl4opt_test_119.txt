import math

# Problem data (initialized directly)
capacity_regular = 20    # pieces per regular trip
capacity_speed = 30      # pieces per speed trip
gas_regular = 10         # liters per regular trip
gas_speed = 20           # liters per speed trip
max_regular_trips = 20
required_mail = 1000

# Brute-force integer search (small feasible space)
best_gas = None
best_r = None
best_s = None

# Upper bound for speed trips: enough to cover demand alone
max_speed_trips = math.ceil(required_mail / capacity_speed) + 10

for r in range(0, max_regular_trips + 1):
    for s in range(r, max_speed_trips + 1):  # s >= r (at least 50% of trips by speed boats)
        if capacity_regular * r + capacity_speed * s >= required_mail:
            gas = gas_regular * r + gas_speed * s
            if best_gas is None or gas < best_gas:
                best_gas = gas
                best_r = r
                best_s = s

# Output results
print(f"Optimal regular trips: {best_r}")
print(f"Optimal speed trips: {best_s}")
print(f"Total gas consumed: {best_gas} liters")
print(f"FINAL_OBJ: {best_gas}")
