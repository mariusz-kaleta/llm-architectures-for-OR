import math

# Data initialization
TOTAL_WATER_ML = 250000
GLASS_VOLUME = 500    # ml per glass bottle
PLASTIC_VOLUME = 750  # ml per plastic bottle
MIN_GLASS = 20
PLASTIC_TO_GLASS_RATIO = 3  # P >= 3 * G

best_obj = -1
best_solution = None

# Search integer feasible solutions
max_possible_glass = TOTAL_WATER_ML // GLASS_VOLUME
for G in range(MIN_GLASS, max_possible_glass + 1):
    remaining_water = TOTAL_WATER_ML - GLASS_VOLUME * G
    if remaining_water < 0:
        break
    P_max = remaining_water // PLASTIC_VOLUME
    P_min = PLASTIC_TO_GLASS_RATIO * G
    if P_max < P_min:
        continue  # no feasible P for this G
    # For fixed G, objective G+P is maximized at P = P_max
    P = P_max
    obj = G + P
    # Tie-breaker: prefer solution with more glass bottles if same objective
    if obj > best_obj or (obj == best_obj and G > best_solution[0]):
        best_obj = obj
        best_solution = (G, P)

if best_solution is None:
    print("No feasible solution found.")
else:
    G_opt, P_opt = best_solution
    water_used = GLASS_VOLUME * G_opt + PLASTIC_VOLUME * P_opt
    print(f"Optimal glass bottles: {G_opt}")
    print(f"Optimal plastic bottles: {P_opt}")
    print(f"Total bottles: {best_obj}")
    print(f"Water used (ml): {water_used}")

# Must print final objective value in the required format as the last line
print(f"FINAL_OBJ: {best_obj}")
