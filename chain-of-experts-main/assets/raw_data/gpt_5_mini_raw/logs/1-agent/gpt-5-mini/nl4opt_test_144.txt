import pulp

# Data initialization based on problem statement
# Variables: servings of vegetables (v) and fruits (f)
cost_veg = 3   # $ per serving of vegetables
cost_fruit = 5 # $ per serving of fruit

# Nutrient content per serving
vitamin_veg = 2
vitamin_fruit = 4

mineral_veg = 3
mineral_fruit = 1

# Requirements
min_vitamins = 20
min_minerals = 30

# Define LP problem: minimize cost
prob = pulp.LpProblem("Minimize_Cost_of_Servings", pulp.LpMinimize)

# Decision variables (continuous, non-negative)
v = pulp.LpVariable("vegetable_servings", lowBound=0, cat='Continuous')
f = pulp.LpVariable("fruit_servings", lowBound=0, cat='Continuous')

# Objective
prob += cost_veg * v + cost_fruit * f

# Constraints
prob += vitamin_veg * v + vitamin_fruit * f >= min_vitamins, "Vitamin_requirement"
prob += mineral_veg * v + mineral_fruit * f >= min_minerals, "Mineral_requirement"

# Solve
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Retrieve solution
veg_servings = v.varValue if v.varValue is not None else 0.0
fruit_servings = f.varValue if f.varValue is not None else 0.0
total_cost = cost_veg * veg_servings + cost_fruit * fruit_servings

# Print servings (optional), final line must be the required format
print(f"Vegetable_servings: {veg_servings}")
print(f"Fruit_servings: {fruit_servings}")
print(f"Total_cost: {total_cost}")
print(f"FINAL_OBJ: {total_cost}")
