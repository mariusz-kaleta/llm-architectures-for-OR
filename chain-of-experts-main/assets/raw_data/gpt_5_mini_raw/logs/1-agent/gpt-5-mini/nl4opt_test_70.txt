import math

# Problem data (initialized directly)
capacity_new = 50        # gifts per trip by new company
capacity_old = 70        # gifts per trip by old company
diesel_new = 30          # liters per trip by new company
diesel_old = 40          # liters per trip by old company
required_gifts = 1000
max_new_trips = 15
# At least 40% of all trips must be by the old company.
# Equivalent integer constraint: 3*y >= 2*x  (since y >= 0.4*(x+y) -> 3y >= 2x)

# Brute-force search over integer trip counts (small search space)
best_obj = None
best_x = None
best_y = None

# Upper bound for y: sufficient to consider up to required_gifts (very conservative)
max_y = required_gifts

for x in range(0, max_new_trips + 1):
    for y in range(0, max_y + 1):
        # Gift requirement
        if capacity_new * x + capacity_old * y < required_gifts:
            continue
        # Old-company share requirement: 3*y >= 2*x
        if 3 * y < 2 * x:
            continue
        diesel = diesel_new * x + diesel_old * y
        if (best_obj is None) or (diesel < best_obj):
            best_obj = diesel
            best_x = x
            best_y = y

# Print results
print(f"New company trips (x): {best_x}")
print(f"Old company trips (y): {best_y}")
print(f"Total diesel used (liters): {best_obj}")
print(f"FINAL_OBJ: {best_obj}")
