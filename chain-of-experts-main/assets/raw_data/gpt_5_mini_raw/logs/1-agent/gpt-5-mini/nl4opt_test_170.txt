import pulp

# Define the problem
prob = pulp.LpProblem("Maximize_Honey_Transport", pulp.LpMaximize)

# Decision variables
s = pulp.LpVariable('s', lowBound=0, upBound=300, cat='Integer')  # small bottles
l = pulp.LpVariable('l', lowBound=0, upBound=100, cat='Integer')  # large bottles

# Objective: maximize total honey transported (5 units per small, 20 per large)
prob += 5 * s + 20 * l

# Constraints
prob += s <= 300
prob += l <= 100
prob += s >= 2 * l            # at least twice as many small as large
prob += s + l <= 200          # at most 200 bottles total
prob += l >= 50               # at least 50 large bottles

# Solve
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract results
s_val = int(pulp.value(s))
l_val = int(pulp.value(l))
obj_val = int(pulp.value(prob.objective))

# Print solution
print(f"Small bottles (s): {s_val}")
print(f"Large bottles (l): {l_val}")
print(f"Total honey transported: {obj_val} units")
print(f"FINAL_OBJ: {obj_val}")
