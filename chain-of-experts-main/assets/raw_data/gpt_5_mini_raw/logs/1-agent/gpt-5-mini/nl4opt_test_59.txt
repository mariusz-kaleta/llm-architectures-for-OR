import pulp

# Define the problem
prob = pulp.LpProblem("Maximize_Electricity", pulp.LpMaximize)

# Decision variables: number of experiments alpha (a) and beta (b)
a = pulp.LpVariable('alpha_experiments', lowBound=0, cat='Continuous')
b = pulp.LpVariable('beta_experiments', lowBound=0, cat='Continuous')

# Objective: maximize electricity produced
# Alpha produces 8 units, Beta produces 10 units
prob += 8*a + 10*b, "Total_Electricity"

# Constraints:
# Metal: 3 units per alpha, 5 units per beta, available 800
prob += 3*a + 5*b <= 800, "Metal_Constraint"
# Acid: 5 units per alpha, 4 units per beta, available 750
prob += 5*a + 4*b <= 750, "Acid_Constraint"

# Solve the problem
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Retrieve results
alpha_opt = a.varValue
beta_opt = b.varValue
value = pulp.value(prob.objective)

# Print results (decision variables)
print(f"Optimal alpha experiments: {alpha_opt}")
print(f"Optimal beta experiments: {beta_opt}")

# Final objective value printed in the required format
print(f"FINAL_OBJ: {value}")
