import pulp

# Problem data (from the problem statement)
# Each wrap requires: 5 units meat, 3 units rice, 10 minutes
# Each platter requires: 7 units meat, 5 units rice, 8 minutes
# Requirements: at least 3000 units meat, 2500 units rice
# At least 3 times as many wraps as platters: W >= 3 * P

# Create LP problem (minimize total production time)
prob = pulp.LpProblem("Minimize_Production_Time", pulp.LpMinimize)

# Decision variables: number of wraps (W) and platters (P)
# Variables are non-negative and can be continuous (large production scale)
W = pulp.LpVariable("Wraps", lowBound=0)
P = pulp.LpVariable("Platters", lowBound=0)

# Objective: minimize total production time (minutes)
prob += 10 * W + 8 * P, "Total_Time"

# Constraints
prob += 5 * W + 7 * P >= 3000, "Meat_requirement"
prob += 3 * W + 5 * P >= 2500, "Rice_requirement"
prob += W >= 3 * P, "Wraps_at_least_three_times_platters"

# Solve
solver = pulp.PULP_CBC_CMD(msg=False)
prob.solve(solver)

# Retrieve results
wraps_opt = pulp.value(W)
platters_opt = pulp.value(P)
obj_opt = pulp.value(prob.objective)

# Print solution details
print(f"Optimal wraps (W): {wraps_opt}")
print(f"Optimal platters (P): {platters_opt}")
print(f"Total production time (minutes): {obj_opt}")

# Final required print (must be the last line)
value = obj_opt
print(f"FINAL_OBJ: {value}")
