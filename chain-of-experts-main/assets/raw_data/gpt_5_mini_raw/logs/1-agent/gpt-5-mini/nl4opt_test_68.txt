import pulp

# Data (from the problem statement)
# Production rates (pills per hour)
f1_allergy = 20
f1_fever = 15
f2_allergy = 10
f2_fever = 30

# Compound usage (units per hour)
f1_compound = 20
f2_compound = 30
compound_available = 1000

# Demand requirements (pills)
required_allergy = 700
required_fever = 600

# Decision variables: hours to run each factory
h1 = pulp.LpVariable("h1_hours_factory1", lowBound=0, cat="Continuous")
h2 = pulp.LpVariable("h2_hours_factory2", lowBound=0, cat="Continuous")

# Define the LP problem (minimize total hours)
prob = pulp.LpProblem("Minimize_Total_Hours", pulp.LpMinimize)
prob += h1 + h2  # objective

# Constraints
prob += f1_allergy * h1 + f2_allergy * h2 >= required_allergy, "Allergy_requirement"
prob += f1_fever * h1 + f2_fever * h2 >= required_fever, "Fever_requirement"
prob += f1_compound * h1 + f2_compound * h2 <= compound_available, "Compound_limit"

# Solve
prob.solve()

# Retrieve results
h1_val = pulp.value(h1)
h2_val = pulp.value(h2)
obj_val = pulp.value(prob.objective)

# Print hours for each factory (informational)
print(f"Hours_factory1: {h1_val}")
print(f"Hours_factory2: {h2_val}")

# Final required print in specified format
print(f"FINAL_OBJ: {obj_val}")
