import math

# Problem data (from the problem statement)
total_employees = 500
taxi_capacity = 2
car_capacity = 3
max_car_fraction = 0.60  # at most 60% of rides can be company car rides
min_company_cars = 30

# We search for integers x (taxi rides) and y (company car rides)
# Objective: minimize x
solution_found = False
x_sol = None
y_sol = None

# Upper bound for taxi rides: if only taxis used, need ceil(500/2)=250 taxis
for x in range(0, 251):
    # Maximum company cars allowed by fraction constraint: y <= 1.5 * x
    y_max = math.floor((max_car_fraction / (1 - max_car_fraction)) * x) if x > 0 else 0
    # Minimum company cars required to meet capacity (and at least min_company_cars)
    remaining_employees = max(0, total_employees - taxi_capacity * x)
    y_needed_for_capacity = math.ceil(remaining_employees / car_capacity) if remaining_employees > 0 else 0
    y_min = max(min_company_cars, y_needed_for_capacity)
    if y_min <= y_max:
        x_sol = x
        y_sol = y_min
        solution_found = True
        break

if not solution_found:
    raise ValueError("No feasible solution found with given constraints.")

print(f"Taxi rides: {x_sol}")
print(f"Company car rides: {y_sol}")
print(f"Total employees transported: {taxi_capacity*x_sol + car_capacity*y_sol}")
print(f"Fraction company car rides: {y_sol/(x_sol+y_sol):.4f}")

# Final required output line (must be the last line)
print(f"FINAL_OBJ: {x_sol}")
