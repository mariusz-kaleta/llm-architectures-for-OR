import pulp

# Data
capacity_van = 50      # pairs per van
capacity_truck = 100   # pairs per truck
demand = 2000          # required pairs

# Define problem: minimize number of vans
prob = pulp.LpProblem("Minimize_Vans", pulp.LpMinimize)

# Decision variables: number of vans and trucks (integers, >=0)
v = pulp.LpVariable('vans', lowBound=0, cat='Integer')
t = pulp.LpVariable('trucks', lowBound=0, cat='Integer')

# Objective: minimize vans
prob += v

# Constraints
prob += capacity_van * v + capacity_truck * t >= demand   # meet demand
prob += t <= v                                            # trucks cannot exceed vans

# Solve
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Get optimal number of vans (as integer)
v_opt = int(pulp.value(v))

print(f"FINAL_OBJ: {v_opt}")
