import pulp

# Data initialization
costs = {'C': 2, 'Y': 3}
nitrous = {'C': 1.5, 'Y': 5}    # units of nitrous oxide per kg
vitamin = {'C': 3, 'Y': 1}      # units of vitamin mix per kg
req_nitrous = 5
req_vitamin = 8

# Define the LP problem: minimize cost
prob = pulp.LpProblem("Minimize_Cost", pulp.LpMinimize)

# Decision variables: kilograms of C and Y (nonnegative)
C = pulp.LpVariable('C', lowBound=0)
Y = pulp.LpVariable('Y', lowBound=0)

# Objective
prob += costs['C'] * C + costs['Y'] * Y

# Constraints
prob += nitrous['C'] * C + nitrous['Y'] * Y >= req_nitrous
prob += vitamin['C'] * C + vitamin['Y'] * Y >= req_vitamin

# Solve
prob.solve()

# Extract final objective value
final_obj = pulp.value(prob.objective)

print(f"FINAL_OBJ: {final_obj}")
