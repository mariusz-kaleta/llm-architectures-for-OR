import pulp

# Define the problem
prob = pulp.LpProblem("Medication_Time_Minimization", pulp.LpMinimize)

# Decision variables: number of units (integers)
A = pulp.LpVariable("Anxiety_units", lowBound=0, cat='Integer')      # anxiety medication units
D = pulp.LpVariable("Antidepressant_units", lowBound=0, cat='Integer')  # anti-depressant units

# Objective: minimize total time (minutes)
prob += 3 * A + 5 * D, "Total_Effective_Time"

# Constraints
prob += A + D >= 100        # at least 100 units total
prob += A >= 30             # at least 30 should be anxiety medication
prob += A <= 2 * D          # at most twice the amount of anxiety as anti-depressants

# Solve
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Retrieve results
A_val = int(pulp.value(A))
D_val = int(pulp.value(D))
obj_val = pulp.value(prob.objective)

# Print solution details
print(f"Anxiety units (A): {A_val}")
print(f"Antidepressant units (D): {D_val}")
print(f"Total time (minutes): {obj_val}")

# Final required output line
value = obj_val
print(f"FINAL_OBJ: {value}")
