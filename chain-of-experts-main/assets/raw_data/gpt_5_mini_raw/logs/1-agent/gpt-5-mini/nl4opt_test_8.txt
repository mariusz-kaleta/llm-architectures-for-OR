# Optimization for scooter and bike production to maximize profit
# Problem data initialized directly

from pulp import LpProblem, LpVariable, LpMaximize, lpSum, value, PULP_CBC_CMD

# Data
profit_scooter = 200  # profit per scooter
profit_bike = 300     # profit per bike

design_hours_per_scooter = 2
design_hours_per_bike = 4
engineering_hours_per_scooter = 3
engineering_hours_per_bike = 5

design_capacity = 5000
engineering_capacity = 6000

# Define the LP problem
prob = LpProblem("Scooter_Bike_Profit_Maximization", LpMaximize)

# Decision variables: number of scooters (x) and bikes (y)
x = LpVariable("scooters", lowBound=0, cat="Continuous")
y = LpVariable("bikes", lowBound=0, cat="Continuous")

# Objective: maximize profit
prob += profit_scooter * x + profit_bike * y

# Constraints
prob += design_hours_per_scooter * x + design_hours_per_bike * y <= design_capacity, "DesignCapacity"
prob += engineering_hours_per_scooter * x + engineering_hours_per_bike * y <= engineering_capacity, "EngineeringCapacity"

# Solve
prob.solve(PULP_CBC_CMD(msg=0))

# Retrieve results
x_val = value(x)
y_val = value(y)
obj_val = value(prob.objective)

# Print solution details
print(f"Optimal scooters: {x_val}")
print(f"Optimal bikes: {y_val}")

# Final objective printed in required format
print(f"FINAL_OBJ: {obj_val}")
