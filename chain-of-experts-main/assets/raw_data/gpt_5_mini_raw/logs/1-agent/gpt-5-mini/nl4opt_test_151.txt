import pulp

# Define the LP problem: maximize zinc intake
prob = pulp.LpProblem("Maximize_Zinc", pulp.LpMaximize)

# Decision variables
G = pulp.LpVariable("Gummies", lowBound=0, cat="Continuous")  # number of gummies
P = pulp.LpVariable("Pills", lowBound=0, cat="Continuous")    # number of pills

# Objective: maximize zinc = 4*G + 5*P
prob += 4 * G + 5 * P

# Constraints
prob += 3 * G + 2 * P <= 200    # magnesium limit
prob += P >= 10                 # at least 10 pills
prob += G >= 3 * P              # at least 3 times as many gummies as pills

# Solve
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract results
g_opt = G.varValue
p_opt = P.varValue
obj = pulp.value(prob.objective)

# Print results
print(f"Optimal number of gummies: {g_opt}")
print(f"Optimal number of pills: {p_opt}")
print(f"Maximum zinc intake: {obj}")

# Final required output line
print(f"FINAL_OBJ: {obj}")
