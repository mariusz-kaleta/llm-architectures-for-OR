# Solve the suitcase packing problem by enumeration (integer variables)
# Data initialization based on the problem statement
small_capacity_per = 50   # snacks per small suitcase
large_capacity_per = 80   # snacks per large suitcase

max_small_available = 70
max_large_available = 50

min_large_required = 15
max_total_suitcases = 70

# Constraints: at least twice as many small suitcases as large suitcases: s >= 2*l
# Variables s and l are nonnegative integers.

best_obj = -1
best_s = None
best_l = None

# Enumerate feasible integer choices
for l in range(min_large_required, min(max_large_available, max_total_suitcases) + 1):
    for s in range(0, max_small_available + 1):
        # Check constraints
        if s < 2 * l:
            continue
        if s + l > max_total_suitcases:
            continue
        # feasible, compute objective
        obj = small_capacity_per * s + large_capacity_per * l
        if obj > best_obj:
            best_obj = obj
            best_s = s
            best_l = l

# Output solution
print(f"Small suitcases: {best_s}")
print(f"Large suitcases: {best_l}")
print(f"Total snacks: {best_obj}")
print(f"FINAL_OBJ: {best_obj}")
