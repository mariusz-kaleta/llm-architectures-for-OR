import pulp

# Problem data (initialized directly)
# Variables: x1 = minutes of Beam 1, x2 = minutes of Beam 2
# Pancreas dose rates (objective): 0.3*x1 + 0.2*x2 (to minimize)
# Skin dose constraint: 0.2*x1 + 0.1*x2 <= 4
# Tumor dose constraint: 0.6*x1 + 0.4*x2 >= 3
# Nonnegativity: x1, x2 >= 0

# Create the LP problem (minimization)
prob = pulp.LpProblem("Minimize_Pancreas_Dose", pulp.LpMinimize)

# Decision variables
x1 = pulp.LpVariable("x1_minutes_beam1", lowBound=0, cat='Continuous')
x2 = pulp.LpVariable("x2_minutes_beam2", lowBound=0, cat='Continuous')

# Objective
prob += 0.3 * x1 + 0.2 * x2, "Pancreas_Dose"

# Constraints
prob += 0.2 * x1 + 0.1 * x2 <= 4, "Skin_Max_4"
prob += 0.6 * x1 + 0.4 * x2 >= 3, "Tumor_Min_3"

# Solve
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract results
x1_val = float(pulp.value(x1))
x2_val = float(pulp.value(x2))
obj_val = float(pulp.value(prob.objective))

# Print solution details
print(f"x1 (minutes of Beam 1) = {x1_val}")
print(f"x2 (minutes of Beam 2) = {x2_val}")
print(f"Pancreas dose (objective) = {obj_val}")

# Final required print (must be the last line)
print(f"FINAL_OBJ: {obj_val}")
