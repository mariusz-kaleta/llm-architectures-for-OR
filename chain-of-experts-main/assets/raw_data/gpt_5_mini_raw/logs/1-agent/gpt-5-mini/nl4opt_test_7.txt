import pulp

# Problem data (from the prompt)
profit_chair = 43
profit_dresser = 52

stain_available = 17.0  # gallons
oak_available = 11.0    # lengths

stain_per_chair = 1.4
stain_per_dresser = 1.1

oak_per_chair = 2.0
oak_per_dresser = 3.0

# Define LP problem (maximization)
prob = pulp.LpProblem("Elm_Furniture_Profit_Maximization", pulp.LpMaximize)

# Decision variables: number of chairs and dressers (continuous, >= 0)
chairs = pulp.LpVariable("chairs", lowBound=0)
dressers = pulp.LpVariable("dressers", lowBound=0)

# Objective: maximize profit
prob += profit_chair * chairs + profit_dresser * dressers

# Constraints
prob += stain_per_chair * chairs + stain_per_dresser * dressers <= stain_available, "StainConstraint"
prob += oak_per_chair * chairs + oak_per_dresser * dressers <= oak_available, "OakConstraint"

# Solve
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Get final objective value
final_obj = pulp.value(prob.objective)

print(f"FINAL_OBJ: {final_obj}")
