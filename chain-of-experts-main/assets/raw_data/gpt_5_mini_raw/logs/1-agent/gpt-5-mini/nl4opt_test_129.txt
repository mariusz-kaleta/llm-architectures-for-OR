# Optimization by enumeration for the airport vehicle purchase problem

# Data initialization (from problem statement)
cap_luggage_4 = 60    # luggage per day by a 4-wheeler
cap_luggage_3 = 40    # luggage per day by a 3-wheeler
pollut_4 = 30         # pollution per day by a 4-wheeler
pollut_3 = 15         # pollution per day by a 3-wheeler

required_luggage = 1000
max_pollution = 430

# Reasonable bounds for enumeration
# Maximum number of vehicles of each type won't exceed the number needed if only that type is used
max_4 = (required_luggage + cap_luggage_4 - 1) // cap_luggage_4 + 10  # +10 margin
max_3 = (required_luggage + cap_luggage_3 - 1) // cap_luggage_3 + 10

best_solutions = []
best_total = None

for x in range(0, max_4 + 1):        # x = number of 4-wheelers
    for y in range(0, max_3 + 1):    # y = number of 3-wheelers
        luggage = cap_luggage_4 * x + cap_luggage_3 * y
        pollution = pollut_4 * x + pollut_3 * y
        if luggage >= required_luggage and pollution <= max_pollution:
            total = x + y
            if (best_total is None) or (total < best_total):
                best_total = total
                best_solutions = [(x, y, luggage, pollution)]
            elif total == best_total:
                best_solutions.append((x, y, luggage, pollution))

# If multiple optimal solutions exist, choose the one with minimal pollution (tie-breaker)
if not best_solutions:
    print("No feasible solution found.")
else:
    best_solutions.sort(key=lambda t: (t[3], -t[0]))  # sort by pollution asc, then more 4-wheelers
    best_x, best_y, best_luggage, best_pollution = best_solutions[0]

    print(f"Optimal number of 4-wheelers (x): {best_x}")
    print(f"Optimal number of 3-wheelers (y): {best_y}")
    print(f"Total vehicles: {best_total}")
    print(f"Total luggage capacity per day: {best_luggage}")
    print(f"Total pollution per day: {best_pollution}")

# Final required print (must be the last line)
print(f"FINAL_OBJ: {best_total}")
