import pulp

# Define problem
prob = pulp.LpProblem("Cleaning_Time_Minimization", pulp.LpMinimize)

# Decision variables
# x = units of cleansing chemical
# y = units of odor-removing chemical
x = pulp.LpVariable('x', lowBound=100, cat='Continuous')  # at least 100 units of cleansing chemical
y = pulp.LpVariable('y', lowBound=0, cat='Continuous')

# Objective: minimize total time = 4*x + 6*y
prob += 4 * x + 6 * y, "Total_Cleaning_Time"

# Constraints:
prob += x + y >= 300, "Total_chemicals_at_least_300"
prob += x - 2 * y <= 0, "Cleansing_at_most_twice_odor"

# Solve
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Retrieve solution
x_val = x.varValue
y_val = y.varValue
obj_val = pulp.value(prob.objective)

# Print solution details
print(f"Units of cleansing chemical (x): {x_val}")
print(f"Units of odor-removing chemical (y): {y_val}")
print(f"Total cleaning time (objective): {obj_val}")

# Final required output line
print(f"FINAL_OBJ: {obj_val}")
