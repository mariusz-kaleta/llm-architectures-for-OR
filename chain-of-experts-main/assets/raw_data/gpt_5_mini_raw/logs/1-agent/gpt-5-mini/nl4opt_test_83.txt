import pulp

# Decision variables:
# x = number of runners used (each makes one delivery/trip)
# y = number of canoers used (each makes one delivery/trip)
# Both are integer counts (cannot use fractional persons/trips)
x = pulp.LpVariable('runners', lowBound=0, cat='Integer')
y = pulp.LpVariable('canoers', lowBound=0, cat='Integer')

# Problem: maximize total mail delivered
# Runners deliver 3 bags each, canoers deliver 10 bags each
prob = pulp.LpProblem('MaximizeMail', pulp.LpMaximize)
prob += 3 * x + 10 * y  # objective

# Constraints:
# 1) Time: runners take 4 hours each, canoers take 2 hours each; total available <= 200
prob += 4 * x + 2 * y <= 200

# 2) At most 33% of deliveries can be by canoe:
#    y <= 0.33 * (x + y)  -> 0.67*y <= 0.33*x  -> 67*y - 33*x <= 0  (scaled to avoid float precision)
prob += 67 * y - 33 * x <= 0

# 3) At least 4 runners must be used
prob += x >= 4

# Solve
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract results
runners_used = int(pulp.value(x))
canoers_used = int(pulp.value(y))
objective_value = pulp.value(prob.objective)

# Print results
print(f"Runners used: {runners_used}")
print(f"Canoers used: {canoers_used}")
print(f"Total mail delivered (bags): {objective_value}")

# Final required output line (must be last)
value = objective_value
print(f"FINAL_OBJ: {value}")
