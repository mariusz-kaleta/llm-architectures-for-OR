# Water keg selection optimization (brute-force integer search)

# Data from problem
small_capacity = 40   # liters per small keg
large_capacity = 100  # liters per large keg

max_small_available = 30
max_large_available = 10

total_keg_limit = 25

# Constraints
# - At least twice as many small kegs as large kegs: s >= 2*l
# - At least 5 kegs must be large: l >= 5

best_value = -1
best_s = None
best_l = None

# Search integer feasible combinations
for l in range(5, max_large_available + 1):  # large kegs at least 5
    for s in range(0, max_small_available + 1):
        if s + l > total_keg_limit:
            continue
        if s < 2 * l:
            continue
        # feasible
        liters = small_capacity * s + large_capacity * l
        if liters > best_value:
            best_value = liters
            best_s = s
            best_l = l

# Output results
if best_s is None:
    print("No feasible solution found.")
else:
    print(f"Small kegs used: {best_s}")
    print(f"Large kegs used: {best_l}")
    print(f"Total liters transported: {best_value}")

# Required final line
print(f"FINAL_OBJ: {best_value}")
