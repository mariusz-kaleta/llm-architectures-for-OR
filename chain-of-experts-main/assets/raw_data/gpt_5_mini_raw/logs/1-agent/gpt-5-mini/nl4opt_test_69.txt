import math

# Problem data
capacity_wide = 25  # units per minute per wide pipe
capacity_narrow = 15  # units per minute per narrow pipe
required_capacity = 900  # required units per minute
min_wide = 5  # at least this many wide pipes
# Constraint: number of wide pipes at most a third the number of narrow pipes -> 3*W <= N

# Bounds for brute-force search
max_wide = math.ceil(required_capacity / capacity_wide)  # no need to consider more wide pipes than this
max_narrow = math.ceil(required_capacity / capacity_narrow)

best_total = None
best_solution = None

for W in range(min_wide, max_wide + 1):
    # N must satisfy 3W <= N, so start from that lower bound
    N_min_by_ratio = 3 * W
    # Also N cannot be negative; upper bound set by max_narrow
    for N in range(N_min_by_ratio, max_narrow + 1):
        if capacity_wide * W + capacity_narrow * N >= required_capacity:
            total = W + N
            if best_total is None or total < best_total:
                best_total = total
                best_solution = (W, N)

# Print final objective value as required
print(f"FINAL_OBJ: {best_total}")
