import pulp

# Define the problem
prob = pulp.LpProblem("Maximize_Back_Medicine", pulp.LpMaximize)

# Decision variables: doses of pain killer 1 and 2 (can be continuous, non-negative)
x1 = pulp.LpVariable("doses_pain_killer_1", lowBound=0, cat="Continuous")
x2 = pulp.LpVariable("doses_pain_killer_2", lowBound=0, cat="Continuous")

# Coefficients from the problem statement
# Back medicine per dose: pain killer 1 -> 0.8, pain killer 2 -> 0.4
prob += 0.8 * x1 + 0.4 * x2, "Total_Back_Medicine"

# Sleep medicine constraint: 0.3*x1 + 0.6*x2 <= 8
prob += 0.3 * x1 + 0.6 * x2 <= 8, "Sleep_Medicine_Upper_Bound"

# Legs medicine constraint: 0.5*x1 + 0.7*x2 >= 4
prob += 0.5 * x1 + 0.7 * x2 >= 4, "Legs_Medicine_Lower_Bound"

# Solve
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract solution
x1_val = x1.varValue if x1.varValue is not None else 0.0
x2_val = x2.varValue if x2.varValue is not None else 0.0
obj_val = pulp.value(prob.objective)

# Print solution (variables), then final objective as required (final line)
print(f"doses_pain_killer_1 = {x1_val}")
print(f"doses_pain_killer_2 = {x2_val}")
print(f"Total_back_medicine = {obj_val}")
print(f"FINAL_OBJ: {obj_val}")
