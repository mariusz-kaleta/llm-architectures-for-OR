# Factory cart optimization (brute-force integer search)

# Parameters (initialized directly from the problem statement)
horses_available = 60
horses_per_medium = 2
horses_per_large = 4
cap_per_medium = 30
cap_per_large = 70

# Additional constraints
# number of medium carts must be three times number of large carts: m = 3 * l
min_medium = 5
min_large = 5

best_obj = None
best_solution = None

# We'll search over possible integers for large carts (l) and derive medium carts (m = 3*l)
# Bound l from 0..horses_available to be safe
for l in range(0, horses_available + 1):
    m = 3 * l
    # Apply minimum counts
    if m < min_medium or l < min_large:
        continue
    # Horse availability constraint
    if horses_per_medium * m + horses_per_large * l > horses_available:
        continue
    # Objective: total rice capacity
    total_capacity = cap_per_medium * m + cap_per_large * l
    if best_obj is None or total_capacity > best_obj:
        best_obj = total_capacity
        best_solution = (m, l)

# Output the results
if best_solution is None:
    print("No feasible solution found.")
else:
    m_opt, l_opt = best_solution
    print(f"Optimal number of medium carts: {m_opt}")
    print(f"Optimal number of large carts: {l_opt}")
    print(f"Total rice transported (kg): {best_obj}")

# CRITICAL final line required by the task:
print(f"FINAL_OBJ: {best_obj}")
