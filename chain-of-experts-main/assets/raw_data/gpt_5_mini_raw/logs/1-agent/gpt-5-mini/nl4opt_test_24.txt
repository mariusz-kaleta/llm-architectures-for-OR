import pulp

# Data (from problem statement)
required_people = 80
cap_large = 6         # people per large unit
cap_small = 2         # people per small unit
spots_large = 2       # parking spots per large unit
spots_small = 1       # parking spots per small unit
min_small_units = 5   # at least 5 small units
# Large units must be at least 75% of all vehicles -> L >= 0.75*(L+S) -> L >= 3*S

# Define the problem
prob = pulp.LpProblem("Minimize_Parking_Spots", pulp.LpMinimize)

# Decision variables: number of large (L) and small (S) mobile production units
L = pulp.LpVariable('LargeUnits', lowBound=0, cat='Integer')
S = pulp.LpVariable('SmallUnits', lowBound=min_small_units, cat='Integer')  # enforce S >= 5 directly

# Objective: minimize total parking spots
prob += spots_large * L + spots_small * S, "Total_Parking_Spots"

# Constraints
prob += cap_large * L + cap_small * S >= required_people, "CapacityRequirement"
prob += L >= 3 * S, "LargeUnitsAtLeast75Percent"

# Solve
prob.solve(pulp.PULP_CBC_CMD(msg=0))

# Extract results
L_opt = int(pulp.value(L))
S_opt = int(pulp.value(S))
total_spots = int(pulp.value(prob.objective))

# Optional prints (kept minimal). Final required print must be last.
print(f"Large units: {L_opt}")
print(f"Small units: {S_opt}")
print(f"Total parking spots: {total_spots}")
print(f"FINAL_OBJ: {total_spots}")
