import pulp

# Data initialization
fat_burger = 10      # units of fat per burger
fat_pizza = 8        # units of fat per pizza slice
cal_burger = 300     # calories per burger
cal_pizza = 250      # calories per pizza slice
chol_burger = 12     # cholesterol units per burger
chol_pizza = 10      # cholesterol units per pizza slice

fat_req = 130        # minimum fat required
cal_req = 3000       # minimum calories required

# Decision variables: number of burgers (B) and pizza slices (P)
# We use integer variables because you can't practically eat fractional burgers/slices.
B = pulp.LpVariable('Burgers', lowBound=0, cat='Integer')
P = pulp.LpVariable('Pizzas', lowBound=0, cat='Integer')

# Problem definition: minimize total cholesterol
prob = pulp.LpProblem('Minimize_Cholesterol', pulp.LpMinimize)
prob += chol_burger * B + chol_pizza * P, "Total_Cholesterol"

# Constraints
prob += fat_burger * B + fat_pizza * P >= fat_req, "Fat_requirement"
prob += cal_burger * B + cal_pizza * P >= cal_req, "Calorie_requirement"
prob += P >= 2 * B, "Pizza_at_least_twice_burgers"

# Solve
prob.solve()

# Extract results
num_burgers = int(pulp.value(B))
num_pizzas = int(pulp.value(P))
final_obj = pulp.value(prob.objective)

# Print results
print(f"Number of burgers: {num_burgers}")
print(f"Number of pizza slices: {num_pizzas}")
print(f"Total cholesterol: {final_obj}")

# Final required line (must be the last output)
print(f"FINAL_OBJ: {final_obj}")
