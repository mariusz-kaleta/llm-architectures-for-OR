[MONOLITH]:
import pulp

# Data (initialized directly)
iron_A = 5      # iron units in supplement A
calcium_A = 10  # calcium units in supplement A
iron_B = 4      # iron units in supplement B
calcium_B = 15  # calcium units in supplement B

min_iron = 40
min_calcium = 50

cost_A = 2
cost_B = 3

# Define the problem
prob = pulp.LpProblem("Minimize_Supplement_Cost", pulp.LpMinimize)

# Decision variables: number of pills (integer, non-negative)
A = pulp.LpVariable("A_pills", lowBound=0, cat='Integer')
B = pulp.LpVariable("B_pills", lowBound=0, cat='Integer')

# Objective: minimize total cost
prob += cost_A * A + cost_B * B, "Total_Cost"

# Constraints
prob += iron_A * A + iron_B * B >= min_iron, "IronRequirement"
prob += calcium_A * A + calcium_B * B >= min_calcium, "CalciumRequirement"

# Solve
prob.solve(pulp.PULP_CBC_CMD(msg=False))

# Extract results
A_val = int(pulp.value(A))
B_val = int(pulp.value(B))
value = pulp.value(prob.objective)

# Print solution
print(f"A_pills: {A_val}")
print(f"B_pills: {B_val}")
print(f"Total_cost: {value}")

# Final required print (must be last)
print(f"FINAL_OBJ: {value}")
