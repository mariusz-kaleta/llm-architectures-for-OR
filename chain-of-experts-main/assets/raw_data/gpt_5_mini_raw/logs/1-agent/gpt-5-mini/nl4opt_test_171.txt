import pulp

# Data initialization based on the problem statement
capacity_glass = 250     # ml per glass jar
capacity_plastic = 300   # ml per plastic jar
total_honey = 20000      # total ml available
min_glass = 20           # at least 20 glass jars
# Constraint: number of plastic jars must be at least twice number of glass jars

# Define the problem
prob = pulp.LpProblem("Maximize_Number_of_Jars", pulp.LpMaximize)

# Decision variables: integer number of glass (G) and plastic (P) jars
G = pulp.LpVariable("G", lowBound=min_glass, cat=pulp.LpInteger)  # glass jars, at least min_glass
P = pulp.LpVariable("P", lowBound=0, cat=pulp.LpInteger)         # plastic jars

# Objective: maximize total number of jars
prob += G + P, "Total_Jars"

# Constraints
prob += P >= 2 * G, "Plastic_at_least_twice_glass"
prob += capacity_glass * G + capacity_plastic * P <= total_honey, "Honey_capacity"

# Solve the problem
solver = pulp.PULP_CBC_CMD(msg=False)
prob.solve(solver)

# Retrieve solution values (rounded to int for safety)
G_val = int(round(pulp.value(G)))
P_val = int(round(pulp.value(P)))
total_jars = int(round(pulp.value(prob.objective)))

# Print solution details
print(f"Glass_jars: {G_val}")
print(f"Plastic_jars: {P_val}")
print(f"Total_jars: {total_jars}")

# Final required output line (must be the last printed line)
value = total_jars
print(f"FINAL_OBJ: {value}")
