# Integer optimization by enumeration for the fruit problem
# Decision variables:
#   x = number of oranges (nonnegative integer)
#   y = number of grapefruits (nonnegative integer)
# Objective:
#   Minimize S = 5*x + 6*y
# Constraints:
#   5*x + 7*y >= 80   (Vitamin C)
#   3*x + 5*y >= 70   (Vitamin A)
#   x >= 2*y          (preference)

# Brute-force search over a reasonable range
max_x = 100
max_y = 100

best_S = None
best_x = None
best_y = None

for x in range(max_x + 1):
    for y in range(max_y + 1):
        if x < 2*y:
            continue
        if 5*x + 7*y < 80:
            continue
        if 3*x + 5*y < 70:
            continue
        S = 5*x + 6*y
        if best_S is None or S < best_S:
            best_S = S
            best_x = x
            best_y = y
        # optional tie-breaker: prefer fewer fruits total if same sugar
        elif S == best_S:
            if (x + y) < (best_x + best_y):
                best_x = x
                best_y = y

if best_S is None:
    raise RuntimeError("No feasible integer solution found in search range.")

# Print solution summary (optional)
print(f"Optimal integer solution found: x={best_x} oranges, y={best_y} grapefruits, Sugar={best_S} g")
# Final required line (must be exactly this format)
print(f"FINAL_OBJ: {best_S}")
