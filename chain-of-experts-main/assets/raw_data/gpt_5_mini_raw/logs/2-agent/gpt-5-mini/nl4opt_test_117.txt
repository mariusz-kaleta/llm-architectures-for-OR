import numpy as np

# Problem data
# Maximize Z = c^T x
c = np.array([5.0, 3.0])

# Constraints A x <= b
A = np.array([
    [4.0, 6.0],  # Flour
    [6.0, 3.0],  # Special liquid
    [4.0, 2.0],  # Waste
])
b = np.array([150.0, 100.0, 30.0])

tol = 1e-9

candidates = []

# Add the origin
candidates.append(np.array([0.0, 0.0]))

# Intersections with axes (x1=0 and x2=0)
for i in range(A.shape[0]):
    a1, a2 = A[i]
    bi = b[i]
    # x1 = 0 -> solve for x2 if possible
    if abs(a2) > tol:
        x = np.array([0.0, bi / a2])
        candidates.append(x)
    # x2 = 0 -> solve for x1 if possible
    if abs(a1) > tol:
        x = np.array([bi / a1, 0.0])
        candidates.append(x)

# Intersections of pairs of constraint lines (set equalities)
m = A.shape[0]
for i in range(m):
    for j in range(i + 1, m):
        Ai = A[[i, j], :]
        bi = b[[i, j]]
        det = np.linalg.det(Ai)
        if abs(det) < tol:
            continue
        x = np.linalg.solve(Ai, bi)
        candidates.append(x)

# Evaluate candidates, keep feasible ones
best_obj = -np.inf
best_x = None

for x in candidates:
    x1, x2 = x
    # feasibility check
    if x1 < -tol or x2 < -tol:
        continue
    lhs = A.dot(x)
    if np.all(lhs <= b + 1e-7):
        obj = c.dot(x)
        if obj > best_obj + 1e-12:
            best_obj = obj
            best_x = x.copy()

# As a fallback (shouldn't be needed), also check the corners of the bounding box
# derived from simple variable upper bounds computed from each constraint assuming the other var = 0
if best_x is None:
    # numeric fallback: sample grid

    x2_max = max(0.0, max(b[i] / A[i, 1] if A[i, 1] > tol else 0.0 for i in range(m)))
    xs1 = np.linspace(0, x1_max, 201)
    xs2 = np.linspace(0, x2_max, 201)

            if np.all(A.dot(x) <= b + 1e-7):

                    best_x = x.copy()
