import math

# Problem data and constraints:
# Variables: x (small bins), y (large bins), integers >= 0
# Objective: Maximize Z = 25x + 60y
# Constraints:
#   1) 2x + 5y <= 100
#   2) x = 3y
#   3) x >= 10, y >= 4
# Integrality: x,y integers

# From x = 3y, substitute into constraints.
# Compute feasible integer y values given constraints.
y_min_from_x10 = math.ceil(10 / 3)  # ensures x = 3y >= 10
y_min = max(4, y_min_from_x10)
y_max = 100 // 11  # from 11y <= 100

# Enumerate feasible y and pick the one maximizing the objective.
best_obj = None
best_x = None
best_y = None

for y in range(y_min, y_max + 1):
    x = 3 * y
    if 2 * x + 5 * y <= 100 and x >= 10 and y >= 4:
        obj = 25 * x + 60 * y
        if best_obj is None or obj > best_obj:
            best_obj = obj
            best_x = x
            best_y = y

# best_obj should be found; per analytical solution, it is 1215 with x=27, y=9
value = best_obj
print(f"FINAL_OBJ: {value}")
