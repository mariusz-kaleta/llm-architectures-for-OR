3x + 5y = 800

Check corner solutions:
- x = 0 → y = min(800/5, 750/4) = 160 → Z = 1600
- y = 0 → x = min(800/3, 750/5) = 150 (acid binds) → Z = 1200
The interior intersection above gives the maximum for the continuous problem.

#!/usr/bin/env python3
import numpy as np

# Problem data
# Variables: x, y >= 0 (integers optional)
# Maximize Z = 8x + 10y
# Constraints:
#   3x + 5y <= 800
#   5x + 4y <= 750

# Continuous solution (solve the two equality constraints)
A = np.array([[3.0, 5.0],
              [5.0, 4.0]])
b = np.array([800.0, 750.0])

cont_solution = np.linalg.solve(A, b)
x_cont, y_cont = cont_solution
z_cont = 8.0 * x_cont + 10.0 * y_cont

# Corner solutions (axis intersections)
# x = 0:
y_at_x0 = min(800.0 / 5.0, 750.0 / 4.0)
z_x0 = 8.0 * 0.0 + 10.0 * y_at_x0

# y = 0:
x_at_y0 = min(800.0 / 3.0, 750.0 / 5.0)
z_y0 = 8.0 * x_at_y0 + 10.0 * 0.0

# Integer (integrality) version: brute-force search for integer x,y
# Determine reasonable bounds for search
max_x_bound = int(min(800 // 3, 750 // 5))  # when y = 0
max_y_bound = int(min(800 // 5, 750 // 4))  # when x = 0

best_int_obj = None
best_int_pairs = []

for x in range(max_x_bound + 1):
    # For each x, y can range from 0 to max_y_bound, but we can limit by metal constraint too
    # Solve for max y allowed by metal and acid constraints:
    # 3x + 5y <= 800 -> y <= (800 - 3x)/5
    # 5x + 4y <= 750 -> y <= (750 - 5x)/4
    y_max_metal = (800 - 3 * x) // 5
    y_max_acid = (750 - 5 * x) // 4
    y_max = int(min(y_max_metal, y_max_acid, max_y_bound))
    if y_max < 0:
        continue
    for y in range(y_max + 1):
        if 3 * x + 5 * y <= 800 and 5 * x + 4 * y <= 750:
            z = 8 * x + 10 * y
            if (best_int_obj is None) or (z > best_int_obj):
                best_int_obj = z
                best_int_pairs = [(x, y)]
            elif z == best_int_obj:
                best_int_pairs.append((x, y))

# Print results (informational)
print("Continuous solution (intersection of resource constraints):")
print(f" x = {x_cont:.10f}, y = {y_cont:.10f}, Z = {z_cont:.10f}")
print("")
print("Corner solutions:")
print(f" x=0 -> y={y_at_x0:.10f}, Z={z_x0:.10f}")
print(f" y=0 -> x={x_at_y0:.10f}, Z={z_y0:.10f}")
print("")
print("Integer (brute-force) optimum:")
print(f" Best integer objective Z = {best_int_obj}")
print(f" Achieved at integer (x,y) pairs: {best_int_pairs}")
print("")

# Final required print (only this line must have this exact form)
print(f"FINAL_OBJ: {best_int_obj}")
