import numpy as np
from scipy.optimize import linprog

# Problem data
c = np.array([5.0, 2.5])          # objective coefficients
A = np.array([[600.0, 250.0],
              [1.5,   5.0]])      # left-hand side coefficients (>= constraints)
b = np.array([2000.0, 16.0])      # right-hand side

# Convert >= constraints to <= by multiplying by -1
A_ub = -A
b_ub = -b

bounds = [(0, None), (0, None)]  # x >= 0, y >= 0

# Solve LP using scipy.optimize.linprog (Highs)
res = linprog(c, A_ub=A_ub, b_ub=b_ub, bounds=bounds, method='highs')

if res.success:
    x_opt, y_opt = res.x
    obj = float(np.dot(c, res.x))
else:
    # Fallback: solve the two equality constraints (both binding) analytically
    # 600x + 250y = 2000
    # 1.5x + 5y = 16
    try:
        sol = np.linalg.solve(A, b)
        x_opt, y_opt = sol
        obj = float(np.dot(c, sol))
    except Exception:
        # As a last resort, set to zeros
        x_opt, y_opt = 0.0, 0.0
        obj = float(np.dot(c, np.array([x_opt, y_opt])))

# Optional prints (not the final required line)
print(f"x (noodles servings) = {x_opt}")
print(f"y (protein bars servings) = {y_opt}")
print(f"Objective (cost) computed = {obj}")

# Final required output line
print(f"FINAL_OBJ: {obj}")
