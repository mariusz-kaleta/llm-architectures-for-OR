# Self-contained solver for the milk tea production integer program (enumeration)

# Problem data (initialized directly)
profit_per_B = 7.5  # profit for black milk tea
profit_per_M = 5.0  # profit for matcha milk tea

# Resource usage per bottle
milk_per_B = 600    # grams
milk_per_M = 525    # grams
milk_available = 30000  # grams

honey_per_B = 10    # grams
honey_per_M = 5     # grams
honey_available = 500   # grams

# Determine reasonable upper bounds for enumeration (integer variables)
max_xB_by_milk = milk_available // milk_per_B
max_xB_by_honey = honey_available // honey_per_B
max_xB = int(min(max_xB_by_milk, max_xB_by_honey))

max_xM_by_milk = milk_available // milk_per_M
max_xM_by_honey = honey_available // honey_per_M
max_xM = int(min(max_xM_by_milk, max_xM_by_honey))

best_profit = None
best_xB = 0
best_xM = 0

# Enumerate integer feasible solutions
for xB in range(0, max_xB + 1):
    # For each xB, we can bound xM more tightly by remaining resources
    remaining_milk = milk_available - milk_per_B * xB
    remaining_honey = honey_available - honey_per_B * xB
    if remaining_milk < 0 or remaining_honey < 0:
        continue
    # Upper bound for xM given remaining resources
    ub_xM_by_milk = remaining_milk // milk_per_M
    ub_xM_by_honey = remaining_honey // honey_per_M
    ub_xM = int(min(ub_xM_by_milk, ub_xM_by_honey, max_xM))
    for xM in range(0, ub_xM + 1):
        # Check constraints (redundant given bounds but kept for safety)
        if milk_per_B * xB + milk_per_M * xM > milk_available:
            continue
        if honey_per_B * xB + honey_per_M * xM > honey_available:
            continue
        profit = profit_per_B * xB + profit_per_M * xM
        if (best_profit is None) or (profit > best_profit):
            best_profit = profit
            best_xB = xB
            best_xM = xM

# Print chosen solution (optional informative output)
print(f"Optimal integer solution: x_B = {best_xB}, x_M = {best_xM}, profit = {best_profit}")
