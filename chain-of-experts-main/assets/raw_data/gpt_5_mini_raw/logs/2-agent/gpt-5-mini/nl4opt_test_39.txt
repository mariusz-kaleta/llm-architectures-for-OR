x_min, x_max = 5.0, 10.0
y_min, y_max = 5.0, 8.0
# resource: x + 2y <= 30

# Function to solve intersection of two equations
def intersect(eq1, eq2):
    # returns (x,y) or None if no unique intersection
    if eq1[0] == 'x' and eq2[0] == 'y':
        return (float(eq1[1]), float(eq2[1]))
    if eq1[0] == 'y' and eq2[0] == 'x':
        return (float(eq2[1]), float(eq1[1]))
    if eq1[0] == 'x' and eq2[0] == 'x':
        # x = val1 and x = val2 -> if equal, infinite; else none
        if abs(eq1[1] - eq2[1]) < tol:
            return (float(eq1[1]), None)  # degenerate
        return None
    if eq1[0] == 'y' and eq2[0] == 'y':
        if abs(eq1[1] - eq2[1]) < tol:
            return (None, float(eq1[1]))
