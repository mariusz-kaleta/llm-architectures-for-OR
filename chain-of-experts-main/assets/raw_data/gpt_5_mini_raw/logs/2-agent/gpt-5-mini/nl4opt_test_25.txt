#!/usr/bin/env python3
import numpy as np

# Problem data (initialized directly)
# Decision variables: x (Zodiac pills per day), y (Sunny pills per day)
# Objective: minimize C = 1*x + 3*y
# Constraints:
#   1.3 x + 1.2 y >= 5
#   1.5 x + 5.0 y >= 10
#   x, y >= 0
A = np.array([[1.3, 1.2],
              [1.5, 5.0]])
b = np.array([5.0, 10.0])

# 1) Continuous LP relaxation: solve the two equalities (intersection)
try:
    sol = np.linalg.solve(A, b)
    x_cont, y_cont = sol[0], sol[1]
    cost_cont = x_cont + 3.0 * y_cont
except np.linalg.LinAlgError:
    x_cont = y_cont = cost_cont = None

# Also check axis intersections (feasible points where x=0 or y=0)
# For x = 0, y must satisfy both constraints: y >= max(5/1.2, 10/5)
y_req_x0 = max(5.0 / 1.2, 10.0 / 5.0)
pt_x0 = (0.0, y_req_x0)
cost_x0 = pt_x0[0] + 3.0 * pt_x0[1]

# For y = 0, x must satisfy both constraints: x >= max(5/1.3, 10/1.5)
x_req_y0 = max(5.0 / 1.3, 10.0 / 1.5)
pt_y0 = (x_req_y0, 0.0)
cost_y0 = pt_y0[0] + 3.0 * pt_y0[1]

# Determine continuous optimal among corner candidates (intersection and axis points)
candidates = []
if x_cont is not None and y_cont is not None:
    # Verify feasibility (numerical tolerance)
    if (1.3 * x_cont + 1.2 * y_cont >= 5 - 1e-9) and (1.5 * x_cont + 5.0 * y_cont >= 10 - 1e-9) and x_cont >= -1e-9 and y_cont >= -1e-9:
        candidates.append(("intersection", x_cont, y_cont, cost_cont))
candidates.append(("x=0", pt_x0[0], pt_x0[1], cost_x0))
candidates.append(("y=0", pt_y0[0], pt_y0[1], cost_y0))

# Choose minimal cost among continuous candidates
best_cont = min(candidates, key=lambda t: t[3])

# 2) Integer MILP: brute-force search over reasonable bounds
# Determine search bounds: we can cap at a value where cost would clearly exceed known candidates.
# We'll pick bounds conservatively.
x_max = 30
y_max = 30

best_int_cost = float('inf')
best_int_solutions = []

for xi in range(0, x_max + 1):
    for yi in range(0, y_max + 1):
        if (1.3 * xi + 1.2 * yi >= 5 - 1e-9) and (1.5 * xi + 5.0 * yi >= 10 - 1e-9):
            cost = xi + 3 * yi
            if cost < best_int_cost - 1e-9:
                best_int_cost = cost
                best_int_solutions = [(xi, yi)]
            elif abs(cost - best_int_cost) <= 1e-9:
                best_int_solutions.append((xi, yi))

# Print results (informational). Final required line printed last exactly as specified.
print("Continuous LP relaxation solution (intersection of equalities):")
if x_cont is not None:
    print(f" x = {x_cont:.8f}, y = {y_cont:.8f}, cost = {cost_cont:.8f}")
else:
    print(" No unique intersection (singular system).")
print()
print("Other candidate continuous feasible corner points:")
print(f" x=0 point -> y = {pt_x0[1]:.8f}, cost = {cost_x0:.8f}")
print(f" y=0 point -> x = {pt_y0[0]:.8f}, cost = {cost_y0:.8f}")
print()
print("Chosen continuous optimal candidate:")
print(f" type = {best_cont[0]}, x = {best_cont[1]:.8f}, y = {best_cont[2]:.8f}, cost = {best_cont[3]:.8f}")
print()
print("Integer (x,y ∈ Z>=0) optimal solutions found by enumeration:")
print(f" best integer cost = {best_int_cost}")
print(" solutions (x,y):")
for soln in best_int_solutions:
    print(f"  {soln}")

# FINAL required output line (only this exact text on the line)
# We'll print the integer optimal objective value as the final objective value.
# Convert to int if it is integer-valued to match expected formatting (e.g., 7 instead of 7.0)
final_value = int(best_int_cost) if abs(best_int_cost - round(best_int_cost)) < 1e-9 else best_int_cost
print(f"FINAL_OBJ: {final_value}")
