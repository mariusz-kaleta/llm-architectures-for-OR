#!/usr/bin/env python3
import math

# Problem data
# Coefficients for constraints: a1*x + a2*y <= b
constraints = [
    (46.0, 13.0, 1345.0),  # A constraint
    (43.0,  4.0,  346.0),  # B constraint
    (56.0, 45.0, 1643.0),  # C constraint
]
# Objective coefficients (maximize)
p_x = 10.0
p_y = 15.0

tol = 1e-9

# Helper: check feasibility of a point (x,y)
def feasible(x, y, cons):
    if x < -tol or y < -tol:
        return False
    for (a1, a2, b) in cons:
        if a1 * x + a2 * y - b > tol:
            return False
    return True

# Generate candidate corner points for continuous LP:
candidates = []
# origin
candidates.append((0.0, 0.0))

# intersections with axes (x=0 or y=0)
for (a1, a2, b) in constraints:
    if abs(a2) > tol:
        candidates.append((0.0, b / a2))
    if abs(a1) > tol:
        candidates.append((b / a1, 0.0))

# intersections of pairs of constraint lines
for i in range(len(constraints)):
    a1, a2, b1 = constraints[i]
    for j in range(i + 1, len(constraints)):
        c1, c2, b2 = constraints[j]
        # Solve:
        # a1*x + a2*y = b1
        # c1*x + c2*y = b2
        det = a1 * c2 - a2 * c1
        if abs(det) < tol:
            # parallel or nearly so; skip
            continue
        x = (b1 * c2 - a2 * b2) / det
        y = (a1 * b2 - b1 * c1) / det
        candidates.append((x, y))

# Filter feasible candidates and evaluate objective
best_cont_x = None
best_cont_y = None
best_cont_obj = -math.inf

for (x, y) in candidates:
    if feasible(x, y, constraints):
        obj = p_x * x + p_y * y
        if obj > best_cont_obj + tol:
            best_cont_obj = obj
            best_cont_x = x
            best_cont_y = y

# Round small negatives to zero for display
if best_cont_x is not None:
    if abs(best_cont_x) < 1e-12:
        best_cont_x = 0.0
    if abs(best_cont_y) < 1e-12:
        best_cont_y = 0.0

# Now solve the integer version by brute force search
# bounds for search
x_bound = math.floor(min(
    (b / a1) if a1 > 0 else math.inf
    for (a1, a2, b) in constraints
))
y_bound = math.floor(min(
    (b / a2) if a2 > 0 else math.inf
    for (a1, a2, b) in constraints
))

# safety caps
x_bound = max(0, x_bound)
y_bound = max(0, y_bound)

best_int_x = 0
best_int_y = 0
best_int_obj = -math.inf

for xi in range(0, x_bound + 1):
    for yi in range(0, y_bound + 1):
        if feasible(xi, yi, constraints):
            obj = p_x * xi + p_y * yi
            if obj > best_int_obj + tol:
                best_int_obj = obj
                best_int_x = xi
                best_int_y = yi

# Print results (informational)
print("Continuous LP optimum (may be fractional):")
print(f" x = {best_cont_x:.8f}, y = {best_cont_y:.8f}, profit = {best_cont_obj:.8f}")
print("Integer (practical) optimum:")
print(f" x = {best_int_x}, y = {best_int_y}, profit = {best_int_obj}")

# Final required line (must be the last printed line)
print(f"FINAL_OBJ: {int(best_int_obj)}")
