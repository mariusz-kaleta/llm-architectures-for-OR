# Self-contained script to solve the integer LP described in the prompt by enumeration.

# Decision variables:
# x = number of pills of supplement A (nonnegative integer)
# y = number of pills of supplement B (nonnegative integer)
#
# Objective:
# Minimize z = 2*x + 3*y
#
# Constraints:
# 5*x + 4*y >= 40   (Iron)
# 10*x + 15*y >= 50 (Calcium)

# We perform a complete integer enumeration over a reasonable bounding box to find the optimal integer solution.

import math

# Coefficients and requirements
cost_coeffs = (2, 3)  # cost = 2*x + 3*y
iron_coeffs = (5, 4)  # 5*x + 4*y >= 40
calcium_coeffs = (10, 15)  # 10*x + 15*y >= 50
iron_req = 40
calcium_req = 50

# Search bounds for enumeration (large enough to include feasible optima)
MAX_X = 100
MAX_Y = 100

best_cost = None
best_x = None
best_y = None

for x in range(0, MAX_X + 1):
    for y in range(0, MAX_Y + 1):
        if iron_coeffs[0]*x + iron_coeffs[1]*y >= iron_req and calcium_coeffs[0]*x + calcium_coeffs[1]*y >= calcium_req:
            cost = cost_coeffs[0]*x + cost_coeffs[1]*y
            if best_cost is None or cost < best_cost:
                best_cost = cost
                best_x = x
                best_y = y

if best_cost is None:
    raise RuntimeError("No feasible integer solution found within search bounds.")

# Final objective value
value = best_cost

print(f"FINAL_OBJ: {value}")
