# Self-contained script to analyze the slicer purchase MILP and find solutions by brute-force integer search.

from typing import Optional, Tuple, Dict

def solve_bruteforce(grease_limit: int,
                     strict_manual_lt_auto: bool,
                     max_m: int = 50,
                     max_a: int = 50) -> Dict[str, Optional[int]]:
    """
    Brute-force search over integer nonnegative (m,a) to minimize m+a subject to:
      5*m + 8*a >= 50    (production)
      3*m + 6*a <= grease_limit
      if strict_manual_lt_auto: m + 1 <= a  (enforces m < a for integers)
    Returns a dict with status, m, a, objective.
    """
    best: Optional[Tuple[int,int,int]] = None  # (objective, m, a)
    for m in range(0, max_m+1):
        for a in range(0, max_a+1):
            if strict_manual_lt_auto:
                if not (m + 1 <= a):
                    continue
            # production constraint
            if 5*m + 8*a < 50:
                continue
            # grease constraint
            if 3*m + 6*a > grease_limit:
                continue
            obj = m + a
            if best is None or obj < best[0] or (obj == best[0] and (a < best[2] or (a == best[2] and m < best[1]))):
                best = (obj, m, a)
    if best is None:
        return {'status': 'Infeasible', 'm': None, 'a': None, 'objective': None}
    else:
        return {'status': 'Optimal', 'm': best[1], 'a': best[2], 'objective': best[0]}
