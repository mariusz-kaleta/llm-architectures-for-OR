# Self-contained script to analyze the LP described in the prompt.
# It first attempts to solve the original LP (minimize x1+x2 with 1500 worker-hours).

from pulp import LpProblem, LpVariable, lpSum, LpMinimize, LpMaximize, LpStatus, value

# Parameters
heart_require = 20000.0
lung_require = 30000.0
worker_hours_available = 1500.0

# Coefficients: production per hour
heart_per_x1 = 20.0
heart_per_x2 = 30.0
lung_per_x1 = 30.0
lung_per_x2 = 40.0

# 1) Original LP: minimize total time T = x1 + x2
prob_min_time = LpProblem("Minimize_Total_Time", LpMinimize)
x1 = LpVariable("x1", lowBound=0)
x2 = LpVariable("x2", lowBound=0)
prob_min_time += x1 + x2
# Production constraints (must meet minima)
prob_min_time += heart_per_x1 * x1 + heart_per_x2 * x2 >= heart_require
prob_min_time += lung_per_x1 * x1 + lung_per_x2 * x2 >= lung_require
# Worker-hours constraint
prob_min_time += 3.0 * x1 + 5.0 * x2 <= worker_hours_available

status_min_time = prob_min_time.solve()
status_text = LpStatus[prob_min_time.status]
print("Original LP status:", status_text)
if status_text == "Optimal":
    x1_val = x1.varValue
    x2_val = x2.varValue
    total_time = x1_val + x2_val
    print(f"Original LP optimal x1={x1_val:.6f}, x2={x2_val:.6f}, total_time={total_time:.6f}")

x2_f = LpVariable("x2_f", lowBound=0)
t = LpVariable("t", lowBound=0)

prob_max_fraction += heart_per_x1 * x1_f + heart_per_x2 * x2_f >= heart_require * t
prob_max_fraction += lung_per_x1 * x1_f + lung_per_x2 * x2_f >= lung_require * t
prob_max_fraction += 3.0 * x1_f + 5.0 * x2_f <= worker_hours_available

status_frac = prob_max_fraction.solve()

worker_used = 3.0 * x1_f_val + 5.0 * x2_f_val

print(f"  t = {t_val:.6f} (fraction of both targets achievable)")
