final_value = None  # will hold the final objective or the string "INFEASIBLE"

if PULP_AVAILABLE:
    # Build MILP with PuLP
    prob = pulp.LpProblem("HydrogenTransport", pulp.LpMinimize)
    x = pulp.LpVariable("x", lowBound=0, cat='Integer')
    y = pulp.LpVariable("y", lowBound=0, cat='Integer')

    # Objective: minimize x + y
    prob += x + y

    # Constraints
    prob += capacity_per_x * x + capacity_per_y * y >= required_capacity
    prob += cost_x * x + cost_y * y <= budget
    prob += x + 1 <= y

    # Solve (suppress solver output)
    prob.solve(pulp.PULP_CBC_CMD(msg=False))

    status = pulp.LpStatus.get(prob.status, "Unknown")

    max_y_by_capacity = ceil(required_capacity / capacity_per_y) + 5

    x_max = max(max_x_by_budget, max_x_by_capacity, 50)
    y_max = max(max_y_by_budget, max_y_by_capacity, 50)

    best_obj = None
    best_solution = None
    for xi in range(0, x_max + 1):
