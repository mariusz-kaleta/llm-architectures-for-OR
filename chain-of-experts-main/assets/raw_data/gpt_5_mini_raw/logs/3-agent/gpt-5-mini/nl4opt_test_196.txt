minimum cost = 5,000·15 + 9,150·35 = 75,000 + 320,250 = 395,250

from pulp import LpProblem, LpVariable, LpMaximize, LpInteger, LpStatus, value

# create problem
prob = LpProblem("Ads", LpMaximize)
r = LpVariable('r', lowBound=15, upBound=40, cat=LpInteger)

prob += 5000*r + 9150*s <= 250000 # budget
prob.solve()
print(LpStatus[prob.status])
