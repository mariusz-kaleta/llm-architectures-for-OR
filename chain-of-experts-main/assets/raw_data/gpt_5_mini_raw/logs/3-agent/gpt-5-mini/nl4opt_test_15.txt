#!/usr/bin/env python3
# Self-contained script to solve the ambulance/van scheduling MILP using PuLP

from pulp import LpProblem, LpMinimize, LpVariable, LpInteger, PULP_CBC_CMD, value

# Data
cost_a = 820  # cost per Type II ambulance shift
cost_v = 550  # cost per hospital van shift

cap_a = 20    # patients per ambulance shift
cap_v = 15    # patients per van shift

required_capacity = 320

# Build the problem
prob = LpProblem("Ambulance_Van_Scheduling", LpMinimize)

# Decision variables: number of shifts (nonnegative integers)
a = LpVariable("a", lowBound=0, cat=LpInteger)  # type II ambulances
v = LpVariable("v", lowBound=0, cat=LpInteger)  # hospital vans

# Objective: minimize total cost
prob += cost_a * a + cost_v * v

# Constraints
prob += cap_a * a + cap_v * v >= required_capacity  # capacity requirement
# Union share constraint: v <= 0.6*(a + v)  -> equivalent to v <= 1.5*a
prob += v <= 1.5 * a

# Solve with the default CBC solver (silenced)
solver = PULP_CBC_CMD(msg=False)
prob.solve(solver)

# Extract integer solutions
a_val = int(a.varValue + 0.0)
v_val = int(v.varValue + 0.0)
total_cost = int(value(prob.objective) + 0.0)

# (Optional) Print solution details
print(f"a (ambulance shifts) = {a_val}")
print(f"v (van shifts)       = {v_val}")
print(f"Total capacity       = {cap_a * a_val + cap_v * v_val}")
print(f"Total daily cost     = ${total_cost}")

# Final required output line (exact format)
print(f"FINAL_OBJ: {total_cost}")
