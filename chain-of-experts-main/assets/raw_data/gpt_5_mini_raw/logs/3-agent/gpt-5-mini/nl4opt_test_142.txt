- Continuous (LP) solution: solve the LP relaxation (L,S real). - Integer (practical) solution: pieces must be integer. Checking nearby integer feasible points gives optimal integer solutions:

#!/usr/bin/env python3
# Self-contained solver for the art-piece production integer optimization
# Decision variables:
#   L = number of large pieces (integer, >=5)
#   S = number of small pieces (integer, >=5)
# Objective: maximize 30*L + 15*S
# Constraints:
#   4L + 2S <= 100   (paint)
#   3L + 1S <= 50    (glitter)
#   5L + 2S <= 70    (glue)

def main():
    # Bounds derived from resource single-item limits to restrict search space
    L_min = 5
    S_min = 5
    # Individual upper bounds (safe overestimates)
    L_max = min(100 // 4, 50 // 3, 70 // 5)  # = min(25,16,14) = 14
    S_max = min(100 // 2, 50 // 1, 70 // 2)  # = min(50,50,35) = 35

    best_obj = None
    best_solutions = []

    for L in range(L_min, L_max + 1):
        for S in range(S_min, S_max + 1):
            if 4 * L + 2 * S > 100:
                continue
            if 3 * L + 1 * S > 50:
                continue
            if 5 * L + 2 * S > 70:
                continue
            obj = 30 * L + 15 * S
            if (best_obj is None) or (obj > best_obj):
                best_obj = obj
                best_solutions = [(L, S)]
            elif obj == best_obj:
                best_solutions.append((L, S))

    # According to enumeration, best_obj should be 480 (e.g., (5,22) and (6,20)).
    if best_obj is None:
        best_obj = 0

    # Final required print line (no other text on this line)
    print(f"FINAL_OBJ: {int(best_obj)}")

if __name__ == "__main__":
    main()
