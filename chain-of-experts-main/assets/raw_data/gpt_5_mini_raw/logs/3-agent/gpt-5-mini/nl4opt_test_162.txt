# Self-contained script to analyze the staffing ILP described by the user,
# check infeasibility under the exact 2:1 technician:researcher ratio,

#-ratio integer program (minimize total shifts subject to hours and budget).

# At the very end this script prints the final objective value in the required format:

import math

# Data (initialized directly)
tech_shift_hours = 8
res_shift_hours = 5
tech_cost = 300
res_cost = 100
required_hours = 500
available_budget = 14000

# 1) Check model with regulatory ratio x = 2y
# Using substitution x = 2y, derive constraints in terms of y:
# Hours: 8*(2y) + 5*y = 21*y >= 500  => y >= 500/21
# Budget: 300*(2y) + 100*y = 700*y <= 14000 => y <= 14000/700
y_hours_req = required_hours / (tech_shift_hours * 2 + res_shift_hours)  # 500 / 21
y_budget_max = available_budget / (tech_cost * 2 + res_cost)            # 14000 / 700

y_hours_req_val = required_hours / 21.0
y_budget_max_val = available_budget / 700.0

# Smallest integer y that satisfies hours
y_min_hours_int = math.ceil(y_hours_req_val)
y_max_budget_int = math.floor(y_budget_max_val)

# Feasibility check
feasible_with_ratio = (y_min_hours_int <= y_max_budget_int)

print("=== Check with regulatory ratio x = 2y ===")
print(f"Needed y from hours constraint (ceil): {y_min_hours_int}")
print(f"Allowed y from budget constraint (floor): {y_max_budget_int}")
if feasible_with_ratio:
    x_candidate = 2 * y_min_hours_int
    y_candidate = y_min_hours_int
    hours = tech_shift_hours * x_candidate + res_shift_hours * y_candidate
    cost = tech_cost * x_candidate + res_cost * y_candidate
    print("Model with x = 2y is feasible.")
    print(f"Example feasible integer solution: x={x_candidate}, y={y_candidate}, hours={hours}, cost=${cost}")
else:
    print("Model with x = 2y is INFEASIBLE under the current budget and hours requirement.")
    # Compute minimum budget needed to satisfy hours with ratio enforced
    # smallest integer y satisfying hours is y_min_hours_int
    y_needed = y_min_hours_int
    min_total_cost_with_ratio = (tech_cost * 2 + res_cost) * y_needed  # 700 * y_needed
    extra_budget_needed = max(0, min_total_cost_with_ratio - available_budget)

        if hours >= required_hours and cost <= available_budget:

if not feasible_solutions:

    final_obj = None

    print(f"FINAL_OBJ: None")
