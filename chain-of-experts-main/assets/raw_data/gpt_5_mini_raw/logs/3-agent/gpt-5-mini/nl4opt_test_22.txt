- palladium‑heavy uses 15 units Pt and 25 units Pd, converts at 5 units/s.

Constraints
- Platinum constraint: 15x + 20y ≤ 450.
- Palladium constraint: 25x + 14y ≤ 390.
- Nonnegativity: x ≥ 0, y ≥ 0.
- (Optional integrality: x, y integers.)

Objective

3x + 4y = 90 (from dividing platinum constraint by 5) and 25x + 14y = 390.

Thus the continuous optimum is x ≈ 5.172, y ≈ 18.621 (use fractional catalysts only if allowed).

```python
import math
from scipy.optimize import linprog

# Data
pt_available = 450.0  # platinum units
pd_available = 390.0  # palladium units

# Resource coefficients
# For x (palladium-heavy): Pt 15, Pd 25, value 5
# For y (platinum-heavy): Pt 20, Pd 14, value 4
A_ub = [
    [15.0, 20.0],  # platinum constraint: 15x + 20y <= 450
    [25.0, 14.0],  # palladium constraint: 25x + 14y <= 390
]
b_ub = [pt_available, pd_available]

# Continuous LP (maximize 5x + 4y) using scipy.optimize.linprog (which minimizes)
c = [-5.0, -4.0]  # minimize -5x -4y => maximize 5x + 4y
bounds = [(0, None), (0, None)]

res = linprog(c, A_ub=A_ub, b_ub=b_ub, bounds=bounds, method="highs")

if not res.success:
    raise RuntimeError("Continuous LP solver failed: " + (res.message if hasattr(res, "message") else str(res)))

x_cont, y_cont = res.x
z_cont = 5.0 * x_cont + 4.0 * y_cont

print(f"Continuous solution: x = {x_cont:.6f}, y = {y_cont:.6f}, Z = {z_cont:.6f}")

# Integer solution (brute-force search over reasonable bounds)
# Determine safe upper bounds for search
x_max_by_pt = int(math.floor(pt_available / 15.0))
x_max_by_pd = int(math.floor(pd_available / 25.0))
x_max = max(x_max_by_pt, x_max_by_pd, 0)  # safe upper bound for x

y_max_by_pt = int(math.floor(pt_available / 20.0))
y_max_by_pd = int(math.floor(pd_available / 14.0))
y_max = max(y_max_by_pt, y_max_by_pd, 0)  # safe upper bound for y

best_obj_int = -math.inf
best_x_int = None
best_y_int = None

for xi in range(0, x_max + 1):
    for yi in range(0, y_max + 1):
        if 15 * xi + 20 * yi <= pt_available and 25 * xi + 14 * yi <= pd_available:
            obj = 5 * xi + 4 * yi
            if obj > best_obj_int:
                best_obj_int = obj
                best_x_int = xi
                best_y_int = yi

if best_x_int is None:
    raise RuntimeError("No feasible integer solution found.")

print(f"Integer solution: x = {best_x_int}, y = {best_y_int}, Z = {best_obj_int}")

# Final required print line (must be the very last line)
print(f"FINAL_OBJ: {best_obj_int}")
```
