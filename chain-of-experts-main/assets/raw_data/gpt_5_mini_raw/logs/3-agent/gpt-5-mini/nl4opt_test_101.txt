import math

# Problem data
water_top = 6      # units per topsoil bag
water_sub = 10     # units per subsoil bag
capacity = 150     # truck capacity (bags)
min_top = 10       # at least 10 topsoil bags
top_frac = 0.30    # at most 30% of total bags can be topsoil

# Helper: check percentage constraint T <= 0.30*(T+S)
def pct_ok(T, S):
    return T <= top_frac * (T + S)

# A) Interpretation A: total must equal 150 (T + S = 150)
best_eq = None  # tuple (Z, T, S)
for T in range(min_top, capacity + 1):
    S = capacity - T
    if S < 0:
        continue
    if not pct_ok(T, S):
        continue
    Z = water_top * T + water_sub * S
    if best_eq is None or Z < best_eq[0]:
        best_eq = (Z, T, S)

# B) Interpretation B: total may be at most 150 (T + S <= 150)
best_at_most = None  # tuple (Z, T, S)
# We'll search over feasible T and S
for T in range(min_top, capacity + 1):
    # minimal S from percentage constraint S >= (7/3)T
    S_min = math.ceil((7/3) * T)
    S_max = capacity - T  # because T + S <= capacity
    if S_min > S_max:
        continue
    # For fixed T, Z increases with S (coeff positive), so pick minimal S
    S = S_min
    Z = water_top * T + water_sub * S
    if best_at_most is None or Z < best_at_most[0]:
        best_at_most = (Z, T, S)

# Print human-readable results (not the final required line)
if best_eq is not None:
    Z_eq, T_eq, S_eq = best_eq
    print(f"Interpretation A (T+S = {capacity}): T = {T_eq}, S = {S_eq}, Z = {Z_eq}")
else:
    print(f"Interpretation A (T+S = {capacity}): No feasible solution")

if best_at_most is not None:
    Z_am, T_am, S_am = best_at_most
    print(f"Interpretation B (T+S <= {capacity}): T = {T_am}, S = {S_am}, Z = {Z_am}")
else:
    print(f"Interpretation B (T+S <= {capacity}): No feasible solution")

# Final object containing both objective values
final_obj = {"equal_150": (best_eq[0] if best_eq is not None else None),
             "at_most_150": (best_at_most[0] if best_at_most is not None else None)}

print(f"FINAL_OBJ: {final_obj}")
