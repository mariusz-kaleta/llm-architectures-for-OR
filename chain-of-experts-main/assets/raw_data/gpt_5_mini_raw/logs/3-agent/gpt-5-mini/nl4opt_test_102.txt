Objective

Constraints
- Calories requirement: 300R + 200K >= 2200
- Protein requirement: 4.5R + 4K >= 30
- Nonnegativity: R >= 0, K >= 0

Mathematical formulation
Minimize
  Z = 3R + 2K
subject to
  300R + 200K >= 2200
  4.5R + 4K >= 30
  R, K >= 0

import math

# Problem data
c = [3.0, 2.0]  # objective coefficients for R, K
# Constraints (>=) converted to <= for scipy.linprog by multiplying by -1:
# 300 R + 200 K >= 2200  ->  -300 R -200 K <= -2200
# 4.5 R + 4 K >= 30      ->  -4.5 R -4 K <= -30
A_ub = [[-300.0, -200.0],
        [-4.5,   -4.0]]
b_ub = [-2200.0, -30.0]
bounds = [(0, None), (0, None)]  # R, K >= 0

value = None

# Try to solve with scipy.linprog if available
try:
    from scipy.optimize import linprog
    res = linprog(c=c, A_ub=A_ub, b_ub=b_ub, bounds=bounds, method='highs')
    if res.success:
        value = res.fun
except Exception:
    value = None

# Fallback analytic solution if solver not available or failed
if value is None:
    # Evaluate corner (axis) feasible points:
    # If K = 0, R must satisfy both constraints: R >= max(2200/300, 30/4.5)
    R_k0 = max(2200.0/300.0, 30.0/4.5)
    K_k0 = 0.0
    cost_k0 = 3.0 * R_k0 + 2.0 * K_k0

    # If R = 0, K must satisfy both constraints: K >= max(2200/200, 30/4)
    R_r0 = 0.0
    K_r0 = max(2200.0/200.0, 30.0/4.0)
    cost_r0 = 3.0 * R_r0 + 2.0 * K_r0

    # Intersection of equalities (may be infeasible if negative)
    # Solve:
    # 300 R + 200 K = 2200
    # 4.5 R + 4 K = 30
    det = 300.0 * 4.0 - 200.0 * 4.5  # = 1200 - 900 = 300
    if abs(det) > 1e-12:
        R_int = (2200.0 * 4.0 - 200.0 * 30.0) / det
        K_int = (300.0 * 30.0 - 2200.0 * 4.5) / det
        if R_int >= -1e-12 and K_int >= -1e-12:
            cost_int = 3.0 * R_int + 2.0 * K_int
        else:
            cost_int = float('inf')
    else:
        cost_int = float('inf')

    value = min(cost_k0, cost_r0, cost_int)

# Print final objective value as required
print(f"FINAL_OBJ: {value}")
