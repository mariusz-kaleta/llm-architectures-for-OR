import numpy as np
from scipy.optimize import linprog

# Problem data
c = np.array([-25.0, -18.0])  # linprog minimizes; negate to maximize
A_ub = np.array([
    [10.0, 12.0],  # Mint
    [20.0, 15.0],  # Active ingredient
    [5.0, 3.0],    # Black tar
])
b_ub = np.array([120.0, 100.0, 50.0])
bounds = [(0, None), (0, None)]  # x1 >=0, x2 >=0

# Solve continuous LP with scipy
res = linprog(c, A_ub=A_ub, b_ub=b_ub, bounds=bounds, method="highs")

def evaluate_objective(x):
    return 25.0 * x[0] + 18.0 * x[1]

if res.success:
    x_opt = res.x
    value = evaluate_objective(x_opt)
else:
    # Fallback: enumerate intersections of constraint lines and axes to find best corner
    # Lines: 10x1+12x2=120, 20x1+15x2=100, 5x1+3x2=50, x1=0, x2=0
    lines = [
        (np.array([10.0, 12.0]), 120.0),
        (np.array([20.0, 15.0]), 100.0),
        (np.array([5.0, 3.0]), 50.0),
        ("x1_axis", 0.0),
        ("x2_axis", 0.0),
    ]
    candidates = []
    n = len(lines)
    for i in range(n):
        for j in range(i+1, n):
            li, bi = lines[i]
            lj, bj = lines[j]
            # Handle axis intersections
            if li == "x1_axis" and lj == "x2_axis":
                x = np.array([0.0, 0.0])
            elif li == "x1_axis":
                # x1 = 0 => solve lj * x = bj for x2
                A = np.array([[lj[1]]])  # coefficient for x2
                rhs = np.array([bj - lj[0]*0.0])
                if abs(A[0,0]) < 1e-12:
                    continue
                x2 = rhs[0] / A[0,0]
                x = np.array([0.0, x2])
            elif lj == "x1_axis":
                A = np.array([[li[1]]])
                rhs = np.array([bi - li[0]*0.0])
                if abs(A[0,0]) < 1e-12:
                    continue
                x2 = rhs[0] / A[0,0]
                x = np.array([0.0, x2])
            elif li == "x2_axis":
                # x2 = 0
                A = np.array([[lj[0]]])
                rhs = np.array([bj - lj[1]*0.0])
                if abs(A[0,0]) < 1e-12:
                    continue
                x1 = rhs[0] / A[0,0]
                x = np.array([x1, 0.0])
            elif lj == "x2_axis":
                A = np.array([[li[0]]])
                rhs = np.array([bi - li[1]*0.0])
                if abs(A[0,0]) < 1e-12:
                    continue
                x1 = rhs[0] / A[0,0]
                x = np.array([x1, 0.0])
            else:
                # Solve 2x2 system [li; lj] x = [bi; bj]
                A = np.vstack([li, lj])
                b = np.array([bi, bj])
                det = np.linalg.det(A)
                if abs(det) < 1e-12:
                    continue
                x = np.linalg.solve(A, b)
            # Check feasibility (allow small numerical tolerance)
            if x is None:
                continue
            if np.any(x < -1e-8):
                continue
            if np.all(A_ub.dot(x) - b_ub <= 1e-8):
                candidates.append((x, evaluate_objective(x)))
    # Also consider origin
    if np.all(A_ub.dot(np.array([0.0, 0.0])) - b_ub <= 1e-8):

    else:
