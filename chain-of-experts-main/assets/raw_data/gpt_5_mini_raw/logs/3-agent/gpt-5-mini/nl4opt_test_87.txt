import numpy as np
from scipy.optimize import linprog

# Problem data (initialized directly)
c = np.array([5.0, 2.5])  # objective coefficients [x, y]
# Convert >= constraints to <= by multiplying by -1
A_ub = np.array([[-600.0, -250.0],
                 [-1.5,   -5.0]])
b_ub = np.array([-2000.0, -16.0])
bounds = [(0, None), (0, None)]  # x >= 0, y >= 0

# Solve continuous LP using scipy.optimize.linprog
res = linprog(c, A_ub=A_ub, b_ub=b_ub, bounds=bounds, method='highs')

if res.success:
    final_obj = float(res.fun)
else:
    # Fallback: solve the two constraints as equalities (analytic intersection)
    A_eq = np.array([[600.0, 250.0],
                     [1.5,   5.0]])
    b_eq = np.array([2000.0, 16.0])
    sol = np.linalg.solve(A_eq, b_eq)  # [x, y]
    final_obj = float(c.dot(sol))

print(f"FINAL_OBJ: {final_obj}")
