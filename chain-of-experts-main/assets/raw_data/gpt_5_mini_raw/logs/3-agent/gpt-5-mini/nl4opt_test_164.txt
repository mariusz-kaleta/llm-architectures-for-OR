import math

# Data
profit_R = 400
profit_S = 260

# Constraints (left-hand coefficients and right-hand side)
# 60 R + 45 S <= 20000
# 20 R + 25 S <= 13000
max_mover = 20000
max_setup = 13000

# Upper bounds for brute-force search based on mover constraint
R_max = max_mover // 60  # floor(20000/60) = 333
S_max = max_mover // 45  # floor(20000/45) = 444

best_obj = None
best_R = None
best_S = None

for R in range(R_max + 1):
    # For given R, we can tighten S range by both constraints
    # From mover time: 45 S <= max_mover - 60 R -> S <= floor((max_mover - 60R)/45)
    rem_mover = max_mover - 60 * R
    if rem_mover < 0:
        continue
    S_upper_mover = rem_mover // 45

    # From setup time: 25 S <= max_setup - 20 R -> S <= floor((max_setup - 20R)/25)
    rem_setup = max_setup - 20 * R
    if rem_setup < 0:
        continue
    S_upper_setup = rem_setup // 25

    S_upper = min(S_upper_mover, S_upper_setup, S_max)

    for S in range(S_upper + 1):
        # constraints already enforced by S range, but check for safety
        if 60 * R + 45 * S > max_mover:
            continue
        if 20 * R + 25 * S > max_setup:
            continue

        obj = profit_R * R + profit_S * S
        if (best_obj is None) or (obj > best_obj):
            best_obj = obj
            best_R = R
            best_S = S

# Final objective value (integer)
if best_obj is None:
    best_obj = 0

print(f"FINAL_OBJ: {best_obj}")
