import math

# Parameters
cool_L = 12     # cooling per low-power unit
cool_H = 17     # cooling per high-power unit
elec_L = 150    # electricity per low-power unit
elec_H = 250    # electricity per high-power unit
cool_req = 250  # required cooling
elec_cap = 3400 # electricity capacity
H_min = 7
H_max = elec_cap // elec_H  # 3400//250 = 13

# Helper to compute feasible L range for a given H
def feasible_L_range(H):
    # From cooling: 12L + 17H >= 250 => L >= ceil((250 - 17H)/12)
    numerator = cool_req - cool_H * H
    if numerator <= 0:
        L_low = 0
    else:
        L_low = math.ceil(numerator / cool_L)
    # From electricity: 150L + 250H <= 3400 => L <= floor((3400 - 250H)/150)
    denom = elec_cap - elec_H * H
    if denom < 0:
        return None  # no feasible L because electricity already exceeded
    L_high = math.floor(denom / elec_L)
    if L_high < 0:
        return None
    if L_low > L_high:
        return None
    return L_low, L_high

# 1) Check feasibility with 30% aesthetic constraint:
# L <= 0.30 (L + H)  <=>  0.7 L <= 0.3 H  <=>  7 L <= 3 H  (integer-safe)
feasible_with_30 = []
for H in range(H_min, H_max + 1):
    rng = feasible_L_range(H)
    if rng is None:
        continue
    L_low, L_high = rng
    for L in range(L_low, L_high + 1):
        if 7 * L <= 3 * H:
            feasible_with_30.append((L, H))

# Print diagnostic about 30% constraint feasibility
if len(feasible_with_30) == 0:
    print("Model with strict 30% aesthetic constraint: INFEASIBLE (no integer (L,H) found).")
else:
    print("Feasible solutions with 30% constraint (L,H):")
    for sol in feasible_with_30:
        L, H = sol
        N = L + H
        cooling = cool_L * L + cool_H * H
        electricity = elec_L * L + elec_H * H
        print(f" L={L}, H={H}, N={N}, cooling={cooling}, electricity={electricity}")

# 2) Solve the ILP without the 30% constraint by exhaustive search (small search space)
feasible_no30 = []
for H in range(H_min, H_max + 1):
    rng = feasible_L_range(H)
    if rng is None:
        continue
    L_low, L_high = rng
    for L in range(L_low, L_high + 1):
        # All other constraints already satisfied by range; add to list
        cooling = cool_L * L + cool_H * H
        electricity = elec_L * L + elec_H * H
        if cooling >= cool_req and electricity <= elec_cap:
            feasible_no30.append((L, H))

if len(feasible_no30) == 0:
    print("Model without 30% constraint: INFEASIBLE (unexpected).")
    best_N = None
else:
    # Find minimal N = L + H
    best_N = min(L + H for (L, H) in feasible_no30)
    best_solutions = [(L, H) for (L, H) in feasible_no30 if L + H == best_N]
    print("Optimal solutions without the 30% constraint (minimize N = L+H):")
    for L, H in best_solutions:
        cooling = cool_L * L + cool_H * H
        electricity = elec_L * L + elec_H * H
        print(f" L={L}, H={H}, N={L+H}, cooling={cooling}, electricity={electricity}")

# Final required print (only this exact line at the very end)
print(f"FINAL_OBJ: {best_N}")
