x1, x2, x3 ∈ Z_{\ge 0}

#!/usr/bin/env python3
# Complete, self-contained script to solve the ILP by enumeration

# Problem data
c1 = 1200  # cost Pi TV
c2 = 2000  # cost Beta Video
c3 = 4000  # cost Gamma Live

v1 = 2000  # viewers Pi TV
v2 = 5000  # viewers Beta Video
v3 = 9000  # viewers Gamma Live

BUDGET = 20000

# Bounds for enumeration (safe upper bounds)
max_x1 = BUDGET // c1  # at most floor(20000/1200) = 16
max_x2 = 8              # given
max_x3 = BUDGET // c3  # at most floor(20000/4000) = 5

best_obj = -1
best_solution = None

for x2 in range(0, max_x2 + 1):
    for x3 in range(0, max_x3 + 1):
        # Compute maximum possible x1 given budget to cut enumeration a bit
        remaining_budget_after_23 = BUDGET - (c2 * x2 + c3 * x3)
        if remaining_budget_after_23 < 0:
            continue
        max_x1_local = remaining_budget_after_23 // c1
        for x1 in range(0, max_x1_local + 1):
            # Constraints:
            # Budget:
            total_cost = c1 * x1 + c2 * x2 + c3 * x3
            if total_cost > BUDGET:
                continue
            # Gamma at most one-third of total commercials: 2*x3 <= x1 + x2
            if 2 * x3 > x1 + x2:
                continue
            # Pi at least 20%: 4*x1 >= x2 + x3
            if 4 * x1 < x2 + x3:
                continue
            # All variables are nonnegative integers by construction

            # Objective: total viewers
            obj = v1 * x1 + v2 * x2 + v3 * x3

            if obj > best_obj:
                best_obj = obj
                best_solution = (x1, x2, x3)

# If no feasible solution found, set objective to 0
if best_obj < 0:
    best_obj = 0
    best_solution = (0, 0, 0)

# The required final print (only this line must be exactly in this format)
print(f"FINAL_OBJ: {best_obj}")
