# Self-contained solver for the described LP:
# Maximize Z = 200*x + 300*y
# subject to:
#   x + 2y <= 30
#   5 <= x <= 10
#   5 <= y <= 8

import math

# Problem data
profit_x = 200.0
profit_y = 300.0

x_min, x_max = 5.0, 10.0
y_min, y_max = 5.0, 8.0

# Time constraint function: returns True if (x,y) satisfies x + 2y <= 30 (with tolerance)
def time_feasible(x, y, tol=1e-9):
    return x + 2.0 * y <= 30.0 + tol

# Generate candidate corner points:
candidates = []

# Rectangle corners
candidates += [
    (x_min, y_min),
    (x_min, y_max),
    (x_max, y_min),
    (x_max, y_max),
]

# Intersections of time line x + 2y = 30 with x = x_min and x = x_max
for x_val in (x_min, x_max):
    y_val = (30.0 - x_val) / 2.0
    if y_min - 1e-9 <= y_val <= y_max + 1e-9:
        candidates.append((x_val, y_val))

# Intersections of time line with y = y_min and y = y_max
for y_val in (y_min, y_max):
    x_val = 30.0 - 2.0 * y_val
    if x_min - 1e-9 <= x_val <= x_max + 1e-9:
        candidates.append((x_val, y_val))

# Filter feasible candidates (respecting bounds and time constraint)
feasible = []
for x, y in candidates:
    if x_min - 1e-9 <= x <= x_max + 1e-9 and y_min - 1e-9 <= y <= y_max + 1e-9 and time_feasible(x, y):
        feasible.append((x, y))

# Ensure we have feasible points (should be true)
if not feasible:
    raise RuntimeError("No feasible points found. Check problem data.")

# Evaluate objective and pick best
best_obj = -math.inf
best_point = None
for x, y in feasible:
    obj = profit_x * x + profit_y * y
    if obj > best_obj + 1e-9:
        best_obj = obj
        best_point = (x, y)

# Round objective to nearest integer if it's very close to an integer
if abs(best_obj - round(best_obj)) < 1e-9:
    best_obj = int(round(best_obj))

# Final required print (only this final line must follow the exact format)
print(f"FINAL_OBJ: {best_obj}")
