#!/usr/bin/env python3
# Brute-force integer search for the given ILP:
# Maximize 10*O + 12*A
# s.t. 8*O + 6*A <= 300, A >= 3*O, O >= 3, O,A integers >= 0

best_obj = None
best_solution = None

# Upper bound for O: when A = 0, 8*O <= 300 -> O <= 300//8
max_O = 300 // 8

for O in range(3, max_O + 1):
    # A must be at least 3*O and also satisfy 6*A <= 300 - 8*O
    max_A = (300 - 8 * O) // 6
    min_A = 3 * O
    if min_A > max_A:
        continue
    # For objective 10*O + 12*A, it increases with A, so take A = max_A
    A = max_A
    # Ensure A meets the preference constraint (it should, as checked)
    if A < min_A:
        continue
    obj = 10 * O + 12 * A
    if (best_obj is None) or (obj > best_obj):
        best_obj = obj
        best_solution = (O, A)

# Fallback if no feasible solution found (shouldn't happen here)
if best_obj is None:
    value = 0
else:
    value = best_obj

print(f"FINAL_OBJ: {value}")
