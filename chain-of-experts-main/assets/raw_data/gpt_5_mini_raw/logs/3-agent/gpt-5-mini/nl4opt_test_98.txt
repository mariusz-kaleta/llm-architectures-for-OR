#!/usr/bin/env python3
# Self-contained script to solve the ILP described in the prompt.
# Uses PuLP (CBC) to minimize total vehicle-hours: minimize h + 3b
# Subject to:
#   5h + 8b >= 120
#   7h - 3b >= 0
#   0 <= b <= 10
#   h, b integers >= 0

import pulp

# Define the problem
prob = pulp.LpProblem("Transport_Patients", pulp.LpMinimize)

# Decision variables: integer non-negative
h = pulp.LpVariable("h", lowBound=0, cat="Integer")
b = pulp.LpVariable("b", lowBound=0, upBound=10, cat="Integer")  # b has an upper bound of 10

# Objective: minimize total time (vehicle-hours)
prob += h + 3 * b

# Constraints
prob += 5 * h + 8 * b >= 120        # patient capacity
prob += 7 * h - 3 * b >= 0          # at least 30% of trips by helicopter

# Solve with CBC (default solver in PuLP). Suppress solver output.
solver = pulp.PULP_CBC_CMD(msg=False)
prob.solve(solver)

# Retrieve objective value
obj_val = pulp.value(prob.objective)

# If objective is effectively integer, cast to int for clean output
if obj_val is None:
    final_value = None
else:
    if abs(obj_val - round(obj_val)) < 1e-9:
        final_value = int(round(obj_val))
    else:
        final_value = float(obj_val)

print(f"FINAL_OBJ: {final_value}")
