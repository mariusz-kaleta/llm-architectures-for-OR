import math

# Problem data
# Decision variables:
# x = number of runs of experiment alpha (>= 0)
# y = number of runs of experiment beta (>= 0)
#
# Objective:
# maximize Z = 8x + 10y
#
# Constraints:
# 3x + 5y <= 800   (metal)
# 5x + 4y <= 750   (acid)
# x, y >= 0
#
# We'll compute:
# 1) Continuous optimal solution (solve the two binding constraints analytically)
# 2) Integer optimal solution by simple enumeration (small problem)

# Coefficients and supplies
a1, b1, supply1 = 3, 5, 800   # metal: 3x + 5y <= 800
a2, b2, supply2 = 5, 4, 750   # acid:  5x + 4y <= 750
c1, c2 = 8, 10                # objective: 8x + 10y

# 1) Continuous (fractional) solution: solve
# a1*x + b1*y = supply1
# a2*x + b2*y = supply2
det = a1 * b2 - a2 * b1
if det == 0:
    raise RuntimeError("Linear system is singular; no unique intersection of constraints.")
det_x = supply1 * b2 - b1 * supply2
det_y = a1 * supply2 - supply1 * a2

x_cont = det_x / det
y_cont = det_y / det
z_cont = c1 * x_cont + c2 * y_cont

# 2) Integer solution by enumeration
# Determine reasonable bounds for x
max_x_by_metal = supply1 // a1
max_x_by_acid = supply2 // a2
max_x = int(min(max_x_by_metal, max_x_by_acid))

best_obj = -math.inf
best_x = None
best_y = None

for x in range(0, max_x + 1):
    # compute maximum feasible y for this x (from both constraints)
    rem1 = supply1 - a1 * x
    rem2 = supply2 - a2 * x
    if rem1 < 0 or rem2 < 0:
        continue
    y_max1 = rem1 / b1
    y_max2 = rem2 / b2
    y_max = math.floor(min(y_max1, y_max2))
    if y_max < 0:
        continue
    y = y_max
    obj = c1 * x + c2 * y
    if obj > best_obj:
        best_obj = obj
        best_x = x
        best_y = y

# Print results (informational), then final required line
print(f"Continuous solution (fractional): x = {x_cont:.6f}, y = {y_cont:.6f}, Z = {z_cont:.6f}")
print(f"Integer solution (practical): x = {best_x}, y = {best_y}, Z = {int(best_obj)}")

# Final required print line (must be the last line)
print(f"FINAL_OBJ: {int(best_obj)}")
