#!/usr/bin/env python3
import math

# Problem data
# Maximize 3r + 4s
# Subject to:
# 2r + 3s <= 40
# 3r + 5s <= 70
# r, s >= 0 and integer

coef_r = 3
coef_s = 4

# Compute integer bounds for brute-force search
r_max = min(40 // 2, 70 // 3)  # floor bounds from each constraint
s_max = min(40 // 3, 70 // 5)

best_profit = None
best_r = None
best_s = None

for r in range(r_max + 1):
    for s in range(s_max + 1):
        if 2 * r + 3 * s <= 40 and 3 * r + 5 * s <= 70:
            profit = coef_r * r + coef_s * s
            if (best_profit is None) or (profit > best_profit):
                best_profit = profit
                best_r = r
                best_s = s

# If no feasible solution found (should not happen), set to 0
if best_profit is None:
    best_profit = 0
    best_r = 0
    best_s = 0

# Final output (must be the last printed line in the exact format)
print(f"FINAL_OBJ: {best_profit}")
